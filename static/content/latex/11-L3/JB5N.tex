\begin{enumerate}
    \item \question{Soit $U$ le plan $\Rr^2$ privé de l'origine.
Montrer que l'application 
\begin{eqnarray*}
  f : U&\to& U\\
(x,y)&\mapsto& (x^2-y^2,2xy)
 \end{eqnarray*}
est un difféomorphisme au voisinage de chacun des points de $U$.}
    \item \question{Est-ce un difféomorphisme de $U$ ?}
\end{enumerate}
