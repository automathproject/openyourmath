\texte{
Soit $\Sigma$ le graphe de la fonction $f(x,y) = xy$ de $\Rr^2$ dans $\Rr$.
}
\begin{enumerate}
    \item \question{Est-ce une surface régulière ?}
    \item \question{Déterminer une paramétrisation de $\Sigma$.}
    \item \question{Dans cette paramétrisation, calculer la première et la seconde forme fondamentale.}
    \item \question{Déterminer la courbure de Gauss et la courbure moyenne.}
\end{enumerate}
