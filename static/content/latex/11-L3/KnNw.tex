\chapitre{Champ de vecteurs}
\sousChapitre{Champ de vecteurs}

\contenu{
\texte{
Soit $Y:\Rr^2 \to \Rr^2$ le champ de vecteurs $Y(x,y) =
(x^2y,xy^2)$.
}
\begin{enumerate}
    \item \question{Si $\gamma(t) = (x(t), y(t))$ est solution de
l'équation $\gamma'(t) = Y(\gamma(t))$, que peut-on dire
de la dérivée de
$y/x$ par rapport
à
$t$~?}
    \item \question{Trouver les solutions maximales du champ $Y$
(y compris leur domaine de définition).}
    \item \question{Esquisser le portrait de phase du champ $Y$.}
    \item \question{Déterminer le flot du champ $Y$ (y compris
son domaine de définition).}
\end{enumerate}
}
