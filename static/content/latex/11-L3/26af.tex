\sousChapitre{Sous-variété}

\contenu{
\texte{
Soit $$\mathcal{C}:=\{(x,y)\in\Rr^2, x^3+y^3-3xy=0\}.$$
}
\begin{enumerate}
    \item \question{Au voisinage de quels points de $\mathcal{C}$, cette équation définit-elle $y$ comme fonction de $x$ ?
Quelle est alors la dérivée de cette fonction ?}
    \item \question{Paramétrer $\mathcal{C}$ à l'aide de $t$ tel que $y=tx$.}
    \item \question{Représenter $\mathcal{C}$ avec ses asymptotes.}
    \item \question{Le sous-ensemble $\mathcal{C}$ est-il une sous-variété différentiable de $\Rr^2$ ?}
\end{enumerate}
}
