\contenu{
\texte{
On considère la sphère $S$ unité dans $\Rr^3$.
}
\begin{enumerate}
    \item \question{Déterminer par une expression en coordonnées de
la projection stéréographique d'un ouvert $U$ de $S$ (à déterminer)
depuis le pôle nord sur le plan d'équation $z=0$.
Vérifier qu'il s'agit bien de paramétrage de $U$.}
    \item \question{Fa\^ites de même depuis le pôle sud.}
    \item \question{Montrer que le changement de paramétrage est de classe $\mathcal{C}^\infty$.}
    \item \question{Interpréter les coordonnées sphériques comme un paramétrage de la sphère $S$.}
\end{enumerate}
}
