\chapitre{Sous-variété}
\sousChapitre{Sous-variété}

\contenu{
\texte{
Déterminer, parmi les sous-ensembles définis ci-dessous, ceux
qui sont des sous-variétés:
}
\begin{enumerate}
    \item \question{$\{(x,y,z)\in \Rr^3 \; ; \; x^3+y^3+ z^3 -3xyz =1\}$;}
    \item \question{$\{(x,y) \in \Rr^2 \; ; \; xy=0\}$;}
    \item \question{$\{(x,y,z)\in \Rr^3 \; ; \; x^2+y^2+z^2 =1 \; et \; x^2+y^2-x=0\}$;}
    \item \question{$\{(x,y) \in \Rr^2 \; ; \; y^2 = x^3\}$;}
    \item \question{$\{(x,y,z)\in \Rr^3 \; ; \; x^2 + y^2 = \tan (\alpha ) z^2\}$;}
\end{enumerate}
}
