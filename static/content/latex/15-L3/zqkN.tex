\chapitre{Espace topologique, espace métrique}
\sousChapitre{Espace topologique, espace métrique}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Donner la définition d'un espace topologique $T_1$.}
    \item \question{Montrer qu'un espace topologique $X$  est $T_1$
si et seulement si~: $\forall x\in X\ :\ \{x\}$ est
fermé.}
    \item \question{Soit $X$ un espace topologique contenant un
nombre fini de points. Montrer que si $X$ est $T_1$,
alors sa toplogie est la topologie discrète.}
    \item \question{Soit $X$ un espace topoogique $T_1$ ayant la
propriété~: $\forall x\in X, \forall A\subset X\ :
\text{ $A$ fermé et $x\notin A$ } \Longrightarrow
\exists U,V\text{ ouverts } : x\in U,\  A\subset V,\  U\cap
V = \emptyset$. Montrer que $X$ est $T_2$.}
\end{enumerate}
