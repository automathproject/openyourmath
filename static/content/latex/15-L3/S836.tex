\contenu{
\texte{
On désigne par $E$ l'espace $C([-1,1])$ muni de la norme uniforme et
par $T$ la forme linéaire
 définie par $$Tf=\int_{-1}^1 \sin(\pi t)\ f(t)\ dt $$
pour $f\in E$. Vérifier que $T$ est continue et calculer la norme de $T$.
}
}
