\chapitre{Théorème du point fixe}
\sousChapitre{Théorème du point fixe}

\contenu{
\texte{
Soit $E$ un espace métrique compact, $f$ une application continue de $E$ dans
$E$ et on note $\Omega$ l'ensemble de ses points fixes.
}
\begin{enumerate}
    \item \question{Montrer que $\Omega$ est un compact, qui est non vide dans le cas où
$E=[a,b]$.}
    \item \question{Si $\Omega=\emptyset$, montrer qu'il existe $r>0$ tel que $d(x,f(x))\geq r$
pour tout $x\in E$.}
    \item \question{On suppose que $d(f(x),f(y))<d(x,y)$ pour tous $x\not=y$ de $E$. Montrer que 
$\Omega$ est réduit à un point $a$ et que pour tout choix initial de $x_0\in
E$, la suite récurrente $x_{n+1}=f(x_n)$ converge vers $a$.}
\end{enumerate}
}
