\contenu{
\texte{
Soit $(u_n)$ une série positive divergente telle que $u_n$ décroit vers
$0$ et on pose $A=\{\pm u_1\pm u_2\pm\cdots\pm u_n,\ n\geq1\}$. Montrer
que $\overline A=\R$.
}
}
