\chapitre{Théorème du point fixe}
\sousChapitre{Théorème du point fixe}

\contenu{
\texte{
Pour $x,y\in X=]0,+\infty[$ on pose $\delta(x,y)=\vert \log x-\log y\vert$
}
\begin{enumerate}
    \item \question{Montrer que $X$ muni de $\delta$ est complet alors qu'il ne l'est pas pour
la métrique usuelle de $\Rr$.}
    \item \question{Soit $f$ une application de classe $C^1$ de $X$ dans $X$ vérifiant pour
tout $x\in X$  
$$x\vert f'(x)\vert\leq k f(x)$$
 où $k$ est un réel de $]0,1[$ fixé.
Montrer que $f$ a un seul point fixe dans $X$.}
\end{enumerate}
}
