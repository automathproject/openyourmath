\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Montrer que si deux fonctions continues sur un espace topologique $X$
co\"\i ncident sur un ensemble dense dans $X$, elles sont égales.}
    \item \question{Soit $f$ une fonction réelle définie continue sur $[-1,1]$. Montrer que si
pour tout $n$, $\int_{-1}^1f(x)\ x^n\ dx$ est nulle, alors $f$ est nulle.

(\emph{Indication :} Considérer l'application $g\to \int_{-1}^1f(x)\ g(x)\ dx$.)}
\end{enumerate}
}
