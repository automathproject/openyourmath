\contenu{
\texte{
Soit $X$ un espace topologique, et $D$ un sous-ensemble dense dans $X$.
Montrer qu'il est aussi équivalent de dire
}
\begin{enumerate}
    \item \question{Le complémentaire de $D$ est d'intérieur vide.}
    \item \question{Si $F$ est un fermé contenant $D$, alors $F=X$.}
    \item \question{$D$ rencontre tout ouvert de $X$.}
\end{enumerate}
}
