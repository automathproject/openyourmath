\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{
Soit $f$ une injection continue de $\Rr$ dans $\Rr$.
}
\begin{enumerate}
    \item \question{Montrer à l'aide
du théorème des valeurs intermédiaires que
$f$ est stric\-tement monotone.}
    \item \question{Montrer que l'image par $f$ d'un intervalle ouvert est encore un
intervalle ouvert; en déduire que $f$ est ouverte et donc un homéomorphisme
de $\Rr$ sur
$f(\Rr)$.}
\end{enumerate}
}
