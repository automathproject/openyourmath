\chapitre{Espace métrique complet, espace de Banach}
\sousChapitre{Espace métrique complet, espace de Banach}

\contenu{
\texte{
On considère sur $C^1([0,1],\Rr)$ les normes suivantes :
}
\begin{enumerate}
    \item \question{$\Vert f\Vert=\sup_{[0,1]} \vert f(x)\vert$}
    \item \question{$\Vert f\Vert=\sup_{[0,1]} \vert f'(x)\vert + \vert f(0)\vert$}
    \item \question{$\Vert f\Vert=\sup_{[0,1]} \vert f'(x)+f(x)\vert + \vert f(0)\vert$}
\end{enumerate}
}
