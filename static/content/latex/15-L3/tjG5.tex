\contenu{
\texte{
Soit $(X,d)$ un espace métrique, $A$ et $B$ deux parties de $X$. On pose
$d(A,B)=\inf_{a\in A, b\in B} d(a,b)$.
}
\begin{enumerate}
    \item \question{Si $A$ et $B$ sont disjointes, l'une compacte et l'autre fermée, montrer que
$d(A,B)>0$.}
    \item \question{Montrer, par un contre-exemple, que ceci peut être faux si les deux parties
sont seulement fermées.}
\end{enumerate}
}
