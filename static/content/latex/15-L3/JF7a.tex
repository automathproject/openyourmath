\chapitre{Compacité}

\contenu{
\texte{
Soit $E=C_b(\Rr)$ muni de la norme uniforme; pour $f\in E$, on note
$f_a$ la translatée de $f$ par $a$, ie la fonction $x\to f(x-a)$, et $O_f$
l'ensemble des translatées de $f$.
}
\begin{enumerate}
    \item \question{Montrer que si $f$ est périodique, $O_f$ est compact (considérer
l'application $a\to f_a$).}
    \item \question{Soit $f$ une limite uniforme sur $\Rr$ de fonctions périodiques; montrer
que $O_f$ est précompact.}
    \item \question{On suppose cette fois $O_f$ précompact; on va montrer que $f$ est
uniformé\-ment continue.
  \begin{enumerate}}
    \item \question{De toute suite $(f_{a_n})$ de $O_f$ on peut extraire une sous-suite
convergente dans $E$.}
    \item \question{Si $x_n-y_n$ tend vers $0$, montrer que $(f_{x_n-y_n})$ n'a qu'une
valeur d'adhé\-rence $f$; en déduire que $f(x_n)-f(y_n)$ tend vers $0$.}
    \item \question{Montrer que $f$ est uniformément continue.}
\end{enumerate}
}
