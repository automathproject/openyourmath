\contenu{
\texte{
Soit $A$ un ouvert connexe non-vide de $\Rr^n$ et $a\in A$. Soit $G_a$ l'ensemble des points de $A$
pouvant être reliés à $a$ par un chemin contenu
dans $A$.
}
\begin{enumerate}
    \item \question{Montrer que $G_a$ et $A\setminus G_a$ sont
ouverts.}
    \item \question{En déduire que $A$ est connexe par arcs.}
\end{enumerate}
}
