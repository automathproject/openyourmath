\chapitre{Espace vectoriel normé}
\sousChapitre{Espace vectoriel normé}

\contenu{
\texte{
Soit $E$ l'espace vectoriel des fonctions continues de $[-1,1]$ à
valeurs dans $\R$ muni de la norme $$\|f\|_1=\int_0^1 |f(x)| \,dx.$$
On considère l'application $L:E\rightarrow \R$ définie par $L(f)=f(1)$.
}
\begin{enumerate}
    \item \question{Montrer que $L$ est une application linéaire.}
    \item \question{En considérant les fonctions $f_n:x\mapsto \sqrt n x^n$,
montrer que $L$ n'est pas continue.}
\end{enumerate}
}
