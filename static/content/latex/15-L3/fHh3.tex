\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{
Soit $X$ un espace topologique et $Y$ un espace
topologique séparé. Soit $f,g : X \to Y$  deux
applications continues.
}
\begin{enumerate}
    \item \question{Démontrer que $U = \{x\in X \mid f(x) \neq
g(x)\}$ est un ouvert de $X$.}
    \item \question{Soit $D\subset X$ une partie dense.
Démontrer que si $f\vert_D = g\vert_D$, alors $f=g$.}
\end{enumerate}
}
