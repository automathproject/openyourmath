\chapitre{Espace topologique, espace métrique}
\sousChapitre{Espace topologique, espace métrique}

\contenu{
\texte{
Si $A$ est une partie bornée d'un espace métrique $(E,d)$, on pose
$\hbox{diam} A=\sup_{a,b\in A}d(a,b)$.
}
\begin{enumerate}
    \item \question{Montrer que  $\hbox{diam} A=\hbox{diam} \overline A$.}
    \item \question{Trouver le diamètre de $\{f\in C([0,1])\ ;\  0\leq f\leq 1\}$; de $\{f\in
C([0,1])\ ;\ 0\leq f\leq 1,\ f(0)=0\}$, $C$ étant muni de la métrique $d_1$.}
\end{enumerate}
}
