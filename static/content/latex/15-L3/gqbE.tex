\chapitre{Espace métrique complet, espace de Banach}
\sousChapitre{Espace métrique complet, espace de Banach}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Montrer que l'espace $C([0,1])$ est complet pour la norme uniforme mais pas
pour la norme $||.||_1$.}
    \item \question{Montrer que l'espace $S$ des suites réelles nulles à partir d'un certain rang,
muni de la norme uniforme, n'est pas complet. Trouver un espace métrique complet
contenant $S$ comme sous-espace dense.}
\end{enumerate}
