\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Donner la {\it définition\/} d'un intervalle
dans $\Rr$ et montrer que si $A\subset \Rr$ est
connexe, alors $A$ est un intervalle.}
    \item \question{En utilisant la notion de connexité, montrer
qu'il n'existe pas un homéo\-mor\-phisme $f: \Rr^2
\to \Rr$.}
\end{enumerate}
