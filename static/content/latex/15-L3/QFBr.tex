\chapitre{Espace vectoriel normé}

\contenu{
\texte{
Soit $\left(\begin{array}{cccl}
N:&\R^n&\longrightarrow&\R\\
&x&\longmapsto &\sum_{k=1}^n |\sum_{i=1}^k x_i|
\end{array}\right)$. Montrer que $N$ est une norme.
}
}
