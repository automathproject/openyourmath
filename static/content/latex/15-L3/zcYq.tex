\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{
Soit $(E,d)$ un espace métrique et $A$ une partie de $E$; retrouver les
propriétés de la fonction $d_A : x\to d(x,A)$ :
}
\begin{enumerate}
    \item \question{$d_A$ est $1$-lipschitzienne;
$d(x,A)=d(x,\overline A)$ et $d_A(x)=0$ si et seulement si
$x\in\overline A$.}
    \item \question{Montrer que $\{x\in E\ ; \
d(x, A)< \varepsilon\}$ est un ouvert contenant $A$.}
    \item \question{Montrer que tout fermé de $E$ est un $G_\delta$ et que tout
ouvert est un $F_\sigma$.}
\end{enumerate}
}
