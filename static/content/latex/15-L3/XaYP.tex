\begin{enumerate}
    \item \question{Montrer que la boule unité fermée d'un evn de dimension finie est compacte.}
    \item \question{Soit $E$ un espace vectoriel normé et $F$ un sous-espace de $E$ de dimension
finie. Montrer que $d(x,F)=\inf\{d(x,y),\ y\in F,\ ||y||\leq2||x||\}$; en
déduire que $F$ est fermé dans $E$.}
    \item \question{Soit $(f_n)$ une suite de polyn\^omes qui converge uniformément sur $[0,1]$ vers
une fonction qui n'est pas un polyn\^ome. Montrer que la suite des degrés tend
vers l'infini (raisonner par l'absurde).}
\end{enumerate}
