\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{
Soit $f:E\to\R$ une fonction continue
définie sur un espace topologique $E$. On appelle support (fermé) de $f$,
$S=S(f)=\overline{\{x\in E\ ;\ f(x)\not=0\}}$.
}
\begin{enumerate}
    \item \question{Montrer que $S=\overline{\stackrel\circ{S}}$.}
    \item \question{Réciproque. On suppose $E$ métrique et $A\subset E$ fermé vérifiant
$A=\overline{\stackrel\circ{A}}$. Montrer qu'il existe $f:E\to\R$ une
fonction continue telle que $A=S(f)$.}
\end{enumerate}
}
