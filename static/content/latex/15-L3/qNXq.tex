\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{
Soit $F$ l'application de ${\Rr}^+$ dans ${\Cc}^2$ qui à $x$ associe
$(\exp(2i\pi x), \exp(2i\pi x\sqrt2))$ dont l'image est la courbe $\gamma$.
}
\begin{enumerate}
    \item \question{Montrer que $F$ est continue injective.}
    \item \question{Montrer que l'adhérence de $\gamma$ dans ${\Cc}^2$ est
$\mathbb{S}^1\times\mathbb{S}^1$.}
    \item \question{Montrer que $F^{-1}$ n'est continue en aucun point de $\gamma$.}
\end{enumerate}
}
