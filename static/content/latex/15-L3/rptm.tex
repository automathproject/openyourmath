\chapitre{Compacité}
\sousChapitre{Compacité}

\contenu{
\texte{
Soit $(X,d)$ un espace métrique. On suppose que {\it
toutes\/} les boules fermées (c'est-à-dire les
ensembles de la forme $\{y\in X \mid d(x,y) \le r\}$) sont
compactes.
}
\begin{enumerate}
    \item \question{Démontrer que $X$ est complet.}
    \item \question{Démontrer que si $A \subset X$ est fermé
et borné, alors $A$ est compact.}
\end{enumerate}
}
