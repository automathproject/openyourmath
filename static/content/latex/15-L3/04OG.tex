\contenu{
\texte{
Soit $K$ un convexe compact de $\Rr^2$.
}
\begin{enumerate}
    \item \question{Si $K$ est d'intérieur vide, montrer que $K$ est homéomorphe au segment
$[0,1]$.}
    \item \question{Si $K$ n'est pas d'intérieur vide, montrer que $K$ est homéomorphe au disque
unité fermé en considérant l'application $p(x)=\inf\{a>0\ ;\ {x\over a}\in
K\}$; on montrera que $0$ est un point intérieur, que $\delta ||x||\leq
p(x)\leq C||x||$ puis que $p$ est continue.}
\end{enumerate}
}
