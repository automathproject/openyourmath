\texte{
\label{gijsexopr}
Etablir les propriétés suivantes de l'adhérence d'un ensemble dans un
espace topologique :
}
\begin{enumerate}
    \item \question{$\overline{\overline A}=\overline A$}
    \item \question{Si $A\subset B$ alors $\overline A\subset \overline B$.}
    \item \question{$\overline{A\cup B}=\overline A\cup \overline B$}
\end{enumerate}
}
