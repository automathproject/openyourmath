\chapitre{Espace métrique complet, espace de Banach}
\sousChapitre{Espace métrique complet, espace de Banach}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Soit $f$ un homéomorphisme d'espaces métriques $f:X\to Y$; montrer que $X$
peut être complet sans que $Y$ le soit.}
    \item \question{On suppose de plus que $f$ est uniformément
continue. Montrer que si $Y$ est complet, $X$ l'est aussi.}
    \item \question{On considère $E=\{f\in C^1([0,1])\ ;\ f(0)=0\}$, muni de la métrique
$d(f,g)=\inf(1,\sup|f'(t)-g'(t)|)$. Montrer que $E$ est complet pour cette
métrique.}
\end{enumerate}
}
