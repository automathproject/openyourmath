\contenu{
\texte{
Soit $(E, 
\langle , \rangle )$ un espace euclidien et $S=\{ x \in E ; \Vert x 
\Vert =1  \} .$
}
\begin{enumerate}
    \item \question{Soient $x, y\in E $ et $I$ le segment $[x,y]$. Calculer $S\cap I .$}
    \item \question{Les normes $\Vert \quad \Vert _{1} $ et $\Vert \quad \Vert 
_{\infty } $ de $\Rr ^n$ sont-elles euclidiennes ?}
\end{enumerate}
}
