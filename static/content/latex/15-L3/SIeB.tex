\chapitre{Théorème du point fixe}
\sousChapitre{Théorème du point fixe}

\contenu{
\texte{
On désigne par $E$ l'espace $C([0,1])$ muni de la norme uniforme et l'opérateur
$A$ défini par $$Af(x)=\int_0^x tf(t)\ dt + x\int_x^1 f(t)\ dt$$
pour $f\in E$ et $x\in [0,1]$.
}
\begin{enumerate}
    \item \question{Vérifier que $A$ est continu et calculer sa norme opérateur.}
    \item \question{L'équation $Af=f$ a-t-elle dans $E$ des solutions $f$ non nulles ?}
\end{enumerate}
}
