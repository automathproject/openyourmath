\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Montrer qu'une fonction de $(X,d)$ dans $(Y,\delta)$ n'est pas uniformément
continue, si et seulement si on peut trouver $\varepsilon>0$ et deux suites de
points de
$X$,
$(x_n)$ et $(y_n)$ vérifiant

(i) $d(x_n,y_n)$ tend vers $0$.

(ii) $\delta(f(x_n),f(y_n))\geq\varepsilon$.}
    \item \question{Parmi les fonctions de variable réelle suivantes, lesquelles sont
unifor\-mément continues : $\sin(x^2)$, $x\sin x$, $\sin {1\over
x},\  x\neq0$ ?}
\end{enumerate}
}
