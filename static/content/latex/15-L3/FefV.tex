\chapitre{Continuité, uniforme continuité}
\sousChapitre{Continuité, uniforme continuité}

\contenu{
\texte{
Soit $f$ une application continue de $\R$ dans $\R$. On rappelle que $t$
est une période de $f$ si $f(x+t)=f(x)$ pour tout $x$ réel. Soit $E$ le
groupe des périodes de $f$, supposé non vide et $T=\inf \{t\in E\ ;\
t>0,\}$.
}
\begin{enumerate}
    \item \question{Montrer que si $T=0$ alors $f$ est constante.}
    \item \question{Si $T>0$, $f$ est $T$-périodique et $E={\Zz}.T$.}
\end{enumerate}
}
