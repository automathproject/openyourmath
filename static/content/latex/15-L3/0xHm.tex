\chapitre{Connexité}
\sousChapitre{Connexité}

\contenu{
\texte{
$A$ et $B$ sont des parties d'un espace topologique $X$. Vrai ou faux ?
}
\begin{enumerate}
    \item \question{Si $A$ est connexe, $\partial A$ est connexe ?}
    \item \question{Si $\overline A$ est connexe, $A$ est connexe ?}
    \item \question{Si $A$ et $B$ sont connexes et $A\cap B\not=\emptyset$, $A\cap B$
est connexe ?}
    \item \question{Si $X$ est un evn et $A$ et $B$ convexes avec $A\cap B\not=\emptyset$,
$A\cap B$ est connexe ?}
    \item \question{Si $A$ et $B$ sont connexes, $A\cup B$
est connexe ?}
    \item \question{Soit $f$ continue de $X$ dans $Y$ espace topologique. Si $A$ est connexe
par arcs , $f(A)$ est connexe par arcs ?}
    \item \question{Soit $f$ continue de $X$ dans $Y$ evn. Si $A$ est convexe, $f(A)$ est
convexe ?}
\end{enumerate}
}
