\contenu{
\texte{
Soit $X$ un espace topologique compact et $f_1,f_2,\ldots,f_n$, $n$ fonctions
continues réelles qui séparent les points de $X$. Montrer que $X$ est homéomorphe
à une partie de $\Rr^n$.
}
}
