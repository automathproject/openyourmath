\contenu{
\texte{
Soit $E=\Rr^n$ muni de sa norme euclidienne et $f$ une application continue de
$[0,1]$ dans
$\Rr^n$.

Montrer que  $||\int_{[0,1]}f(t)\ dt||\leq \int_{[0,1]}||f(t)||\ dt$.
}
}
