\chapitre{Espace vectoriel}
\sousChapitre{Dimension}

\contenu{
\texte{
Soit $E$ un espace vectoriel de dimension $3$ et $f \in \mathcal{L} (E)$ telle
que $f^2 \neq 0$ et $f^3 = 0$. Soit $x_0 \in E / f^2 (x_0) \neq 0$.
}
\begin{enumerate}
    \item \question{Montrer que $ (x_0, f (x_0), f^2 (x_0))$ est une base.}
    \item \question{Montrer que l'ensemble des endomorphismes qui commutent avec $f$ est un
sous-espace vectoriel de $\mathcal{L} (E)$ de base $ (id, f, f^2)$.}
\end{enumerate}
}
