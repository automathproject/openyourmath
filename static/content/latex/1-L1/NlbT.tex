\texte{
Compl\'eter les pointill\'es par le connecteur logique qui s'impose : $\Leftrightarrow ,\ \Leftarrow,\ \Rightarrow.$
}
\begin{enumerate}
    \item \question{$  x\in \Rr\ \ x^2=4 \ \ldots\ldots \ x=2$ ;}
\reponse{$\Leftarrow$}
    \item \question{$  z\in \Cc\ \ z=\overline{z} \ \ldots\ldots \ z\in\Rr$ ;}
\reponse{$\Leftrightarrow$}
    \item \question{$ x\in \Rr\ \ x=\pi \ \ldots\ldots \ e^{2ix}=1$.}
\reponse{$\Rightarrow$}
\end{enumerate}
}
