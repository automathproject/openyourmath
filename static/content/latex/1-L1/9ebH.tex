\chapitre{Polynôme, fraction rationnelle}
\sousChapitre{Pgcd}

\contenu{
\texte{
Soit les polynômes de $\mathbb{R}[X]$~:
\begin{eqnarray*}
A & = & (X+3)^2(X+1)(X^2+1)^3 \\
B & = & (X+3)^2(X+2)^2(X^2+1) \\
C & = & (X+3)(X+2)(X^2+1)^2.
\end{eqnarray*}
}
\begin{enumerate}
    \item \question{Combien $A$ possède-t-il de diviseurs normalisés~? et $B$~? et $C$~?}
    \item \question{Écrire le pgcd et le ppcm de $A$ et~$B$.}
    \item \question{Écrire le pgcd et le ppcm des trois polynômes~$A$, $B$ et~$C$.}
\end{enumerate}
}
