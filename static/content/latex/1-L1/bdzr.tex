    \item \question{Soit $(x,y,z)\in \Nn^3$ .
Montrer que :
\begin{eqnarray*}
    x^2+y^2=z^2\Leftrightarrow \exists (x',y',z')\in \Nn^3,\exists n\in \Nn \text{ tq } \\
     \text{ pgcd}(x',y',z')=1 \\
    {x'}^2+{y'}^2={z'}^2 \\
    x=nx' \text{ et }y=ny' \text{ et } z=nz'.
\end{eqnarray*}}
    \item \question{Soit$(x,y,z)\in \Nn^3$ tels que $x^2+y^2=z^2$.
On suppose que pgcd$(x,y,z)=1$
    \begin{enumerate}}
    \item \question{Montrer que $x$ et $y$ ne sont pas de m\^emes parit\'e.}
    \item \question{On suppose $x$ pair et $y$ impair.
    On pose :
    $$x=2u,\ z-y=2v,\ z+y=2w$$
    avec $(u,v)\in \N^*$. Montrer que $v$ et $w$ sont premiers entre eux.}
    \item \question{Montrer que
    $$x=2mn,\ y=m^2-n^2,\ z=m^2+n^2$$
    avec $m$ et $n$ entiers naturels de parit\'e diff\'erentes.}
    \item \question{Montrer que si
    $$x=2mn,\ y=m^2-n^2,\ z=m^2+n^2$$
    alors
    $$x^2+y^2=z^2.$$}
