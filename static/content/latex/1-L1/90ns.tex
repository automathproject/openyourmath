\contenu{
\texte{
Pour tout nombre complexe $Z$, on pose $P(Z)=Z^4-1$.
}
\begin{enumerate}
    \item \question{Factoriser $P(Z)$ et en déduire les solutions dans $\mathbb{C}$ de 
l'équation $P(Z)=0$.}
    \item \question{Déduire de 1. les solutions de l'équation d'inconnue $z$ :
$$((2z+1)/(z-1))^4=1$$}
\end{enumerate}
}
