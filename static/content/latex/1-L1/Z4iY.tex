\contenu{
\texte{
Soit $E$ l'ensemble des polyn\^omes \`a coefficients r\'eels de
degr\'e inf\'erieur ou \'egal \`a $n$. \'Etant donn\'e deux r\'eels $a,b$, on
note $u$ l'application qui \`a tout polyn\^ome $P$ de $E$ fait
correspondre 
$$ u(P) = (X-a)(X-b)P' -n XP $$
}
\begin{enumerate}
    \item \question{V\'erifier que $u$ est un endomorphisme de $E$.}
    \item \question{Trouver les valeurs propres et vecteurs propres de $u$.}
    \item \question{Trouver le noyau de $u$, ainsi que l'ensemble des
polyn\^omes qui v\'erifient $u(P) = 1$.}
\end{enumerate}
}
