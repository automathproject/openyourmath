\texte{
Soit $E$ un $ K$-ev de dimension finie et $p_1,\dots,p_n$ des projecteurs
tels que $p_1 + \dots + p_n = \mathrm{id}_E$.
}
\begin{enumerate}
    \item \question{Montrer que $\mathrm{tr}(p_i) = \mathrm{rg}(p_i)$.}
    \item \question{Montrer que $E = \Im(p_1) \oplus \dots \oplus \Im(p_n)$.}
\end{enumerate}
}
