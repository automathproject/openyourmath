\chapitre{Injection, surjection, bijection}
\sousChapitre{Application}

\contenu{
\texte{
Soient $E \xrightarrow{f} F \xrightarrow{g} G
\xrightarrow{h} E$ trois applications telles que $h\circ g\circ f$ et
$g\circ f\circ h$ sont injectives et $f\circ h\circ g$ est surjective.
Montrer que $f,g,h$ sont bijectives.
}
}
