\begin{enumerate}
    \item \question{Écrire l'ensemble de définition de chacune des fonctions numériques
suivantes : $x\mapsto\sqrt x$, $x\mapsto\frac{1}{x-1}$, $x\mapsto\sqrt 
x+\frac{1}{x-1}$.}
    \item \question{Simplifier $[1,3]\cap[2,4]$ et $[1,3]\cup[2,4]$.}
    \item \question{Pour tout $n\in\Nn$, on note $n\Zz$ l'ensemble des entiers 
relatifs multiples de $n$ : $n\Zz=\{np \mid\nolinebreak p\in\Zz\}$. 
Simplifier $2\Zz\cap3\Zz$.}
\end{enumerate}
