\chapitre{Espace vectoriel}
\sousChapitre{Base}

\contenu{
\texte{
Soit
$$ E =\left\{ f_{\alpha,A} \in {\mathcal F}(\R,\R);  (\alpha,A)\in \R^2,\quad f_{\alpha,A}(x) =
A\cos(x+\alpha)\right\}.
$$
Montrer que $E$ est un sous-espace vectoriel de    ${\mathcal F}(\R,\R)$ et en donner une base.
}
}
