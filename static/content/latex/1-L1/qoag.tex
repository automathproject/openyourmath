\contenu{
\texte{
Soit $n \in \N$. Calculer $\displaystyle\sum_{i+j=n}ij$ et $\displaystyle\sum_{i+j+k=n}ijk$.
}
\reponse{
$\frac {n(n^2-1)}6$, $\frac {n(n^2-1)(3n^2-12)}{360}$.
}
}
