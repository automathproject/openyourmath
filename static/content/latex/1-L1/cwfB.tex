\begin{enumerate}
    \item \question{Soit $\theta$ et $\theta'$ deux nombres réels.
En utilisant les formules d'addition pour le calcul de $\cos(\theta+\theta')$ et $\sin(\theta+\theta')$,
montrer que $$e^{i(\theta+\theta')}=e^{i\theta} e^{i\theta'}.$$}
    \item \question{Soit $z$ et $z'$ deux nombres complexes. Déterminer $\arg (zz')$ en fonction de $\arg (z)$ et $\arg(z')$.}
    \item \question{En déduire l'écriture polaire de $z^n$, puis ses parties réelles et imaginaires.}
    \item \question{Déterminer $\cos (5\theta)$ en fonction de $\cos(\theta)$ et de $\sin(\theta)$.}
\end{enumerate}
