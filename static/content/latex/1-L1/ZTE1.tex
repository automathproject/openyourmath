\contenu{
\texte{
\'Ecrire la n\'egation des assertions suivantes o\`u $P, Q, R, S$ sont des
propositions.
}
\begin{enumerate}
    \item \question{$P \Rightarrow Q$,}
\reponse{$P$ et non $Q$;}
    \item \question{$P$ et non $Q$,}
\reponse{``non P ou $Q$" ce qui la même chose que ``$P \Rightarrow Q$";}
    \item \question{$P$ et ($Q$ et $R$),}
\reponse{(non $P$) ou ((non $Q$) ou (non $R$)) (on peut supprimer les parenthèses);}
    \item \question{$P$ ou ($Q$ et $R$),}
\reponse{non $P$ et (non $Q$ ou non $R$) (ici les parenthèses sont importantes);}
    \item \question{($P$ et $Q$) $\Rightarrow \ (R \Rightarrow S)$.}
\reponse{$P$ et $Q$ et $R$ et non $S$;}
\end{enumerate}
}
