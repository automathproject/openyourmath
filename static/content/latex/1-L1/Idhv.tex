\chapitre{Matrice}
\sousChapitre{Propriétés élémentaires, généralités}

\contenu{
\texte{
Soit $A$ une matrice carr\'ee d'ordre $n$ v\'erifiant pour
tout $i\in\{1,\ldots,n\}$
$$ | a_{i,i} | > |a_{i,1}| + |a_{i,2}| +\ldots+ |a_{i,i-1}|+
|a_{i,i+1}| +\ldots+|a_{i,n}|.$$
Montrer que $A$ est inversible.
}
}
