\chapitre{Nombres complexes}
\sousChapitre{Forme cartésienne, forme polaire}

\contenu{
\texte{
Calculer le module et l'argument des nombres complexes suivants, ainsi
que de leurs conjugués :
}
\begin{enumerate}
    \item \question{$1+i(1+\sqrt2)$.}
    \item \question{$\sqrt{10+2\sqrt5}+i(1-\sqrt5)$.}
    \item \question{$\frac{\tan\varphi -i}{\tan\varphi +i}$ où $\varphi$ est un angle donné.}
\reponse{
$\rho =\sqrt{4+2\sqrt2}$, $\theta ={3\pi \over8}$; \qquad
$\rho =4$, $\theta =-{\pi \over 10}$; \qquad
$\rho =1$, $\theta =2\varphi +\pi $.
}
\end{enumerate}
}
