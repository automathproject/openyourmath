\contenu{
\texte{
Soit $E$ l'ensemble des fonctions de $\Nn$ dans $\left\{ 1, 2, 3\right\}$.
Pour $i = 1, 2, 3$ on pose $A_i = \left\{ f \in E / f (0) = i \right\}$.
Montrer que les $A_i$ forment une partition de $E$.
}
}
