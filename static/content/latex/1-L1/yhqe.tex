\texte{
Soient $f,g \in \mathcal{L}(E)$ tels que $f\circ g \circ f = f$ et $g \circ f\circ g = g$.
}
\begin{enumerate}
    \item \question{Montrer que $E = \mathrm{Ker} f \oplus \Im g$.}
\reponse{$\vec u = (\vec u - g\circ f(\vec u)) + g\circ f(\vec u)$.}
    \item \question{Montrer que $f(\Im g) = \Im f$.}
\reponse{$f(\Im g) \subset \Im f$.\par
             $f = (f\circ g) \circ f  \Rightarrow  \Im f \subset \Im(f\circ g)= f(\Im g)$.}
\end{enumerate}
}
