\contenu{
\texte{
Soit $P \in {\Z[X]}$. On appelle {\it contenu de $P$\/} le pgcd des coefficients
de $P$ (notation : cont($P$)).
}
\begin{enumerate}
    \item \question{Soient $P,Q \in {\Z[X]}$ avec cont$(P)=1$, et $R = PQ$. Soit $p$ un facteur
    premier de cont($R$).
  \begin{enumerate}}
    \item \question{Si $p$ est premier avec le coefficient constant de $P$, D{\'e}montrer que
          $p$ divise tous les coefficients de $Q$.}
    \item \question{Si $p$ divise le coefficient constant de $P$, se ramener au cas
          pr{\'e}c{\'e}dent.}
    \item \question{En d{\'e}duire que cont($Q$) = cont($R$).}
\end{enumerate}
}
