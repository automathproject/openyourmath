\chapitre{Nombres complexes}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $u$, $v$ des nombres complexes non r\'eels tels que $|u|=|v|=1$ et $uv\not= -1$.
Montrer que $\frac{u+v}{1+uv}$ est r\'eel.
}
}
