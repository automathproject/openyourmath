\chapitre{Arithmétique dans Z}
\sousChapitre{Divisibilité, division euclidienne}

\contenu{
\texte{
Soit $p \in \N^*$ premier et $n \in \N^*$, $n < p$.
Montrer que $\frac {(p-1)(p-2) \dots (p-n)}{n!} - (-1)^n$ est  un entier
divisible par~$p$.
}
}
