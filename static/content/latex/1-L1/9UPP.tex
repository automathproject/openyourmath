\contenu{
\texte{
Soit $ E $ un espace vectoriel de dimension finie $ \varphi  $ et
$ \psi  $ deux applications lin\'eaires  de $ E $ dans lui-m\^eme
telles que $ \varphi \circ \psi =\hbox{id}_{E} .$ Montrer que $ \psi \circ
\varphi =\hbox{id}_{E} .$
}
}
