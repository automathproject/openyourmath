\chapitre{Arithmétique dans Z}
\sousChapitre{Divisibilité, division euclidienne}

\contenu{
\texte{
Soient $m\ge 1$ et $n\ge 2$ des entiers ; montrer que :
}
\begin{enumerate}
    \item \question{$n-1 |n^m-1$ ;}
    \item \question{$(n-1)^2|n^m-1 \text{ si et seulement si } n-1|m$.}
\end{enumerate}
}
