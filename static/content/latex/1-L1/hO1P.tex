\texte{
Soit $f : {\C\setminus\{i\}} \to {\C\setminus\{1\}}, z \mapsto {\frac {z+i}{z-i}}$
}
\begin{enumerate}
    \item \question{Montrer que $f$ est bijective.}
    \item \question{D{\'e}terminer $f(\R)$, $f(\mathbb{U}\setminus\{i\})$, $f(i\R\setminus\{i\})$.}
\reponse{
$\mathbb{U}\setminus\{1\}$, $i\R$, $\R\setminus\{1\}$.
}
\end{enumerate}
}
