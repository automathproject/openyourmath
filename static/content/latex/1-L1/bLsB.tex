\texte{
Soit $\cal R$ une relation sur $E$ r{\'e}flexive et transitive. On d{\'e}finit la
relation : $x \sim y \iff x {\cal R} y$ et $y {\cal R} x$.
}

    \item \question{Montrer que $\sim$ est une relation d'{\'e}quivalence sur $E$.\par

    choisis.}

\end{enumerate}
