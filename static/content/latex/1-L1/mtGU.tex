\texte{
Pour $n\in \mathbb{N}$ on considère la propriété suivante~:
$$P_{n}: \quad 2^n>n^2$$
}
\begin{enumerate}
    \item \question{Pour quelles valeurs de $n$ l'implication $P_{n} \Longrightarrow P_{n+1}$
est-elle vraie~?}
    \item \question{Pour quelles valeurs de $n$ la propriété $P_{n}$ est-elle vraie~?}
\end{enumerate}
}
