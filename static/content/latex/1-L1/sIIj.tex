\sousChapitre{Racine carrée, équation du second degré}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Soient $u,v \in \C$. Montrer que $|u+v|^2 + |u-v|^2 = 2|u|^2 + 2|v|^2$.}
    \item \question{Soient $\alpha,\beta \in \C$, $m = \frac {\alpha+\beta}2$ et
    $\mu$ une racine carr{\'e}e de $\alpha\beta$.
    Montrer que $|\alpha|+|\beta|  = |m+\mu| + |m-\mu|$.}
\reponse{
{\'E}l{\'e}ver au carr{\'e} :
             $|\alpha|^2 + |\beta|^2 + 2\underbrace{|\alpha\beta|}_{|\mu|^2} =
             \underbrace{|m-\mu|^2 + |m+\mu|^2}_{2|m|^2 + 2|\mu|^2} +
             2\underbrace{|m^2-\mu^2|}_{|\alpha-\beta|^2/4}$.
}
\end{enumerate}
