\begin{enumerate}
    \item \question{Soit $A$ une matrice triangulaire à diagonale nulle.
    Montrer que $A$ est nilpotente.}
    \item \question{Soit $A \in \mathcal{M}_n(K)$ une matrice nilpotente d'indice $n$ et $\phi$
    l'endomorphisme de $ K^n$ associé.

    On note $E_i = \mathrm{Ker} \phi^i$, et $\vec e_i$ un vecteur quelconque choisi
    dans $E_i \setminus E_{i-1}$ $(\vec e_1 \in E_1 \setminus \{\vec 0 \})$.
  \begin{enumerate}}
    \item \question{Justifier l'existence de $\vec e_i$.}
    \item \question{Montrer que la famille $(\vec e_i)$ est une base de $ K^n$.}
    \item \question{En déduire que $A$ est semblable à une matrice triangulaire à diagonale nulle.}
\end{enumerate}
