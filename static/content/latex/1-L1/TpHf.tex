\contenu{
\texte{
Donner une condition sur $\lambda$ pour que l'{\'e}quation :
$x^4-2x^3+\lambda x^2+2x-1 = 0$ ait une racine au moins triple.
}
\reponse{
$\lambda = 0$, $x=1$.
}
