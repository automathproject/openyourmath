\contenu{
\texte{
Soient $m,n\in\Nn^*$ et $p\in\Nn$. En utilisant la formule du bin\^ome,
d\'emontrer que $m^{2p+1}+n^{2p+1}$ est divisible par $m+n$.
}
}
