\contenu{
\texte{
Montrer que, pour $a$ réel positif et $n$ entier naturel donnés, $(1+a)^n\geq1+na$.
}
\reponse{
$(1+a)^n=(1+a)\ldots(1+a)=1+na+...\geq1+na$.
}
}
