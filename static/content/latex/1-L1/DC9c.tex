\texte{
Soient $a,b \in \N$. On pose $m = \text{ppcm}(a,b)$ et
$d = \text{pgcd}(a,b)$.
}
\begin{enumerate}
    \item \question{Soit $x$ un multiple commun {\`a} $a$ et $b$. En {\'e}crivant la division
     euclidienne de $x$ par $m$, montrer que $m \mid x$.}
    \item \question{Soit $x$ un diviseur commun {\`a} $a$ et $b$. Montrer que $\text{ppcm}(x,d)$ est
     aussi un diviseur commun {\`a} $a$ et $b$. En d{\'e}duire $x \mid d$.}
    \item \question{Comment qualifier $m$ et $d$ pour la relation d'ordre de divisibilit{\'e} ?}
\end{enumerate}
}
