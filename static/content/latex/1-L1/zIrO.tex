\contenu{
\texte{
Soit $A$ une matrice carr\'ee.
}
\begin{enumerate}
    \item \question{Montrer que $A^2 = I$ si et seulement si $(I-A)(I+A)=0$.
Montrer que dans ce cas $A$ est inversible.}
    \item \question{Montrer que si $A$ est idempotente ($A^2 = A$), alors
$B=I-A$ l'est aussi et que $AB=BA=0$.}
    \item \question{Montrer que $I$ est la seule matrice idempotente
inversible.}
\end{enumerate}
}
