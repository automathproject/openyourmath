\contenu{
\texte{
Calculer le pgcd des nombres suivants :
}
\begin{enumerate}
    \item \question{126, 230.}
\reponse{$126 = 2.3^2.7$ et $230 = 2.5.23$ donc le pgcd de $126$ et $230$ est $2$.}
    \item \question{390, 720, 450.}
\reponse{$390 = 2.3.5.13$, $720 = 2^4.3^2.5$, $450 = 2.3^2.5^2$ et donc le pgcd de ces trois nombres est
$2.3.5=30$.}
    \item \question{180, 606, 750.}
\reponse{$\pgcd(180,606,750) = 6$.}
\end{enumerate}
