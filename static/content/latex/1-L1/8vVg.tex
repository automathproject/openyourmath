\chapitre{Arithmétique dans Z}
\sousChapitre{Sous-groupes de Z}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Déterminer $2\mathbb{Z}\cup 3\mathbb{Z}$. Est-ce un sous-groupe de
$\mathbb{Z}$~?}
    \item \question{Déterminer~:
$7\mathbb{Z}\cup 49\mathbb{Z}$~; $5\mathbb{Z}\cup 45\mathbb{Z}$~;
$\bigcup_{n=1}^{28} 2^n \mathbb{Z}$.
Ces ensembles sont-ils des sous-groupes de $\mathbb{Z}$~?}
    \item \question{Trouver une condition nécessaire et suffisante pour qu'une réunion de
deux sous-groupes de $\mathbb{Z}$ soit un sous-groupe de~$\mathbb{Z}$.}
\end{enumerate}
