\chapitre{Polynôme, fraction rationnelle}

\contenu{
\texte{
Pour $n\in \mathbb{N}$, montrer que le polynôme $(X-1)^{n+2}+X^{2n+1}$ est 
divisible par $X^2-X+1$. Trouver le quotient si $n=2$.
}
}
