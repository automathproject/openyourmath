\contenu{
\texte{
Soit $P \in {\C[X]}$ de racines $x_1,x_2,\dots,x_n$ avec les multiplicit{\'e}s
$m_1,m_2,\dots,m_n$.
}
\begin{enumerate}
    \item \question{D{\'e}composer en {\'e}l{\'e}ments simples $\frac {P'}P$.}
\reponse{$P' = \sum_{i=1}^n \frac {m_iP}{X-x_i}  \Rightarrow 
             \frac {P'}P = \sum_{i=1}^n \frac {m_i}{X-x_i}$.}
    \item \question{En d{\'e}duire que les racines de $P'$ sont dans l'enveloppe convexe
    de $x_1,\dots,x_n$.}
\reponse{$P'(z) = 0 \Leftrightarrow \sum_{i=1}^n m_i\frac {\overline{z-x_i}}{|z-x_i|^2} = 0
             \Leftrightarrow z = \text{Bar}\biggl( x_i, \frac {m_i}{|z-x_i|^2} \biggr)$.}
\end{enumerate}
}
