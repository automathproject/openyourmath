\chapitre{Application linéaire}
\sousChapitre{Image et noyau, théorème du rang}

\contenu{
\texte{
Soient $F$ et $G$ deux sous-espaces vectoriels de $\Rr^n$, 
on d\'efinit l'application 
$f:F \times G \rightarrow \Rr^n$ par $f(x_1,x_2)= x_1+x_2$.
}
\begin{enumerate}
    \item \question{Montrer que f est lin\'eaire.}
    \item \question{D\'eterminer le noyau et l'image de $f$.}
\end{enumerate}
}
