\chapitre{Injection, surjection, bijection}
\sousChapitre{Injection, surjection}

\contenu{
\texte{
Soit $f : X \rightarrow Y$.On note ${\hat{f}} : {\mathcal{P} (X)} \to {\mathcal{P} (Y)},
{A} \mapsto {f (A)}$ et ${\tilde{f}} : {\mathcal{P} (Y)} \to {\mathcal{P} (X)},
{B} \mapsto {f^{-1} (B)}$. Montrer que :
}
\begin{enumerate}
    \item \question{$f$ est injective ssi $\hat{f}$ est injective.}
    \item \question{$f$ est surjective ssi $\tilde{f}$ est injective.}
\end{enumerate}
}
