\contenu{
\texte{
Soient $n \in \N^*$, $\omega = e^{2i\pi/n}$ et
$Z = \sum_{k=0}^{n-1}\,\omega^{k^2}$. On demande de calculer $|Z|^2$.
Pour cela $\ldots$
}
\begin{enumerate}
    \item \question{{\'E}crire $|Z|^2$ comme une somme double.}
    \item \question{Regrouper les termes diagonalement en tenant compte de la p{\'e}riodicit{\'e} de
    la fonction $k \mapsto \omega^k$.}
    \item \question{Terminer le calcul.}
\reponse{
$n$ impair $ \Rightarrow  |Z|^2 = n$,\par
         $n$ pair $ \Rightarrow  |Z|^2 = n(1+(-1)^{n/2})$.
}
\end{enumerate}
}
