\chapitre{Polynôme, fraction rationnelle}
\sousChapitre{Racine, décomposition en facteurs irréductibles}

\contenu{
\texte{
Soit $P \in {\Z}[X]$ et $n \in {\Z}$ ; notons $m=P(n)$ ;
 (deg($P$) $\ge 1$).
}
\begin{enumerate}
    \item \question{Montrer que : $\forall k \in {\Z}, m$ divise $P(n+km)$.}
    \item \question{Montrer qu'il n'existe pas de polyn\^ome $P$ dans ${\Z}[X]$, non constant,
    tel que pour tout $n \in {\Z}$, $P(n)$ soit premier.}
\end{enumerate}
}
