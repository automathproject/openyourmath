\contenu{
\texte{
Soient $E,F$ deux espaces vectoriels et $f : E \to F$ linéaire.
}
\begin{enumerate}
    \item \question{Montrer que $f$ est injective si et seulement si $f$ transforme
    toute famille libre de $E$ en une famille libre de~$F$.}
    \item \question{Montrer que $f$ est surjective si et seulement s'il existe une famille
    génératrice de $E$ transformée par $f$ en une famille génératrice de $F$.}
\end{enumerate}
}
