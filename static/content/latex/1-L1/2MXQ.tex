\contenu{
\texte{
Soit $f : E \to F$ une application, $A,A' \subset E$ et $B,B' \subset F$.
}
\begin{enumerate}
    \item \question{Simplifier $f(f^{-1}(f(A)))$ et $f^{-1}(f(f^{-1}(B)))$.}
    \item \question{Montrer que $f(A\cap f^{-1}(B)) = f(A)\cap B$.}
    \item \question{Comparer $f(A \mathop{\Delta} A')$ et $f(A) \mathop{\Delta} f(A')$.}
    \item \question{Comparer $f^{-1}(B \mathop{\Delta} B')$ et
    $f^{-1}(B) \mathop{\Delta} f^{-1}(B')$.}
    \item \question{A quelle condition sur $f$ a-t-on :
    $\forall\ A \subset E,\ f(E\setminus A) = F\setminus f(A)$ ?}
\end{enumerate}
}
