\contenu{
\texte{
Soit $E$ un $ K$-ev, $E_1,\dots,E_n$ des sev tels que
$E_1 \oplus \dots \oplus E_n = E$, $F$ un autre sev de $E$, et
$F_i = E_i \cap F$.
}
\begin{enumerate}
    \item \question{Montrer que la somme $G = F_1 + \dots + F_n$ est directe.}
    \item \question{Comparer $F$ et $G$.}
\end{enumerate}
}
