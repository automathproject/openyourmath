\texte{
Calculer $A^n$ pour tout $n\in\mathbb{Z}$, avec successivement
$$
A = \left(\begin{array}{cc}\cos(a) & -\sin(a)\\ \sin(a) & \cos(a)\end{array}\right), \quad\quad\left(\begin{array}{cc}\cosh(a) & \sinh(a)\\ \sinh(a) & \cosh(a)\end{array}\right).
$$
}
