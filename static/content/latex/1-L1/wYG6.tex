\contenu{
\texte{
Soit $\theta$ un nombre réel.
}
\begin{enumerate}
    \item \question{\`A l'aide des formules d'addition, calculer $\cos 2\theta$ et $\sin 2\theta$ en fonction de $\cos\theta$ et $\sin\theta$.}
    \item \question{Vérifier la validité des formules obtenues pour $\theta=\pi/2$ et $\theta=\pi/3$.}
    \item \question{Calculer $\cos 3\theta$ et $\sin 3\theta$ en fonction de $\cos\theta$ et $\sin\theta$.}
    \item \question{Vérifier la validité des formules obtenues pour $\theta=\pi/2$ et $\theta=\pi/3$.}
\end{enumerate}
