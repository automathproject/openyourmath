\texte{
Pour chaque couple de matrices $(A_i, b_i)$, $1\leq i\leq 5$, ci-dessous
}
\begin{enumerate}
    \item \question{donner la nature de l'ensemble des solutions du syst\`eme $A_i X = b_i$ ;}
    \item \question{donner une repr\'esentation param\'etrique de l'ensemble des solutions de $A_i X = b_i$ ;}
    \item \question{donner une base de l'image et une base du noyau de $A_i$.}
\end{enumerate}
}
