\chapitre{Espace vectoriel}
\sousChapitre{Système de vecteurs}

\contenu{
\texte{
On considère les vecteurs de $ K^3$ :
$\vec a = (1, 2, 1)$,
$\vec b = (1, 3, 2)$,
$\vec c = (1, 1, 0)$,
$\vec d = (3, 8, 5)$.

Soient $F = \text{vect}(\vec a,\vec b\,)$ et
       $G = \text{vect}(\vec c,\vec d\,)$.
Comparer $F$ et $G$.
}
\reponse{
$F = G$.
}
}
