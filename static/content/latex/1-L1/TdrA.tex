\contenu{
\texte{
Soit $E$ de dimension finie et $f \in \mathcal{L} (E)$. Montrer l'\'equivalence
des trois propri\'et\'es :
}
\begin{enumerate}
    \item \question{[(i)] $\ker f = \ker f^2$.}
    \item \question{[(ii)] $\text{Im}f = \text{Im}f^2$.}
    \item \question{[(iii)] $E = \ker f \oplus \text{Im}f$.}
\end{enumerate}
}
