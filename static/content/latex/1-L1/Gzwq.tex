\contenu{
\texte{
D\'eterminer le module et l'argument de
$\frac{1+i}{1-i}$. Calculer $(\frac{1+i}{1-i})^{32}$.
}
\reponse{
$$\frac{1+i}{1-i} = \frac{\sqrt{2}e^{i\pi/4}}{\sqrt{2}e^{-i\pi/4}}
= e^{i\pi/2} = i.$$ On remarque $1 = i^0 = i^4 = i^8 = \cdots =
i^{32}$.
}
}
