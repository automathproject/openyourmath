\chapitre{Injection, surjection, bijection}

\contenu{
\texte{
Soit $f : X \rightarrow Y$. Montrer que les trois propositions suivantes sont
\'equivalentes :
}
\begin{enumerate}
    \item \question{[i. ] $f$ est injective.}
    \item \question{[ii. ] $\forall A, B \subset X \, \, f (A \cap B) = f (A) \cap f (B)$.}
    \item \question{[iii. ]$\forall A, B \subset X \, \, A \cap B = \emptyset \Rrightarrow
f (A) \cap f (B) = \emptyset$.}
\end{enumerate}
}
