\texte{
Donner une CNS
sur $p,q \in \C$ pour que $X^8 + X^4 + 1$ divise $X^{8n} + pX^{4n} +
q$ ($n \in \N^*$ fix{\'e}).
}
\reponse{
$\iff X^2 + X + 1\ |\ X^{2n} + pX^n + q \iff j^{2n} + pj^n + q = 0$.
}
