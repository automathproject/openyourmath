\chapitre{Espace vectoriel}
\sousChapitre{Système de vecteurs}

\contenu{
\texte{
Soit $E$ un espace vectoriel sur $\mathbb{R}$ et $x,y,z,t$ une famille libre
d'éléments de $E$, les familles suivantes sont-elles libres~?
}
\begin{enumerate}
    \item \question{$x,2y,z$.}
    \item \question{$x,z$.}
    \item \question{$x,2x+t,t$.}
    \item \question{$3x+z, z, y+z$.}
    \item \question{$2x+y,x-3y,t, y-x$.}
\end{enumerate}
}
