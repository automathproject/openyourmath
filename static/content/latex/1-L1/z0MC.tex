\texte{
Ecrire la n\'egation des phrases suivantes :
}
\begin{enumerate}
    \item \question{$ (\forall x) (\exists n)/ (x \leq n)$.}
    \item \question{$ (\exists M) / (\forall n) (\left|u_n\right| \leq M)$.}
    \item \question{$ (\forall x) (\forall y) (xy = yx)$.}
    \item \question{$ (\forall x) (\exists y)/ (yxy^{-1} = x)$.}
    \item \question{$ (\forall \epsilon>0) (\exists N \in \Nn)/ (\forall n \geq N) (\left|u_n\right|
< \epsilon)$.}
    \item \question{$(\forall x \in \Rr) (\forall \epsilon >0) (\exists \alpha >0)/ (\forall f \in \mathcal{F})
 (\forall y \in \Rr) (\left|x-y\right|<\alpha \Rightarrow \left|f (x)-f (y)\right|< \epsilon)$.}
\end{enumerate}
}
