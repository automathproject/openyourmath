\chapitre{Application linéaire}
\sousChapitre{Image et noyau, théorème du rang}

\contenu{
\texte{
Soient  $E$  un espace vectoriel et  $F$  un sous-espace vectoriel
de  $E$  de dimension finie. Soit  $f$  une application lin\' eaire de
$E$  dans lui-m\^eme.
}
\begin{enumerate}
    \item \question{Montrer que, si $F\subset f(F)$  alors  $f(F)=F$.}
    \item \question{Montrer que, si  $f$  est injective et
$f(F)\subset F$  alors  $f(F)=F$.}
\end{enumerate}
}
