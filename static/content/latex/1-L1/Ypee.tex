\chapitre{Arithmétique dans Z}
\sousChapitre{Divisibilité, division euclidienne}

\contenu{
\texte{
Soit $a \in \N$ premier {\`a} 10.
}
\begin{enumerate}
    \item \question{Montrer que $a^4 \equiv 1 (\mathrm{mod}\,{10})$.}
    \item \question{Montrer que pour tout entier $k \in \N$, $a^{4\times10^k} \equiv 1 (\mathrm{mod}\, {10^{k+1}})$.}
    \item \question{En d{\'e}duire qu'il existe un nombre $x \in \N$ tel que $x^3$ se termine par
    $123456789$ en base 10.}
\reponse{
R{\'e}currence : $a^{4\times 10^{k+1}} - 1 = (a^{4\times10^k} - 1)
                           (a^{4\times10^k\times 9} + \dots + a^{4\times10^k\times 0})$.
$x = 123456789^{800000001/3}$.
}
\end{enumerate}
}
