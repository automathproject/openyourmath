\texte{
Soient les polyn\^omes complexes $P_1=X^3-2$,
$P_2=X^4+4$ et  $P_3=X^4+4X^3+8$.
}
\begin{enumerate}
    \item \question{\'Etudier leur irr\'eductibilit\'e sur $\Cc$ et sur $\Rr$.}
    \item \question{Montrer que $P_1$ est irr\'eductible sur $\Q$ (on utilisera que $\sqrt[3]{2}\notin \Q$).}
    \item \question{Montrer que $P_2$ est r\'eductible sur $\Zz$.}
    \item \question{Montrer que $P_3$ est irr\'eductible sur $\Zz$.}
\end{enumerate}
