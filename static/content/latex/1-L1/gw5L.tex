\texte{
Soit $n \in \N$. Chercher $(n^3+n) \wedge (2n+1)$.
}
\reponse{
$8(n^3 + n) = (2n+1)(4n^2-2n+5) - 5  \Rightarrow  d = (2n+1) \wedge 5
  \Rightarrow  \begin{cases} \text{si } n \equiv     2 (\mathrm{mod}\, 5), &$d = 5$ \cr
             \text{si } n \not\equiv 2 (\mathrm{mod}\, 5), &$d = 1$.\end{cases}$
}
