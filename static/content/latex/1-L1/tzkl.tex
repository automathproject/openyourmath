\contenu{
\texte{
Démontrer par récurrence que pour tout $k\in\mathbb{N}$, $k!$ 
divise le produit de $k$ entiers consécutifs~:
$$\forall n\in \mathbb{N},\; k! \mid n(n+1)\cdots(n+k-1)$$
}
}
