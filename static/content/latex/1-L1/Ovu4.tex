\texte{
R\'esoudre dans $\mathbb{R}$ les \'equations suivantes:
}
\begin{enumerate}
    \item \question{$\cos^2(x)-\sin^2(x)=\sin(3x)$.}
\reponse{$\cos^2(x)-\sin^2(x)=\sin(3x)$ ssi $x=\pi/2+2k\pi$ ou
 $x=-\pi/10+2k\pi/5$, avec $k\in\mathbb{Z}$.}
    \item \question{$\cos^4(x)-\sin^4(x)=1$.}
\reponse{$\cos^4(x)-\sin^4(x)=1$ ssi $x=k\pi$, avec $k\in\mathbb{Z}$.}
\end{enumerate}
}
