\texte{
Soit $F \in { K(X)}$ non constante et $P \in { K[X]}$, $P \ne 0$.
}
\begin{enumerate}
    \item \question{Montrer que $P\circ F \ne 0$.}
    \item \question{Montrer que l'application${ K(X)} \to { K(X)},  G \mapsto  {G\circ F}$ est un morphisme
    injectif d'alg{\`e}bre.}
    \item \question{A quelle condition est-il surjectif ?}
    \item \question{Montrer que tous les isomorphismes de corps de ${ K(X)}$ sont de cette forme.}
\reponse{
ssi $\exists\ G \in { K(X)}$ tel que $G\circ F = X  \Rightarrow  P\circ F = XQ\circ F$.
             \par
             $F = \frac AB,\ A \wedge B = 1  \Rightarrow 
              \begin{cases}A\mid(p_0 - Xq_0)\cr B\mid(p_n - Xq_n)\end{cases}  \Rightarrow 
              F$ est homographique.
$F = \phi(X)$.
}
\end{enumerate}
