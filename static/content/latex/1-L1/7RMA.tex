\texte{
Soit $E$ un espace vectoriel et $p,q \in \mathcal{L}(E)$ deux projections.
}
\begin{enumerate}
    \item \question{Montrer que $p$ et $q$ ont même base si et seulement si :
    $p\circ q = q$ et $q \circ p = p$.}
    \item \question{Donner une condition analogue pour que $p$ et $q$ aient même direction.}
\end{enumerate}
}
