\texte{
Pour $n \in \N^*$, on note $d_n$ le nombre de diviseurs positifs de $n$.
}
\begin{enumerate}
    \item \question{Montrer que si $n = ab$ avec $a \wedge b = 1$, alors $d_n$ = $d_a d_b$.}
    \item \question{Montrer que $n$ est un carr{\'e} parfait si et seulement si $d_n$ est impair.}
    \item \question{Montrer que : $\prod_{d|n}d = {\sqrt n}^{d_n}$.}
\end{enumerate}
