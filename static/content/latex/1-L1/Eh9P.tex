\texte{
Montrer que les polyn{\^o}mes $P$ et $Q$ suivants sont premiers entre eux.
Trouver $U,V \in { K[X]}$ tels que ${UP + VQ = 1}$.
}
\begin{enumerate}
    \item \question{$P=X^4 + X^3 -2X +1$\par
    $Q=X^2 + X + 1$}
\reponse{$7U= X+3,\quad 7V= -X^3-3X^2+X+4$}
    \item \question{$P=X^3 + X^2 +1$\par
    $Q=X^3 + X + 1$}
\reponse{$3U= 2X^2-X+1,\quad 3V = -2X^2-X+2$}
\end{enumerate}
}
