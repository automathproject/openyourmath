\chapitre{Application linéaire}
\sousChapitre{Image et noyau, théorème du rang}

\contenu{
\texte{
Soit $E$ un espace vectoriel de dimension finie, $f$ une application
linéaire de~$E$ dans~$E$~; montrer que les propriétés (1) à (3) sont
équivalentes~:
}
\begin{enumerate}
    \item \question{[(1)] $ E=\mbox{Im }f \oplus\ker f$,}
    \item \question{[(2)] $ \mbox{Im }f = \mbox{Im }f^2$,}
    \item \question{[(3)] $\ker f=\ker f^2$.}
\end{enumerate}
}
