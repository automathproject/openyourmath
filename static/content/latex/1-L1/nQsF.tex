\contenu{
\texte{
Dans $\R^4$, trouver le rang de la famille de vecteurs :

$$\vec a = (3,2,1,0),\quad
  \vec b = (2,3,4,5),\quad
  \vec c = (0,1,2,3),\quad
  \vec d = (1,2,1,2),\quad
  \vec e = (0,-1,2,1).$$
}
\reponse{
$r = 3$,\quad $2\vec a - 3\vec b + 5\vec c = \vec 0$,
                 \quad $\vec b - 2\vec d - \vec e = \vec 0$.
}
}
