\sousChapitre{Binôme de Newton et combinaison}

\contenu{
\texte{
Soit $u_n = \sum_{p=0}^n C_{n-p}^p$. Montrer que $u_0 = u_1 = 1$ et :
$\forall\ n \in \N,\ u_{n+2} = u_{n+1} + u_n$
({\it suite de Fibonacci\/}).
}
}
