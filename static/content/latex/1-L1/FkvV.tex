\texte{
% Bernoulli amélioré
}
\begin{enumerate}
    \item \question{Soit $a \in \R_+$. Montrer $ \forall n\in\N^*,\:
(1+a)^n \geq 1+na + \frac{n(n-1)}{2}a^2$.}
    \item \question{Soit $(u_n)_{n\in\N}$ la suite définie par $u_n = \frac{3n}{3^n}$. Montrer que pour tout $n\in\N^*$, on a $0\leq u_n \leq\frac{3n}{2n^2+1}$.}
\end{enumerate}
