\sousChapitre{Racine, décomposition en facteurs irréductibles}

\contenu{
\texte{
Soit $P \in {\R[X]}$ dont toutes les racines sont r{\'e}elles.
}
\begin{enumerate}
    \item \question{D{\'e}montrer que les racines de $P'$ sont aussi r{\'e}elles.}
    \item \question{En d{\'e}duire que : $\forall\ a \in \R^*$, les racines de $P^2+a^2$ sont
    simples.}
\end{enumerate}
}
