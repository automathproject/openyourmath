\texte{
Soit $p \in \N^*$, et $n = 2^p$.
}
\begin{enumerate}
    \item \question{Soit $ k \in \{1,\dots, n-1\}$. V{\'e}rifier que $kC_n^k = nC_{n-1}^{k-1}$.}
    \item \question{En d{\'e}duire que : $\forall\ k \in \{1,\dots, n-1\}$, $C_n^k$ est pair.}
    \item \question{En d{\'e}duire que : $\forall\ k \in \{0,\dots, n-1\}$, $C_{n-1}^k$ est impair.}
\end{enumerate}
}
