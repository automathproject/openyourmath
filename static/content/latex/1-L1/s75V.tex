\texte{
Soit $z = \exp\frac {2i\pi}7$ et $u = z + z^2 + z^4$, $v = z^3 + z^5 + z^6$.
}
\begin{enumerate}
    \item \question{Calculer $u+v$ et $u^2$.}
\reponse{$u+v = -1$, $u^2 = u+2v = -2-u$.}
    \item \question{En d{\'e}duire $\sin\frac {2\pi}7 + \sin\frac {4\pi}7 + \sin\frac {8\pi}7$.}
\reponse{$\Sigma = \Im(u) = \frac {\sqrt7}2$.}
\end{enumerate}
}
