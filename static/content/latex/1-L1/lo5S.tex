\texte{
Soit $f$ qui, à $P\in\Rr_{2n}[X]$ associe $f(P)=X(X+1)P'-2kXP$.
Trouver $k$ tel que $f\in\mathcal{L}(\Rr_{2n}[X])$ puis, pour cette valeur de $k$, trouver tous les polynômes $P$ non nuls tels que la famille $(P,f(P))$ soit liée.
}
}
