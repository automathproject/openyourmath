\texte{
Montrer par récurrence que pour tout entier $n\in\mathbb{N}$, 
$$\left(a+b\right)^n=\sum_{k=0}^{n}C_n^k a^kb^{n-k},$$
pour tout réel $a$ et $b$.
}
}
