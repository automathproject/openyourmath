\contenu{
\texte{
Soient $\vec{v_1}(1,2,3,4), \vec{v_2}(2,2,2,6), 
\vec{v_3}(0,2,4,4), \vec{v_4}(1,0,-1,2), 
\vec{v_5}(2,3,0,1)$ dans $\Rr^4$. 
Soient $F=Vect\{\vec{v_1}, \vec{v_2}, \vec{v_3}\}$ 
et $G=Vect\{\vec{v_4}, \vec{v_5} \}$. D\'eterminer une base des 
sous-espaces 
$F\cap G,$ $F,$ $G$ et $F+G.$
}
}
