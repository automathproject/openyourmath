\chapitre{Arithmétique dans Z}
\sousChapitre{Divisibilité, division euclidienne}

\contenu{
\texte{
Soient $a,b \in \Z$ et $n \in \N^*$.
Montrer que : $a \equiv b (\mathrm{mod}\, n)  \Rightarrow  a^n \equiv b^n (\mathrm{mod}\, {n^2})$.
}
}
