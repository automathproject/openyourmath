\contenu{
\texte{
Soit $f\in L(E)$ o\`{u} $E$ est un $K-$espace vectoriel.\ On suppose :
$$\forall x\in E,\exists \lambda \in K,f(x)=\lambda x.$$
Montrer :
$$\exists \mu \in K,f=\mu id.$$
}
}
