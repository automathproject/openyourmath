\chapitre{Nombres complexes}
\sousChapitre{Autre}

\contenu{
\texte{
Soit  $f$ la fonction de $\mathbb C$ dans $\mathbb C$ d\'efinie
 par $f(z) ={1+z\over 1-z}$.
}
\begin{enumerate}
    \item \question{Calculer les points fixes de la fonction $f$, c'est \`a dire les nombres 
complexes $z$ tels que $f(z) =z$.}
    \item \question{D\'eterminer les nombres complexes $z$ pour lesquels 
$f(z)$ est r\'eel.}
\end{enumerate}
}
