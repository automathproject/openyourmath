\contenu{
\texte{
Soient $E$, $F$ deux ev de dimensions finies et $u,v \in \mathcal{L}(E,F)$.

Montrer que
$\dim(\mathrm{Ker}(u+v)) \le \dim(\mathrm{Ker} u\cap\mathrm{Ker} v) + \dim(\Im u\cap \Im v)$.

(considérer $w = u_{|\mathrm{Ker}(u+v)}$)
}
}
