\contenu{
\texte{
Soient $a,b,c \in \C$ distincts. Montrer que les propositions suivantes sont
{\'e}quivalentes :
}
\begin{enumerate}
    \item \question{$\{a,b,c\}$ est un triangle {\'e}quilat{\'e}ral.}
    \item \question{$j$ ou $j^2$ est racine de $az^2 + bz + c = 0$.}
    \item \question{$a^2 + b^2 + c^2 = ab + ac + bc$.}
    \item \question{$\frac 1{a-b} + \frac 1{b-c} + \frac 1{c-a} = 0$.}
\end{enumerate}
}
