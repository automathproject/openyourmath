\contenu{
\texte{
\noindent Trouver $a$ et $b$ entiers naturels tels que
}
\begin{enumerate}
    \item \question{$a+b=2070$ et $\mbox{ppcm}(a,b)=9180$ ;}
    \item \question{$a^2+b^2=5409$ et $\mbox{ppcm}(a,b)=360$ (on pourra commencer par montrer que 
$\mbox{pgcd}(a,b)$
divise $\mbox{pgcd}(5409,360)$ et considérer ensuite différents cas).}
\end{enumerate}
}
