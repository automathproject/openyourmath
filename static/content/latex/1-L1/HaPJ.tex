\texte{
Montrer que $A = \begin{pmatrix} 1 &2 &3 \cr
                    0 &1 &2 \cr
                    0 &0 &1 \cr\end{pmatrix}$
et  $B = \begin{pmatrix} 3 &1 &2 \cr
                    2 &0 &1 \cr
                    1 &0 &0 \cr\end{pmatrix}$
ne sont pas semblables.
}
\reponse{
$B-I$ est inversible.
}
