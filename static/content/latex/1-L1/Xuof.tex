\contenu{
\texte{
Soient $f$ et $g$, applications de ${ \Cc}$ dans ${ \Cc}$, d\'efinies par
$f(z)=\bar{z}$ et $g(z)=\Re (z)$. Montrer que $f$ et $g$ sont lin\'eaires sur ${ \Cc}$ en tant
que ${ \Rr}$-e.v., et non lin\'eaires sur ${ \Cc}$ en tant que ${ \Cc}$-e.v.
}
}
