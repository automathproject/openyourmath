\chapitre{Polynôme, fraction rationnelle}
\sousChapitre{Racine, décomposition en facteurs irréductibles}

\contenu{
\texte{
Soit le polynôme $P=X^8+2X^6+3X^4+2X^2+1$.
}
\begin{enumerate}
    \item \question{Montrer que $j$ est racine de ce polynôme. 
Déterminer son ordre de multiplicité.}
    \item \question{Quelle conséquence peut-on tirer de la parité de $P$~?}
    \item \question{Décomposer $P$ en facteurs irréductibles dans $\mathbb{C}[X]$ et dans 
$\mathbb{R}[X]$.}
\end{enumerate}
}
