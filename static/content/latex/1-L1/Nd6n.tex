\texte{
Soit $A = \begin{pmatrix} 1&0&0\cr 0&1&1\cr 1&0&1 \cr\end{pmatrix}$.
En écrivant $A = I + J$, calculer $A^n$, $n \in \Z$.
}
\reponse{
$A^n = \begin{pmatrix} 1 & 0 & 0 \cr C_n^2 & 1 & n \cr n & 0 & 1 \cr\end{pmatrix}$.
}
}
