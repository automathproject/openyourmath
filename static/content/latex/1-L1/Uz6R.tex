\contenu{
\texte{
Soit $f\in L(E)$ non nul; montrer que $f$ est injective si et seulement si
pour tout couple $(E_{1},E_{2})$ de sous-espaces suppl\'{e}mentaires de $E$,
la somme $f(E_{1})+f(E_{2})$ est directe (i.e. $f(E_{1})$ et $f(E_{2})$ sont
suppl\'{e}mentaires).
}
}
