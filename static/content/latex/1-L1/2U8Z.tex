\contenu{
\texte{
Soient $u,v \in \mathcal{L}(E)$. Déterminer le rang de l'endomorphisme de $\mathcal{L}(E)$ :
$f  \mapsto u \circ f \circ v$.
}
\reponse{
$(\mathrm{rg} u)(\mathrm{rg} v)$.
}
