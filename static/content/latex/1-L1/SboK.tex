\chapitre{Espace vectoriel}

\contenu{
\texte{
Soit $E =\R^3$. On d\'efinit le syst\`eme $$S=\left\{ {\mathbf e}_1 =(1,1,1), {\mathbf e}_2
=(1,1,2), {\mathbf e}_3 =(1,2,3)\right\}$$
}
\begin{enumerate}
    \item \question{Montrer que $S$ est une base de $E$.}
    \item \question{Calculer les coordonn\'ees de ${\mathbf v} = (5,7,12)$ dans cette base.}
\end{enumerate}
}
