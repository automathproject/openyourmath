\texte{
On
suppose que la proposition $P$ est vraie ainsi que les propositions suivantes :
}
\begin{enumerate}
    \item \question{$ (\neg Q) \wedge P \Rightarrow \neg S$.}
    \item \question{$S \Rightarrow (\neg P) \vee Q$.}
    \item \question{$P \Rightarrow R \vee S$.}
    \item \question{$ S \wedge Q \Rightarrow \neg P$.}
    \item \question{$R \wedge \neg (S \vee Q) \Rightarrow T$.}
    \item \question{$R \Rightarrow (\neg P) \vee (\neg Q)$.}
\end{enumerate}
}
