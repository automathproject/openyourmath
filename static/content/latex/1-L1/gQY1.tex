\contenu{
\texte{
On considère que $\R$ est un $\Q$-espace vectoriel.
}
\begin{enumerate}
    \item \question{Montrer que la famille $\big(1,\sqrt2,\sqrt3\big)$ est libre.}
    \item \question{Montrer que la famille $(\ln p)$ où $p$ décrit l'ensemble des nombres
    premiers positifs est libre.}
\end{enumerate}
}
