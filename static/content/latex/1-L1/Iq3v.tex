\texte{
Soit $\omega = \exp\frac {2i\pi}n$. Calculer :
}
\begin{enumerate}
    \item \question{$\sum_{k=0}^{n-1} (1+\omega^k)^n$.}
\reponse{D{\'e}velopper. $S=2n$.}
    \item \question{$\sum_{k=0}^{n-1}\sum_{\ell=k}^{n-1} C_\ell^k \omega^{k+\ell}$.}
\reponse{$\frac {1 - (1+\omega)^n}{1 - \omega - \omega^2} =
              \frac {1 + (2\cos(\pi/n))^n}{1 - \omega - \omega^2}$.}
\end{enumerate}
}
