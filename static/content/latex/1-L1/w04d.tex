\texte{
On considère que $\C$ est un $\R$-espace vectoriel.
}
\begin{enumerate}
    \item \question{Donner une base de $\C$.}
    \item \question{Montrer que tout endomorphisme de $\C$ peut se mettre sous la forme :
    $f(z) = az + b\overline{z}$, avec $a,b \in \C$.}
    \item \question{CNS sur $a$ et $b$ pour que $f$ soit bijectif ?}
\reponse{
3. $|a| \ne |b|$.
}
\end{enumerate}
}
