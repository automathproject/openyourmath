\contenu{
\texte{
Soit $f: E\to F$.
Que pensez-vous des affirmations suivantes ?
}
\begin{enumerate}
    \item \question{$\forall\ x \in E     \quad \forall\ y \in F     \quad f(x)=y.$}
    \item \question{$\forall\ x \in E     \quad \exists\ y \in F \text{ tel que } \quad f(x)=y.$}
    \item \question{$\exists\ x \in E \text{ tel que } \quad \forall\ y \in F     \quad f(x)=y.$}
    \item \question{$\exists\ x \in E \text{ tel que } \quad \exists\ y \in F \text{ tel que } \quad f(x)=y.$}
    \item \question{$\forall\ y \in F     \quad \forall\ x \in E     \quad f(x)=y.$}
    \item \question{$\forall\ y \in F     \quad \exists\ x \in E \text{ tel que } \quad f(x)=y.$}
    \item \question{$\exists\ y \in F \text{ tel que } \quad \forall\ x \in E     \quad f(x)=y.$}
    \item \question{$\exists\ y \in F \text{ tel que } \quad \exists\ x \in E \text{ tel que } \quad f(x)=y.$}
\end{enumerate}
}
