\chapitre{Polynôme, fraction rationnelle}
\sousChapitre{Racine, décomposition en facteurs irréductibles}

\contenu{
\texte{
Soit $P \in \Rr[X]$ scind\'e sur $\Rr$ \`a racines simples.
}
\begin{enumerate}
    \item \question{Montrer qu'il en est de même de $P'$.}
    \item \question{Montrer que le polynôme $P^2 + 1$ n'a que des racines simples dans $\Cc$.}
\end{enumerate}
}
