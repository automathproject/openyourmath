\contenu{
\texte{
Soient $z,z' \in \Cc$ tels que $\text{Arg}(z)-\text{Arg}(z')=\frac{\pi}{2}$.
}
\begin{enumerate}
    \item \question{Montrer que $z\overline{z'}+\overline{z}z' = 0$.}
    \item \question{Montrer que $|z+z'|^2=|z-z'|^2=|z|^2+|z'|^2$.}
\end{enumerate}
}
