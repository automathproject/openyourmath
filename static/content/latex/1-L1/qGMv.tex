\texte{
Soit $A = \begin{pmatrix} 0&0&0 \\ -2&1&-1 \\ 2&0&2 \end{pmatrix}$.
}
\begin{enumerate}
    \item \question{Calculer $A^3-3A^2 + 2A$.}
    \item \question{Quel est le reste de la division euclidienne de $X^n$ par $X^3-3X^2 + 2X$ ?}
    \item \question{Calculer $A^n$ pour $n \in \Nn$.}
    \item \question{$A$ est-elle inversible ?}
\end{enumerate}
