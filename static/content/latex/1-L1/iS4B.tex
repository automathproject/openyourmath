\chapitre{Polynôme, fraction rationnelle}

\contenu{
\texte{
Soit $P$ un polynôme dont le reste de la division euclidienne par $X-1$ est $7$
et par $X + 5$ est $3$. Quel est le reste de la division euclidienne de $P$
par $X^2 + 4X -5$ ?
}
}
