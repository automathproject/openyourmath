\contenu{
\texte{
Soit $  N\in M_n(\C )  $ une matrice nilpotente. Calculer $
\hbox{det}(I+N)  .$ Si $  A\in M_n(\C )  $ commute avec $  N  ,$ montrer que $
\hbox{det}(A+N)=\hbox{det}(A)  .$ (on pourra commencer par \'etudier le cas o\`u $  A  $ est inversible.)
}
}
