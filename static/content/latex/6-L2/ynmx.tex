\texte{
Soit ${\cal E}$ un ellipse de centre $O$ et de dimensions $a$, $b$.
Soient $M,P \in {\cal E}$ tels que $OMP$ soit rectangle en $O$.
}
\begin{enumerate}
    \item \question{Montrer que $\frac{1}{OM^2} + \frac1{OP^2} = \frac1{a^2} + \frac1{b^2}$.}
    \item \question{En déduire que $(MP)$ reste tangente à un cercle fixe de centre $O$.}
\end{enumerate}
}
