\texte{
Déterminer l'équation de la sphère contenant les cercles d'équations
    $\begin{cases} x^2+y^2 = 9\cr z=0\cr \end{cases}$ et
    $\begin{cases} x^2+y^2 = 25\cr z=2.\cr \end{cases}$
}

$x^2+y^2+z^2-10z=9$.
}
