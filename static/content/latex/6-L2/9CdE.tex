\contenu{
\texte{
\'Equation de la surface de révolution engendrée par la rotation de $\Gamma$
    autour de $Oz$ où $\Gamma$ est la courbe d'équations paramétriques :
    $\begin{cases}x=a\cos^3u\cr y=a\sin^3u\cr z=a\cos2u.\cr\end{cases}$ ($a > 0$)
}
\reponse{
$4x^2 + 4y^2 - 3z^2 = a^2$.
}
}
