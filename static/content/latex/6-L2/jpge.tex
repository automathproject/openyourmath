\texte{
% pythagore, trinôme
On considère un carré $ABCD$, et un cercle $\mathcal C$ passant par $A$ et $B$ et tangent à $[CD]$.
}
\begin{enumerate}
    \item \question{Montrer que le point de tangence est le milieu de $[CD]$.}
    \item \question{Montrer que si le rayon vaut $r=10$ alors $AB=16$, et réciproquement.
%}
    \item \question{Montrer que $AB = \frac{8}{5} r$.}
\indication{Pythagore.}
\end{enumerate}
