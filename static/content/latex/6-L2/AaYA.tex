\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Les plans suivants 
sont-ils parall\`eles ou s\'ecants? Dans ce dernier cas, donner un vecteur
directeur de la droite $(D)=(P) \cap (P')$.
}
\begin{enumerate}
    \item \question{$(P): 5x-y-1=0$ et $(P'): z=3$.}
    \item \question{$(P): x+y+z+1=0$ et $(P'): 2x-y+3z+2=0$.}
    \item \question{$(P): 2x-z+1=0$ et $(P'): 4x-3y+2z+5=0$.}
    \item \question{$(P): 4x-6y+8z-1=0$ et $(P'): -6x+12y-9z+11=0$.}
\end{enumerate}
}
