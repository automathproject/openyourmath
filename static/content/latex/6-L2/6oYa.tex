\contenu{
\texte{
% tags : corde, médiatrice, symétrique, milieu, triangle rectangle, cercle circonscrit
On donne un cercle $\mathcal C$ (sans son centre).
}
\begin{enumerate}
    \item \question{Tracer son centre, si possible de plusieurs façons.}
    \item \question{On donne un point $P$ à l'extérieur du cercle. Tracer les tangentes à $\mathcal C$ passant par $P$.}
    \item \question{Même question si $P$ est sur le cercle.}
\indication{Utiliser une ou plusieurs cordes.}
\end{enumerate}
}
