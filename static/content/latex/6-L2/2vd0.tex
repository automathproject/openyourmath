\chapitre{Courbes planes}
\sousChapitre{Courbes paramétrées}

\contenu{
\texte{
\'Etudier les branches infinies des courbes paramétrées suivantes :
}
\begin{enumerate}
    \item \question{$x=t^5-t^3+\frac t4$,\quad           $y=\frac{3t}{3t^2+1}$.}
    \item \question{$x=2\cos^2t+\ln|\sin t|$,\quad        $y=\sin2t$.}
    \item \question{$x=\sqrt{\frac{t^2-2}{t^4-1}}$,\quad $y=tx$.
    L'aire comprise entre la courbe et ses asymptotes est-elle finie ?}
    \item \question{$x=\frac{t^3-t}{2t-1}$,\quad         $y=tx$.}
    \item \question{$x=\frac1t+\frac1{t+1}$,\quad       $y=\frac1t+\frac1{(t+1)^2}$.}
    \item \question{$x=\frac1t+\frac1{t+1}$,\quad       $y=\frac1t-\frac1{t+1}$.}
    \item \question{$x=\frac{3t}{1+t^3}$,\quad           $y=tx$.}
    \item \question{$x=\frac{te^t}{t+1}$,\quad           $y=\frac{e^t}{t+1}$.}
    \item \question{$x=2t^3+3t^2$,\quad                   $y=3t^2+6t$.}
    \item \question{$x=t^3-3t$,\quad                      $y=t^3-t^2-t+1$.}
    \item \question{$x=\frac t{t^2-1}$,\quad             $y=\frac{t^2}{t-1}$.}
\end{enumerate}
}
