\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Dans un plan affine, soit $ABC$ et $A'B'C'$ deux triangles de centre
de gravité $G$ et $G'$ respectivement.
}
\begin{enumerate}
    \item \question{Calculer $\vec{AB'}+\vec{BC'}+\vec{CA'}$ en fonction de $G$ et $G'$.}
    \item \question{Montrer que $ABC$ et $A'B'C'$ ont le même centre de gravité si
 et seulement s'il existe un point $D$ tel que $DBA'C$ et $DB'AC'$
 soient des parallélogrammes.}
\end{enumerate}
}
