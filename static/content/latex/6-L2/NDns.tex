\chapitre{Géométrie affine euclidienne}

\contenu{
\texte{
Soient $(O,\vec{OA}, \vec{OB}, \vec{OC})$ et $(O,\vec{OA'}, \vec{OB'}, \vec{OC'})$
deux repères orthonormés directs de l'espace.

Montrer que $\vec{AA'}$, $\vec{BB'}$, $\vec{CC'}$ sont coplanaires.
}
}
