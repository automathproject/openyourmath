\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Dans un espace affine de dimension $3$ muni d'un repère affine.
}
\begin{enumerate}
    \item \question{Montrer que les droites $D$ et $D'$ d'équations cartésiennes
$$D ~:~x+y-z-2=0\quad \text{ et } \quad 2x-y+3z-1=0$$
et $$D' ~:~x-2y-3\quad \text{ et } \quad 3x+6y-1$$
sont concourantes.}
    \item \question{Trouver une équation cartésienne du plan qu'elles déterminent.}
\end{enumerate}
}
