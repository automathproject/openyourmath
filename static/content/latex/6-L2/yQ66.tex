\begin{enumerate}
    \item \question{Soit $C$ un cercle et $\vec{u}$ un vecteur.
Construire une corde $[AB]$ du cercle $C$ telle que $\vec{AB}=\vec{u}$.}
    \item \question{Construire un segment $[AB]$ connaissant son milieu $I$ et sachant que $A$ appartient à une droite donnée $d$ et $B$ à un cercle donné $C$.}
    \item \question{Construire un carré $ABCD$ sachant que $A$ et $C$ sont sur une droite donnée $d_1$ que $B$ est sur une droite donnée $d_2$ et que $D$ est sur une droite donnée $d_3$.}
\end{enumerate}
