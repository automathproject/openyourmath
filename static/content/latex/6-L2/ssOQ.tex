\begin{enumerate}
    \item \question{Soit $G$ un sous-groupe fini de déplacements du plan.
  \begin{enumerate}}
    \item \question{Montrer que $G$ est constitué uniquement de rotations.}
    \item \question{Soient $f,g \in G$. Montrer que $f$ et $g$ ont même centre
          (étudier $f \circ g \circ f^{-1} \circ g^{-1}$).}
    \item \question{Prouver enfin que $G$ est cyclique.}
\end{enumerate}
