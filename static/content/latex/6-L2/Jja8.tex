\contenu{
\texte{
Soient ${\cal E} : \frac {x^2}{4a^2} + \frac {y^2}{4b^2} = 1$,
et ${\cal E}' : \frac {x^2}{a^2} + \frac {y^2}{b^2} = 1$.
}
\begin{enumerate}
    \item \question{CNS sur $u,v,w$ pour que la droite d'équation $ux+vy+w=0$ soit
    tangente à ${\cal E}'$ ?}
\reponse{$a^2u^2+b^2v^2 -w^2=0$.}
    \item \question{Soient $(MP)$, $(MQ)$ deux tangentes à ${\cal E}'$ avec $M,P,Q \in {\cal E}$.
    Montrer que $(PQ)$ est aussi tangente à ${\cal E}$.}
\reponse{$M : \begin{pmatrix}2a\cos\theta\cr2a\sin\theta\end{pmatrix}$,
             $P : \begin{pmatrix}2a\cos\alpha\cr2a\sin\alpha\end{pmatrix}$ :
             $(MP)$ est tangente à ${\cal E}'
             \Leftrightarrow \theta \equiv \alpha \pm \frac{2\pi}3\ [2\pi]$.}
\end{enumerate}
}
