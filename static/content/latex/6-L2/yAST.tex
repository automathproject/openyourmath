\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Soit les droites $D$ et $D'$ non coplanaires~:
$$(D) \;\left\{
\begin{array}{rcl}
    x-y+z+1 & = & 0  \\
    2x+y-z  & = & 0
\end{array}\right.\quad
\mbox{et}\quad (D') \; \left\{
\begin{array}{rcl}
    x+2y+z & = & 0  \\
    2x-2y-2z-1 & = & 0
\end{array}\right.$$
Trouver des équations de leur perpendiculaire commune.
}
}
