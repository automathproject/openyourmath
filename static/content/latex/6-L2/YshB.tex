\begin{enumerate}
    \item \question{Soit $(AB)$ une droite dans un espace affine déterminée par deux points distincts $A$ et $B$. Décrire à l'aide des coordonnées barycentriques dans le repère $AB$ les trois régions de la droite découpées par les points $A$ et $B$.}
    \item \question{Soit $ABC$ un triangle non plat dans un plan affine $E$.
Décrire à l'aide des coordonnées barycentriques dans le repère $ABC$ les sept régions découpées par les droites qui portent les cotés du triangle $ABC$.}
\end{enumerate}
