\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Soit $E$ un espace affine, $f$ une application affine de $E$ dans $E$ et
 $$F = \left\{M\in E /f (M) = M \right\}.$$
 On suppose que $F \neq \emptyset$.
}
\begin{enumerate}
    \item \question{Montrer que $\vec{F} = \ker (\vec{f}-id)$.}
    \item \question{On suppose que $\vec{f}\circ \vec{f} = \vec{f}$. Soit $s$ la projection
affine sur $F$ parall\`element \`a $\ker (\vec{f})$. Montrer que
$f = s$.}
    \item \question{Faire la même chose si $\vec{f}\circ \vec{f} = id$.}
\end{enumerate}
}
