\chapitre{Courbes planes}
\sousChapitre{Courbes paramétrées}

\contenu{
\texte{
Soit $\mathcal{C}$ la courbe d'équations paramétriques~:
$x(t) = \frac{t^2+1}{t^3-1},
 y(t) = \frac{2t}{t^3-1}$.
}
\begin{enumerate}
    \item \question{Montrer que les points de paramètres $t,u,v$ (distincts) sont alignés si et
    seulement si $tuv = t + u + v + 1$.}
    \item \question{Prouver que $\mathcal{C}$ admet exactement trois points d'inflexion et qu'ils
    sont alignés.}
\reponse{
$\frac{m'(t)}{m(t)} =\frac{-2(t-1)(t^2+t+1)(t^3-3t-1)}{t(2t^3+1)(t^3+2+3t)}$.
}
\end{enumerate}
}
