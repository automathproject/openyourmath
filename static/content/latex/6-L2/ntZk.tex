\texte{
Soit $\gamma :[0,1]\rightarrow \Cc $ continue, et $z\in \Cc$
quelconque. Montrer :
\begin{eqnarray*}
\forall \epsilon &>&0,\exists \gamma ^{\prime }\in C([0,1],\Cc)
\text{ tel que:} \\
1 &:&\forall t\in [0,1],\left| \gamma (t)-\gamma ^{\prime }(t)\right|
<\epsilon , \\
2 &:&\forall t\in [0,1],\gamma (t)\neq z.
\end{eqnarray*}
}
}
