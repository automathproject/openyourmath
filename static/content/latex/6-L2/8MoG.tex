\contenu{
\texte{
Soit ${\cal E}$ un ellipsoïde d'équation
$\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2} = 1$
et $P$ un plan d'équation $ux + vy + wz = 1$.
Montrer que $P$ est tangent à $E$ si et seulement si
$a^2u^2 + b^2v^2 + c^2w^2 = 1$.
}
}
