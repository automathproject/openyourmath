\contenu{
\texte{
Soit $\cal P$ la parabole d'équation $y^2 = 2px$.
}
\begin{enumerate}
    \item \question{Un rayon incident arrive suivant une parallèle à $Ox$ et se réfléchit à
    ``l'intérieur'' de $\cal P$ avec le même angle.
    Trouver l'enveloppe des rayons réfléchis.}
\reponse{Foyer.}
    \item \question{Même question, mais le rayon incident est parallèle à $Oy$.}
\reponse{Point d'impact : $\left(\frac{t^2}{2p},t\right)$
             Point caractéristique :
             $\left(\frac{3t^2}{2p},\frac{t(3p^2-t^2)}{2p^2}\right)$.}
\end{enumerate}
}
