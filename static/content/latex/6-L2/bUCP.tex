\chapitre{Géométrie affine euclidienne}
\sousChapitre{Géométrie affine euclidienne du plan}

\contenu{
\texte{
Soit $A, B ,C$ un repère affine d'un plan affine $E$.
}
\begin{enumerate}
    \item \question{Déterminer les équations barycentriques des médianes du triangle $ABC$.}
\reponse{Comme $A$ a pour coordonnées barycentriques $(1,0,0)$ et le milieu $A'$ de $[BC]$ $(0,1,1)$, la médiane $(AA')$ de $ABC$ a pour équation barycentrique 
    $$\det\left(\begin{array}{ccc}1&0&a\\ 0&1&b\\0&1&c
    \end{array}\right)=c-b=0.$$
    De même la médiane $(BB')$ a pour équation barycentrique $a=c$ et $(CC')$
    $a=b$.}
    \item \question{En utilisant la question précédente, montrer que les médianes du triangle $ABC$ sont concourantes.}
\reponse{Les médianes du triangle $ABC$ sont concourantes au point de coordonnées barycentriques $(1,1,1)$..}
\end{enumerate}
}
