\contenu{
\texte{
Soit $R$ le rotation vectorielle d'angle $\theta$ et
 d'axe orient\'e par le vecteur unitaire $k$. Montrer que $\forall x\in \Rr^{3}$
 $ R (x) = (\cos\theta)x + (\sin\theta)k\wedge x + 2 (x|k)\sin^{2} (\tfrac{\theta}{2})k$.
}
}
