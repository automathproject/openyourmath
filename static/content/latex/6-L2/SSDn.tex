\chapitre{Géométrie affine euclidienne}
\sousChapitre{Géométrie affine euclidienne du plan}

\contenu{
\texte{
Soit $E$ un plan euclidien et $A$ et $B$ deux points de $E$
}
\begin{enumerate}
    \item \question{Déterminer la composée de l'homothétie de centre $A$ et de rapport $2$
avec l'homothétie de centre $B$ et de rapport $1/2$.
On pourra construire l'image de quelques points.}
    \item \question{Déterminer la composée de l'homothétie de centre $A$ et de rapport $2$
avec l'homothétie de centre $B$ et de rapport $3$.}
    \item \question{Que dire en général de la composé de deux homothéties ?}
    \item \question{Que dire de la composée d'une homothétie et d'une translation ?}
\end{enumerate}
}
