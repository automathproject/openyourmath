\contenu{
\texte{
Quel est l'angle entre deux faces d'un dodécaèdre régulier ?
(on donne : $4\sin\frac\pi5 = \sqrt{10-2\sqrt5}$)
}
\reponse{
$\cos\theta = 1 - \frac{1}{2\sin^2(\pi/5)} = -\frac{1}{\sqrt5}$.
}
}
