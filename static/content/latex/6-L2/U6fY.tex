\contenu{
\texte{
Soit $F$ un point, $D$ une droite ne passant pas par $F$, et
$a > \frac12d(F,D)$.

Trouver l'ensemble des points $M$ tels que $MF + d(M,D) = 2a$.
}
\reponse{
arcs de paraboles de foyer $F$ et de directrices $\Delta,\Delta'$,
         parallèles à $D$ à la distance $2a$ de $D$.
}
}
