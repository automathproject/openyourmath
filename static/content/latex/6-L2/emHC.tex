\chapitre{Courbes planes}
\sousChapitre{Propriétés métriques : longueur, courbure,...}

\contenu{
\texte{
\def\arc#1{\mathop{#1}\limits^\frown}
Déterminer la longueur d'un arc $\arc{M_0M_t}$
ou $\arc{M_0M_\theta}$ pour les courbes :
}
\begin{enumerate}
    \item \question{$x = t - \ch t\sh t$,\quad $y = 2\ch t$}
\reponse{$\sh^2t$.}
    \item \question{$\rho = \tanh \frac\theta2$.}
\reponse{$\theta-\tanh\frac\theta2$.}
\end{enumerate}
}
