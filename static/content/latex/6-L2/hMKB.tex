\texte{

$\mathcal{C}$ et  $\mathcal{C}'$ les cercles de centres $M$ et $M'$ de rayons

[FI]$ et $J$ le deuxi\`eme point d'intersection de $\mathcal{C}$ et  $\mathcal{C}'$.
}
\begin{enumerate}
    \item \question{Montrer que $ (MM')$ est bissectrice de l'angle $F'MJ$.}

\end{enumerate}
