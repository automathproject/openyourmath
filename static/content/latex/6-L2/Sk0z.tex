\chapitre{Géométrie affine euclidienne}
\sousChapitre{Géométrie affine euclidienne de l'espace}

\contenu{
\texte{
Soit $E$ un espace affine euclidien de dimension trois et $(O,\vec{\imath},\vec{\jmath}, \vec{k})$ un repère.
Soit $A(1,2,3)$ et $B(3,2,1)$ deux points de $E$.
}
\begin{enumerate}
    \item \question{Déterminer l'expression analytique dans le repère $(O,\vec{\imath},\vec{\jmath}, \vec{k})$
de la symétrie ortho\-gonale par rapport au plan média\-teur 
du segment $[AB]$.}
    \item \question{Vérifier votre résultat en déterminant l'image du point $A$.}
\end{enumerate}
}
