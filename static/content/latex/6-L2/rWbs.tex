\texte{
Soit ${\cal S}$ la surface d'équation cartésienne :
$\frac1{(x-y)^2} + \frac1{(y-z)^2} = \frac1{(x-z)^2}$.
}
\begin{enumerate}
    \item \question{Montrer que ${\cal S}$ est à la fois un cylindre et un cône.}
    \item \question{Comment est-ce possible ?}
\end{enumerate}
}
