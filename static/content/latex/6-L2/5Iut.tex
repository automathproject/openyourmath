\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
On considère le plan muni d'un un repère orthonormé ($O, \overrightarrow {\imath},\overrightarrow{\jmath}$) et la courbe $(C)$ d'équation 

\begin{center}$2x^{2} + 3xy -2y^{2} - 10 = 0$ \end{center}
}
\begin{enumerate}
    \item \question{Vérifier que $O$ est un centre de symétrie. Trouver une base orthonormale tel que l'équation de $(C)$ soit de la forme $\displaystyle \frac {x^{2}} {a^{2}} - \frac {y^{2}} {b^{2}} = 1$ $(a,b$ réels).}
    \item \question{Donner l'équation des asymptotes de $(C)$ et tracer $(C)$ dans le repère orthonormé ($O, \overrightarrow {\imath},\overrightarrow{\jmath}$).}
\end{enumerate}
}
