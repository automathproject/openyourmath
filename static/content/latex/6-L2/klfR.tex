\texte{
On donne un segment $[AB]$.
}
\begin{enumerate}
    \item \question{Diviser le segment en quatre.}
    \item \question{Diviser le segment en trois parties égales sans utiliser le théorème de Thalès.}
    \item \question{Diviser le segment en sept parties égales.}
    \item \question{Construire le barycentre de $(A,3)$ et $(B,4)$.}
\indication{2. Penser à l'intersection des médianes.}
\end{enumerate}
