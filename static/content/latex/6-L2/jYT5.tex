\contenu{
\texte{
Soit ${\cal S}$ la surface d'équations paramétriques :
$\begin{cases} x = u/(u^2+v^2)\cr
         y = v/(u^2+v^2)\cr
         z = 1/(u^2+v^2).\cr \end{cases}$
}
\begin{enumerate}
    \item \question{Donner une équation cartésienne de ${\cal S}$.}
\reponse{$z=x^2+y^2$.}
    \item \question{Déterminer l'ensemble $\mathcal{C}$ des points de ${\cal S}$ où le plan tangent est
    parallèle à la droite $D$ d'équations : ${x=y=z}$.}
\reponse{$x+y=\frac12$.}
    \item \question{Déterminer l'équation cartésienne du cylindre de génératrices parallèles à
    $D$ s'appuyant sur $\mathcal{C}$. (Cylindre circonscrit à ${\cal S}$)}
\reponse{$(x-y+\frac12)^2 = 2(z-y+\frac14)$.}
\end{enumerate}
}
