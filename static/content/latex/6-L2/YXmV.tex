\texte{
Soit $\mathcal{C}$ la courbe d'équations :
$\begin{cases} x^2 - y^2 - 4x + 2 = 0 \cr x + z = 1.\cr\end{cases}$
}
\begin{enumerate}
    \item \question{Déterminer la nature et les éléments remarquables de $\mathcal{C}$.}
\reponse{Hyperbole équilatère d'asymptotes :
             $\begin{cases} x-z-1 = \pm y\sqrt2 \cr x+z = 1.\cr\end{cases}$}
    \item \question{Chercher une équation cartésienne de la surface ${\cal S}$ engendrée par la
    rotation de $\mathcal{C}$ autour de $Oz$ et reconnaître~${\cal S}$.}
\reponse{$x^2+y^2 = 2z^2$, cône de révolution.}
\end{enumerate}
}
