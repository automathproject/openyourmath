\texte{
On consid\`ere le champ de vecteurs $P:\R^2\rightarrow \R^2$ d\'efini par
$$P(x,y)=(2x {\mathrm e}^{x^2-2y};-2 {\mathrm e}^{x^2-2y}).$$
}
\begin{enumerate}
    \item \question{V\'erifier que la forme diff\'erentielle associ\'ee \`a $P$ est ferm\'ee.}
    \item \question{En d\'eduire que $P$ est un champ de gradients et en d\'eterminer un
potentiel.}
    \item \question{Calculer la circulation de $P$ le long du chemin $$\gamma:[0,1]\rightarrow
\R^2;\quad t\mapsto (\ln (1+t);{\mathrm e}^t +1).$$}
\end{enumerate}
}
