\texte{
Dans un \emph{rond} on donne la droite
$D  : \begin{cases} x +2y - z = -3 \cr  x - y +2z = -4\cr\end{cases}$ et $M(x,y,z)$.
Calculer $d(M,D)$.
}
\reponse{
$d^2 = \frac{(x+2y-z+3)^2}6 + \frac{(3x+3z+11)^2}9$.
}
