\texte{
\def\arc#1{\mathop{#1}\limits^\frown}
Calculer la longueur totale des courbes suivantes :
}
\begin{enumerate}
    \item \question{$x = (1+\cos^2t)\sin t$,\quad $y=\sin^2t\cos t$.}
\reponse{$4\sqrt2+4\arccos\frac1{\sqrt3}-\pi = 4\sqrt2+4\Arctan\sqrt2-\pi$.}
    \item \question{$\rho = \sin^2\frac\theta2$.}
\reponse{$4$.}
\end{enumerate}
