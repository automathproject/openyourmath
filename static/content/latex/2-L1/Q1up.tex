\contenu{
\texte{
Soit $ (1)$ l'\'equation $x-E (x) = \frac 1 {x^2}$.
}
\begin{enumerate}
    \item \question{Montrer que pour tout $n \in \Nn^*$ il existe un unique $x_n \in [n, n + 1[$
solution de $ (1)$.}
    \item \question{D\'eterminer un \'equivalent de $x_n$.}
    \item \question{Faire un DAS de $x_n-n$ en $ + \infty$ en fonction de $\frac 1n$ \`a l'ordre $5$.}
\end{enumerate}
}
