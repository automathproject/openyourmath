\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Calculer $ \int_{t=0}^{\pi/4} \ln(1+\tan t)\,d t$.
On remarquera que $\cos t + \sin t = \sqrt 2 \cos\left(\frac \pi4 - t\right)$.
}
}
