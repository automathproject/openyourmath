\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : {[a,b]} \to \R$ dérivable telle que $f'(a) = f'(b)$.

Montrer qu'il existe $c \in {]a,b[}$ tel que $f'(c) = \frac {f(c) - f(a)}{c-a}$.
}
}
