\contenu{
\texte{
Soient $(u_{n})_{n\in \Nn}$ et $(v_{n})_{n\in \Nn} $ deux suites r\'{e}elles de
limite $+\infty  $ telles que $u_{n}=o(v_{n}).$
Montrer qu'il existe une suite $(w_{n})_{n\in \Nn} $ de limite $+\infty  $
telle que $u_{n}=o(w_{n})$ et $w_{n}=o(v_{n}).$
}
}
