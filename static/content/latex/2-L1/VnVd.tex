\chapitre{Propriétés de R}
\sousChapitre{Maximum, minimum, borne supérieure}

\contenu{
\texte{
Soit $A$ une partie majorée de $\mathbb{R}$ d'au moins deux éléments et $x$
un élément de $A$.
}
\begin{enumerate}
    \item \question{Montrer que si $x<\sup A$, alors $\sup(A\setminus\{x\})=\sup A$.}
    \item \question{Montrer que si $\sup(A\setminus\{x\})<\sup A$, alors $x=\sup A$.}
\end{enumerate}
}
