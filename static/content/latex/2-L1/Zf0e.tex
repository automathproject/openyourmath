\contenu{
\texte{
On note $E (x)$ la partie enti\`ere d'un r\'eel $x$.
}
\begin{enumerate}
    \item \question{Montrer que $\forall (x, y) \in \Rr^2 \; \; E (x) + E (y) \leq E (x + y)
\leq E (x) + E (y) + 1$.}
    \item \question{Calculer $E (x) + E (-x)$ pour $x \in \Rr$.}
    \item \question{Montrer que $\forall n \in \Nn^*$ et $\forall x \in \Rr$ $E (x) =E(\dfrac{E (nx)}
{n})$.}
\end{enumerate}
}
