\chapitre{Calcul d'intégrales}

\contenu{
\texte{
Calculer $\lim_{n\to\infty} \frac 1{\sqrt{n^2-1}} + \frac 1{\sqrt{n^2-4}}
          + \dots + \frac 1{\sqrt{n^2-(n-1)^2}}$.
}
\reponse{
$ \int_{t=0}^1 \frac {d t}{\sqrt{1-t^2}} = \frac \pi2$.
}
