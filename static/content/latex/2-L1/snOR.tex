\chapitre{Suite}
\sousChapitre{Suite définie par une relation de récurrence}

\contenu{
\texte{
Soit $(u_n)$ une suite r{\'e}elle telle que pour tout entier $n$ :
$u_n \in\,[0,1]$ et $u_{n+1}(1-u_n) > \frac 14$.\par
Montrer que cette suite converge vers $\frac 12$.
}
}
