\texte{
On pose :
$f(x) = 1/(1+x)$,
$g(x) = e^{-x}$,
$h(x)  = \sqrt{1-2\sin x}$,
$k(x)  = \cos(\sqrt{2x})$.

Préciser les positions relatives de $\mathcal{C}_f$, $\mathcal{C}_g$, $\mathcal{C}_h$, $\mathcal{C}_k$ au voisinage
de $0$.
}
\reponse{
$h \le k \le g \le f$.
}
