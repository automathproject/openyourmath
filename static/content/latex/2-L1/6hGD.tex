\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f : {[0,1]} \to \R$ continue positive. On pose $A =  \int_{t=0}^1 f(t)\,d t$.

Montrer que $\sqrt{1+A^2} \le  \int_{t=0}^1 \sqrt{1+f^2(t)}\,d t \le 1+A$.
}
}
