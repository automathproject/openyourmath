\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f$ continue sur $[0,\pi ]$ telle que $\int_{0}^{\pi }f(u)\cos
(u)du=\int_{0}^{\pi }f(u)\sin (u)du=0,$ montrer que $f$ s'annulle au moins
deux fois sur $]0,\pi [.$
}
}
