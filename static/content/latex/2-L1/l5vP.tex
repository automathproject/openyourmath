\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f : {[a,b]} \to \R$ continue telle que pour toute fonction
$g : {[a,b]} \to \R$ en escalier, $ \int_{t=a}^b f(t)g(t)\,d t = 0$.
Démontrer que $f = 0$.
}
}
