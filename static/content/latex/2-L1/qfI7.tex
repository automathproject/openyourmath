\texte{
Soit $\alpha >0$. Étudier la série $\sum u_n$, avec 
$u_n=\frac{(-1)^n}{\sqrt{n^{\alpha}+(-1)^n}}\cdotp$
}
\reponse{
$u_n = \frac{(-1)^n}{n^{\alpha/2}}-\frac1{2n^{3\alpha/2}}
+  o\Bigl(\frac1{n^{3\alpha/2}}\Bigr)$,
il y a convergence ssi $\alpha>\frac23$.
}
}
