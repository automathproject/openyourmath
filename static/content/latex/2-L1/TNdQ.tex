\texte{
Calculer pour tout $n \in \N^*$~:
$I_n =  \int_{x=0}^\pi \frac{dx}{1+\cos^2(nx)}$.
}
\reponse{
Couper en intervalles de $k\pi/n$.
         On obtient $I_n = \frac\pi{\sqrt2}$ pour tout $n\ge 1$.
}
}
