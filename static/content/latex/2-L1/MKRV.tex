\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : {[a,+\infty[} \to \R$ dérivable telle que $f(x) \to f(a)$ lorsque $x\to {+\infty}$.
Montrer qu'il existe $x \in {]a,+\infty[}$ tel que $f'(x) = 0$.
}
}
