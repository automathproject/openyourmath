\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Calculs}

\contenu{
\texte{
Soit $f:\Rr_+\rightarrow \Rr_+^*$ une fonction 
born\'ee deux fois d\'erivable et telle qu'il existe $\alpha > 0$
tel que, pour tout $x\in \Rr_+ ,$ on ait $\alpha f(x)\leq f''(x).$
}
\begin{enumerate}
    \item \question{\begin{enumerate}}
    \item \question{Montrer que $f'$ a une limite en $+\infty .$ Quelle est 
la valeur de cette limite~?}
    \item \question{Montrer que $f$ est d\'ecroissante et que $\displaystyle{ \lim 
_{+\infty} f(x)=0}.$}
\end{enumerate}
}
