\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f$ continue de $[0,1]$ dans ${\Rr}$ telle que $\int_{0}^{1}f^{n}(u)du$
ne prenne qu'un nombre fini de valeurs quand $n$ d\'{e}crit $\Nn.$
Montrer que $f=-1$ ou $f=0$ ou $f=1.$
}
}
