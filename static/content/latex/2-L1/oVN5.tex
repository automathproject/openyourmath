\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Etude de fonctions}

\contenu{
\texte{
Soit $f : \R \to \R$ croissante telle que :
$\forall\ x \in \R,\ f\circ f(x) = x$.
Montrer que : $\forall\ x \in \R,\ f(x) = x$.
}
}
