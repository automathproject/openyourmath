\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Fonctions convexes}

\contenu{
\texte{
Soit  $f  : {\R^{+*}} \to  \R$ convexe et $g : {\R^{+*}} \to \R$ affine.
On suppose : $\begin{cases}\forall\ x > 0,\ f(x) \le g(x),\cr f(1) = g(1).\cr\end{cases}$

Montrer que $f = g$.
}
}
