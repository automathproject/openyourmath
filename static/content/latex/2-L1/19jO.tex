\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Calculer $\frac {d^n}{d t^n}\left(t^{n-1}\ln t\right)$, et
$\frac {d^n}{d t^n}\left(t^{n-1}\exp(1/t)\right)$
(essayer $n=1,2,3$).
}
\reponse{
$\frac {(n-1)!}t$,
  \qquad
    $\frac {(-1)^n}{t^{n+1}}\exp(1/t)$.
}
}
