\texte{
Soient $n\in \Nn^{*},$ et $(x_{1},x_{2},...,x_{n})\in [0,1]^{n}$, montrer que :
$$\prod\limits_{i=1}^{n}(1-x_{i})\geq 1-\sum\limits_{i=1}^{n}x_{i}.$$
}
}
