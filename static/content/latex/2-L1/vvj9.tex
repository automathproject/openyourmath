\texte{
On note $x_n = (\sqrt3 + 1)^{2n+1}$, $y_n = (\sqrt3 - 1)^{2n+1}$, et
$z_n = [x_n]$.
}
\begin{enumerate}
    \item \question{Montrer que $z_n = x_n - y_n$.}
    \item \question{En d{\'e}duire que $2^{n+1}$ divise $z_n$.}
\end{enumerate}
}
