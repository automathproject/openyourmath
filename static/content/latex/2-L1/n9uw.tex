\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Trouver les fonctions $f:{\Rr\rightarrow \Rr}$ d\'{e}rivables en $0 $  telles que :
$$\exists \lambda \in {\Rr}^{+}-\{1\},\forall x\in {\Rr},f(\lambda
x)=\lambda f(x).
$$
}
}
