\begin{enumerate}
    \item \question{Donner un exemple de suite born\'{e}e divergente, puis de suite divergente
telle que
$$\forall k\in \Nn,\lim_{n\rightarrow \infty }x_{n+k}-x_{n}=0.$$}
    \item \question{Donner un exemple de suite divergente qui a une seule valeur d'adh\'{e}rence
(i.e. telle qu'il existe une seule extraction $\phi  $ telle que
$x_{\phi(n)\text{ }}$ converge).}
    \item \question{Donner un exemple de suite $(x_{n})_{n\in \Nn} $ divergente telle que $\forall
k\geq 2,(x_{nk})_{n\in \Nn}$ converge.}
\end{enumerate}
