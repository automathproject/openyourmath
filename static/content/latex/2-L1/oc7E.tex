\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f \in C^2 (\Rr)$ telle que $\forall (x, y) \in \Rr^2 \, \, f (x + y)f (x-y)
\leq f (x)^2$.
Montrer que $\forall x \in \Rr \, \, f (x)f'' (x) \leq f' (x)^2$.
}
}
