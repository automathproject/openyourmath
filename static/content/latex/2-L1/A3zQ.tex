\contenu{
\texte{
Soit $\alpha \in {]0,1[}$.
}
\begin{enumerate}
    \item \question{Montrer que : $\forall n \in \N^*,\
           \frac\alpha{(n+1)^{1-\alpha}} \le (n+1)^\alpha - n^\alpha \le
           \frac\alpha{n^{1-\alpha}}$.}
    \item \question{En déduire que $\sum_{k=1}^n \frac1{k^{1-\alpha}} \sim \frac{n^\alpha}{\alpha}$ pour $n\to\infty$.}
\end{enumerate}
}
