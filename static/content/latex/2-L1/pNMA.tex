\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Continuité : pratique}

\contenu{
\texte{
Etudier la continuité sur $\mathbb{R}$ des fonctions suivantes :
}
\begin{enumerate}
    \item \question{$f_1(x)=x^2 \cos \frac{1}{x} \mbox{ si } x\not=0, \mbox{ et } 
f_1(0)=0$ ;}
    \item \question{$f_2(x)= \sin x\sin \frac{1}{x} \mbox{ si } x\not=0, \mbox{ et } f_2(0)=0$ ;}
    \item \question{$f_3(x)=x E(x) $ ;}
    \item \question{$f_4(x) = E(x) \sin(\pi x) $.}
\end{enumerate}
}
