\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $\varphi : {\R_n[X]} \to {\R^{n+1}}, P \mapsto {(a_0,\dots,a_n)}$
avec $a_k =  \int_{t=0}^{+\infty} e^{-t}t^kP(t)\,d t$.
}
\begin{enumerate}
    \item \question{Justifier l'existence de $\varphi$.}
    \item \question{Montrer que $\varphi$ est un isomorphisme d'espace vectoriel.}
\end{enumerate}
}
