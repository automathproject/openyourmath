\texte{
D\'eterminer la limite en $0$ de $\dfrac{\arctan x
- \sin x}{\tan x - \arcsin x}$.
}
\reponse{
$$\lim _{x\rightarrow 0}{\frac {\arctan(x)-\sin(x)}{\tan(x)-\arcsin(x)}}
=-1.$$
}
}
