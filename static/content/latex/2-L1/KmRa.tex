\contenu{
\texte{
Les énoncés suivants sont-ils vrais ou faux~?
}
\begin{enumerate}
    \item \question{Si une suite positive est non majorée, elle tend vers $+\infty$.}
    \item \question{Si une suite d'entiers converge, elle est stationnaire.}
    \item \question{Si une suite a un nombre fini de valeurs, elle converge si et seulement
si elle est stationnaire.}
    \item \question{Une suite est convergente si et seulement si elle est bornée.}
    \item \question{Si une suite n'est pas majorée, elle est minorée.}
\end{enumerate}
}
