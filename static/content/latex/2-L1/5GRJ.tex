\chapitre{Propriétés de R}
\sousChapitre{Maximum, minimum, borne supérieure}

\contenu{
\texte{
Soit $A$ l'ensemble des nombres réels qui peuvent s'écrire
$ x= \frac{2p^2-3q}{p^2 +q}$ pour $p$ et $q$ entiers vérifiant
$0<p<q$.
}
\begin{enumerate}
    \item \question{Montrer que $A$ est minorée par $-3$ et majorée par $2$.}
    \item \question{Déterminer $\inf A$ et $\sup A$ (pour la borne supérieure on pourra
prendre $q=p+1$).}
\end{enumerate}
}
