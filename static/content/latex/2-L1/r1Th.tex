\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Fonctions convexes}

\contenu{
\texte{
Soit $f : \R \to \R$ deux fois dérivable telle que :
$f \ge 0$, $f' \ge 0$, $f'' \ge 0$.
}
\begin{enumerate}
    \item \question{\'Etudier l'existence des limites (dans $\overline\R\,$)
    en $+\infty$ de $f(x)$, $f'(x)$, $\frac {f(x)}x$.}
    \item \question{Même question pour les limites en $-\infty$ de $f(x)$, $f'(x)$, et $xf'(x)$.}
\reponse{
$f(x) \to \ell \in\R$,\quad
             $f'(x) \to 0$.\par
             TAF entre $x$ et $x/2$ $ \Rightarrow  2\left(f(x) - f\left(\frac x2\right)\right) \le xf'(x) \le 0
              \Rightarrow  xf'(x) \to 0$.
}
\end{enumerate}
}
