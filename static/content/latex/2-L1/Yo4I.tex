\begin{enumerate}
    \item \question{Montrer que pour tout $0<\epsilon <1$ et pour $x \in \mathbb{R},$ on a:
$$ |x-1| < \frac{\epsilon}{4} \Rightarrow |x^2+x-2| < \epsilon.$$}
    \item \question{En déduire:
$$ \lim_{x \rightarrow 1}x^2+x-1\;\mbox{ et }\; \lim_{x \rightarrow 1}(x^2+x-2)\cos x.$$}
\end{enumerate}
