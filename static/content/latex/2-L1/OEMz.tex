\chapitre{Calcul d'intégrales}
\sousChapitre{Intégration par parties}

\contenu{
\texte{
Soit $f \in C^{2} ([a, b], \Rr)$.
}
\begin{enumerate}
    \item \question{Montrer que $\int_{a}^{b}f (t)dt = \frac{b-a}2 (f (a) + f (b)) +
  \frac 12 \int_a^b f'' (x) (a-x) (b-x)dx$.}
    \item \question{En d\'eduire un encadrement de  $\int_{a}^{b}f (t)dt $ si $\forall x\in [a, b]
 \,\,  m \leq f'' (x) \leq M$.}
\end{enumerate}
}
