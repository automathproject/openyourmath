\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : {[a,b]} \to \R$ dérivable.
}
\begin{enumerate}
    \item \question{On suppose que : $\forall\ x \in {[a,b]},\ f'(x) \ne 0$. Montrer que $f'$ est
      de signe constant.}
    \item \question{Dans le cas général, montrer que $f'([a,b])$ est un intervalle.}
\end{enumerate}
}
