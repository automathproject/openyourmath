\sousChapitre{Maximum, minimum, borne supérieure}

\contenu{
\texte{
Soit $A$ et $B$ deux parties born\'ees de $\Rr$.
\textbf{Vrai} ou \textbf{faux} ?
}
\begin{enumerate}
    \item \question{$A \subset B \Rightarrow \sup A \leqslant \sup B$,}
\reponse{Vrai.}
    \item \question{$A \subset B \Rightarrow \inf A \leqslant \inf B$,}
\reponse{Faux. C'est vrai avec l'hypothèse $B \subset A$ et non $A \subset B$.}
    \item \question{$\sup (A\cup B) = \max(\sup A,\sup B)$,}
\reponse{Vrai.}
    \item \question{$\sup(A+B) < \sup A + \sup B$,}
\reponse{Faux. Il y a égalité.}
    \item \question{$\sup(-A) = -\inf A$,}
\reponse{Vrai.}
    \item \question{$\sup A +\inf B \leqslant \sup(A+B)$.}
\reponse{Vrai.}
\indication{Deux propositions sont fausses...}
\end{enumerate}
}
