\chapitre{Calcul d'intégrales}
\sousChapitre{Autre}

\contenu{
\texte{
Soient ${f,g} : {[a,b]} \to {\R}$ continues, $f$ décroissante, et
$0 \le g \le 1$.
On note $G(x) = a +  \int_{t=a}^x g(t)\,d t$.

Démontrer que $ \int_{t=a}^b fg(t)\,d t \le  \int_{t=a}^{G(b)} f(t)\,d t$.
}
}
