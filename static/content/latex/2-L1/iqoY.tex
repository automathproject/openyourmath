\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f : {[a,b]} \to \R$ continue et $g : \R \to \R$ continue convexe.

Démontrer que $g\left(\frac 1{b-a} \int_{t=a}^b f(t)\,d t\right)
      \le \frac 1{b-a} \int_{t=a}^b g(f(t))\,d t$.
}
}
