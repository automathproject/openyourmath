\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Continuité : théorie}

\contenu{
\texte{
Soit $f : \Rr \rightarrow \Rr^ + $ continue telle que $f (0) = 1$,
$\lim\limits_{ -\infty}f = 0$ et $\lim\limits_{  + \infty}f = 0$.
}
\begin{enumerate}
    \item \question{Montrer qu'il existe $a>0$ tel que si $\left|x\right|>a$ alors $f (x) \leq \frac 12$.}
    \item \question{Montrer que $f$ est born\'ee et poss\`ede un maximum.}
\end{enumerate}
}
