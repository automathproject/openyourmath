\sousChapitre{Equivalents}

\contenu{
\texte{
Chercher un équivalent simple
en $0^+$ de $f_k(x) = x^{x^{.^{.^{.^{\scriptstyle x}}}}}$
($k$ fois $x$).
}
\reponse{
$\sim x$ pour $k$ impair, et
         $= 1 + x\ln(x) + o_{x\to0}(x\ln x) \sim 1$ si $k$ est pair $\ge 2$.
}
}
