\texte{
Par comparaison à une intégrale, donner un équivalent de :
}
\begin{enumerate}
    \item \question{$\sum_{k=n+1}^{2n} \frac 1{\sqrt k}$.}
\reponse{$2(\sqrt 2-1)\sqrt n$.}
    \item \question{$\sum_{k=2}^{n} \frac 1{k\ln k}$.}
\reponse{$\ln(\ln n)$.}
\end{enumerate}
}
