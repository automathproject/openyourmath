\texte{
D{\'e}terminer $x \in \Q$ sachant que :
}
\begin{enumerate}
    \item \question{$2x^3-x^2+x+1 = 0$.}
\reponse{$x=-\frac 12$.}
    \item \question{$6x^5+11x^4-x^3+5x-6 = 0$.}
\reponse{$x=\frac 23$.}
    \item \question{$2x^3-x-4 = 0$.}
\reponse{Pas de solution.}
\end{enumerate}
}
