\texte{
Soit $A$ une partie born\'ee de $\Rr$ et $x$ un r\'eel.
}
\begin{enumerate}
    \item \question{Montrer que $x = \sup (A)$ ssi ($x$ majore $A$ et il existe une suite
$ (x_n)_{n \in \Nn}$ de A qui converge vers $x$).}
    \item \question{\'Enoncer un r\'esultat analogue pour $\inf (A)$.}
\end{enumerate}
}
