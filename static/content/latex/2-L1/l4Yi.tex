\texte{
Simplifier $\sum_{k=0}^{n-1} 3^k\sin^3\left(\frac\alpha{3^{k+1}}\right)$.
}
\reponse{
linéariser :
         $\sum = \frac 14\left(3^n\sin\frac\alpha{3^n} - \sin\alpha\right)$.
}
}
