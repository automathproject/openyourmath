\contenu{
\texte{
Soit $H$ un sous-groupe additif de $\R$, $H \ne \{0\}$.
On pose $H^{+*} = H\cap\R^{+*}$, et $\alpha = \inf(H^{+*})$.
}
\begin{enumerate}
    \item \question{Si $\alpha \in H^{+*}$, montrer que $H = \alpha \Z$.}
    \item \question{Si $\alpha \notin H^{+*}$, montrer que $\alpha = 0$ et en d{\'e}duire que $H$
    est dense dans $\R$.}
\end{enumerate}
}
