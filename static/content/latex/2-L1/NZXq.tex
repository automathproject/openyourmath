\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f:\R \to \R$ de classe $\mathcal{C}^\infty$ bornée.
}
\begin{enumerate}
    \item \question{Montrer que si une dérivée, $f^{(k)}$, $k\ge 2$, admet un nombre fini de zéros,
    alors les dérivées précédentes, $f^{(p)}$, $1\le p < k$, tendent vers $0$
    en $\pm\infty$.}
    \item \question{En déduire que pour tout $k \ge 2$, $f^{(k)}$ s'annule au
    moins $k-1$ fois sur $\R$.}
\end{enumerate}
}
