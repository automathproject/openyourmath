\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Calculs}

\contenu{
\texte{
Soit $f:{\Rr}\rightarrow {\Rr}$ d\'{e}finie par :
$$\forall x\neq 0,f(x)=e^{-\frac{1}{x^{2}}},f(0)=0. $$
Montrer que $f\in C^{\infty }({\Rr},{\Rr})$ et calculer ses
d\'{e}riv\'{e}es en 0.
}
}
