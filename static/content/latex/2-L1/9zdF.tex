\chapitre{Suite}
\sousChapitre{Suite de Cauchy}

\contenu{
\texte{
Montrer que la suite définie par
$$u_n=1+\frac{\cos1}{1!} + \frac{\cos2}{2!}+\cdots+\frac{\cos n}{n!}$$
est une suite de Cauchy. En déduire sa convergence.
}
