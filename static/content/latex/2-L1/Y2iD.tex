\contenu{
\texte{
Pour $0<a<b$, déterminez $\lim_{x\to 0^+} \int_{t=ax}^{bx} \frac{1-\cos u}{u^3}\,d u$.
}
\reponse{
DL de $1-\cos u \Rightarrow \lim = \frac12\ln(b/a)$.
}
}
