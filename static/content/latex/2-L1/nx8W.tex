\chapitre{Calcul d'intégrales}

\contenu{
\texte{
Soit $f$ une fonction positive d\'{e}croissante de ${\Rr}^{+}$ dans ${\Rr},$
 telle que $\int_{0}^{\infty }f$ existe. Montrer que :
$$f(x)=o(\frac{1}{x}) $$
quand $x\rightarrow \infty .$
}
}
