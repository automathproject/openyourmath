\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f : {[0,+\infty[} \to {\R^+}$ continue telle que
$ \int_{t=0}^{+\infty} tf(t) \,d t$ converge.
On pose $F(x) =  \int_{t=x}^{+\infty} f(t)\,d t$.
}
\begin{enumerate}
    \item \question{Justifier l'existence de $F(x)$, et montrer que $F(x) = o \Bigl({\frac1x }\Bigr)$ pour $x\to+\infty$.}
    \item \question{Montrer que $ \int_{t=0}^{+\infty} F(t) \,d t =  \int_{t=0}^{+\infty} tf(t) d t$.}
\end{enumerate}
}
