\texte{
Soit $u_n = 1 + \frac1{\sqrt2} + \dots + \frac1{\sqrt n}$.
}
\begin{enumerate}
    \item \question{Chercher $\lim_{n\to\infty} (u_{2n}-u_n)$, puis $\lim_{n\to\infty} u_n$.}
    \item \question{Comparer $\frac 1{2\sqrt k}$, $\sqrt{k+1} - \sqrt k$,
    et $\sqrt k - \sqrt{k-1}$.
    En d{\'e}duire que la suite $\bigl(u_n - 2\sqrt n\bigr)$ est convergente.}
\end{enumerate}
}
