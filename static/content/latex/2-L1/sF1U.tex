\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f(x) = e^{x\sqrt3}\sin x$. Calculer $f^{(n)}(x)$ pour $n\in\N$.
}
\reponse{
$f^{(n)}(x) = 2^ne^{x\sqrt3}\sin(x+n\frac\pi6)$.
}
}
