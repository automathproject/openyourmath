\contenu{
\texte{
Soit $f : {[0,2\pi]} \to \R$ convexe de classe $\mathcal{C}^2$.
Quel est le signe de $I =  \int_{t=0}^{2\pi} f(t)\cos t\,d t$ ?
}
\reponse{
$I = \Bigl[ f'(t)(1+\cos t) \Bigr]_0^{2\pi} +
      \int_{t=0}^{2\pi} f''(t)(1+\cos t)\,d t \ge 0$.
}
}
