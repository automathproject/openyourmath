\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Trouver toutes les applications $f:\R \to \R$ dérivables telles que :
$\begin{cases}\forall\ x \in \R,\ f'(x)f'(f(x)) = 1 \cr
        f(0) = 0 \text{ et } f'(0) > 0.\cr\end{cases}$
}
\reponse{
$f = \mathrm{id}$.
}
}
