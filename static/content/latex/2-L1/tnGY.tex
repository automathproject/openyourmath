\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f : {[0,+\infty[} \to \R$ continue décroissante telle que
$ \int_{}^{+\infty} f(t)\,d t$ converge.
}
\begin{enumerate}
    \item \question{Montrer que la série $\sum_{k=0}^\infty f(k)$ converge et encadrer le reste :
     $\sum_{k=n}^\infty f(k)$ à l'aide d'intégrales de $f$.}
    \item \question{Application : Pour $\alpha > 1$, donner un équivalent pour $n \to \infty$ de
    $\sum_{k=n}^\infty \frac1{k^\alpha} $.}
\end{enumerate}
}
