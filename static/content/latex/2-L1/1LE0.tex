\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Calculer la dérivée $n^{\text{ème}}$ de $x \mapsto (x^3+2x^2-5)e^{-x}$.
}
\reponse{
$(-1)^ne^{-x}\bigl( x^3 + (2-3n)x^2 + (3n^2-7n)x + (-n^3+5n^2-4n-5)\bigr)$.
}
}
