Calculer les limites suivantes~:
}
\begin{enumerate}
    \item \question{$\displaystyle\lim_{n\to\infty}
    \frac{1+\sqrt2+\sqrt3+\cdots+\sqrt n}{n\sqrt n}$.}

    \item \question{$\displaystyle\lim_{n\to\infty}\frac{1}{n}\sum_{p=1}^n
\ln\frac{(3n+6p-4)(n+2p)^2}{3n^3}$.}
\end{enumerate}
