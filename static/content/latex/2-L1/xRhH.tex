\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f:]0;1] \rightarrow {\Rr } $ continue et monotone, telle que $\int_{0}^{1}f$
existe. Calculer
$$\lim\limits_{n\rightarrow \infty }\frac{1}{n}\sum_{k=1}^{n}f\left( \frac{k}{n}\right).$$
}
}
