\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Limite de fonctions}

\contenu{
\texte{
\noindent D\'eterminer les limites suivantes, en justifiant vos calculs.
}
\begin{enumerate}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}{x+2 \over x^2 \ln x}$}
\reponse{$- \infty$}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}2x \ln(x+\sqrt x)$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{x^3-2x^2+3 \over x \ln x}$}
\reponse{$+ \infty$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{e^{\sqrt x+1} \over x+2}$}
\reponse{$+ \infty$}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}{\ln(3x+1) \over2x}$}
\reponse{${3 \over2}$}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}{x^x-1 \over\ln(x+1)}$}
\reponse{$- \infty$}
    \item \question{$\displaystyle\lim_{x \rightarrow- \infty}{2 \over x+1}\ln
\Bigl({x^3+4 \over1-x^2}\Bigr)$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow{(-1)}^+}(x^2-1) \ln(7x^3+4x^2+3)$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow2^+}{(x-2)}^2 \ln(x^3-8)$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}{x(x^x-1) \over\ln(x+1)}$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}(x \ln x -x \ln(x+2))$}
\reponse{$-2$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{e^x-e^{x^2} \over x^2-x}$}
\reponse{$- \infty$}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}{{(1+x)}^{\ln x}}$}
\reponse{$1$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{\Bigl({x+1 \over x-3}\Bigr)^x}$}
\reponse{$e^4$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{\Bigl({x^3+5 \over
x^2+2}\Bigr)^{x+1 \over x^2+1}}$}
\reponse{$1$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{\Bigl({e^x+1 \over
x+2}\Bigr)^{1 \over x+1}}$}
\reponse{$e$}
    \item \question{$\displaystyle\lim_{x \rightarrow0^+}\bigl(\ln(1+x)\bigr)^{1\over\ln x}$}
\reponse{$e$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{x^{(x^{x-1})} \over
x^{(x^{x})}}$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{(x+1)^x \over x^{x+1}}$}
\reponse{$0$}
    \item \question{$\displaystyle\lim_{x \rightarrow+ \infty}{x \sqrt{\ln(x^2+1)} \over
1+e^{x-3}}$}
\reponse{$0$}
\end{enumerate}
}
