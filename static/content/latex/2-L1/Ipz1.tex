\chapitre{Suite}
\sousChapitre{Convergence}

\contenu{
\texte{
Soient $(a_n)$, $(b_n)$ deux suites convergeant vers $a,b$.
Montrer que $\frac {a_0b_n + a_1b_{n-1} + \dots + a_nb_0}{n+1}
\xrightarrow[n\to\infty]{} ab$.
}
}
