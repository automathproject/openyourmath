\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Les fonctions suivantes sont-elles int\'egrables au sens de Riemann?
}
\begin{enumerate}
    \item \question{$f(x)=[x]$ sur $[0,2]$}
\reponse{Oui.}
    \item \question{$g: [0,1] \rightarrow \R,\quad  g(x)=
\begin{cases}
 \left[\frac 1x\right] &\text{ si $0<x\leq 1$, }\\
 1&\text{si $x=0$ } 
\end{cases} $}
\reponse{Non.}
    \item \question{$h: [0,1] \rightarrow \R,\quad  h(x)=
\begin{cases}
 \frac 1x\sin \left(\frac 1x\right) &\text{ si $0<x\leq 1$, }\\
 1&\text{si $x=0$ } 
\end{cases}$}
\reponse{Non.}
    \item \question{$k: [0,1] \rightarrow \R,\quad  k_1(x)=
\begin{cases}
1&\text{ si $x\in [0,1]\cap \Q$, }\\
0&\text{ si $x\in [0,1]\backslash \Q$}
\end{cases}$}
\reponse{Non.}
\end{enumerate}
}
