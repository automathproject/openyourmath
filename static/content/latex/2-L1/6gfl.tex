\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : \Rr^ + \rightarrow \Rr$ d\'erivable telle que $\lim\limits_{ + \infty}
f' = l$. Montrer qu'alors $\lim\limits_{ + \infty} \dfrac{f (x)}x  = l$.
}
}
