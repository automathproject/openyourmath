\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Fonctions convexes}

\contenu{
\texte{
Soit $f : \R \to \R$ dérivable telle que :
$\forall\ a,b \in \R \text{ tq } a < b,\ \exists!\ c \in ]a,b[ \text{ tq }
 f(b) - f(a) = (b-a) f'(c)$.
}
\begin{enumerate}
    \item \question{Montrer que pour tout $a \in \R$, la fonction $b \longmapsto \frac {f(b)-f(a)}{b-a}$
est monotone sur $]-\infty,a[$ et sur $]a,+\infty[$.}
    \item \question{En déduire que $f$ est strictement convexe ou strictement concave.}
\end{enumerate}
}
