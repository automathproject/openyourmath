\chapitre{Propriétés de R}
\sousChapitre{Maximum, minimum, borne supérieure}

\contenu{
\texte{
Soit
$$
I = \left\lbrace x \in \Rr \  | -2 < x +\frac{1}{2x}  \le 2 \right\rbrace.
$$
}
\begin{enumerate}
    \item \question{Montrer que $I$ est la r\'eunion de deux
intervalles.}
    \item \question{D\'eterminer (s'ils existent) : les majorants,
les minorants, la borne sup\'erieure, la borne
inf\'erieure, le plus grand \'el\'ement, le plus petit
\'el\'ement de $I$.}
\end{enumerate}
}
