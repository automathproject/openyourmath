\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Fonctions convexes}

\contenu{
\texte{
Soit $f : {[0,+\infty[} \to {\R}$ deux fois dérivable
telle que $f(x) \to f(0)$
lorsque $x\to{+\infty}$.
Montrer qu'il existe $c \in {]0,+\infty[}$ tel que $f''(c) = 0$.
}
}
