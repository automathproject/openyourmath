\contenu{
\texte{
On donne $u_0,v_0$. R{\'e}soudre le syst{\`e}me :
$\begin{cases} 5u_n = 2u_{n-1} + 3v_{n-1} \cr 5v_n = 2v_{n-1} + 3u_{n-1}.\end{cases}$
}
\reponse{
$2u_n = u_0 + v_0 + \left(-\frac 15\right)^n (u_0 - v_0)$,\quad
$2v_n = u_0 + v_0 - \left(-\frac 15\right)^n (u_0 - v_0)$.
}
}
