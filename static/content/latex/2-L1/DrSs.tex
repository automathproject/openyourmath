\contenu{
\texte{
Montrer que
$\sum_{k=0}^\infty \Arctan\Bigl(\frac1{k^2+k+1}\Bigr) = \frac \pi2$.
(On pourra calculer $\tan s_n$)
}
\reponse{
$\tan s_n = n+1$ par récurrence et
         $s_n \le \sum_{k=0}^\infty \frac 1{k^2+k+1} \le 1
         + \sum_{k=0}^\infty \frac1{n(n+1)} = 2$.
}
}
