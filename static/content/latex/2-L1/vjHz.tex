\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Continuité : pratique}

\contenu{
\texte{
\'Etudier la continuit\'e des fonctions suivantes :
}
\begin{enumerate}
    \item \question{$f_1(x)=x^2\cos \frac{1}{x} \text{\ \  si }x\not=0 \qquad f_1(0)=0$ ;}
    \item \question{$f_2(x)= \sin x\sin \frac{1}{x} \text{\ \  si }x\not=0 \qquad f_2(0)=0$ ;}
    \item \question{$f_3(x)=xE(x) \text{ sur } \Rr$ ;}
    \item \question{$f_4(x) = \left[ x-E(x)\right]^2 \text{\ \  et\ \  } f_5(x) = E(x)+f_4(x)$.}
\end{enumerate}
}
