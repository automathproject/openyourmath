\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Limite de fonctions}

\contenu{
\texte{
Soit $f : \R \to \R$ continue telle que
$f(x+1) - f(x) \to a \in \overline{\R}$ lorsque $x\to+\infty$.
}
\begin{enumerate}
    \item \question{Montrer que $\frac {f(n)}n \to a$ lorsque $n\to\infty$.}
    \item \question{Montrer que $\frac {f(x)}x \to a$ lorsque $x\to\infty$.}
\end{enumerate}
}
