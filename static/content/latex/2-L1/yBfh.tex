\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f$ une application continue de ${\Rr}^{+}$ dans ${\Rr}$ telle
que $\int_{0}^{\infty }f^{2}$ existe.\ Montrer que quand $x\rightarrow\infty :$
$$\int_{0}^{x}f(t)dt=o(\sqrt{x}). $$
}
}
