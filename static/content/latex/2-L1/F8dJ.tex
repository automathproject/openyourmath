\sousChapitre{Suite définie par une relation de récurrence}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{On donne $A\geq 0$, $B\geq 0$, $u_0\geq 0$~; étudier la suite définie
par la relation de récurrence
$\displaystyle u_{n+1}=\frac{A}{n+1} + B u_n$.}
    \item \question{Étudier la suite définie par $u_0=0$ et
$\displaystyle u_{n+1}=\frac{\frac{4n}{n+1} -u_n}{2+u_n}$
(on pourra utiliser la question précédente pour terminer).}
\end{enumerate}
