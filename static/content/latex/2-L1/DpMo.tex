\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f \in C^0 (\Rr)$. On d\'efinit ${g} : {\Rr^*} \to {\Rr}$, ${x}\mapsto {\frac 1x \int_0^x f (t)dt}$.
}
\begin{enumerate}
    \item \question{Montrer que $g$ se prolonge par continuit\'e en $0$.}
    \item \question{Montrer que si $f$ est p\'eriodique, $g$ admet une limite en $ + \infty$.}
\end{enumerate}
}
