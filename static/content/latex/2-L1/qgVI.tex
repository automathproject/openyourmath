\contenu{
\texte{
Étudier la suite définie par $u_{n+1}= \frac{2u_n^3+7}{3(u_n^2+1)}$ 
et $u_0=2$. En déduire une valeur approchée à $10^{-8}$ près de la 
racine réelle du polynôme $X^3+3X-7$.
}
}
