\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
On veut calculer $\ell = \lim_{n \to \infty} \frac 1{n+1} + \dots + \frac 1{2n}$.
}
\begin{enumerate}
    \item \question{Montrer l'existence de $\ell$.}
    \item \question{Soit $f : {[0,+\infty[} \to \R$ dérivable telle que $f(0) = 0$.
      Montrer que $f\left(\frac 1{n+1}\right) + \dots + f\left(\frac 1{2n}\right) \to \ell f'(0)$ lorsque $n\to \infty$.}
    \item \question{On prend $f(x) = \ln(1+x)$. Déterminer $\ell$.}
\end{enumerate}
}
