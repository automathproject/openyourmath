\texte{
Pour chacune des courbes suivantes,
déterminer la tangente pour $x=0$ et la position de la courbe par rapport à
cette tangente.
}
\begin{enumerate}
    \item \question{$y = \frac{e^{\sin x}-1}x$.}
\reponse{$y = 1 + \frac x2 - \frac{x^3}8$.}
    \item \question{$y = \frac1{\sh x} - \frac 1x$.}
\reponse{$y = -\frac x6 + \frac{7x^3}{360}$.}
    \item \question{$y = \frac1{\arcsin x} - \frac 1x$.}
\reponse{$y = -\frac x6 - \frac{17x^3}{360}$.}
    \item \question{$y = (2e^x - e^{-x})^{1/x}$.}
\reponse{$y = e^3(1 - 4x + 16x^2)$.}
    \item \question{$y = \frac 2{e^{2x}-1} - \frac1x$.}
\reponse{$y = -1 + \frac x3 - \frac{x^3}{45}$.}
    \item \question{$y = \sqrt{1+\sqrt{1+x}}$.}
\reponse{$y = \sqrt2\left(1 + \frac x8 - \frac{5x^2}{128}\right)$.}
\end{enumerate}
}
