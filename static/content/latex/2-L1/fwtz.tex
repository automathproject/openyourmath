\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f:\R \to \R$ continue telle que
$\frac {f(2t)-f(t)}t \to a$ (si $t\to 0$. Montrer que $f$ est dérivable en $0$ et $f'(0) = a$.
}
}
