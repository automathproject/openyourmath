\texte{
Soit $(a,b)\in\R^2$, r\'esoudre le syst\`eme 
$\left\lbrace
\begin{array}{c}
 \mathrm{ch}x+\mathrm{sh}y=a  \\
 \mathrm{sh}x+\mathrm{ch}y=b
\end{array}\right.$.
}
