\texte{
Soient $a \neq 0$ un r\'eel, et $y>x>0$.
}
\begin{enumerate}
    \item \question{Calculer la valeur de 
$$ I(x,y) = \int_x^y{ \ln \left( 1+ {a^2 \over x^2} \right) dt}. $$}
    \item \question{Montrer que $I(x,y)$ a une limite $I_0(y)$ quand $x$ tend
vers z\'ero et la calculer.}
    \item \question{Montrer que $I_0(y)$ a une limite quand $y$ tend vers
$+\infty$ et la calculer.}
\end{enumerate}
