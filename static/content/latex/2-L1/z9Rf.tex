\texte{
Soit $f : \R \to \R$ un morphisme de corps.
}
\begin{enumerate}
    \item \question{Montrer que : $\forall\ x \in \Q,\ f(x) = x$.}
    \item \question{Montrer que $f$ est une application croissante.}
    \item \question{En d{\'e}duire que $f = \mathrm{id}_\R$.}
\end{enumerate}
}
