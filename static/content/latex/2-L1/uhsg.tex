\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f:[1,\infty [\rightarrow {\Rr}^{+} $ continue telle que
$$\int_{1}^{\infty }f(t)dt $$ converge.
Montrer que
$$\lim\limits_{x\rightarrow \infty }\frac{1}{x}\int_{1}^{x}tf(t)dt=0. $$
}
}
