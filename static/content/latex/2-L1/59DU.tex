\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Calculer $I =  \int_{t=0}^\pi \ \frac t{1+\sin t}\,d t$.
}
\reponse{
$u = \pi-t  \Rightarrow  I = \frac \pi2 \int_{t=0}^\pi \ \frac 1{1+\sin t}\,d t
= \frac \pi2  \int_{t=-\pi/2}{\pi/2} \ \frac 1{1+\cos t}\,d t = \pi$.
}
}
