\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
\label{limfprime}
Soit $f:\R \to \R$ dérivable et bornée telle que
$f'(x) \to \ell$ lorsque $x \to +\infty$. Montrer que $\ell = 0$.
}
}
