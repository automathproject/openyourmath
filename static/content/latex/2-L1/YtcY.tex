\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f$ continue sur $[0,1]$ \`{a} valeurs dans $[a,b]$. On suppose $a<0<b$
et $\int_{0}^{1}f(t)dt=0.$
Montrer que :
$$\int_{0}^{1}f^{2}(t)dt\leq -ab. $$
}
}
