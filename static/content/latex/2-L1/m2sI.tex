\texte{
Soit $k\in\C$ fix{\'e}. {\'E}tudier la convergence de la suite $(a_n)$ d{\'e}finie par :
$a_0\in \C$, $a_{n+1} = k a_n^2$.
}
\reponse{
CV (vers 0) ssi $|ka_0| < 1$.
}
