\texte{
Soit $f : {\R} \to {\R^+}$ une fonction de classe $\mathcal{C}^2$.
On suppose : $\forall\ x \in \R,\ |f''(x)| \le M$.
}
\begin{enumerate}
    \item \question{Montrer que : $\forall\ x,y \in \R,\ f(x) + yf'(x) + \frac {y^2}2M \ge 0$.}
\reponse{$= f(x+y) + \frac {y^2}2(M-f''(z))$.}
    \item \question{En déduire que : $\forall\ x \in \R,\ |f'(x)| \le \sqrt{2Mf(x)}$.}
\reponse{$\Delta \le 0$.}
    \item \question{On suppose que : $\forall\ x \in \R,\ |f'(x)| = \sqrt{2Mf(x)}$.
    Que pouvez-vous dire de $f$ ?}
\reponse{$\sqrt f$ est affine.}
\end{enumerate}
}
