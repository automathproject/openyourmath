\chapitre{Propriétés de R}
\sousChapitre{Maximum, minimum, borne supérieure}

\contenu{
\texte{
Les ensembles suivants ont-ils une borne supérieure, un plus 
grand élément, une borne inférieure, un plus petit élément, 
dans $\mathbb{D}$, dans $\mathbb{Q}$, dans $\mathbb{R}$, 
(si la question se pose) ?
}
\begin{enumerate}
    \item \question{$[0,3\mathclose[$,}
    \item \question{$\{0\}\cup\mathopen]1,2]$,}
    \item \question{$\mathbb{D} \cap[0,1/3]$,}
    \item \question{$\{x \mid \exists n \in\mathbb{N},\; x=1/n\}$,}
    \item \question{$\{x\in\mathbb{Q}\mid x^2<2 \}$.}
\end{enumerate}
}
