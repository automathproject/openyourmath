\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f$ continue de $[0,1]$ dans ${\Rr}, n\in \Nn$ tels que :
$$\forall k\in \{0,...,n\},\int_{0}^{1}f(u)u^{k}du=0. $$
Montrer que $f$ admet au moins $n+1$ z\'{e}ros distincts dans $]0,1[.$
}
}
