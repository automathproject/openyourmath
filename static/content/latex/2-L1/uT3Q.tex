\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Etude de fonctions}

\contenu{
\texte{
Montrer que les fonctions : $x  \mapsto \ch(n\Argch(x))$ et
$x  \mapsto \frac{\sh(n\Argch(x))}{\sqrt{x^2-1}}$ $(n \in \N)$ sont
polynomiales.
}
}
