\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f:{\Rr}^{+}\rightarrow {\Rr}$ continue, tendant vers $\ell$ en $+\infty , $
montrer alors :
$$\lim\limits_{n\rightarrow \infty }\int_{0}^{\infty }\frac{f(t)n}{n^{2}+t^{2}}dt
=\frac{\pi }{2}\ell. $$
}
}
