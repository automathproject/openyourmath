\texte{
Soit $u_n = \frac{(-1)^n}{n^{3/4}+\cos n}$.
}
\begin{enumerate}
    \item \question{La série $\sum u_n$ est-elle absolument convergente ?}
    \item \question{En écrivant $u_n = \frac{(-1)^n}{n^{3/4}} + v_n$, étudier la convergence
    de $\sum u_n$.}
\reponse{
$|v_n| =  O(n^{-3/2})  \Rightarrow $ CV.
}
\end{enumerate}
