\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f:[0,1]\rightarrow {\Rr}$ une application continue strictement
croissante telle que :
$$f(0)=0,\  f(1)=1. $$
Calculer :
$$\lim\limits_{n\rightarrow \infty }\int_{0}^{1}f^{n}(t)dt. $$
}
}
