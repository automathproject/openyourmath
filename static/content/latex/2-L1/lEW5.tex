\contenu{
\texte{
Soit $u_n = \sqrt{ n + \sqrt{ n-1 + \dots + \sqrt 1 }}$.
}
\begin{enumerate}
    \item \question{Montrer que la suite $\left(\frac{u_n}{\sqrt n}\right)$ est born{\'e}e.}
\reponse{Par r{\'e}currence, $\sqrt n \le u_n \le \sqrt{2n}$.}
    \item \question{D{\'e}terminer $\lim_{n\to\infty} \left(\frac{u_n}{\sqrt n}\right)$.}
\reponse{$\sqrt n \le u_n \le \sqrt{n+\sqrt{2(n-1)}}  \Rightarrow  \lim = 1$.}
    \item \question{D{\'e}terminer $\lim_{n\to\infty}(u_n-\sqrt n\,)$.}
\reponse{$\sqrt{n+\sqrt{n-1}} \le u_n \le \sqrt{n+\sqrt{n-1+\sqrt{2(n-2)}}}
               \Rightarrow  \lim = \frac 12$.}
\end{enumerate}
}
