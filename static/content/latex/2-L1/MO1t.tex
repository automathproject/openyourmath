\chapitre{Calcul d'intégrales}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : {[0,1]} \to {[0,1]}$ définie par
$f(0,a_1a_2a_3\dots) = 0,a_2a_1a_3\dots$ (échange des deux $1^{\text{ères}}$
décimales).

Montrer que $f$ est continue par morceaux et calculer $ \int_{t=0}^1 f(t)\,d t$.
}
}
