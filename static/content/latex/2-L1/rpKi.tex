\contenu{
\texte{
Etudier la limite des suites suivantes :
$a_n=\cos{(\dfrac{2^n}{n!})}$; $b_n=\sqrt[n]{3-\sin{n}^2}$;
$c_n=\dfrac{n^3+2^n}{3^n}$;
$d_n=\dfrac{n^2+(-1)^n}{n^2+\sqrt{n}}$;
$e_n=(\cos{n})\sin{\dfrac{(-1)^n}{\sqrt{n}}}$.
}
}
