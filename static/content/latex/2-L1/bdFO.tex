\texte{
Soient $M,M'$ deux points du plan d'affixes $z = x+iy$ et $z' = x'+iy'$.
}
\begin{enumerate}
    \item \question{On suppose que $z$ et $z'$ sont liés par la relation : $z' = e^z$.
    \'Etudier la courbe décrite par $M'$ lorsque $M$ décrit~:
  \begin{enumerate}}
    \item \question{une droite $x = $ cste. \label{droite verticale}}
    \item \question{une droite $y = $ cste. \label{droite horizontale}}
    \item \question{une droite quelconque.}
\end{enumerate}
}
