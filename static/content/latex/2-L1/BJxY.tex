\contenu{
\texte{
Montrer que pour $x\in\C$, $|x|<1$, on a l'égalité :
$\sum_{n=0}^{+\infty}\frac{x^{2n+1}}{1-x^{2n+1}}=
\sum_{n=1}^{+\infty}\frac{x^{n}}{1-x^{2n}}$.
}
\reponse{
$\sum_{n=0}^{+\infty}\frac{x^{2n+1}}{1-x^{2n+1}}=
\sum_{(n,p)\in\N}^{+\infty}x^{(p+1)(2n+1)}=\sum_{p=0}^{+\infty}\frac{x^{p+1}}{1-x^{2p+2}}$.
}
}
