\chapitre{Développement limité}
\sousChapitre{Calculs}

\contenu{
\texte{
Calculer le développement limité de $\left(\frac{\tan x}x\right)^{1/x^2}$
en~$0$ à l'ordre~$3$.
}
\reponse{
$e^{1/3}\left(1+\frac7{90}x^2 +  o(x^3)\right)$.
}
}
