\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Continuité : pratique}

\contenu{
\texte{
On admet que pour tout $x\in \mathbb{R} , \vert \sin x \vert \leq \vert x\vert .$
}
\begin{enumerate}
    \item \question{Montrer que $x\mapsto \sin x $ est continue en $0$ puis sur
$\mathbb{R}$ tout entier.}
    \item \question{En déduire que $x\mapsto \cos x $ est continue sur $\mathbb{R}$.}
\end{enumerate}
}
