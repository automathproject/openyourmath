\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Calculer par récurrence :
$I_n =  \int_{t=0}^1 \frac{t^n\,d t}{\sqrt[4]{t^3(1-t)}} $.
}
\reponse{
$I_n = \pi\sqrt2 \prod_{k=1}^n \bigl( 1-\frac3{4k} \bigr) $.
}
}
