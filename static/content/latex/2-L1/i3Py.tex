\contenu{
\texte{
On réordonne les termes de la série harmonique alternée en prenant tour
à tour $p$ termes positifs puis $q$ termes négatifs, $p,q\ge 1$.
Calculer la somme de la série correspondante.
}
\reponse{
$\ln2 + \frac12\ln(p/q)$.
}
}
