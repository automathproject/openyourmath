\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : {[a,b]} \to {\R}$ dérivable telle que
$f(a) = f(b) = 0$, et $f'(a) > 0$, $f'(b) > 0$.

Montrer qu'il existe $c \in {]a,b[}$ tel que $f(c) = 0$, et $f'(c) \le 0$.
}
}
