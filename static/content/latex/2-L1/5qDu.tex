\contenu{
\texte{
\'Etudier la convergence des séries de terme général :
}
\begin{enumerate}
    \item \question{$\left(1+\frac 1n\right)^n - e$.}
\reponse{$\sim -\frac e{2n}  \Rightarrow $ DV.}
    \item \question{$\ch^\alpha n - \sh^\alpha n$.}
\reponse{$\sim \frac \alpha{2^{\alpha-1}}e^{n(\alpha-2)}  \Rightarrow $
             CV ssi $\alpha < 2$.}
    \item \question{$2\ln(n^3+1) - 3\ln(n^2+1)$.}
\reponse{$\sim -\frac 3{n^2}  \Rightarrow $ CV.}
    \item \question{$\sqrt[n] {n+1} - \sqrt[n] n$.}
\reponse{$\sim \frac 1{n^2}  \Rightarrow $ CV.}
    \item \question{$\arccos\left(\frac {n^3+1}{n^3+2} \right)$.}
\reponse{$\sim \sqrt{\frac 2{n^3}}  \Rightarrow $ CV.}
    \item \question{$\frac {a^n}{1+a^{2n}}$.}
\reponse{cv ssi $|a| \ne 1$.}
    \item \question{$\frac{(-1)^n}{\sqrt{n^2+n}}$.}
\reponse{Série alternée $ \Rightarrow $ CV.}
    \item \question{$\frac{(-1)^n}{\ln n}$.}
\reponse{Série alternée $ \Rightarrow $ CV.}
    \item \question{$\frac{1+(-1)^n\sqrt n}n$.}
\reponse{Harmonique + alternée $ \Rightarrow $ DV.}
    \item \question{$\frac{2.4.6\dots(2n)}{n^n}$.}
\reponse{d'Alembert $ \Rightarrow $ CV.}
    \item \question{$\frac{1!+2!+\dots+n!}{(n+2)!}$.}
\reponse{$\le \frac{(n-1)(n-1)!+n!}{(n+2)!} \le \frac 2{(n+1)(n+2)}  \Rightarrow $ CV.}
    \item \question{$\frac{1!-2!+\dots\pm n!}{(n+1)!}$.}
\reponse{$= \frac{(-1)^{n-1}}{n+1} +  O\Bigl(\frac1{n^2}\Bigr)  \Rightarrow $ CV.}
    \item \question{$\frac{(-1)^n}{\ln n + \sin(2n\pi/3)}$.}
\reponse{Décomposition en 3 séries alternées $ \Rightarrow $ CV.}
    \item \question{$\sqrt{1+\frac{(-1)^n}{\sqrt n}}-1$.}
\reponse{$=\frac{(-1)^n}{2\sqrt n} - \frac1{8n} +  O(n^{-3/2})  \Rightarrow $ DV.}
    \item \question{$\frac{(-1)^n}{\sqrt n + (-1)^n}$.}
\reponse{Regroupement de termes $ \Rightarrow $ DV.}
    \item \question{$\frac{(-1)^{[\sqrt n\,]}}n$.}
\reponse{Regroupement par paquets + CSI $ \Rightarrow $ CV.}
    \item \question{$\frac{(\ln n)^n}{n^{\ln n}}$.}
\reponse{Terme général ne tend pas vers zéro, DV.}
    \item \question{$\frac1{(\ln n)^{\ln n}}$.}
\reponse{$= \frac1{n^{\ln\ln n}}  \Rightarrow $ CV.}
\end{enumerate}
}
