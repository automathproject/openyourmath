\sousChapitre{Suite définie par une relation de récurrence}

\contenu{
\texte{
On pose $u_n = \sqrt{1 + \sqrt{2 + \sqrt{ \dots + \sqrt{n-1 +\sqrt n}}}}$
     et $v_n = \sqrt{1 + \sqrt{2 + \sqrt{ \dots + \sqrt{n-1 +\sqrt{2n}}}}}$.
}
\begin{enumerate}
    \item \question{Montrer que ces suites sont convergentes.}
    \item \question{On note $\lambda = \lim_{n\to\infty} u_n$.
    Montrer que $\lambda-u_n \le \frac n{2^n\sqrt{n!}}$.}
\indication{\begin{enumerate}
\item
\item $\sqrt a - \sqrt b = \frac {a-b}{\sqrt
      a + \sqrt b}$
\end{enumerate}}
\end{enumerate}
}
