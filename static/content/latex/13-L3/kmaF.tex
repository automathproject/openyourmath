\chapitre{Solution maximale}
\sousChapitre{Solution maximale}

\contenu{
\texte{
On considère l'équation différentielle $x'= |x|+|t|$.
}
\begin{enumerate}
    \item \question{Montrer que pour tout réel $x_0$, il existe une solution
maximale $(\varphi , J)$ telle que $\varphi (0) = x_0$.}
    \item \question{Déterminer la solution maximale correspondant  à $x_0
=1$, en distinguant les cas $t\geq 0$ et $t<0$, et vérifier
qu'elle est définie sur $\Rr$ tout entier. Combien de fois
est-elle dérivable?}
\end{enumerate}
}
