\chapitre{Equation aux dérivées partielles}

\contenu{
\texte{
Soit $f : \Rr^3 \rightarrow \Rr$ d\'erivable. On pose $F (x, y, z) = f (x-y, y-z, z-x)$.\\
Calculer $\frac{\partial F}{\partial x} + \frac{\partial F}{\partial y} +
\frac{\partial F}{\partial z}$.
}
}
