\contenu{
\texte{
Pour une fonction {\bf réelle}  $x$ de la variable $t$ on
considère le système 
\begin{equation}
\begin{cases}
x^{(4)} - 2x^{(3)} + 2x'' - 2x' + x = 0 
\\ 
x(0) = 0 \quad,\quad x''(0) = -2 \quad ,\quad  x^{(3)}(0) =
0 
\end{cases}
\tag{*}
\end{equation}
}
\begin{enumerate}
    \item \question{Le système \thetag* a-t-il une solution
unique~? Pourquoi~?

\medskip
\noindent
Si votre réponse à la question 1. est oui,
répondre à la question 2., si votre réponse à la
question 1. est non, répondre à la question 3..}
    \item \question{Trouver la solution unique de \thetag*.
Est-elle périodique~?}
    \item \question{Trouver toutes les solutions de \thetag*.
Y-a-t-il des solutions périodiques~? Si oui, les
expliciter.}
\end{enumerate}
}
