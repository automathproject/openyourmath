\titre{ Résolution d'un système linéaire }
\theme{matrices}
\auteur{}
\organisation{AMSCC}

\contenu{
    \texte{
        Soit la matrice $M = \begin{pmatrix}
            0&1&2\\
            1&1&2\\
            0&2&3
            \end{pmatrix}$.
    }
\begin{enumerate}
    \item \question{ Montrer que $M$ est inversible et calculer $M^{-1}$. }
    \item \question{ Résoudre le système linéaire suivant : 
        $$
        \left\{\begin{aligned}
        y+2z & =1 \\
        x+y+2z & =2 \\
        2y+3z & =3
        \end{aligned}\right.
        $$
    }
\end{enumerate}
}
