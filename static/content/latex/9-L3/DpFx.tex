\begin{enumerate}
    \item \question{Montrer que le plan projectif $P_1$
et la droite $P_2$ dans $\mathbb{P}^3$ soit se coupent en un point soit $P_2\subset
P_1.$}
    \item \question{Soient $P_i=P(M_i)$ deux sous-espaces projectifs $(i=1,2)$. Montrer
  que si $P_1\cap P_2=\emptyset$ alors la somme $M_1+M_2$ est directe.}
\end{enumerate}
