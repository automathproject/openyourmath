\chapitre{Géométrie et trigonométrie hyperbolique}
\sousChapitre{Géométrie et trigonométrie hyperbolique}

\contenu{
\texte{
Montrer que si un élément $g\in M$ n'est pas strictement
loxodromique conjugué à

$z\to ke^{i\theta}\ (\theta\not= \pi+\pi m,\ k>0)$ alors il existe deux
familles ${\cal C}_i\ (i=1,2)$ de cercles généralisés vérifiant les conditions
suivantes :
}
\begin{enumerate}
    \item \question{$\forall\ C\in {\cal C}_1\ :\ g(C)=C$}
    \item \question{$\forall\ C\in {\cal C}_2\ :\ g(C)= {\cal C}_2\setminus \{C\}$}
    \item \question{$\forall\ C_1\in {\cal C}_1,\ \forall\ C_2\in {\cal C}_2\ :\ C_2\ \bot\ C_1.$}
\end{enumerate}
}
