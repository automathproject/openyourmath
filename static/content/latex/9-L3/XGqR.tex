\begin{enumerate}
    \item \question{Notons $l \subset \Rr^2$ une droite affine et $ \tau_l$ la réflexion
par rapport à $l$. Montrer que si $f \in Iso (\Rr^2)$ vérifie $f_{\vert
l}\equiv id$ alors soit $f = id $ soit $ f= \tau _l$.}
    \item \question{Soient $l$ et $m$ deux droites affines dans $\Rr^2$.
  
  \begin{enumerate}}
    \item \question{Montrer qu'il existe $\alpha \in Iso (\Rr^2)$ telles que
  $\alpha \tau_l \alpha^{-1}= \tau_m. $}
    \item \question{Montrer que $\tau _l .\tau _m $ est une translation si et seulement si $l$ et
  $m$ sont parallèles.}
\end{enumerate}
