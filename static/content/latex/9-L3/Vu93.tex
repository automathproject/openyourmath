\contenu{
\texte{
Montrer que la fibration de Hopf $FH:S^3\to S^2$ ( $FH^{-1}(x)$ est un  grand
cercles de $S^3\ (\forall\ x\in S^2)$) s'écrit comme ceci :

$$FH(z,z')=(2z'\overline{z}, \vert z'\vert^2-\vert z\vert^2).$$
}
}
