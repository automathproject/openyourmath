\chapitre{Géométrie et trigonométrie sphérique}
\sousChapitre{Géométrie et trigonométrie sphérique}

\contenu{
\texte{
Soit $\alpha:[a,b]\to S^n$ une courbe  avec $b-a < \pi.$ Montrer l'équivalence des
conditions suivantes :
}
\begin{enumerate}
    \item \question{$\alpha$ est une courbe géodésique.}
    \item \question{Il existe deux vecteurs orthogonaux $A, B\in S^n$ tels que
$\alpha(t)= A\cos (t-a) + B \sin(t-a).$}
    \item \question{La courbe $\alpha$ vérifie l'équation $\alpha''+\alpha =0.$}
\end{enumerate}
}
