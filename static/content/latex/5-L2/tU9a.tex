\chapitre{Equation différentielle}
\sousChapitre{Equations différentielles non linéaires}

\contenu{
\texte{
$\begin{cases} y'' + \sin y = 0 \cr  y(0) = 0,\ y'(0) = \alpha \ge 0.\cr\end{cases}$

Soit $y$ la solution maximale.
On a l'intégrale première : $\frac {y'^2}2 - \cos y = C = \alpha^2 - 1$.
}
\begin{enumerate}
    \item \question{\begin{enumerate}}
    \item \question{Montrer que $y$ est définie sur $\R$.}
    \item \question{Montrer que $y$ est impaire.}
\end{enumerate}
}
