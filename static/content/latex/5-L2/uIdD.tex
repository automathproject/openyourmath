\contenu{
\texte{
Soit $f:\R \to \R$ continue, ayant même limite finie~$\ell$ en $\pm\infty$.
Montrer que $f$ est limite uniforme sur~$\R$ de fractions rationnelles.
}
\reponse{
$g = x \mapsto f(\tan(x/2))$ est limite uniforme de polynômes trigonométriques.
}
}
