\texte{
Calculer les coordonn\'ees du centre d'inertie (de gravit\'e) du domaine $D$ :
}
\begin{enumerate}
    \item \question{$D=\left\{ (x,y)\in \R^2 \mid \frac{x^2}{a^2}+\frac{y^2}{b^2} \leq 1\, ,\, x\geq 0\, ,\, y\geq 0 \right\}$ (le quart d'ellipse) ;}
    \item \question{$D=\left\{ (x,y)\in \R^2 \mid x^2+y^2 \leq 1\, ,\, x\geq 0 \, ,\, |y|\leq ax \right\}$ ;}
    \item \question{$D=\{ (x,y)\in \R^2 \mid  x^2+y^2\leq 9\, ,\, (x-1)^2+y^2\geq 1 \}$.}
\end{enumerate}
}
