\texte{
Montrer que la longueur d'une ellipse de demi-axes $a,b$ est :
$L = 2\pi\sqrt{\frac{a^2+b^2}2}\sum_{p=0}^\infty
\left(\frac{a^2-b^2}{a^2+b^2}\right)^{2p}\frac{C_{4p}^{2p} C_{2p}^p}
      {4^{3p}(1-4p)}$.
}
}
