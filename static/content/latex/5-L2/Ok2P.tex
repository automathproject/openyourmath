\sousChapitre{Intégrale de Riemann dépendant d'un paramètre}

\contenu{
\texte{
Montrer qu'il existe un unique réel $x \in {[0,\pi]}$ tel que
$ \int_{\theta=0}^\pi \cos(x\sin\theta)\, d\theta = 0$.
Calculer une valeur approchée de $x$ à $10^{-2}$ près.
}
\reponse{
$2.40 < x < 2.41$.
}
}
