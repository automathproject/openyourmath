\contenu{
\texte{
Résoudre sur $(\R^{+*})^2$ : $x\frac{\partial f}{\partial x} = y\frac{\partial f}{\partial y}$,
en posant $\begin{cases} u = xy \cr v = \frac xy. \end{cases}$
}
\reponse{
$f(x,y) = g(xy)$.
}
