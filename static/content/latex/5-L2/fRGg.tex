\contenu{
\texte{
Soit $f:\R \to \R$ de classe $\mathcal{C}^2$ et $g(x,y,z) = \frac {f(r)}r$
avec $r = \sqrt{x^2+y^2+z^2}$.
Déterminer $f$ de sorte que $\frac{\partial^2 g}{\partial x^2} + \frac{\partial^2 g}{\partial y^2} + \frac{\partial^2 g}{\partial z^2} = -g$.
}
\reponse{
$f(r) = A\cos r + B\sin r$.
}
}
