\chapitre{Fonction de plusieurs variables}
\sousChapitre{Extremums locaux}

\contenu{
\texte{
Soit $ABC$ un triangle de cotés $a,b,c$.
}
\begin{enumerate}
    \item \question{Calculer l'aire, $S$, de $ABC$ en fonction de $a,b,c$.}
\reponse{$4S=\sqrt{2a^2b^2 + 2a^2c^2 + 2b^2c^2 - a^4 - b^4 - c^4}$.}
    \item \question{Montrer que $\frac{S}{a^2+b^2+c^2}$ est maximal lorsque $ABC$ est équilatéral.}
\reponse{max = $\frac4{\sqrt3}$.}
\end{enumerate}
