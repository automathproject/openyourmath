\contenu{
\texte{
Développer en série entière $\frac{e^x}{1-x}$ puis $\frac{e^{x^2}}{1-x}$.
}
\reponse{
$\frac{e^{x^2}}{1-x} = (1+x)\frac{e^{x^2}}{1-x^2}
          = \sum_{n=0}^\infty \left(1+\frac1{1!} + \dots
          + \frac1{n!}\right)(x^{2n}+x^{2n+1})$.
}
}
