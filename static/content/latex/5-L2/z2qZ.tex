\begin{enumerate}
    \item \question{Calculer $\varphi(a) =  \int_{t=0}^1 \frac{d t}{1+at}$.}
\reponse{$\frac{\ln(1+a)}a$.}
    \item \question{En déduire la valeur de $ \int_{t=0}^1 \frac{td t}{(1+at)^2}$.}
\reponse{$=-\varphi'(a) = \frac{\ln(1+a)}{a^2} - \frac1{a(1+a)}$.}
\end{enumerate}
