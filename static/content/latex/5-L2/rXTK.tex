\contenu{
\texte{
Soit $f_n(x) = n\cos^nx\sin x$.
}
\begin{enumerate}
    \item \question{Chercher la limite simple, $f$, des fonctions $f_n$.}
    \item \question{Vérifier que $ \int_{t=0}^{\pi/2} f(t)\,d t\ne
          \lim_{n\to\infty}  \int_{t=0}^{\pi/2} f_n(t)\,d t$.}
\end{enumerate}
}
