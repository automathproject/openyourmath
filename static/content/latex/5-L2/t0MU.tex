\begin{enumerate}
    \item \question{Calculer $f(a) =  \int_{t=0}^{+\infty} e^{-t^2}\cos(at)\,d t$.}
\reponse{$f'(a) = -\frac a2f(a)  \Rightarrow  f(a) = \frac{\sqrt\pi}2\exp(-a^2/4)$.}
    \item \question{Soit $g(a) =  \int_{t=0}^{+\infty} e^{-t^2}\frac{\sin(at)}t\,d t$~;
    calculer $\lim_{a\to+\infty} g(a)$.}
\reponse{$g'(a) = f(a)  \Rightarrow  g(a) \to \frac\pi2$ lorsque $a\to+\infty$.}
\end{enumerate}
