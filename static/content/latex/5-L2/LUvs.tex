\contenu{
\texte{
Calculer $\iint_D^{} f(x,y)\,d xd y$ :
}
\begin{enumerate}
    \item \question{$D=\{y\ge0, x+y\le1, y-x\le1\}$,                     \par\nobreak $f(x,y)=x^2y$.}
\reponse{$\frac1{30}$.}
    \item \question{$D=\{x^2+y^2\le R^2\}$,                              \par\nobreak $f(x,y)=x^2y$.}
\reponse{$0$.}
    \item \question{$D=\{\frac{x^2}{a^2} + \frac{y^2}{b^2} \le 1\}$,   \par\nobreak $f(x,y)=x^2 + y^2$.}
\reponse{$\frac\pi4 ab(a^2+b^2)$.}
    \item \question{$D=\{0 \le x \le 1-\frac{y^2}4\}$,                  \par\nobreak $f(x,y)=x^2 + y^2$.}
\reponse{$\frac{96}{35}$.}
    \item \question{$D=\{x^2+y^2 \le 1\}$,                               \par\nobreak $f(x,y)=(x+y)^2$.}
\reponse{$\frac\pi2$.}
    \item \question{$D=\{x^2+y^2 \le 1\}$,                               \par\nobreak $f(x,y)=\frac{(x+y)^2}{x^2+y^2+1}$.}
\reponse{$\pi(1-\ln 2)$.}
    \item \question{$D=\{x\ge0, y\ge0, x+y\le1\}$,                       \par\nobreak $f(x,y)=x+y+1$.}
\reponse{$\frac56$.}
    \item \question{$D=\{|x+y|\le1,|x-y|\le1\}$,                         \par\nobreak $f(x,y)=\ln(x+y+1)$.}
\reponse{$2(\ln 2-1)$.}
    \item \question{$D=\{x\ge0, y\ge0, x+y\le\pi\}$,                     \par\nobreak $f(x,y)=(x+y)\sin x\sin y$.}
\reponse{$\frac{3\pi}2$.}
    \item \question{$D=\{|x|\le x^2+y^2\le 1\}$,                         \par\nobreak $f(x,y)=(1+x^2+y^2)^2$.}
\reponse{$\frac{65\pi}{48}$.}
    \item \question{$D=\{x\ge0, y\ge0, x+y\le a\}$,                      \par\nobreak $f(x,y)=x+y+\sqrt{a^2+(x+y)^2}$.}
\reponse{$\frac{2\sqrt2}3 a^3$.}
    \item \question{$D=\{x\ge0, y\ge0, x^2+y^2\le 1\}$,                  \par\nobreak $f(x,y)=xy\sqrt{x^2+4y^2}$.}
\reponse{$\frac7{45}$.}
    \item \question{$D=\{x^2+y^2-2y \le 0\}$,                            \par\nobreak $f(x,y)=y\exp(x^2+y^2-2y)$.}
\reponse{$\pi(1-\frac1e )$.}
    \item \question{$D=\{y^2 \le 2px, x^2 \le 2py\}$,                    \par\nobreak $f(x,y)=\exp\left(\frac{x^3+y^3}{xy}\right)$.}
\reponse{$\frac{(e^{2p}-1)^2}3 \quad (x=u^2v,y=uv^2)$.}
\end{enumerate}
}
