\texte{
Soit $A$ une partie convexe d'un evn $E$.
}
\begin{enumerate}
    \item \question{D{\'e}montrer que $\overline A$ et $\mathring A$ sont aussi convexes
    (pour $\mathring A$ : faire un dessin).}
    \item \question{Montrer que l'application $x \mapsto d(x,A)$ est convexe (c.a.d.
    $d(tx+(1-t)y,A) \le td(x,A) + (1-t)d(y,A)$).}
\end{enumerate}
