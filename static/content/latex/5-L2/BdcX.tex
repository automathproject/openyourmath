\contenu{
\texte{

}
\begin{enumerate}
    \item \question{\'Etudier la convergence simple, uniforme, de la série de fonctions :
    $f(x) = \sum_{n=0}^\infty ne^{-nx}$.}
    \item \question{Calculer $f(x)$ lorsque la série converge (intégrer terme à terme).}
\reponse{
$f(x) = \frac{e^x}{(e^x-1)^2}$.
}
\end{enumerate}
