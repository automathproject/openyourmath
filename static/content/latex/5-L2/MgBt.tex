\chapitre{Suite et série de fonctions}
\sousChapitre{Convergence simple, uniforme, normale}

\contenu{
\texte{
Soit ${f_n} : {[0,+\infty[} \to   {\R}$, $\begin{cases}
                     x \le n             &(1- x/n)^n \cr
                      x > n               &0.         \cr\end{cases}$
}
\begin{enumerate}
    \item \question{Déterminer la limite simple, $f$, des fonctions $f_n$.}
    \item \question{Montrer que : $\forall\ x\in \R^+,\ 0 \le f_n(x) \le f(x)$.}
    \item \question{Montrer que $(f_n)$ converge uniformément vers $f$ sur tout segment $[0,a]$.}
    \item \question{Démontrer que la convergence est uniforme sur $\R^+$.}
\reponse{
$e^{-x}$.
}
\end{enumerate}
}
