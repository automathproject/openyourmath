\texte{
Soit $A$ une partie born{\'e}e de $\R$ ayant un seul point d'accumulation, $a$.
}
\begin{enumerate}
    \item \question{Montrer que $A$ est d{\'e}nombrable.}
    \item \question{On num{\'e}rote les {\'e}l{\'e}ments de $A$ d'une mani{\`e}re quelconque :
    $A = \{ x_1, x_2, \dots, x_n, \dots \}$.

    Montrer que $x_n \xrightarrow[n\to\infty]{} a$.}
\reponse{
$A \setminus ]a-\frac 1n, a+\frac 1n[$ est fini.
}
\end{enumerate}
