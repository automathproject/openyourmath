\chapitre{Fonction de plusieurs variables}
\sousChapitre{Dérivée partielle}

\contenu{
\texte{
Soient $P \in \C[X]$ et $f : {\R^2} \to \C, {(x,y)} \mapsto {P(x+iy).}$
Montrer que $\frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial y^2} = 0$.
}
}
