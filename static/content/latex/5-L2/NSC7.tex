\chapitre{Fonction de plusieurs variables}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f:\R^n\to \R$ une application diff\'erentiable. Montrez que si $\| \nabla f(x)\| 
\leq M,\; \forall x\in \R^n$, alors 
$$
|f(x)-f(y)|\leq M\| x-y\|,\; \forall x,y \in \R^n \; .
$$
}
}
