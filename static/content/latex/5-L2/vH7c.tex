\texte{
Soit $A$ une partie non vide d'un espace vectoriel norm{\'e} $E$.
Pour $x \in E$, on pose $d(x,A) = \inf\{d(x,a)\text{ tq } a\in A\}$.
}
\begin{enumerate}
    \item \question{Montrer que : $\forall\ x,y \in E,\ |d(x,A) - d(y,A)| \le d(x,y)$.}
    \item \question{Montrer que l'application $x  \mapsto d(x,A)$ est continue.}
\end{enumerate}
}
