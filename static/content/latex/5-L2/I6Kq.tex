\chapitre{Suite et série de fonctions}
\sousChapitre{Convergence simple, uniforme, normale}

\contenu{
\texte{
Soit $y_n$ la solution de l'équation :
$(*_n) \Leftrightarrow \left(1+\frac1n\right)y'' - \left(2+\frac1n\right)y' + y = 0$
vérifiant les conditions initiales~: $y(0) = 0$, $y'(0) = 1$.
}
\begin{enumerate}
    \item \question{Calculer explicitement $y_n$.}
    \item \question{Déterminer la limite simple, $y$, des fonctions $y_n$.}
    \item \question{Vérifier que $y$ est solution de l'équation limite de $(*_n)$ avec les mêmes
    conditions initiales.}
\reponse{
$y_n = (n+1)\left(e^x - e^{nx/(n+1)}\right)$.
$y = xe^x$.
}
\end{enumerate}
}
