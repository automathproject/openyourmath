\chapitre{Intégration}
\sousChapitre{Intégrale de Riemann dépendant d'un paramètre}

\contenu{
\texte{
On considère les fonctions définies par :
$f(x) = \left({ \int_{t=0}^x e^{-t^2}\,d t}\right)^2$ et
$g(x) =  \int_{t=0}^1 \frac{e^{-x^2(1+t^2)}}{1+t^2}\,d t$.
}
\begin{enumerate}
    \item \question{Montrer que $f$ et $g$ sont dérivables et calculer $f'$ et $g'$.}
    \item \question{Montrer que $f(x) + g(x) = \frac\pi4$ pour tout $x \in \R^+$.}
    \item \question{En déduire la valeur de $ \int_{t=0}^{+\infty} e^{-t^2}\,d t$.}
\reponse{
$g'(x) =  \int_{t=0}^1 (-2x)e^{-x^2(1+t^2)}\,d t = -2e^{-x^2}\sqrt{f(x)} = -f'(x)$.
}
\end{enumerate}
}
