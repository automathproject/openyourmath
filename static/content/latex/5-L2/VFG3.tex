\chapitre{Equation différentielle}

\contenu{
\texte{
Intégrer :
}
\begin{enumerate}
    \item \question{$y'' -2y' + 2y = xe^x$.}
\reponse{$y = (x + a\cos x + b\sin x) e^x$.}
    \item \question{$y'' - 4y' + 4y = 2(x-2)e^x$.}
\reponse{$y = (ax+b)e^{2x} + 2xe^x$.}
    \item \question{$y'' - 4y' + 13y = 10\cos 2x + 25\sin 2x$.}
\reponse{$y = e^{2x}(a\cos3x + b\sin3x) + 2\cos2x + \sin 2x$.}
    \item \question{$y'' + y = \mathrm{cotan} x$.}
\reponse{$y = \sin x\ln\Bigl|\tan \frac x2\Bigr| + \lambda \cos x + \mu\sin x$
               (variation de la constante avec sin).}
    \item \question{$y'' + 3y' + 2y = \frac{x-1}{x^2} e^{-x}$.}
\reponse{$y = (\lambda + \ln|x|)e^{-x} + \mu e^{-2x}$.}
    \item \question{$y'' + y = P(x)$ où $P$ est un polynôme.}
\reponse{$y = \lambda\cos x + \mu\sin x + \sum_{n=0}^\infty (-1)^n P^{(2n)}(x)$.}
    \item \question{$y'^2 + y^2 = 1$ (dériver).}
\reponse{$y = a\cos x + b\sin x$ avec $a^2 + b^2 = 1$ ou $y = \pm1$.}
\end{enumerate}
}
