\sousChapitre{Intégrale de Riemann dépendant d'un paramètre}

\contenu{
\texte{
On pose $I(x) =  \int_{t=0}^{\pi/2} \ln(\cos^2t + x^2\sin^2t)\,d t$.
}
\begin{enumerate}
    \item \question{Montrer que $I$ est de classe $\mathcal{C}^1$ sur $\R^{+*}$.}
    \item \question{Calculer $I'(x)$ et en déduire $I(x)$.}
\reponse{
$I'(x) = \frac\pi{x+1}$, $I(x) = \pi\ln\left(\frac{x+1}2\right)$.
}
\end{enumerate}
