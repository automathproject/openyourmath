\contenu{
\texte{
Soit $A =\{ n+p\sqrt2 \text{ tq } n,p \in \N,\ n+p\sqrt2 > 0,\ n^2-2p^2 = 1\}$.
Montrer que $A$ est un sous-groupe discret de $\R^{+*}$.
}
\reponse{
$n+p\sqrt2 > 1  \Rightarrow  n>0,\ p>0$, donc $A\cap]1,+\infty[$
         admet un plus petit {\'e}l{\'e}ment : $3+2\sqrt2$.
}
}
