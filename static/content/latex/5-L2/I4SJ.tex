\texte{
Soit $\|.\|$ une norme sur $\R^2$, $A \subset \R^2$ une partie born{\'e}e contenant au moins deux points.
}
\begin{enumerate}
    \item \question{Montrer qu'il existe une boule ferm{\'e}e de rayon minimum contenant $A$.}
    \item \question{Montrer que cette boule n'est pas n{\'e}cessairement unique (on prendra $\|.\| = \|.\|_\infty$).}
    \item \question{Montrer que si $\|.\|$ est une norme euclidienne, alors la boule pr{\'e}c{\'e}dente est unique.}
\end{enumerate}
}
