\contenu{
\texte{
Trouver les fonctions $f:\R \to \R$ $2\pi$-périodiques de classe $\mathcal{C}^2$
telles que $ \int_{t=0}^{2\pi}f(t)\,d t = 0$ et $|f''| \le |f|$.
}
}
