\texte{
Soit $A$ une partie compacte d'un evn $E$ et $f : A \to F$ une fonction
continue et injective ($F$ = evn).
}
\begin{enumerate}
    \item \question{Montrer que ${f^{-1}} : {f(A)} \to A$ est aussi continue.}
    \item \question{Donner un exemple o{\`u} $A$ n'est pas compact et $f^{-1}$ n'est pas continue.}
\end{enumerate}
