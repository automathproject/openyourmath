\chapitre{Fonction de plusieurs variables}
\sousChapitre{Différentiabilité}

\contenu{
\texte{
Soient $\alpha>0$ et $f~:\,\R^2 \to \R$ d\'efinie par
$$\left \{\begin{array}{l}
\displaystyle f(x,y) = \frac{|x|^\alpha y}{x^2+y^4} \quad \mbox{si }(x,y)\neq
(0,0)\\
f(0,0)=0
\end{array}\right.$$
}
\begin{enumerate}
    \item \question{\begin{enumerate}}
    \item \question{Montrer que
$$\displaystyle \forall (x,y)\neq (0,0)\quad \left|f(x,y)\right|\leq
\left(x^2+y^4\right)^{\frac{2\alpha -3}4}.$$}
    \item \question{Calculer $\displaystyle \lim_{
\begin{array}{c}y\to 0\\y\neq 0\end{array}}
\left|f(y^2,y)\right|$.}
    \item \question{\'Etudier la continuit\'e de $f$ en (0,0).}
\end{enumerate}
}
