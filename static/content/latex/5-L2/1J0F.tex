\sousChapitre{Rayon de convergence}

\contenu{
\texte{
Soit $\sum a_n z^n$ et $\sum b_nz^n$ deux séries entières de rayon de convergence respectif $R_1$ et $R_2$.
}
\begin{enumerate}
    \item \question{Montrer que le rayon de convergence de $\sum(a_n+b_n)z^n$ est supérieur à $\min(R_1,R_2)$ avec égalité si $R_2\not=R_1$.}
    \item \question{Montrer que le rayon de convergence de $\sum(a_nb_n)z^n$ est supérieur à $R_1R_2$.}
\end{enumerate}
}
