\texte{
Soit $E$ un evn et $A \subset E$. On suppose que $A$ est {\`a} la fois ouvert
et ferm{\'e}.
}
\begin{enumerate}
    \item \question{Exemples de telles parties ?}
    \item \question{On d{\'e}finit la fonction $f : E \to {\R}$ par
  $\begin{cases}
    \vec x &1 \text{ si } \vec x \in A,\cr
     \vec x &0 \text{ si } \vec x \notin A\cr \end{cases}$
  \begin{enumerate}}
    \item \question{Montrer que $f$ est continue.}
    \item \question{On prend $\vec a \in A$ et $\vec b \notin A$.
        Montrer que $\varphi : {[0,1]} \to {\R}, t \mapsto {f(t\vec a + (1-t)\vec b)}$
        est continue.}
    \item \question{Conclure.}
\end{enumerate}
}
