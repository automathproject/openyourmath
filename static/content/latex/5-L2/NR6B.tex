\contenu{
\texte{
Soient ${f,g} : \R \to \C$ de classe $\mathcal{C}^1$ telles que $f,f',g,g'$ sont
intégrables. Montrer~:
$$ \int_{t=-\infty}^{+\infty}\hat f(t)g(t)\,d t =
   \int_{t=-\infty}^{+\infty} f(t)\hat g(t)\,d t.$$
}
}
