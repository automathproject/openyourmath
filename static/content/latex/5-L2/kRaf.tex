\texte{
On note $T_n$ le nombre de partitions d'un ensemble à $n$ éléments.
}
\begin{enumerate}
    \item \question{Montrer que $T_{n+1} = \sum_{k=0}^n {\binom{n}{k}}T_k$.}
    \item \question{Montrer que $\sum_{n=0}^\infty \frac{T_nx^n}{n!} = e^{e^x-1}$.}
\reponse{
$f'(x) = e^xf(x)$.
}
\end{enumerate}
