\chapitre{Equation différentielle}
\sousChapitre{Equations différentielles linéaires}

\contenu{
\texte{
Intégrer les équations suivantes :
}
\begin{enumerate}
    \item \question{$y''-y'-e^{2x}y = e^{3x}$ (poser $u = e^x$).}
\reponse{$y = -e^x + \lambda e^{e^x} + \mu e^{-e^x}$.}
    \item \question{$y'' - \left(6x+\frac1x\right)y' + 8x^2y = x^4$ (poser $u=x^2$).}
\reponse{$y = \lambda e^{x^2} + \mu e^{2x^2} + \frac{2x^2+3}{16}$.}
    \item \question{$x(1-2\ln x)y'' + (1+2\ln x)y' - \frac4xy = 0$
    (chercher une solution de la forme $y = x^\alpha$).}
\reponse{$y=\lambda x^2 + \mu\ln x$.}
    \item \question{$x^2y''-2xy'+2y = 2+2x^3\sin x$ (poser $u = \ln x$).}
\reponse{$y = ax + bx^2 + 1 - 2x\sin x$.}
    \item \question{$x(x+1)y'' - y' - 2y = 3x^2$
    (chercher une solution de l'équation homogène de la forme $y = x^\alpha$).}
\reponse{$y = x^2\ln|x+1| + \lambda \left(x^2\ln\left|\frac x{x+1}\right| + x-\frac12\right) + \mu x^2$.}
    \item \question{$x^2y'' + 4xy' + (2-x^2)y = 1$ $\Bigl($poser $y = \frac u{x^2}\Bigr)$.}
\reponse{$y = \frac{-1 + a\ch x + b\sh x}{x^2}$.}
    \item \question{$(x^2+3)y'' + xy' - y = 1$ (chercher les solutions polynomiales).}
\reponse{$y = \lambda\sqrt{x^2+3} + \mu x - 1$.}
    \item \question{$xy''-2y'-xy = 0$ (dériver deux fois).}
\reponse{$y^{(4)} - 2y'' + y = 0  \Rightarrow  y = a(\ch x -x\sh x) + b(x\ch x -\sh x)$.}
\end{enumerate}
}
