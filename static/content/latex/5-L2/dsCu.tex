\contenu{
\texte{
Soit $f_n(x) = \sum_{k=1}^n \frac{\sin kx}k$.
}
\begin{enumerate}
    \item \question{Calculer l'abscisse, $x_n$, du premier maximum positif de $f_n$.}
\reponse{$\frac\pi{n+1}$.}
    \item \question{Déterminer $\lim_{n\to\infty} f_n(x_n)$.}
\reponse{Somme de Riemman : $\ell =  \int_{t=0}^\pi \frac{\sin t}t\,d t$.}
\end{enumerate}
