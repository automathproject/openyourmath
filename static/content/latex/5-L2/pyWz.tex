\contenu{
\texte{
Soient $E,F$ deux espaces vectoriels norm{\'e}s et $f : E \to  {F.}$
On note $Gr(f) = \{(x,y) \in E\times F\text{ tq } y = f(x)\}$.
}
\begin{enumerate}
    \item \question{Montrer que si $f$ est continue, alors $Gr(f)$ est ferm{\'e} dans $E\times F$.}
    \item \question{Prouver la r{\'e}ciproque lorsque $f(E)$ est inclus dans un compact de~$F$.}
    \item \question{Donner un contre-exemple si $f(E)$ n'est pas inclus dans un compact.}
\end{enumerate}
}
