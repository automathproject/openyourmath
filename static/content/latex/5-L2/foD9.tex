\chapitre{Equation différentielle}
\sousChapitre{Equations différentielles linéaires}

\contenu{
\texte{
Intégrer les équations suivantes :
}
\begin{enumerate}
    \item \question{$(2+x)y' = 2-y$.}
\reponse{$y = 2 + \frac\lambda{x+2}$.}
    \item \question{$xy'+y = \cos x$.}
\reponse{$y = \frac{C+\sin x}x$.}
    \item \question{$(1+x)y' + y = (1+x)\sin x$.}
\reponse{$y = -\cos x + \frac{\sin x + \lambda}{1+x}$.}
    \item \question{$x^3y' -x^2y = 1$.}
\reponse{$y = \lambda x - \frac 1{3x^2}$.}
    \item \question{$3xy' - 4y = x$.}
\reponse{$y = \lambda x^{4/3} - x$.}
    \item \question{$y'+y = \sin x + 3\sin 2x$.}
\reponse{$y = \frac{\sin x - \cos x}2 + \frac{3\sin 2x - 6\cos 2x}5 + \lambda e^{-x}$.}
    \item \question{$2x(1-x)y' + (1-2x)y = 1$.}
\reponse{$y = \frac{\Argch(1-2x)+\lambda}{2\sqrt{x^2-x}}$ pour $x < 0$\par
               $y = \frac{\arcsin(2x-1)+\mu}{2\sqrt{x-x^2}}$ pour $0 < x < 1$\par
               $y = \frac{-\Argch(2x-1)+\nu}{2\sqrt{x^2-x}}$ pour $1 < x$.}
    \item \question{$x(x+1)y' + y = \arctan x$. \label{arctan}}
\reponse{$y = \frac {x-1}{2x} \arctan x +
       \frac {x+1}{2x} \left(\ln\frac {|x+1|}{\sqrt{x^2+1}} + \lambda \right)$.}
    \item \question{$x(x^2-1)y' + 2y = x\ln x - x^2$.
      

pour \ref{arctan} : \'Etudier les problèmes de raccordement.}
\reponse{$y = \frac{x}{1-x^2}\Bigl( (1+x)\ln x + 1 + \lambda x\Bigr)$.}
\end{enumerate}
}
