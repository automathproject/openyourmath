\contenu{
\texte{
Soit $k \in \R$.
Résoudre l'équation différentielle $y'' + k^2y = \sum_{n=1}^\infty \frac{\cos nx}{n^2}$.
}
\reponse{
$k \notin\Z$ : $y = \sum_{k=1}^\infty \frac{\cos nx}{n^2(k^2-n^2)}
                        + a\cos kx + b\sin kx$.\par
         $k\in\Z$ : remplacer $\frac{\cos kx}{k^2(k^2-k^2)}$ par
                        $\frac{x\cos kx}{2k^3}$.
}
}
