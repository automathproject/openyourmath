\begin{enumerate}
    \item \question{Dans $\R^2$ ou $\R^3$ euclidien muni d'une b.o.n., représenter les
ensembles suivants :
  \begin{itemize}}
    \item \question{$\mathcal{A}=\{(x,y)\in\R^2 \ | \  x^2-y^2>1 \mbox{ et } x^2+y^2<4\}$}
    \item \question{$\mathcal{B}=\{(x,y)\in\R^2 \ | \  (x-1)^2-y^2>1 \mbox{ et } x^2+\frac{y^2}{4}<4\}$}
    \item \question{$\mathcal{C}=\{(x,y,z)\in\R^3 \ | \  1<x+y+z<3 \mbox{ et } x>0 \mbox{ et }
y>0 \mbox{ et } z>0\}$}
    \item \question{$\mathcal{D}=\left\{
\begin{array}{cc}
(x,y,z)\in\R^3&\left|
\begin{array}{cc}
&x+y+z<1\\
\mbox{et }&x-y+z<1\\
\mbox{et }&-x-y+z<1
\end{array}
\right.
\end{array}
\right\}$}
    \item \question{$\mathcal{E}=\{(x,y,z)\in\R^3 \ | \  x^2+y^2-z^2<0 \mbox{ et } 2<z<4\}$}
    \item \question{$\mathcal{F}=\{(x,y,z)\in\R^3\ | \  x^2+y^2+z^2<1 \mbox{ et } x^2+y^2<z^2
  \mbox{ et } z>0\}$}
    \item \question{$\mathcal{G}=\{(x,y,z)\in\R^3\ | \  x^2+y^2=4 \mbox{ et } z=x-1\}$.
\end{itemize}}
    \item \question{Déterminer les projections de $\mathcal{E}$ et $\mathcal{G}$ sur le
plan $(xOy)$.}
\end{enumerate}
