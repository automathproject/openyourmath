\chapitre{Fonction de plusieurs variables}
\sousChapitre{Autre}

\contenu{
\texte{
On consid\`ere la fonction $f(x,y)=2x^2 + y^2$, $(x,y)\in\R^2$.
}
\begin{enumerate}
    \item \question{Tracer les lignes de niveau $f(x,y)=2, f(x,y)=4$.}
    \item \question{Tracer le graphe de la fonction $f$. Expliquer votre dessin en quelques phrases, en identifiant notamment les intersections du graphe de $f$ avec les plans parall\`eles aux trois plans des coordonn\'ees.}
\end{enumerate}
}
