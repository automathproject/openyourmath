\sousChapitre{Calcul de coefficients}

\contenu{
\texte{
Soit $f:\R \to \R$ $2\pi$-périodique continue par morceaux.
Que peut-on dire des coefficients de Fourier de $f$ si l'on~a~:
}
\begin{enumerate}
    \item \question{$\forall\ x \in \R,\ f(x+\pi) = f(x)$ ?}
\reponse{$a_{2p+1} = b_{2p+1} = 0$.}
    \item \question{$\forall\ x \in \R,\ f(x+\pi) = -f(x)$ ?}
\reponse{$a_{2p} = b_{2p} = 0$.}
\end{enumerate}
}
