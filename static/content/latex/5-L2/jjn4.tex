\contenu{
\texte{
Soit $f$ de classe $\mathcal{C}^2$ de $\R$ dans~$\R$ telle que $f(0) = f'(0) = 0$
et pour tout $x$~: $f''(x) \ge f(x) + \frac 2{\ch(x)^3}$.
Montrer pour tout $x$~: $f(x) \ge \frac{\sh(x)^2}{\ch(x)}$.
}
}
