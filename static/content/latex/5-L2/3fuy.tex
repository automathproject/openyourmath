\chapitre{Suite et série de fonctions}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f(x) = \sum_{n=0}^\infty \frac1{x(x+1)\dots(x+n)}$.
}
\begin{enumerate}
    \item \question{\'Etablir l'existence et la continuité de $f$ sur $\R^{+*}$.}
    \item \question{Calculer $f(x+1)$ en fonction de $f(x)$.}
    \item \question{Tracer la courbe de $f$.}
\reponse{
$f(x+1) = xf(x)-1$.
}
\end{enumerate}
}
