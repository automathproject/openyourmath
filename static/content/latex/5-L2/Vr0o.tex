\contenu{
\texte{
Soit $E = \{ \text{fonctions lipchitziennes } f:\R \to \R \}$.
Pour $f \in E$, on pose
$\|f\| = |f(0)| + \sup\limits_{x\ne y} \left|\frac{f(x)-f(y)}{x-y}\right|$.

Montrer que $E$ est complet.
}
}
