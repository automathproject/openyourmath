\contenu{
\texte{
Calcul de $I =  \int_{t=0}^{+\infty} e^{-t^2}\,d t$.
}
\begin{enumerate}
    \item \question{Justifier la convergence de cette intégrale.}
    \item \question{Pour $a > 0$ on note $\Delta_a = [0,a] \times [0,a]$ et
    $C_a$ le quart de disque d'équations : $x^2+y^2 \le a^2$, $x\ge0$, $y\ge0$.
 \begin{enumerate}}
    \item \question{Encadrer l'intégrale sur $\Delta_a$ de $f(x,y) = e^{-x^2-y^2}$ par
        les intégrales de $f$ sur des domaines du type $C_b$.}
    \item \question{Calculer $\iint_{C_b}^{} f(x,y)\,d xd y$ en polaires et en déduire la
        valeur de $I$.}
\end{enumerate}
}
