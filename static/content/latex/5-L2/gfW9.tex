\texte{
R\'esoudre les \'equations diff\'erentielles suivantes :
}
\begin{enumerate}
    \item \question{$y'=y+x$ avec $y(0)=1$,}
    \item \question{$y'= \cos x + y$,}
    \item \question{$y'+2y = (x-2)^2$.}
\end{enumerate}
}
