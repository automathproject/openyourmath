\chapitre{Fonction de plusieurs variables}
\sousChapitre{Différentiabilité}

\contenu{
\texte{
Calculer les d\'eriv\'ees partielles (d'ordre un) des fonctions 
suivantes en un point arbitraire du domaine de d\'efinition.
}
\begin{enumerate}
    \item \question{$f(x,y)=x^2e^{xy}$ ;}
    \item \question{$g(x,y,z)=x^2y^3\sqrt{z}$ ;}
    \item \question{$h(x,y)=\ln (x+\sqrt{x^2+y^2})$.}
\end{enumerate}
}
