\texte{
Soient $A$ et $B$ deux parties non vides et majorées de $\R$. Montrer
les implications suivants :
\begin{itemize}
\item
$\exists M \in\R\;\forall x\in A,\; x<M\Rightarrow\sup A\leq M$
\item
$A\subset B\Rightarrow\sup A\leq\sup B$.
\end{itemize}
}
