\chapitre{Fonction de plusieurs variables}
\sousChapitre{Dérivée partielle}

\contenu{
\texte{
Déterminer les fonctions $f : {D \subset {\R^2}} \to \R$ vérifiant :
}
\begin{enumerate}
    \item \question{$\begin{cases} \frac{\partial f}{\partial x} = \frac {2+x}y \cr
             \frac{\partial f}{\partial y} = \frac {2+y}x \cr\end{cases}$}
\reponse{pas de solution.}
    \item \question{$\begin{cases} \frac{\partial f}{\partial x} = \frac {1-y}{(x+y+1)^2} \cr
             \frac{\partial f}{\partial y} = \frac {2+x}{(x+y+1)^2} \cr\end{cases}$}
\reponse{$f(x,y) = \frac {y-1}{x+y+1} + {}$ cste.}
    \item \question{$\begin{cases} \frac{\partial f}{\partial x} = \frac{y^2}{(x+y)^2} \cr
             \frac{\partial f}{\partial y} = \frac{x^2}{(x+y)^2} \cr \end{cases}$}
\reponse{$f(x,y) = \frac{xy}{x+y} + {}$ cste.}
    \item \question{$\begin{cases} \frac{\partial f}{\partial x} = 2x + \frac 1y \cr
             \frac{\partial f}{\partial y} = 2y - \frac x{y^2} \cr \end{cases}$}
\reponse{$f(x,y) = x^2+y^2+\frac xy + {}$ cste.}
\end{enumerate}
}
