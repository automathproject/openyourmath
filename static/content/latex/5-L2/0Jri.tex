\chapitre{Fonction de plusieurs variables}
\sousChapitre{Différentiabilité}

\contenu{
\texte{
Etudier la diff\'erentiabilit\'e en $(0,0)$ des fonctions d\'efinies par
}
\begin{enumerate}
    \item \question{$$
f(x,y)=\left\{
         \begin{array}{cc}
         \frac{x^3y}{x^4+y^2} & \mbox{ si }(x,y) \neq (0,0) \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .
         $$}
    \item \question{$$
f(x,y)=\left\{
         \begin{array}{cc}
         \frac{xy^3}{x^4+y^2} & \mbox{ si }(x,y) \neq (0,0) \\
         0 & \mbox{ sinon. }
         \end{array}
         \right .
         $$}
\end{enumerate}
}
