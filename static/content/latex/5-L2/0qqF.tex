\chapitre{Fonction de plusieurs variables}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $ f:{\Rr}^{2}\rightarrow {\Rr} $ admettant des d\'{e}riv\'{e}es
partielles continues en $ 0$ et telle que :
$$\forall a\in {\Rr}^{2}-\{0\},\forall t>0,f(ta)=tf(a). $$
Montrer que $ f$ est lin\'{e}aire.
}
}
