\chapitre{Fonction de plusieurs variables}
\sousChapitre{Dérivée partielle}

\contenu{
\texte{
Soit $E$ un espace euclidien et $f : E \to E$ de classe $\mathcal{C}^1$, $\alpha > 0$ vérifiant~:
$$\forall\ x\in E,\ \forall\ h\in E,\ (d f_x(h)\mid h) \ge \alpha\|h\|^2.$$
}
\begin{enumerate}
    \item \question{Montrer pour $x,y\in E$~: $(f(x)-f(y)\mid x-y) \ge \alpha\|x-y\|^2$.
En déduire que $f(E)$ est fermé.}
    \item \question{Montrer que $f(E)$ est ouvert puis que $f$ est un $\mathcal{C}^1$-difféomorphisme
de~$E$ sur~$E$.}
\end{enumerate}
}
