\contenu{
\texte{
Soit $E$ un evn de dimension finie, et $(B_n = B(\vec a_n, r_n))$ une suite de
boules ferm{\'e}es, d{\'e}croissante pour l'inclusion, tq $r_n\xrightarrow[n\to\infty]{} 0$.
}
\begin{enumerate}
    \item \question{Montrer que la suite $(\vec a_n)$ admet une sous-suite convergeant vers $\vec a \in E$.}
    \item \question{Montrer que $\vec a_n \xrightarrow[n\to\infty]{} \vec a$.}
    \item \question{Montrer que $\bigcap_{n\in \N} B_n = \{\vec a \}$.}
\end{enumerate}
}
