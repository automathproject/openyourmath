\chapitre{Equation différentielle}
\sousChapitre{Résolution d'équation différentielle du deuxième ordre}

\contenu{
\texte{
R\'esoudre les \'equations diff\'erentielles du second ordre suivantes :
}
\begin{enumerate}
    \item \question{$y''-y=x^3+x^2$,}
    \item \question{$y''-2y'+y=e^x$,}
    \item \question{$y''-2y'+y= \cos (mx)$ où $m \in \Rr$,}
    \item \question{$y''-2y'+y=x^3e^x+2\cos x+ (x^3+3)$ (utiliser le principe de
superposition).}
\end{enumerate}
}
