\chapitre{Fonction de plusieurs variables}
\sousChapitre{Dérivée partielle}

\contenu{
\texte{
Soit $\varphi : {\R^2} \to {\R^2}$ une isométrie pour la norme $\|\ \|_2$.
}
\begin{enumerate}
    \item \question{Montrer que la matrice jacobienne de $\varphi$ est constante, égale à la matrice
    dans la base canonique de $\R^2$ de la partie linéaire de $\varphi$.}
    \item \question{Soit $f : {\R^2} \to \R$ de classe $\mathcal{C}^2$.
    Montrer que $(\Delta f)\circ \varphi = \Delta(f\circ \varphi)$.}
\end{enumerate}
}
