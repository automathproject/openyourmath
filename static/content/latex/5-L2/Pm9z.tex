\texte{
Soit $E = {\cal C}([0,2\pi])$ muni de la norme $\|.\|_2$.
Pour $n \in \N$, on pose $f_n(x) = \cos(nx)$.
}
\begin{enumerate}
    \item \question{Calculer $\|f_n-f_p\|_2$ pour $n,p \in \N$.}
    \item \question{En d{\'e}duire que $\overline B(0,1)$ n'est pas compacte.}
\end{enumerate}
