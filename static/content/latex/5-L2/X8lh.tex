\contenu{
\texte{
Soit $I =  \int_0^1\frac{\ln(1+x)}{1+x^2}\,d x$.\par
En calculant $J=\iint_D\frac{x\,d x\,d y}{(1+x^2)(1+xy)}$ avec
$D = \{(x,y) \text{ tel que } 0\le x \le 1,\ 0\le y\le 1\}$ de deux façons différentes,
trouver $I$.
}
\reponse{
$I=\frac{\pi\ln2}8$.
}
}
