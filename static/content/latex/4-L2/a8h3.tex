\texte{
Soit $  (E , \langle   ,  \rangle )  $ un espace euclidien, $  F  $
et $  G  $ deux sous-espaces vectoriels de $  E  .$ Montrer que~:
}
\begin{enumerate}
    \item \question{Si $  F\subset G  $ alors $  G^\perp \subset F^\perp   .$}
    \item \question{$  (F+G)^\perp = F^\perp \cap G ^\perp   .$}
    \item \question{$  (F\cap G)^\perp = F^\perp +G ^\perp   .$}
    \item \question{Si $\hbox{dim}(E)  $ est finie, alors $  (F^\perp )^\perp =F  .$}
\end{enumerate}
}
