\texte{
Soit $E=\{a,b\}$ un ensemble à deux éléments.
On cherche à déterminer toutes les structures de groupe sur cet
ensemble.
}
\begin{enumerate}
    \item \question{Si $a$ est l'élément neutre, quel doit être le symétrique de $b$ ?
En déduire la table de multiplication dans ce cas.}
    \item \question{Combien y a-t-il de structures de groupes différentes sur $E$ ?}
\end{enumerate}
}
