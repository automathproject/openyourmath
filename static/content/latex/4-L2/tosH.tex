\texte{
Soit $G$ le groupe-produit $\left(
\Zz/2\Zz\right) \times
\left( \Zz/4\Zz\right) .$
}
\begin{enumerate}
    \item \question{Donner la liste des \'el\'ements de $G$ et d\'eterminer l'ordre de
chacun d'entre eux. $G$ est-il cyclique ?}
    \item \question{Donner la liste des sous-groupes de $G$ et en constuire le treillis.}
\end{enumerate}
}
