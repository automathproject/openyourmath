\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Applications}

\contenu{
\texte{
Soit l'endomorphisme $f\in\mathcal{L}(\Rr^3)$ dont
la matrice dans la base canonique de $\Rr^3$ est :
$${ M=\left(
\begin{smallmatrix}
-2&1&1\\
8&1&-5\\
4&3&-3
\end{smallmatrix}\right).
}$$
}
\begin{enumerate}
    \item \question{D\'eterminer toutes les droites vectorielles de $\Rr^3$ stables par $f$.}
    \item \question{D\'eterminer toutes les plans vectoriels $P$ de $\Rr^3$
stables par $f$ (on commencera par \'etudier le
polyn\^ome caract\'eristique de la restriction de $f$ \`a $P$).}
    \item \question{Donner la liste de tous les sous-espaces vectoriels de $\Rr^3$ stables par $f$.}
\end{enumerate}
}
