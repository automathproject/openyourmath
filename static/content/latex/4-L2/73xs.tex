\begin{enumerate}
    \item \question{Soient $n,p \ge 2$. Montrer que : $n \wedge p = 1 \iff \exists\ k > 0$
    tel que $n^k \equiv 1 (\mathrm{mod}\, p)$.}
    \item \question{Soit $n$ un entier impair non divisible par 5. Montrer qu'il existe un multiple
    de $n$ qui s'{\'e}crit $1...1$ en base~10.}
\end{enumerate}
