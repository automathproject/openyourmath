\chapitre{Déterminant, système linéaire}
\sousChapitre{Applications}

\contenu{
\texte{
Soit $ E $ un espace vectoriel r\'eel de dimension
finie $ n $ et $ \varphi \in \mathcal{L} (E) $ telle que $ \varphi
^2 =-\hbox{id}_E .$
}
\begin{enumerate}
    \item \question{Donner des exemples de telles applications dans le cas $ n=2 $ ou
$ 4 .$}
\reponse{Dans le cas $n=2$, $n=4$ les matrice suivantes conviennent :
$$ J =
\begin{pmatrix}
0&-1\\
1&0\\
\end{pmatrix}, \qquad
J' =
\begin{pmatrix}
J&(0)\\
(0)&J\\
\end{pmatrix}.
$$}
    \item \question{Montrer que de telles applications existent si et seulement si $ n
$ est pair.}
\reponse{Supposons qu'un tel morphisme existe.
Soit $J$ sa matrice pour une base fixée. Alors $J^2 = -I_n$ o\`u
$I_n$ est la matrice identité de taille $n$. En termes de
déterminant nous avons : $\det (J^2)  =  \det I_n $, ce qui
s'écrit $(\det J)^2 = (-1)^n$. Donc $n$ est pair car $(\det J)^2$
est positif.}
\end{enumerate}
}
