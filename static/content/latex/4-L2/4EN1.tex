\contenu{
\texte{
Soit $n \in \N^*$ et $G = \Z/n\Z$. Soit $k \in \Z$ et
$d = k \wedge n$.
}
\begin{enumerate}
    \item \question{D{\'e}terminer l'ordre de $\dot k$ dans $G$.}
    \item \question{Montrer que $\dot k$ et $\dot d$ engendrent le m{\^e}me sous-groupe de $G$.}
    \item \question{Quels sont tous les sous-groupes de $G$ ?}
\end{enumerate}
}
