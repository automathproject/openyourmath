\texte{
Soit $G$ un groupe multiplicatif. On note Aut$(G)$ l'ensemble des
isomorphismes $\phi : G \to  G$.
}
\begin{enumerate}
    \item \question{Montrer que Aut$(G)$ est un groupe pour la loi $\circ$.}
    \item \question{D{\'e}terminer Aut$(\Z)$.}
    \item \question{Pour $a \in G$ on note ${\phi_a} : G \to G, x \mapsto {axa^{-1}}$
    \par
    Montrer que $\phi_a \in \text{Aut}(G)$, et que l'application
    $a \mapsto \phi_a$ est un morphisme de groupes.}
\end{enumerate}
}
