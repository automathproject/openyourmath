\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Polynôme annulateur}

\contenu{
\texte{
Soit $f$ un endomorphisme d'un $K$-espace
vectoriel $E$ de dimension finie tel que
$${f^4=f^2+f.}$$
}
\begin{enumerate}
    \item \question{Montrer que $\ker(f^3-f-\text{Id})\oplus \ker f=E.$}
    \item \question{\begin{enumerate}}
    \item \question{Montrer que $\text{Im\,} f\subseteq \ker(f^3-f-\text{Id}).$}
    \item \question{En d\'eduire que $\text{Im\,} f= \ker(f^3-f-\text{Id}).$}
\end{enumerate}
}
