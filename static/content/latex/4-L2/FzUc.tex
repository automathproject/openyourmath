\texte{
Soient $E$ un $K$-espace vectoriel (o\`u $K$ est
$\Rr$ ou $\Cc$) de dimension finie $n>0$ et $q$
une forme quadratique sur $E$.
}
\begin{enumerate}
    \item \question{$q$ peut-elle \^etre injective ?}
    \item \question{Trouver une condition n\'ecessaire et suffisante sur $q$ pour qu'elle soit surjective.}
\end{enumerate}
