\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit $E$ un espace euclidien de dimension $n$.
Soit $f \in {\cal O}(E)$. On note $F = \mathrm{Ker}(f - \text{id}_E)$
et $p = \text{codim}(F)$.
}
\begin{enumerate}
    \item \question{Montrer qu'on peut décomposer $f$ en produit d'au plus $p$ réflexions.}
    \item \question{Inversement, si $f$ est un produit de $k$ réflexions, démontrer que
    $p \le k$.}
    \item \question{Application : trouver $f\in {\cal O}(E)$ qui se décompose en $n$ réflexions
    et pas moins.}
\end{enumerate}
}
