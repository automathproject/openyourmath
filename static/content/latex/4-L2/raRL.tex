\chapitre{Espace euclidien, espace normé}
\sousChapitre{Endomorphismes auto-adjoints}

\contenu{
\texte{
Soit $E$ un espace vectoriel hermitien et $u\in\mathcal{L}(E)$.
}
\begin{enumerate}
    \item \question{Montrer que $u=u^*$ si et seulement si pour tout~$x\in E$,
    $(u(x)\mid x)$ est réel.}
\reponse{$((u-u^*)(x)\mid x) = 0$.}
    \item \question{On suppose $u$ autoadjoint positif. Montrer~:
    $\forall\ x\in E,\ \|u(x)\|^4 \le (x\mid u(x))\times(u(x)\mid u^2(x))$.}
\reponse{Orthodiagonaliser et appliquer l'inégalité de Cauchy-Schwarz.}
\end{enumerate}
}
