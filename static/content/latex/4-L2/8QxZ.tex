\contenu{
\texte{
Soient $(G,.)$ un groupe et $a,b\in G.$ On suppose
que
$${(1): ab^2=b^3a \quad\text{ et }(2):
ba^2=a^3b.}$$
}
\begin{enumerate}
    \item \question{Montrer, en utilisant seulement (1), que $a^2b^8a^{-2}=b^{18}$
puis que $a^3b^8a^{-3}=b^{27}.$}
    \item \question{En d\'eduire, en utilisant (2), que
$a^3b^8a^{-3}=b^{18}$ et enfin que
 $a=b=1.$}
\end{enumerate}
}
