\contenu{
\texte{
Soit $f$ un morphisme de groupes de $\Zz/15\Zz$
dans $\Zz/18\Zz.$
}
\begin{enumerate}
    \item \question{Montrer que $f$ est caract\'eris\'e par $f(\overline 1).$}
    \item \question{D\'eterminer les ordres possibles de $f(\overline 1).$}
    \item \question{En d\'eduire la liste des morphismes de groupes de $\mathbb{Z}/15\mathbb{Z}$ dans
 $\mathbb{Z}/18\mathbb{Z}.$}
\end{enumerate}
}
