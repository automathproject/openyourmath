\contenu{
\texte{
Les op{\'e}rations suivantes sont-elles des lois de groupe ?
}
\begin{enumerate}
    \item \question{$$\begin{array}{c|ccc}
      &a &b &c \\
      \hline
      a &a &a &a \\
      b &a &b &b \\
      c &a &b &c 
    \end{array}
    $$}
\reponse{Non, $a$ n'est pas r{\'e}gulier.}
    \item \question{$$\begin{array}{c|ccc}
      &a &b &c \\
      \hline
      a &b &c &a \\
      b &c &a &b \\
      c &a &b &c 
    \end{array}
    $$}
\reponse{Oui, $G \approx \Z/3\Z$.}
    \item \question{$$\begin{array}{c|cccc}
      &a &b &c &d \\
      \hline
      a &a &b &c &d \\
      b &b &a &d &c \\
      c &d &c &b &a \\
      d &c &d &a &b 
    \end{array}
    $$}
\reponse{Non, il n'y a pas d'{\'e}l{\'e}ment neutre.}
\end{enumerate}
}
