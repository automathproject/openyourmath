\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soient $E$ un espace vectoriel euclidien, $F$ et
$G$ deux sous-espace vectoriels suppl\'ementaires
de $E$ et $p$ le projecteur de $E$ d'axe $F$ et de
direction $G$.
}
\begin{enumerate}
    \item \question{On suppose que $F\perp G$. Montrer que
$\forall x\in E, \|p(x)\|\le\|x\|.$}
    \item \question{On suppose que $\forall x\in E, \|p(x)\|\le\|x\|$.
    \begin{enumerate}}
    \item \question{Soient $a\in F$ et $b\in G$. Montrer que $\|a+b\|\ge\|a\|$.}
    \item \question{En d\'eduire que $F\perp G$.}
\end{enumerate}
}
