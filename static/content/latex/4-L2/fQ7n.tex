\contenu{
\texte{
Soit $G$ un groupe additif, et $H,K,L$ trois sous-groupes de $G$ v{\'e}rifiant :
$H \subset K$, $H \cap L = K \cap L$, $H + L = K + L$.
D{\'e}montrer que $H = K$.
}
}
