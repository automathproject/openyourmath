\texte{
Le but de l'exercice est de factoriser le polynôme $x^4 + 4x^3 - 81x^2 -16x + 308$.
}
\begin{enumerate}
    \item \question{Vérifier que 2 est racine de $P$ et factoriser $P$ par $x-2$ avec la méthode de Hörner.}
    \item \question{Vérfier que $-2$ est racine du quotient obtenu
dans la question précédente et factoriser le par $x+2$ par la méthode de Hörner.}
    \item \question{Conclure.
%(x - 2)(x + 2)(x - 7)(x + 11)}
\end{enumerate}
