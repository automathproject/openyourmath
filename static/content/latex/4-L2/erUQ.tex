\texte{
Sur l'ensemble $\Rr^2,$ on d\'efinit la loi $\star$ par
$${(x_1,x_2)\star
(y_1,y_2)=(x_1y_1,x_1y_2+x_2y_1).}$$
}
\begin{enumerate}
    \item \question{\begin{enumerate}}
    \item \question{Montrer que $(\Rr^2,+,\star)$ est
un anneau commutatif not\'e $A.$}
    \item \question{Chercher les diviseurs de $0$ de l'anneau $A.$}
\end{enumerate}
}
