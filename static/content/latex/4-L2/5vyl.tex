\contenu{
\texte{
Calculer les inverses des matrices suivantes de deux mani\`eres diff\'erentes~:
$$
A = \left(\begin{array}{ccc}
1 & 2 & 3 \\2 & 5 & 4\\-1 & 0 & 2
\end{array}\right); \quad\quad
B =  \left(\begin{array}{ccc}
1 & 1 & -1\\1 & 2 &3 \\1 & 3 & 6
\end{array}\right);$$ 
$$
C = \left(\begin{array}{cccc}
0 & 1 & 0 & 0 \\
3 & 0 & 2 & 0\\
0 & 2 & 0 & 3\\
0 & 0 & 1 & 0
\end{array}\right);\quad\quad
D = \left(\begin{array}{cccc}
1 & 1 & 1 & 1\\
0 &-1 & -2 &-3\\ 
0 & 0 &1 & 3\\ 
0 & 0 & 0&-1 
\end{array}\right).
$$
}
