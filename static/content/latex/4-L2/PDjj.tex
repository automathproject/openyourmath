\contenu{
\texte{
Soit $E =  K_n[X]$. Soit $\varphi \in E^*$ telle que :
    $\forall\ P \in  K_{n-1}[X],\ \varphi((X-a)P) = 0$.

    Montrer qu'il existe $\lambda \in  K$ tel que :
    $\forall\ P \in E,\ \varphi(P) = \lambda P(a)$.
}
}
