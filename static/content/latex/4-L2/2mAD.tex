\begin{enumerate}
    \item \question{Soient $G$ et $G'$ deux groupes et $f$ un morphisme de $G$ dans $G'$.
    Pour $a \in G$, comparer l'ordre de $a$ et celui de $f(a)$.}
    \item \question{Soient $a,b \in G$. Comparer les ordres de $a$ et de $bab^{-1}$.}
    \item \question{Soient $a,b \in G$. Comparer les ordres de $ab$ et de $ba$.}
\end{enumerate}
