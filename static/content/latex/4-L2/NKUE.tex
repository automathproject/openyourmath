\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $  A\in M_3 (\R )  $ telle que $  A^3=-A  $ et $  A \not = 0  .$ Montrer
que $  A   $ est semblable \`a $  \begin{pmatrix} 0 & 0 & 0 \cr
                                      0 & 0 & -1 \cr
                                      0 & 1 & 0 \cr \end{pmatrix} .$
}
}
