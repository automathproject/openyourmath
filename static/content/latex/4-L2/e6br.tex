\contenu{
\texte{
Soit $\left(E,\;|\;\right)$ un espace vectoriel de
dimension $n$.
}
\begin{enumerate}
    \item \question{Soient $F$ et $G$ des sous-espace vectoriels de $E$. Montrer que $\left(F\cap G\right)^\perp=F^\perp+G^\perp $.}
    \item \question{Soient $\mathcal{B}=(e_1,...,e_n)$ une base orthonormale  de $E$ et
$(a_1,...,a_n)\in\Rr^n\setminus \left\{(0,...,0)\right\} $
et $H$ le sous-espace vectoriel de $E$
d'\'equation cart\'esienne
$\sum_{k=1}^{n}a_kx_k=0$ dans $\mathcal{B}.$
     \begin{enumerate}}
    \item \question{D\'eterminer l'orthogonale de $H$.}
    \item \question{D\'eterminer la distance du vecteur $x=\sum_{k=1}^{n}x_ke_k$ de $E$ au sous-espace vectoriel $H$.}
\end{enumerate}
}
