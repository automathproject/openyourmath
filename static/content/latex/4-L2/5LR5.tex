\texte{
Soient $n = 2$ ou $3$ et $A \in M_n (\Qq)$.
}
\begin{enumerate}
    \item \question{Montrer que si $\forall X \in M_n (\Qq)$ $\det (A + X) = \det (X)$ alors $A = 0$.}
    \item \question{Soit $B \in M_n (\Qq)$ telle que $\forall X \in M_n (\Qq)$ $\det (A + X) =
 \det (B + X)$. Montrer que $A = B$.}
\end{enumerate}
}
