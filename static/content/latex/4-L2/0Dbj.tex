\contenu{
\texte{
Soit $\sigma \in {\cal S}_n$. On note $c$ le nombre de cycles {\`a} supports disjoints
constituant $\sigma$, et $f$ le nombre de points fixes.

Calculer $\epsilon(\sigma)$ en fonction de $n$, $c$, et $f$.
}
\reponse{
$\epsilon(\sigma) = (-1)^{n+c+f}$.
}
}
