\chapitre{Endomorphisme particulier}
\sousChapitre{Endomorphisme orthogonal}

\contenu{
\texte{
Soit $  (E , \langle   ,   \rangle )  $ un espace euclidien 
de dimension $  3  $ et $  u\in \mathcal{O} ^-(E)  .$ On pose $  F=\hbox{Ker}(u+id)  .$
}
\begin{enumerate}
    \item \question{Montrer que $  F\not = \{ 0\}  .$ Montrer que $  F  $ et $  f^\perp  $ sont stables par $  u  .$
Pour quelle raison $  \hbox{dim} (F)\not = 2  ?$}
    \item \question{On suppose $  E\not = F  .$ Montrer que la restriction de $  u  $ \`a $F^\perp$ est une rotation.}
    \item \question{En d\'eduire qu'il existe $  \theta \in \R   $ et une base $  \epsilon$ de $  E  $ tels que~: 
$$ \hbox{Mat} (u, \epsilon )=\begin{pmatrix}
 \cos (\theta ) & \sin (\theta ) & 0  \cr  -\sin (\theta ) & \cos (\theta )  & 0 \cr 0 & 0 & -1
\cr \end{pmatrix}  .$$}
\end{enumerate}
}
