\texte{
On pose\\ $A = \left\{ f : \R \to \R
\text{ de la forme } f(x) = a_0 + \sum_{k=1}^n a_k\cos(kx),\ n \in
\N,\ a_i \in \R \right\}$.
}
\begin{enumerate}
    \item \question{Montrer que $A$ est un sous-anneau de $\R^\R$.}
    \item \question{Soit $f\in A$. Montrer que si $f = 0$, alors les coefficients $a_k$ sont
    tous nuls
    $\left({\text{calculer }  \int_{t=0}^{2\pi} f(t)\cos(nt)\,d t}\right)$.}
    \item \question{En d{\'e}duire que $A$ est int{\`e}gre.}
\end{enumerate}
}
