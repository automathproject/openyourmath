\contenu{
\texte{
Soit $A \in \mathcal{M}_n(\R)$ symétrique définie positive.
}
\begin{enumerate}
    \item \question{Montrer qu'il existe une matrice $T$ triangulaire supérieure telle que
    $A = {}^tTT$.
    Montrer que $T$ est unique si on impose la condition :
    $\forall\ i,\ T_{ii} > 0$.}
    \item \question{Application : Montrer que $\det A \le \prod_{i=1}^n a_{ii}$.}
\end{enumerate}
}
