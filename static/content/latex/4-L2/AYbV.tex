\contenu{
\texte{
Soient $A=\left(a_{ij}\right)_{1\le i,j\le
n}\in\mathcal{O}_n(\Rr)$ et $A_{ij}$ le cofacteur
$(i,j)$ de $A$. Montrer que $\det A>0$ si et
seulement si $a_{ij}$ et $A_{ij}$ sont de m\^eme
signe.
}
}
