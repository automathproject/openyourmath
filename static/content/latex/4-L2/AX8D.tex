\texte{
Soit $G$ un groupe fini et $H$ un sous-groupe de $G$.
On d{\'e}finit une relation sur $G$ par :
$$\forall\ x,y \in G,\ x \sim y \iff \exists\ h \in H \text{ tel que } x = hy.$$
}
\begin{enumerate}
    \item \question{Montrer que $\sim$ est une relation d'{\'e}quivalence. Quelle est la classe de $e$ ?}
    \item \question{Soit $a \in G$. Montrer que $\dot a$ est {\'e}quipotent {\`a} $H$.}
    \item \question{En d{\'e}duire que $\mathrm{Card}\, H$ divise $\mathrm{Card}\, G$ ({\it Th{\'e}or{\`e}me de Lagrange\/}).}
\end{enumerate}
}
