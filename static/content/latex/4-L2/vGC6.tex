\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Applications}

\contenu{
\texte{
Soit $A = \begin{pmatrix} 1 &1 \cr 1 &1 \cr \end{pmatrix}$.
}
\begin{enumerate}
    \item \question{$A$ est-elle diagonalisable ?}
    \item \question{Trouver toutes les matrices $X,Y \in \mathcal{M}_{2}(K)$ telles que $XY = YX = A$.}
\reponse{
2. $X = \frac 12\begin{pmatrix} a+b &b-a \cr b-a &a+b \cr \end{pmatrix}$,
             $Y = \frac 1b\begin{pmatrix} 1 &1 \cr 1 &1 \cr \end{pmatrix}$ ou l'inverse.
}
\end{enumerate}
}
