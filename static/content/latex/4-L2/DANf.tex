\chapitre{Espace euclidien, espace normé}
\sousChapitre{Espaces vectoriels hermitiens}

\contenu{
\texte{
Soit $E$ un espace vectoriel préhilbertien complexe et $f \in \mathcal{L}(E)$ tel
que pour tout vecteur $\vec x \in E$, on a $f(\vec x) \perp \vec x$.
}
\begin{enumerate}
    \item \question{Montrer que pour tous vecteurs $\vec x,\vec y \in E$, on a
    $(f(\vec x)\mid \vec y) = 0$.}
    \item \question{Montrer que $f = 0$.}
    \item \question{Comparer avec le cas réel.}
\reponse{
$(f(\vec x)\mid \vec y) = -(f(\vec y)\mid \vec x)$ et
             $(f(i\vec x)\mid \vec y) = -(f(\vec y)\mid i\vec x)$.
}
\end{enumerate}
}
