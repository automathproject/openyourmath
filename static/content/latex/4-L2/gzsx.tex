\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit $  (E , \langle   ,  \rangle )  $ un espace euclidien et $
  F  $ un sous-espace vectoriel muni d'une base
orthonormale $  (e_1, \ldots, e_m)  .$ Soit $  p  $ la projection orthogonale sur
$  F  .$
}
\begin{enumerate}
    \item \question{Montrer que $  \displaystyle{\forall x \in E  ,   p(x) = \sum_{i=1}^m \langle
x,e_i\rangle e_i}  .$}
    \item \question{Donner de m\^eme l'expression de la sym\'etrie orthogonale par rapport
\`a $  F  $ et la projection orthogonale sur $  F^\perp   .$}
\end{enumerate}
}
