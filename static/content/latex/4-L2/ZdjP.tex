\contenu{
\texte{
Pour tout $x,y \in \R$, on pose $x
\star y = x \sqrt{1+y^2} + y \sqrt{1+x^2}$.
}
\begin{enumerate}
    \item \question{V{\'e}rifier que $\sqrt{1+(x\star y)^2} = \sqrt{1+x^2}\sqrt{1+y^2} + xy$.}
    \item \question{Montrer que $(\R, \star)$ est un groupe.}
    \item \question{Montrer que l'application $\sh$ est un isomorphisme entre $(\R,+)$ et
    $(\R, \star)$.}
\end{enumerate}
}
