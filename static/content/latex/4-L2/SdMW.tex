\texte{
Soit $A \in \mathcal{M}_{n,p}(\R)$.
}
\begin{enumerate}
    \item \question{Montrer que ${}^t\!AA$ est la matrice d'une forme quadratique positive sur
    $\R^p$.}
    \item \question{Déterminer sa signature en fonction de~$\mathrm{rg} A$.}
\end{enumerate}
