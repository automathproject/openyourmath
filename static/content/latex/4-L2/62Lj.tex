\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit
$\alpha=\inf\left\{\int_{-1}^{1}\left(ax^2+bx+c-|{x}|\;\right)^2dx:a,b,c\in\Rr\right\}$.
}
\begin{enumerate}
    \item \question{D\'eterminer un espace vectoriel euclidien $(E,\,|\,)$, un sous-espace vectoriel $F$ de $E$ et $v\in E$ tel que $\alpha=\text{d}(v,F)^2$.}
    \item \question{D\'eterminer $p\in F$ tel que $\alpha=\text{d}(v,p)^2$ et $\alpha$.}
\end{enumerate}
}
