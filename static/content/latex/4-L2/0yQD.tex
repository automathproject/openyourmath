\contenu{
\texte{
Soit $GL_2(\Rr)$ l'ensemble des matrices inversibles $2\times 2$
\`a c\oe fficients r\'eels. $GL_2(\Rr)$ est naturellement muni d'une structure
de groupe par la multiplication usuelle des matrices.
Soit
$$A=\begin{pmatrix}
     1 & 0 \\
     0 & -1 \\
    \end{pmatrix}
\qquad \text{ et } \qquad
B=\begin{pmatrix}
     0 & -1 \\
     1 & 0 \\
    \end{pmatrix}.
$$
}
\begin{enumerate}
    \item \question{Montrer que $A$ et $B$ appartiennent \`a $GL_2(\Rr)$.}
    \item \question{Quels sont les ordres de $A$ et $B$ ?}
    \item \question{Montrer que $AB = -BA$ et en d\'eduire que :
  \begin{enumerate}}
    \item \question{$G = \big\{ I, A, B, AB, -I, -A, -B, -AB \big\}$ est un groupe (pour
la loi multiplicative des matrices ; $I$ esl la matrice identit\'e) ;}
    \item \question{$G$ est le sous-groupe de $GL_2(\Rr)$ engendr\'e par $\{ A,B\}$.}
\end{enumerate}
}
