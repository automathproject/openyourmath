\sousChapitre{Système linéaire, rang}

\contenu{
\texte{
Soit $n\ge 2$ et $A \in \mathcal{M}_n(K)$.
}
\begin{enumerate}
    \item \question{Calculer $\text{com}\,(\text{com}\, A)$ dans le cas où $A$ est inversible.}
    \item \question{Si $\mathrm{rg} A \le n-2$, démontrer que $\text{com}\, A = 0$.}
    \item \question{Si $\mathrm{rg} A = n-1$, démontrer que $\mathrm{rg}(\text{com}\, A)= 1$.}
    \item \question{Dans le cas général, démontrer que $\text{com}\,(\text{com}\, A) = (\det A)^{n-2}A$.}
\end{enumerate}
}
