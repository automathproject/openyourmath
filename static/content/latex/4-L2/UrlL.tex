\texte{
Traduire en termes d'homomorphisme de groupes les
propri\'et\'es traditionnelles suivantes :
}
\begin{enumerate}
    \item \question{$\ln(xy)=\ln x+\ln y$ ;}
    \item \question{$\det(MM')= \det(M)\det(M')$ ;}
    \item \question{$|{zz'}|=|{z}||{z'}|$ ;}
    \item \question{$(xy)^{\frac 12}=x^{\frac 12}y^{\frac 12}$ ;}
    \item \question{$e^{z+z'}= e^{z}e^{z'}$ ;}
    \item \question{$\overline{z+z'}=\overline{z}+\overline{ z'}$.}
\end{enumerate}
}
