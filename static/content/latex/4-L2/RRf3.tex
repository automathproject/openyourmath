\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Applications}

\contenu{
\texte{
Soit $a\in\mathcal{L}(E)$ un endomorphisme d'un
$K$-espace vectoriel de dimension $n$ ayant $n$
valeurs propres distinctes. On pose
$$
\mathcal{C}=\left\{u\in\mathcal{L}(E):au=ua\right\}.
$$
}
\begin{enumerate}
    \item \question{Soit $u\in\mathcal{C}$.
    \begin{enumerate}}
    \item \question{Montrer que tout sous-espace vectoriel propre de $a$ est stable par $u$.}
    \item \question{En d\'eduire que $u$ est diagonalisable.}
\end{enumerate}
}
