\texte{
Soit $\sigma$ un $k$-cycle de ${\cal S}_n$ et $p \in \Z$.
}
\begin{enumerate}
    \item \question{Si $p \mid k$, montrer que $\sigma^p$ est le produit de $p$ cycles {\`a} supports
    disjoints de longueur $\frac kp$.}
    \item \question{Montrer que pour $p \wedge k = 1$, $\sigma^p$ est un $k$-cycle
    (utiliser l'{\'e}galit{\'e} de B{\'e}zout).}
    \item \question{Dans le cas g{\'e}n{\'e}ral, {\'e}tudier la d{\'e}composition en cycles de $\sigma^p$.}
\end{enumerate}
}
