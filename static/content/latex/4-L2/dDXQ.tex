\contenu{
\texte{
Soit $G$ l'ensemble des matrices d'ordre 2 inversibles.
}
\begin{enumerate}
    \item \question{Montrer que $G$ est un groupe pour le produit matriciel.
Est-il commutatif\,?}
    \item \question{Montrer que si $A\in G, B \in G$ v\'erifient $A^2 = B^2 =
ABAB = I$, alors $A = A^{-1}, B = B^{-1}$ et $AB = BA$.}
    \item \question{Trouver deux \'el\'ements de $G$ v\'erifiant $A^2 = B^2 =
I$ et $AB \neq BA$.}
\end{enumerate}
}
