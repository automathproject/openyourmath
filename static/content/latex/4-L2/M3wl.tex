\contenu{
\texte{
Soient $A,B \in \mathcal{M}_n(\R)$ symétriques, $B$ définie positive.
Montrer que les valeurs propres de $AB$ sont réelles.
}
\reponse{
$ABX = \lambda X  \Rightarrow  {}^tX^tBABX = \lambda {}^tXBX$.
}
