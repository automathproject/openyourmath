\begin{enumerate}
    \item \question{Montrer que les ensembles $G$ suivants, munis des lois $\star$ donn\'ees, sont des groupes. Pr\'eciser quel est l'\'el\'ement neutre de $G$ et quel est  l'inverse d'un \'el\'ement quelconque $x\in G$.
\begin{enumerate}}
    \item \question{$G = \mathbb{Z}$, $\star = $ l'addition des nombres ;}
    \item \question{$G = \mathbb{Q}^{*}$ (ensemble des rationnels non nuls), $\star = $la multiplication des nombres  ;}
    \item \question{$G = \mathbb{Q}^{+*}$ (ensemble des rationnels strictement positifs),  $\star = $ la multiplication des nombres  ;}
    \item \question{$G = \mathbb{R}$, $\star = $ l'addition des nombres ;}
    \item \question{$G = \mathbb{R}^{+*}$, $\star = $ la multiplication des nombres ;}
    \item \question{$G = \mathbb{C}$, $\star = $ l'addition des nombres ;}
    \item \question{$G = \mathbb{C}^{*}$, $\star = $ la multiplication des nombres ;}
    \item \question{$G = \{z \in\mathbb{C}, |z| = 1\}$, $\star = $  la multiplication des nombres  ;}
    \item \question{$G = \{e^{i\frac{2\pi k}{n}}, k = 0, 1, \dots, n-1\}$, $\star = $ la multiplication des nombres ($n$ est un entier fix\'e) ;}
    \item \question{$G = \textrm{l'ensemble des bijections d'un ensemble non vide} \,\,\,E$, $\star =$ la composition des applications ;}
    \item \question{$G$ = l'ensemble des isom\'etries de l'espace euclidien $\mathbb{R}^{3}$ (muni du produit scalaire standard), $\star =$ la composition des applications ;}
    \item \question{$G$ = l'ensemble des isom\'etries du plan euclidien $\mathbb{R}^{2}$ (muni du produit scalaire standard) qui pr\'eservent une figure donn\'ee, $\star =$ la composition des applications.}
\end{enumerate}
