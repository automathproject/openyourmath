\contenu{
\texte{
Soit $G$ un groupe multiplicatif.
On note $Z(G) = \{ a \in G$ tq $\forall\ b \in G,$ on a $ab = ba\}$
({\it centre de $G$\/}), et
pour ${a \in G\colon}$ $C(a) = \{ b \in G$ tq $ab = ba\}$
({\it commutant de $a$\/}).

Montrer que $Z(G)$ et $C(a)$ sont des sous-groupes de $G$.
}
}
