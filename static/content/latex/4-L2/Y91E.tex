\contenu{
\texte{
Soit $A \in M_3 (\Rr)$ anti-sym\'etrique. Calculer
$\det (A)$. Ce r\'esultat vaut-il encore pour $A \in M_2 (\Rr)$ ?
}
\reponse{
Soit $$ A =
\begin{pmatrix}
0&a& b\\
-a&0&c\\
-b&-c&0\\
\end{pmatrix}, \qquad
B =
\begin{pmatrix}
0&a\\
-a&0\\
\end{pmatrix}.
$$
Alors $\det A = 0$, mais $\det B = a^2$ est non nul si $a\neq 0$.
}
}
