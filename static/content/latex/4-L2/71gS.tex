\chapitre{Endomorphisme particulier}
\sousChapitre{Endomorphisme auto-adjoint}

\contenu{
\texte{
Soit $ (E, 
\langle   ,  \rangle )  $ un espace euclidien et $f \in \mathcal{L} (E)
$ un endomorphisme autoadjoint.
Montrer que
les trois propri\'et\'es suivantes sont \'equivalentes~:
}
\begin{enumerate}
    \item \question{$\forall x \in E ,\langle f(x), x\rangle \geq 0$.}
    \item \question{Il existe $g\in \mathcal{L} (E)$ tel que $f=g^*g.$}
    \item \question{Il existe $h\in \mathcal{L} (E)$ tel que $h=h^*$ et $f=h^2.$}
\end{enumerate}
}
