\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit $  (E , \langle   ,  \rangle )  $ un espace euclidien et $  F
$ un sous-espace vectoriel de $  E  .$ On note $  p  $ la projection orthogonale sur $  F  $
et on pose, pour tout $  \displaystyle{ x\in E  :   d(x, F)= \inf _{y\in F}\Vert x-y\Vert }  .$
Soit $  z \in F  .$
}
\begin{enumerate}
    \item \question{Montrer que pour tout $  x\in F  , $ les trois conditions sont \'equivalentes~:
\begin{enumerate}}
    \item \question{[(i)] $  d(x,F)=\Vert x-z\Vert   .$}
    \item \question{[(ii)] $  z=p(x)  .$}
    \item \question{[(iii)] $  \forall y \in F  ,   y\perp (x-z)  .$}
\end{enumerate}
}
