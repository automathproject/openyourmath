\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit $E$ un espace euclidien de dimension $n$.
Soit $\vec u$ un vecteur unitaire de matrice $U$ dans une base orthonormée
$\cal B$.
}
\begin{enumerate}
    \item \question{Montrer que $U^tU$ est la matrice dans $\cal B$ de la projection orthogonale
    sur vect$(\vec u)$.}
    \item \question{Trouver la matrice de la symétrie associée.}
\end{enumerate}
}
