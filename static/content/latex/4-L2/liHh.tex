\texte{
Soit $K$ un corps, et $P \in K[X]$. Les assertions suivantes sont elles vraies ou fausses ?
}
\begin{enumerate}
    \item \question{Si $P$ n'a pas de racine dans $K$, alors $P$ est irréductible.}
    \item \question{Si $P$ est irréductible, alors $P$ n'a pas de racine dans $K$.}
\end{enumerate}
}
