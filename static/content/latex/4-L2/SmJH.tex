\sousChapitre{Forme quadratique}

\contenu{
\texte{
Soit $a$ un nombre r\'eel.
Soit $q$ la forme quadratique d\'efinie sur $\Rr^3$ par
$$q(v)=x^2+(1+a)y^2+(1+a+a^2)z^2+2xy-2ayz$$
pour $v=(x,y,z)$.
Soit $f$ la forme bilin\'eaire sym\'etrique associ\'ee \`a $q$.
}
\begin{enumerate}
    \item \question{D\'eterminer une d\'ecomposition de $q$ en combinaison lin\'eaire de carr\'es
de formes lin\'eaires ind\'ependantes.}
    \item \question{Donner le rang et la signature de $q$ suivant les valeurs
de $a$.}
    \item \question{Pour quelles valeurs de $a$, $f$ d\'efinit-elle un produit
scalaire ?}
\end{enumerate}
}
