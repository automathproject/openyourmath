\texte{
Soit le groupe $G$ (additif) $\Zz/40\Zz.$
}
\begin{enumerate}
    \item \question{Soit $H$ le sous-groupe de $G$
engendr\'e par $\overline{12}$ et $\overline{20}.$ Montrer
que $H$ est le sous-groupe de $G$ engendr\'e par $\overline{4}
$ et trouver son ordre.}
    \item \question{Caract\'eriser les g\'en\'erateurs de $G.$ Combien en compte-t-on?}
    \item \question{D\'eterminer l'ordre de $\overline{15}.$}
\end{enumerate}
