\begin{enumerate}
    \item \question{Le polynôme $X^2+1$ est-il irréductible dans $\mathbb{F}_3[X]$ ?}
    \item \question{Quelle est alors la structure de l'ensemble quotient $A=\mathbb{F}_3[X]/(X^2+1)$ ?}
    \item \question{Quelle relation vérifie la classe $\alpha$ du polynôme $X$ dans ce quotient ?}
    \item \question{Donner la liste des éléments de $A$.}
    \item \question{Déterminer l'ordre multiplicatif de $\alpha$ dans $A^\times$.}
    \item \question{Déterminer l'ordre multiplicatif de $a:=\alpha+2$ dans $A^\times$.}
    \item \question{Etablir la table des puissances de $a$.}
    \item \question{Calculer $(2+a) (2+2a)$.}
    \item \question{Calculer $a^3+a^2$ comme puissance de $a$.}
    \item \question{Calculer $(1+2a)^{-1}$.}
\end{enumerate}
