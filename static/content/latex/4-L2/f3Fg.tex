\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Polynôme caractéristique, théorème de Cayley-Hamilton}

\contenu{
\texte{
Soient $A,B \in \mathcal{M}_n(K)$.
}
\begin{enumerate}
    \item \question{Montrer que $AB$ et $BA$ ont les mêmes valeurs propres.}
    \item \question{Montrer que si $A$ ou $B$ est inversible, alors $AB$ et $BA$ ont même
    polynôme caractéristique.}
    \item \question{Dans le cas général, on note
    $M = \begin{pmatrix} BA &-B\cr 0 &0  \end{pmatrix}$,
    $N = \begin{pmatrix} 0  &-B\cr 0 &AB \end{pmatrix}$,
    $P = \begin{pmatrix} I_n  & 0\cr A &I_n \end{pmatrix}$
    ($M,N,P \in \mathcal{M}_{2n}(K)$).

 Vérifier que $MP$ = $PN$, montrer que $P$ est inversible, et conclure.}
\end{enumerate}
}
