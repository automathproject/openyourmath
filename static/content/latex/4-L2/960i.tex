\contenu{
\texte{
Discuter le rang de la matrice suivante en fonction des paramètres
  réels $x$ et $y$~:
$$
 A=
 \begin{pmatrix}
   1 & 2 & y \\
   0 & x & 1 \\
   1 & 0 & 2 \\
   1 & 2 & 1
 \end{pmatrix}
$$
}
