\chapitre{Espace euclidien, espace normé}
\sousChapitre{Forme quadratique}

\contenu{
\texte{
Soit $q$ une forme quadratique sur $\R^n$ et $f$ la forme bilinéaire
symétrique associée.

Montrer que :
$\forall\ \vec x,\vec y \in \R^n,\ dq_{\vec x}(\vec y) = 2f(\vec x,\vec y)$.
}
}
