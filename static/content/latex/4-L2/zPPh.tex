\contenu{
\texte{
Soit $A \in \mathcal{M}_n(\C)$ de valeurs propres $\lambda_1, \dots, \lambda_n$.
Montrer que $AA^* = A^*A \iff \mathrm{tr}(AA^*) = |\lambda_1|^2+ \dots + |\lambda_n|^2$.
}
\reponse{
trigonaliser $A$ dans une base orthonormée.
}
}
