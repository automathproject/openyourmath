\texte{
Soit $\sigma = (1\ 2\ \dots \ n) \in {\cal S}_n$. Trouver toutes les permutations
$\rho \in {\cal S}_n$ commutant avec $\sigma$.
(Reconna{\^\i}tre $\rho \circ \sigma \circ \rho^{-1}$)
}
\reponse{
Les puissances de $\sigma$.
}
}
