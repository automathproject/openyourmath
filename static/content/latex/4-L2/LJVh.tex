\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $E$ un espace vectoriel réel de dimension~$n$ et $f$ un endomorphisme
de~$E$.

 Est-il vrai que~:
$f$ est diagonalisable $\Leftrightarrow$ $\mathrm{Ker} f + \Im f = E$~?
}
}
