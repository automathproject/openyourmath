\texte{
Soit $A$ un anneau commutatif et $(I_n)$ une suite croissante d'id{\'e}aux de $A$.
On pose $I = \bigcup_{n \in \N} I_n$.
}
\begin{enumerate}
    \item \question{Montrer que $I$ est un id{\'e}al de $A$.}
    \item \question{On suppose que $A$ est principal. Montrer qu'il existe $n_0 \in \N$ tel que
    $I = I_{n_0}$.}
    \item \question{En d{\'e}duire que $\R^\R$ n'est pas principal.}
\end{enumerate}
}
