\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Sous-espace stable}

\contenu{
\texte{
Soient $f$ et $g$ deux endomorphismes de $E$ tels
que $fg=gf$.
}
\begin{enumerate}
    \item \question{Soit $x\in E$. Montrer que si $n\in\Nn$ et $f(x)=g(x)$ alors $f^n(x)=g^n(x)$.
\par
{\textit{Dans toute la suite, on suppose $g$ nilpotent.}}}
    \item \question{\begin{enumerate}}
    \item \question{D\'eduire de {1.} que si $f$ est inversible alors $f+g$ est inversible.}
    \item \question{D\'eduire de {(a)} que si  $f+g$ est inversible alors $f$ est inversible.}
\end{enumerate}
}
