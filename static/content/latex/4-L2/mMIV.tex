\chapitre{Déterminant, système linéaire}
\sousChapitre{Calcul de déterminants}

\contenu{
\texte{
Calculer, pour tout $ t\in {\Rr} $ le rang des
matrices $ M_t=\begin{pmatrix}
 1 & t & 1 \cr t & 1 & 1 \cr
 1 & t & 1 \cr \end{pmatrix} $ et
$ N_t=\begin{pmatrix}
 1 & 1 & t \cr 1 & t & 1 \cr
 t & 1 & 1 \cr \end{pmatrix}.$
}
}
