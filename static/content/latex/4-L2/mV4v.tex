\contenu{
\texte{
Soient $f_1,\dots,f_n$ des formes linéaires sur $ K^n$ telles qu'il existe
$\vec x \in  K^n$ non nul tel que $f_1(\vec x) = \dots = f_n(\vec x) = 0$.
Montrer que $(f_1,\dots,f_n)$ est liée.
}
}
