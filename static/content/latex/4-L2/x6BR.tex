\contenu{
\texte{
Montrer l'existence et l'unicité d'une forme linéaire $\Phi$ sur $\R_n[X]$
telle que : $\Phi(1) = 0$, $\Phi(X) = 1$ et $\Phi(P) = 0$ pour tout polynôme
$P \in \R_n[X]$ tel que $P(0) = P(1) = 0$.
}
}
