\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Polynôme annulateur}

\contenu{
\texte{
Soient $J=\left(\begin{smallmatrix}
1&1\\1&1\end{smallmatrix}\right)$ et la matrice
par blocs \`a coefficients r\'eels suivante
$${M=\begin{pmatrix}O&\frac 12 J\\\frac 12
J&O\end{pmatrix}.}$$
}
\begin{enumerate}
    \item \question{Calculer $M^2$ et $M^3$ et en d\'eduire que $M$ est diagonalisable.}
    \item \question{D\'eterminer le polyn\^ome caract\;eristique
 et le polyn\^ome minimal de $M$.}
\end{enumerate}
}
