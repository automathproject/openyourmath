\contenu{
\texte{
Dire pour quelle(s) raison(s) les op\'erations $\star$ suivantes ne munissent pas les ensembles $G$ donn\'es d'une structure de groupe~?
}
\begin{enumerate}
    \item \question{[(a)] $G = \mathbb{N}$, $\star = $ l'addition des nombres ;}
    \item \question{[(b)] $G = \mathbb{N}^{*}$, $\star = $ la multiplication des nombres ;}
    \item \question{[(c)] $G =\mathbb{R}$, $\star = $ la multiplication des nombres ;}
\end{enumerate}
}
