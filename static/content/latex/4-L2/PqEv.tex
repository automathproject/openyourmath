\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit $  (E , \langle   ,  \rangle )  $ un espace euclidien de dimension
sup\'erieure ou \'egale \`a $  2  .$ Soient $  x  $ et $   y \in E  .$ Montrer que~:
}
\begin{enumerate}
    \item \question{Si $  \Vert x\Vert = \Vert y\Vert  ,$ alors il existe un hyperplan $  H  $ de
$  E  $ tel que $  y = s(x)  $ o\`u $  s  $ est la sym\'etrie orthogonale par rapport
\`a $  H  .$}
    \item \question{Si $  \langle x, y\rangle = \Vert y\Vert ^2  ,$ alors il existe un hyperplan $  H  $ de $  E  $
tel que $  y = p(x)  $ o\`u $  p  $ est la projection orthogonale sur $  H  .$}
\end{enumerate}
}
