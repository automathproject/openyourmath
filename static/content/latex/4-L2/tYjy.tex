\contenu{
\texte{
Soit~$A\in\mathcal{M}_n(\R)$ telle que pour tout~$X\in\mathcal{M}_{n,1}(\R)$ on a
$\mathrm{tr}({}^tXAX)\ge 0$. Comparer les noyaux de~$A$ et ~${}^tA$.
}
\reponse{
Ils sont égaux (décomposer $A$ en symétrique $+$ antisymétrique).
}
}
