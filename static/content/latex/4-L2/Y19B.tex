\chapitre{Déterminant, système linéaire}
\sousChapitre{Calcul de déterminants}

\contenu{
\texte{
Soit $\Delta (x) = \det (a_{i, j} (x))$ de taille $n = 2$ ou $3$ avec $a_{i, j}$
 des fonctions d\'erivables.
}
\begin{enumerate}
    \item \question{Montrer que $\Delta' (x)$ est la somme des $n$ d\'eterminants obtenus en
remplaçant successivement dans $\Delta (x)$ chaque colonne par sa d\'eriv\'ee.}
    \item \question{Calculer $\begin{vmatrix} x + a_1 &x&x\\x&x + a_2&x\\x&x&x + a_3 \\
\end{vmatrix}$ et $\begin{vmatrix} 1 + x&1&1\\1&1 + x&1\\1&1&1 + x\\ \end{vmatrix}$.}
\end{enumerate}
}
