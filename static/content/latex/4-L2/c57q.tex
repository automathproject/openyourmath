\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Polynôme annulateur}

\contenu{
\texte{
Soit $A\in\mathcal{M}_n(\C)$ inversible diagonalisable et $B\in \mathcal{M}_n(\C)$, $p\in\N^*$
tels que $B^p=A$.
}
\begin{enumerate}
    \item \question{Montrer que $B$ est diagonalisable.}
\reponse{Polynôme annulateur simple.}
    \item \question{Si $A$ n'est pas inversible la conclusion subsiste-t-elle~?}
\reponse{Non, ctrex = $B$ nilpotent.}
\end{enumerate}
}
