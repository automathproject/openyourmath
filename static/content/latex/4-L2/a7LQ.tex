\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Sous-espace stable}

\contenu{
\texte{
Soit $E$ un $ K$-ev de dimension finie et $f\in\mathcal{L}(E)$.
}
\begin{enumerate}
    \item \question{Soit $F$ un plan vectoriel. Montrer que si $F$ est stable par~$f$ alors
    il existe $P\in K_2[x]$ non nul tel que $F \subset \mathrm{Ker} P(f)$.}
    \item \question{Réciproquement, si $P\in K_2[x]$ est irréductible, montrer que $\mathrm{Ker} P(f)$
    contient un plan stable par~$f$.}
    \item \question{Si $ K = \R$ montrer que $f$ admet toujours une droite ou un plan stable.}
\end{enumerate}
}
