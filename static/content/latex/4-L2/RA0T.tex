\texte{
Soit $n \in \N^*$.
Montrer que ${\cal S}_n$ est engendr{\'e} par les sous-ensembles suivants :
}
\begin{enumerate}
    \item \question{$A = \{ (i,i+1)$ tq $1 \le i < n    \}$.}
    \item \question{$B = \{ (1\ i)$  tq $2 \le i \le n  \}$.}
    \item \question{$C = \{ (1\ 2), (1\ 2\ \cdots\ n)  \}$.}
\end{enumerate}
}
