\texte{
On d{\'e}finit l'op{\'e}ration dans $\Z^2$ : $(a,b) * (a',b') = (aa', ab'+b)$.
}
\begin{enumerate}
    \item \question{{\'E}tudier les propri{\'e}t{\'e}s de cette op{\'e}ration.}
    \item \question{Pour $z \in \Z$, on pose $f_{a,b}(z) = az+b$.

      Montrer que $\phi : {\Z^2} \to {\Z^{\Z}},  {(a,b)} \mapsto {f_{a,b}}$
      est un morphisme pour $*$ et $\circ$.}
    \item \question{Est-ce un isomorphisme ?}
\reponse{
Non commutative, associative, $(1,0)$ = {\'e}lt neutre,\par
               $(a,b)$ est r{\'e}gulier $\iff a \ne 0$. \par
               $(a,b)$ est inversible $\iff a = \pm1$.
}
\end{enumerate}
}
