\begin{enumerate}
    \item \question{Soit $(G,.)$ un groupe. Montrer l'équivalence de:
\begin{description}}
    \item \question{{i)} $G$ est abélien.}
    \item \question{{ii)} Pour tout $a,b\in G$, on a: $(ab)^{2}=a^{2}b^{2}$.}
    \item \question{{iii)} Pour tout $a,b\in G$, on a: $(ab)^{-1}=a^{-1}b^{-1}$.}
    \item \question{{iv)} L'application $f$ de $G$ dans $G$ définie par $f(x)=x^{-1}$ est un automorphisme.
\end{description}}
    \item \question{En déduire que si pour tout $x\in G$, $x^{2}=e$, alors $G$ est abélien.}
\end{enumerate}
