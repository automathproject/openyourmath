\texte{
Calculer les déterminants suivants~:
$$
\left|
  \begin{matrix}
    2 & 3 \\
    -1& 4
  \end{matrix}
\right|
\quad
\left|
  \begin{matrix}
    1 & 0 & 2 \\
    3 & 4 & 5  \\
    5 & 6  & 7
  \end{matrix}
\right|
\quad
\left|
  \begin{matrix}
    1 & 0 & 6 \\
    3 & 4 & 15\\
    5 & 6 & 21
  \end{matrix}
\right|
\quad
\left|
  \begin{matrix}
    1 & 0 & 0 \\
    2 & 3 & 5  \\
    4 & 1 & 3
  \end{matrix}
\right|
$$
}
