\texte{
Pour $\alpha \in \C^*$ et $\beta \in \C$, on note
${f_{\alpha,\beta}} : {\C} \to {\C}, z \mapsto {\alpha z + \beta}$
}
\begin{enumerate}
    \item \question{Montrer que l'ensemble des fonctions $f_{\alpha,\beta}$ est un groupe pour la
    loi $\circ$. Est-il commutatif ?}
    \item \question{A quelle condition sur $\alpha,\beta$, $f_{\alpha,\beta}$ est-elle d'ordre
    fini ?}
\end{enumerate}
}
