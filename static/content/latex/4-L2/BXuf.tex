\texte{
Soient $n\in\Nn\setminus\left\{0\right\}$ et $f$
l'endomorphisme de l'espace vectoriel $\Rr^{2n}$
dont la matrice dans la base canonique est la
matrice par blocs $M=\left(\begin{smallmatrix}
I_n&I_n\\O_n&O_n\end{smallmatrix}\right)\in
\mathcal{M}_{2n}(\Rr)$
.
}
\begin{enumerate}
    \item \question{D\'eterminer le polyn\^ome caract\'eristique de $M$.}
    \item \question{\begin{enumerate}}
    \item \question{D\'eterminer le noyau de $f$.}
    \item \question{Montrer que $f$ est diagonalisable.}
\end{enumerate}
}
