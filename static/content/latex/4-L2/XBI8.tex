\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Sous-espace stable}

\contenu{
\texte{
Soient $E$ un $K$-espace vectoriel de dimension
finie, $f$ un endomorphisme de $E$ et $F$ un
sous-espace vectoriel de $E$ stable par $f.$ On
d\'esigne par $g$ l'endomorphisme de  $F$ induit
par $f$ sur $F$.
}
\begin{enumerate}
    \item \question{Montrer que $\text{Sp}(g)\subseteq \text{Sp}(f)$.}
    \item \question{Montrer que si $P(f)=0$ alors $P(g)=0$. En d\'eduire que le polyn\^ome minimal de $g$ divise celui de $f$.}
\end{enumerate}
}
