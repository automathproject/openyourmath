\contenu{
\texte{
Soit $A$ un anneau, on dit que $x\in A$ est nilpotent si $\exists n\in \Nn$
tel que $x^{n}=0.$
}
\begin{enumerate}
    \item \question{Montrer que si $x$ est nilpotent alors $(1-x)$ est inversible.}
    \item \question{Montrer que si $x$ et $y$ sont nilpotents et commutent alors $xy$ et $x+y$
sont nilpotents.}
    \item \question{Un corps admet-il des \'{e}l\'{e}m{e}nts nilpotents ?}
\end{enumerate}
}
