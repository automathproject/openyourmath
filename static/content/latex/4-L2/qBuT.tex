\contenu{
\texte{
Soit $n\ge 3$. Montrer que $\varphi(n)$ est pair et
que $\sum_{x\wedge n = 1, 1\le x\le n} x = \frac{n\varphi(n)}2$.
}
\reponse{
Regrouper $x$ et $n-x$.
}
}
