\begin{enumerate}
    \item \question{Compléter la matrice $A = \frac 17 \begin{pmatrix} 6 &3 &. \cr
                                                 -2 &6 &. \cr
                                                  3 &. &. \cr\end{pmatrix}$
    en une matrice orthogonale positive.}
\reponse{$\frac 17 \begin{pmatrix} 6 &3 &-2 \cr -2 &6 &3 \cr 3 &-2 &6 \cr\end{pmatrix}$.}
    \item \question{Reconnaître l'application de matrice $A$ dans la base canonique
    de $\R^3$.}
\reponse{Rotation autour de $(1,1,1)$ d'angle $-\Arccos(\frac {11}{14})$.}
\end{enumerate}
