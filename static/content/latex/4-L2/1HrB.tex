\contenu{
\texte{
Soit $E$ un $\R$-espace vectoriel de dimension finie et $f$ une forme bilinéaire symétrique
non dégénérée sur~$E$.
}
\begin{enumerate}
    \item \question{Si $u\in\mathcal{L}(E)$ montrer qu'il existe un unique endomorphisme $v\in\mathcal{L}(E)$
    tel que~: $$\forall\ x,y\in\ E,\ f(u(x),y) = f(x,v(y)).$$
    On note $v=u^*$.}
    \item \question{Montrer que l'application $u \mapsto u^*$ est un anti-isomorphisme involutif
    de l'algèbre~$\mathcal{L}(E)$ (c'est-à-dire un isomorphisme linéaire
    tel que $(u\circ v)^* = v^*\circ u^*$ et $u^{**}=u$).}
\end{enumerate}
}
