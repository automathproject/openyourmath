\chapitre{Espace euclidien, espace normé}
\sousChapitre{Endomorphismes auto-adjoints}

\contenu{
\texte{
Soit $u : E \to E$ telle que :
$\forall\ \vec x,\vec y \in E,\ (u(\vec x)\mid\vec y) = (\vec x\mid u(\vec y))$.
Montrer que $u$ est linéaire.
}
}
