\texte{
Soit $E$ un ensemble fini et $f : {{\cal S}_E} \to {\C^*}$ un morphisme de groupes.
}
\begin{enumerate}
    \item \question{Si $\sigma$ est une transposition, que peut-on dire de $f(\sigma)$ ?}
    \item \question{Montrer que deux permutations conjugu{\'e}es ont m{\^e}me image par $f$.}
    \item \question{En d{\'e}duire que $f$ est la fonction constante $1$, ou bien $f$ est la
    signature.}
\end{enumerate}
}
