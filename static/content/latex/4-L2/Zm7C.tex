\chapitre{Endomorphisme particulier}
\sousChapitre{Autre}

\contenu{
\texte{
Dans $ K^3$ on considère les formes linéaires :
$f_1(\vec x) = x+y-z$,
$f_2(\vec x) = x-y+z$,
$f_3(\vec x) = x+y+z$.
}
\begin{enumerate}
    \item \question{Montrer que $(f_1,f_2,f_3)$ est une base de $( K^3)^*$.}
    \item \question{Trouver la base duale.}
\end{enumerate}
}
