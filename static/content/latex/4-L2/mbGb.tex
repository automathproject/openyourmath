\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Applications}

\contenu{
\texte{
Soient $f\in\mathcal{L}(E)$ un endomorphisme et
$a\in E$ tels que la famille
$(a,f(a),...,f^{n-1}(a))$ est une base de $E$.
}
\begin{enumerate}
    \item \question{Soit $P\in K[X]\setminus\left\{0\right\}$ un polyn\^ome annulateur de $f$. Montrer que $\deg(P)\ge n$ (raisonner par l'absurde).}
    \item \question{En d\'eduire que le polyn\^ome minimal de $f$ est
(au signe pr\`es) le polyn\^ome caract\'eristique de $f$.}
\end{enumerate}
}
