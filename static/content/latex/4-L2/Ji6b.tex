\texte{
Soit $E$ un espace euclidien de dimension $n$ et $x_{1}, \ldots, x_{p}$ des vecteurs
de $E$ tels que si $i \neq j$ alors $\langle x_{i}|x_{j} \rangle <0$.
Montrer par r\'ecurrence sur $n$ que $p \leq n + 1$.
}
