\sousChapitre{Endomorphisme orthogonal}

\contenu{
\texte{
Soit $  (E , \langle   ,   \rangle )  $ un espace euclidien et 
$  u \in \mathcal{O} (E)  .$ On pose $  v =id-u  .$
}
\begin{enumerate}
    \item \question{Montrer que $  \hbox{Ker}(v)=\hbox{Im} (v)^\perp   .$}
    \item \question{Montrer que $  \displaystyle{ \lim _{n \rightarrow \infty }\frac{1}{ n} \sum _{p=0}^{n-1}u^p (x)} 
$ est la projet\'e orthogonal de $  x  $ sur $  \hbox{Ker}(v)  .$}
\end{enumerate}
}
