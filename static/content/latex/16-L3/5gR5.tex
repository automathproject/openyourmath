\contenu{
\texte{
Déterminer les résidus
de chacune des fonctions suivantes, au p\^ole indiqué.
}
\begin{enumerate}
    \item \question{$\frac {z^2}{(z-2)(z^2+1)}, \qquad z=2,\ z=i,\ z=-i,$}
    \item \question{$\frac 1{z(z+2)^3},\qquad z=0,\ z=-2,$}
    \item \question{$\frac {z e^{zt}}{(z -3)^2},\qquad z=3,$}
    \item \question{$\textrm{cotg} z,\qquad z=-5 \pi.$}
\end{enumerate}
}
