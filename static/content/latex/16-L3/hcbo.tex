\contenu{
\texte{
Montrer que, si $f$ est holomorphe au voisinage de
$\overline{D(0,1)}$ et si $f(\partial D(0,1)) \subset D(0,1)$, alors il existe
un $z_0$ et un seul dans $D(0,1)$ tel que l'on ait $f(z_0)=z_0$.
}
}
