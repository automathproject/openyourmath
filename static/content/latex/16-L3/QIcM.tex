\chapitre{Fonction holomorphe}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Déterminer, si possible, une fonction holomorphe sur $\Cc$ dont la partie réelle est $u(x,y)=x^2+y^2$.}
    \item \question{Déterminer, si possible, une fonction holomorphe sur $\Cc$ dont la partie réelle est $u(x,y)=x^2-y^2$.}
    \item \question{Montrer sans calcul que la fonction $f : \Rr^2\to\Rr^2$, $(x,y)\mapsto 2xy$ est harmonique sur $\Rr^2$.}
\end{enumerate}
