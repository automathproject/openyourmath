\contenu{
\texte{
Soit $P(z) = z^n+a_1z^{n-1} + \cdots +a_n$, $n\ge 1$, $a_j\in
\Cc$. Montrer qu'il existe un point $c$ de $\partial D(0,1)$ tel que l'on
ait $|P(c)| \ge 1$.
}
}
