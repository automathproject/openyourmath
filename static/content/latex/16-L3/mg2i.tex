\chapitre{Fonction logarithme et fonction puissance}
\sousChapitre{Fonction logarithme et fonction puissance}

\contenu{
\texte{
On considère l'application
$$\begin{array}{cccc}
 L :& \Cc-\{z\in\Cc, re(z)=0\}&\longrightarrow&\Cc\\
 &z=x+iy&\longmapsto& \frac{1}{2}\log_\Rr(x^2+y^2)+i\arctan (\frac{y}{x}).
\end{array}$$
}
\begin{enumerate}
    \item \question{Montrer que $L$ est holomorphe.}
    \item \question{L'application $L$ coïncide-t-elle avec la branche principale $\log$ du logarithme sur\\ $\{z\in\Cc, re(z)>0\}$ ?}
    \item \question{L'application $L$ est-elle un logarithme sur $\{z\in\Cc, re(z)<0\}$ ?}
\end{enumerate}
}
