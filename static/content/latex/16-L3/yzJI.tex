\contenu{
\texte{
Développer en série entière les fonctions suivantes et préciser le domaine
maximal de convergence ($a,b\not=0$):
$$
f(z)={1\over 1+z+z^2},\quad g(z)={1\over (a-z)(b-z)},\quad 
h(z)={z\sin a\over z^2-2(\cos a)z+1}.$$
}
}
