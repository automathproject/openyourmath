\begin{enumerate}
    \item \question{Démontrer que la partie imaginaire d'une fonction holomorphe est harmonique.}
    \item \question{Soit $f$ une fonction holomorphe sur $\Cc$ non identiquement nulle.
Montrer qu'il existe $\epsilon >0$ tel que $f$ ne s'annule pas sur $\Delta_\epsilon -\{0\}$.}
    \item \question{Soient $\Omega$ un ouvert connexe et $f$ une fonction holomorphe sur $\Omega$. 
L'intégrale de $f$ le long de tout chemin fermé contenu dans $\Omega$ est-elle nulle ?}
\end{enumerate}
