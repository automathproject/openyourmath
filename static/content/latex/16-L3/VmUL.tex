\chapitre{Formule de Cauchy}

\contenu{
\texte{
Calculer les intégrales 
\(\displaystyle{\int_{C(i,\frac{1}{2})}\frac{iz^3-3}{(z-i)^2(z+i)^2}dz}\)
 et 
 \(\displaystyle{ \int_{C(-i,\frac{1}{2})}\frac{iz^3-3}{(z-i)^2(z+i)^2}dz}\).
En déduire 
\(\displaystyle{\int_{C(0,2)}\frac{iz^3-3}{(z-i)^2(z+i)^2}dz}\).
}
}
