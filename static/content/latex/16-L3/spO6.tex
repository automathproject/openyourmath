\contenu{
\texte{
On dit que deux fonctions réelles $u(x,y)$ et $v(x,y)$ sont conjuguées 
har\-mo\-ni\-ques si elles vérifient les équations de Cauchy-Riemann.
}
\begin{enumerate}
    \item \question{Montrer que si $u$ et $v$ sont conjuguées harmoniques, alors $u$ et $v$
sont harmoniques.}
    \item \question{Trouver
les conjuguées har\-mo\-ni\-ques des fonctions har\-mo\-ni\-ques suivantes
dans les ouverts indiqués :
\begin{enumerate}}
    \item \question{$u(x,y)=x^2-y^2+x$ sur $\C$}
    \item \question{$u(x,y)={x\over x^2+y^2}$ sur $\C\setminus \{0\}$}
    \item \question{$u(x,y)={1\over 2}\ln{(x^2+y^2)}$ sur
\begin{enumerate}}
    \item \question{$\C\setminus \{x+iy\vert\ y=0,x\le 0\}$}
    \item \question{$\C\setminus \{0\}$}
\end{enumerate}
}
