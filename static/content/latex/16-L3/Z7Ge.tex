\contenu{
\texte{
Calculer les intégrales suivantes, où les chemins fermés simples
$\gamma $ sont parcourus dans le sens direct.
}
\begin{enumerate}
    \item \question{$\displaystyle \int_\gamma {z\over (z-1)(z-2)^2}dz$ où $\gamma $
est le cercle $\vert z-2\vert ={1\over 2}$ ;}
    \item \question{$\displaystyle \int_\gamma {\exp z\over z^2(z-9)^2}dz$ où $\gamma
$ est le cercle $\vert z\vert =1$ ;}
    \item \question{$\displaystyle \int_\gamma \exp{\left( {1\over z}\right) }\ dz$ où
$\gamma $ est le cercle $\vert z\vert =1$ ;}
    \item \question{$\displaystyle \int_\gamma \sin^2{\left( {1\over z}\right) }\ dz$ où
$\gamma $ est le cercle $\vert z\vert =r$ ;}
    \item \question{$\displaystyle \int_\gamma (z^2+z+1)^{-1/2}\ dz$ où $\gamma $
est le cercle $\vert z\vert =r\ne 1$.}
\end{enumerate}
}
