\texte{
Evaluer
$$\int_{1+i}^{2+4i} z^2 dz$$
}
\begin{enumerate}
    \item \question{le long de la parabole $y = x^2,\ 1 \leq x \leq 2$;}
    \item \question{le long du segment de droite $(1+i), (2+4i)$;}
    \item \question{le long des segments $(1+i), (2+i)$ et $(2+i),(2+4i)$.}
\end{enumerate}
}
