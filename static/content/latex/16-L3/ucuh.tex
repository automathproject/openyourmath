\texte{
Soit $D:=\Cc-[0,1]$ et $f : D\to \Cc$, $z\mapsto\frac{1}{z(z-1)}$.
}
\begin{enumerate}
    \item \question{$D$ est-il un ouvert étoilé ?}
    \item \question{La fonction $F : z\mapsto\log (1-\frac{1}{z})$ est-elle bien définie et continue sur $D$ ?}
    \item \question{Montrer que pour tout chemin orienté $\Gamma$ de $D$,
 $\int_\Gamma f(\zeta)d\zeta=0.$}
\end{enumerate}
}
