\texte{
Démontrer que
}
\begin{enumerate}
    \item \question{$\mathcal{L}(1) (s)= \frac 1 s$}
    \item \question{$\mathcal{L}(t^n)(s) = \frac {n!} {s^{n+1}}$}
    \item \question{$\mathcal{L}(e^{at})(s) = \frac 1 {s-a}$}
    \item \question{$\mathcal{L}(\sin at)(s) = \frac a {s^2+a^2}$}
    \item \question{$\mathcal{L}(\cos at)(s) = \frac s {s^2+a^2}$}
    \item \question{$$\mathcal{L}(U_a)(s) = \frac {e^{-as}} s, \ 
\text{où}\  U_a(t) = \begin{cases} 1,&t\geq a,\\
    0,&t < a,\\
         \end{cases}
         a \in \Rr, a \geq 0.$$}
\end{enumerate}
}
