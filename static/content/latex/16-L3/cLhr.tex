\contenu{
\texte{
Que vaut, en fonction du nombre réel $r>0$, l'intégrale
 $$I_r:=\int_{\Delta_r} \frac{dz}{2z^2-5z+2} \ \ ?$$
 On précisera les valeurs de $r$ exclues.
}
\reponse{
si $0<r<1/2$, $I_r=0$.
si $1/2<r<2$, $I_r=-2i\pi/3$.
si $2<r$, \ \ \ \ \ \ \ \ $I_r=1/3-1/3=0$.
}
}
