\texte{
Démontrer que
$$\int_0^{2\pi} \cos^{2n} \vartheta d\vartheta = 2 \pi \cdot \frac{1
  \cdot 3 \cdot 5 \cdot \dots \cdot (2n-1)} {2 \cdot 4 \cdot 6 \cdot
  \dots \cdot (2n)}, \quad n=1,2,3, \dots .$$
}
