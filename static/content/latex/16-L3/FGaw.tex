\chapitre{Fonction logarithme et fonction puissance}
\sousChapitre{Fonction logarithme et fonction puissance}

\contenu{
\texte{
On considère la branche principale du logarithme, toujours notée $\log$.
}
\begin{enumerate}
    \item \question{Calculer $\log(i)$.}
    \item \question{Calculer $i^i$.}
    \item \question{Démontrer que pour tout $\alpha$ et $\beta$ fixés dans $\Cc$ et $z\in\Cc^-$, 
 $$(z^\alpha)'=\alpha z^{\alpha-1}\quad \text{ et } \quad z^{\alpha+\beta}=z^\alpha z^\beta.$$}
    \item \question{En utilisant la détermination principale du logarithme, on définit les fonctions suivantes : 
\begin{enumerate}}
    \item \question{[a.]$ z \longmapsto z^{1/2} $}
    \item \question{[b.]$ z \longmapsto( 1-z) ^{1/3} $.}
\end{enumerate}
}
