\contenu{
\texte{
Soient $f$ et $g$ deux fonctions entières avec $\forall z\
f(z)g(z) = 0$. Montrer que l'une des deux est identiquement
nulle.
}
\reponse{
Si $f(z)g(z) = 0$ pour tout $z\in \C$, alors au moins une des fonctions $f,g$ a un z\'ero non isol\'e.
}
}
