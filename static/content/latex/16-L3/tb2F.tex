\chapitre{Formule de Cauchy}
\sousChapitre{Formule de Cauchy}

\contenu{
\texte{
On se propose de calculer $I=\int_{-\pi}^\pi \ln|re^{i\theta}-a|\ d\theta$, où
$0<r<|a|$.
}
\begin{enumerate}
    \item \question{Vérifier que $I$ est bien définie.}
    \item \question{On considère $f$ la fonction définie sur $\Omega={\Cc}\backslash {\Rr^-}$
par $f(z)={1\over z}\hbox{ Log}(1-{z\over a})$, où $\hbox {Log}$ est la
détermination principale du logarithme sur $\Omega$. Montrer que $f$ est
holomorphe sur un ouvert contenant le cercle $\{z=re^{i\theta}\}$.}
    \item \question{En déduire $I=2\pi \ln|a|$.}
\end{enumerate}
}
