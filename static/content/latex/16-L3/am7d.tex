\begin{enumerate}
    \item \question{\ Quand dit-on qu'une partie $S$ de $\Rr^n$ est
discrète~?}
    \item \question{Soit $K\subset \Rr^n$ un compact et soit $S \subset
\Rr^n$ une partie discrète et fermée. 
Dé\-mon\-trer que
$K \cap S$ est fini.}
    \item \question{Soit $\Omega\subset \Cc$ un ouvert connexe,
$f:\Omega\to \Cc$ une fonction holomorphe
non-identiquement nulle et soit $K\subset \Omega$ un
compact. Démontrer que $f$ n'a qu'un nombre fini de
zéros dans $K$.}
\end{enumerate}
