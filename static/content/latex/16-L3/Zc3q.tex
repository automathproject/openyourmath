\contenu{
\texte{
Soit $u(x,y)\in\Rr[x,y]$ un polynôme harmonique.
 Montrer que la fonction $$p(z):=2u(\frac{z}{2},\frac{z}{2i})-u(0,0)$$
 est holomorphe de partie réelle $u$.
}
}
