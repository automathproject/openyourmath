\chapitre{Fonction logarithme et fonction puissance}
\sousChapitre{Fonction logarithme et fonction puissance}

\contenu{
\texte{
On définit $\cosh z$ comme la somme de la série $\sum \frac{z^{2n}}{(2n)!}$ et
$\sinh z$ comme la somme de la série $\sum \frac{z^{2n+1}}{(2n+1)!}$.
}
\begin{enumerate}
    \item \question{Déterminer leur rayon de convergence.}
    \item \question{Exprimer $\cosh$ et $\sinh$ à l'aide de la fonction exponentielle.}
    \item \question{Démontrer les formules d'addition pour $\cosh(z+w)$ et $\sinh(z+w)$ pour $z$ et $w$ dans $\Cc$.}
    \item \question{Démontrer que pour tous $x$ et $y$ dans $\Rr$,
 \begin{eqnarray*}
 \cos(x+iy)&=&\cos(x)\cosh(y)-i\sin(x)\sinh(y)\\
 \sin(x+iy)&=&\sin(x)\cosh(y)+i\cos(x)\sinh(y).
 \end{eqnarray*}}
\end{enumerate}
}
