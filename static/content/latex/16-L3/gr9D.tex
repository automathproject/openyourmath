\chapitre{Tranformée de Laplace et de Fourier}

\contenu{
\texte{
Calculer la transformée
inverse de Laplace $\mathcal{L}^{-1}(f)$ pour :
}
\begin{enumerate}
    \item \question{$f(s) = \frac s{(s+1)^3(s-1)^2}$}
    \item \question{$f(s) = \frac 1{(s^2+1)^2}$}
    \item \question{$f(s) = \frac {s+1}{(s+2)(s+3)}$}
    \item \question{$f(s) = \frac {s+1}{(s+2)(s+3)^2(s^2+4s+5)}$}
    \item \question{$f(s) = \frac {1}{(s+3)(s+4)}$}
    \item \question{$f(s) = \frac {s+2}{(s+1)^2(s+3)}$}
    \item \question{$f(s) = \frac {s+1}{s^2 +4s+16}$}
\end{enumerate}
}
