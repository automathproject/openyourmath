\contenu{
\texte{
Etude de la fonction
holomorphe $f(z)= \cos z$.
}
\begin{enumerate}
    \item \question{Trouver l'image d'une droite $x=c$.}
    \item \question{Trouver l'image d'une droite $y=c$.}
    \item \question{Trouver un ouvert maximal $U$ tel que le restriction de $f$ à
  cet ouvert soit injective.}
    \item \question{Trouver un domaine de définition maximal pour la fonction
  réciproque, $\textrm{arccos}$.}
    \item \question{Vérifier que parmi les branches de $\textrm{arccos}$ il y en a
  deux qui s'écrivent sous la forme $\textrm{arccos} (w) = \pm i
  \textrm{Log}(w + \sqrt {w^2-1})$.}
    \item \question{Trouver toutes les branches de la fonction $\textrm{arccos}$.}
\end{enumerate}
}
