\contenu{
\texte{
Trouver les séries de
Laurent par rapport aux singularités indiquées pour chacune des
fonctions suivantes. Caractériser la singularité dans chaque cas
et donner le domaine de convergence de chaque série.
}
\begin{enumerate}
    \item \question{$\frac {e^z}{(z-1)^2}, \qquad z=1,$}
    \item \question{$z \cos \frac 1z,\qquad z=0,$}
    \item \question{$\frac {\sin z}{z -\pi},\qquad z=\pi,$}
    \item \question{$\frac z{(z+1)(z+2)},\qquad z=-1.$}
\end{enumerate}
}
