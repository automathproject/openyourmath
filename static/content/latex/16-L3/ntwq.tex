\chapitre{Formule de Cauchy}

\contenu{
\texte{
Soit $F_1(z)=\int_0^{+\infty}e^{-(z+1)^2t}dt$.
}
\begin{enumerate}
    \item \question{Déterminer l'ouvert $E$ sur lequel $F_1$ définit une fonction
holomorphe.}
    \item \question{Montrer que $F_1$ se prolonge analytiquement en une fonction $F$
sur un ouvert que l'on déterminera. Calculer $F(2-4i)$.}
\end{enumerate}
}
