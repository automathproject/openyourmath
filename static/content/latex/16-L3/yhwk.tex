\chapitre{Formule de Cauchy}

\contenu{
\texte{
Existe-t-il des fonctions $f$ holomorphes sur $D(0,1)$ telles que
pour tout entier $n>0$
}
\begin{enumerate}
    \item \question{$f\left( {1\over 2n}\right) = f\left( {1\over 2n+1}\right)={1\over n}$
?}
    \item \question{$f\left( {1\over n}\right) = f\left( -{1\over n}\right)={1\over n^2}$
?}
    \item \question{$f\left( {1\over n}\right) = f\left( -{1\over n}\right)=-{1\over n^3}$
?}
\end{enumerate}
}
