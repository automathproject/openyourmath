\contenu{
\texte{
Montrer que le rayon de convergence de chacune des séries
   concernées est $1$ et prouver:
}
\begin{enumerate}
    \item \question{$\sum_{n=1}^\infty nz^n$ ne converge en aucun
  point du cercle $|z|=1$.}
    \item \question{$\sum_{n=1}^\infty \frac{z^n}{n^2}$ converge en tout point du cercle
  $|z|=1$.}
    \item \question{$\sum_{n=1}^\infty \frac{z^n}{n}$ converge en tout point du cercle
  $|z|=1$ \textbf{sauf} en $z=1$.}
\end{enumerate}
}
