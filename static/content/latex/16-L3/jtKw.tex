\begin{enumerate}
    \item \question{Trouver le rayon de convergence de la série  $\sum a_nz^n$ où
$a_{2n+1}=a^{2n+1}$ et $a_{2n}=b^{2n}$ avec $0<a,b<1$.}
    \item \question{Pour la série $\sum a_nz^n$, où
$a_{2n}=a^{n}b^{n}$ et $a_{2n+1}=a^{n}b^{n+1}$ avec $a,b>0$, comparer l'inverse
du rayon,
$R^{-1}$, avec
$\limsup {a_{n+1}\over a_n}$.}
\end{enumerate}
