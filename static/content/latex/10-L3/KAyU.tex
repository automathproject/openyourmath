\begin{enumerate}
    \item \question{Soit $(A,B,C;D)$ un repère projectif du plan projectif $P^2$.
{\'E}crire en coordonnées homogènes la projection $p$ de centre $A$
 sur la droite $(BC)$.}
    \item \question{Soit $\mathcal{Q}$ une quadrique projective non singulière
de $P^2$ qui passe par le point $A$. Montrer que la projection $p$
se prolonge en une bijection entre $\mathcal{Q}$ et la droite $(BC)$.}
\end{enumerate}
