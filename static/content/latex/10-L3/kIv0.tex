\contenu{
\texte{
Soit $K=\{ e,a,b,c \}$ un groupe d'ordre $4$ tel que $x^2=e, \forall x$.
}
\begin{enumerate}
    \item \question{Ecrire la table de multiplication de $K$.}
    \item \question{Montrer que $K$ est isomorphe à $(\Z/2 \Z \times \Z/2 \Z ,\; +)$ et n'est pas isomorphe à $\Z /4 \Z$.}
    \item \question{Montrer que tout groupe d'ordre 4 est ou bien isomorphe à $K$ ou bien à $\Z /4 \Z$.}
\end{enumerate}
}
