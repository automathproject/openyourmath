\contenu{
\texte{
Passer $\mod n\ $ avec un  module 
appropri\'e et montrer que chacune des \'equations  suivantes
n'a aucune   solution dans $\Zz$ :
}
\begin{enumerate}
    \item \question{$\quad 3x^2+2=y^2$;}
    \item \question{$\quad x^2+y^2=n\quad$ pour $\quad n=2003$, $2004$;}
    \item \question{$\quad x^2+y^2+z^2=1999$;}
    \item \question{$\quad x^3+y^3+z^3=5$;}
    \item \question{$\quad x_1^4+x_2^4+\dots + x_{15}^4=7936$.}
\end{enumerate}
}
