\texte{
On cherche à décrire le groupe $G$ des isométries d'un plan affine euclidien qui conservent globalement un triangle $\mathcal{T}=ABC$ isocèle en $A$ non équilatéral non aplati.
}
\begin{enumerate}
    \item \question{Déterminer deux éléments différents du groupe $G$.}
    \item \question{Soit $f$ un élément de $G$. 
\begin{enumerate}}
    \item \question{Montrer que $f$ a un point fixe.}
    \item \question{Montrer que $f(A)=A$ et que 
($f(B)=B$ ou $f(B)=C$).}
    \item \question{Montrer que $f$ est soit l'identité soit une réflexion.}
\end{enumerate}
}
