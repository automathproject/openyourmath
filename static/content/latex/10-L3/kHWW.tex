\contenu{
\texte{
Trouver le $\pgcd(f,g)$ dans $\Zz[x]$ de 
$f=x^4+x^3-3x^2-4x-1$ et  $g=x^3+x^2-x-1$.
}
\reponse{
Sur $\Zz[X]$, $\pgcd(x^{4}+x^{3}-3x^{2}-4x-1,x^{3}+x^{2}-x-1)=1$.
}
