\contenu{
\texte{
Soit $G$ un groupe.
}
\begin{enumerate}
    \item \question{Soient $A$, $B$ deux sous-groupes   de $G$.
  \begin{enumerate}}
    \item \question{On suppose que  $A$ est d'indice fini dans $G$, montrer alors que
  $A\cap B$ est d'indice fini dans $B$.}
    \item \question{On suppose que  $A$ et $B$  sont  d'indice fini dans $G$, montrer alors
  que $A\cap B$ est d'indice fini dans $G$, généraliser au cas d'un  nombre fini
  de sous-groupes.}
\end{enumerate}
}
