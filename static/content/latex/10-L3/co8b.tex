\begin{enumerate}
    \item \question{Soit $a\in \Z$ un nombre sans facteurs carrés, $a\not\in \{ -1,\; 0,\; 1\}$. Montrer que $X^n-a$ est irréductible sur $\Q $ pour tout $n\in \N^*$.}
    \item \question{Montrer que $2X^{10}-21$, $3X^5-35,$ $X^5+1000X^2+6$ sont irréductibles sur $\Q $.}
    \item \question{Démontrer l'irréducibilité dans $\Q[X]$ des polyn\^omes $X^4-8X^3+12X^2-6X+2$ et $X^5-12X^3+36X-12$.}
\end{enumerate}
