\contenu{
\texte{
Soit $G$ un groupe et $A$, $B$ deux sous-groupes   de $G$, on note $AB:= \{
g=ab : a\in A, b\in B\}$.
}
\begin{enumerate}
    \item \question{Montrer que $AB$ est un sous-groupe de $G$ si et seulement si $AB=BA$.}
    \item \question{Montrer que si $AB$ est un sous-groupe de $G$ alors $AB= <A,B>$.}
\end{enumerate}
}
