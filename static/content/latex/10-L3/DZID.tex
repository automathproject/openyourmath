\contenu{
\texte{
Soit $H$ et $N$ deux groupes et $\phi ,\; \psi :H \to Aut\, (N)$ deux morphismes. Montrer que
}
\begin{enumerate}
    \item \question{s'il existe $\alpha \in Aut\, (H)$ tel que $\psi =\phi \circ \alpha $, alors $N \rtimes_\phi H \simeq N \rtimes_\psi H$.}
    \item \question{s'il existe $u \in Aut\, (N)$ tel que $\psi(h) =u\phi(h)u^{-1}$, alors $N \rtimes_\phi H \simeq N \rtimes_\psi H$.}
\end{enumerate}
}
