\contenu{
\texte{
Soit $G$ un groupe, $A$ et $B$ deux sous-groupes de $G$ tels que $A\subset
B$. On suppose que $A$ est d'indice fini dans $G$.
Montrer que $ \vert G : A\vert =  \vert G :  B\vert   \vert B : A\vert $.
}
}
