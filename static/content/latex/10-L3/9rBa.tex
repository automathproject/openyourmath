\contenu{
\texte{
Une quadrique d'un espace projectif $P(V)$ est le lieu des zéros d'une forme quadratique $f$ sur $V$.
}
\begin{enumerate}
    \item \question{Montrer que toute quadrique qui contient trois points distincts d'une droite $d$ contient toute la droite $d$.}
    \item \question{Déterminer la dimension de l'espace des quadriques de $P^3(K)$.}
    \item \question{Soit $d_1, d_2, d_3$ trois droites de $P^3(K)$. Montrer qu'il existe
une quadrique qui les contient.}
\end{enumerate}
}
