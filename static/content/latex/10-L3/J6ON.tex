\texte{
Utiliser l'algorithme d'Euclide dans $\Q[X]$ pour exprimer le pgcd demandé sous forme de combinaison linéaire des deux polyn\^omes donnés :
}
\begin{enumerate}
    \item \question{$ (X^4+2X^3-X^2-4X-2,X^4+X^3-X^2-2X-2)$,}
    \item \question{$(X^5+3X^4+X^3+X^2+3X+1,X^4+2X^3+X+2)$,}
    \item \question{$(3X^3-2X^2+X+2,X^2-X+1)$.}
\end{enumerate}
}
