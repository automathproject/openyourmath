\texte{

}
\begin{enumerate}
    \item \question{Si $\mathbb{F} $ est un corps, montrer qu'un polyn\^ome $P$ de degré 2 ou 3 dans $\mathbb{F} [X]$ est irréductible si et seulement si il n'a pas de zéro dans $\mathbb{F}$.}
    \item \question{Décomposer les polyn\^omes suivants en facteurs irréductibles dans $\mathbb{F}_3[X]$.
$$
X^2+X+1,\hspace{1cm}X^3+X+2,\hspace{1cm}X^4+X^3+X+1\; . 
$$}
\end{enumerate}
