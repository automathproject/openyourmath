\chapitre{Groupe fini}

\contenu{
\texte{
Soit $G$ un groupe fini. Soit $p$ le plus petit facteur premier de l'ordre de $G$. Soit $H$ un sous-groupe de $G$ d'indice $p>1$.
}
\begin{enumerate}
    \item \question{Montrez que les orbites de l'action de $H$ sur $G/H$ par translation à gauche sont réduites à des points.}
    \item \question{Montrez que $H$ est distingué.}
\end{enumerate}
}
