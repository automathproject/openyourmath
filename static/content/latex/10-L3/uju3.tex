\texte{
Exprimer à l'aide des polyn\^omes symétriques fondamentaux :
}
\begin{enumerate}
    \item \question{$(x_1^2+x_2^2)(x_1^2+x_3^2)(x_2^2+x_3^2)$ ;}
    \item \question{$(x_1+x_2)(x_1+x_3)(x_1+x_4)(x_2+x_3)(x_2+x_4)(x_3+x_4)$ ;}
    \item \question{$(x_1+x_2-x_3-x_4)(x_1-x_2+x_3-x_4)(x_1-x_2-x_3+x_4)$.}
\end{enumerate}
}
