\texte{
Soit $p=2m+1$ un nombre premier. Montrer que :
}
\begin{enumerate}
    \item \question{$\quad (p-1)!\equiv -1 \mod p$;}
    \item \question{$\quad (m!)^2\equiv (-1)^{m+1} \mod p$.}
\end{enumerate}
}
