\texte{

}
\begin{enumerate}
    \item \question{Montrer que l'image réciproque d'un idéal par un morphisme d'anneaux est un idéal.}
    \item \question{Montrer par un contre-exemple que l'image d'un idéal par un morphisme d'anneaux n'est pas toujours un idéal. Montrer que l'affirmation précédente est toutefois vraie si le morphisme est surjectif.}
    \item \question{Qu'est-ce qu'on peut dire sur l'image réciproque et l'image d'un idéal premier/maximal par un morphisme d'anneaux eventuellement surjectif ?}
\end{enumerate}
