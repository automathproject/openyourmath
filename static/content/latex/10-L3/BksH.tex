\contenu{
\texte{
Démontrer les propriétés suivantes
}
\begin{enumerate}
    \item \question{$(\C / \R ,+) \simeq (\R,+)$ ;}
    \item \question{$\C^*/\R_+^*\simeq \mathbb{S}^1$ ;}
    \item \question{$\C^*/\mathbb{S}^1\simeq \R_+^*\simeq \R^*/\{ \pm 1 \}$ ;}
    \item \question{$\R^*/ \R_+^*\simeq \{ \pm 1 \}$ ;}
    \item \question{$(\C / \Z ,+)\simeq \C^*$ ;}
    \item \question{$\C^* /\R^* \simeq \mathbb{S}^1/\{ \pm 1 \}$.}
\end{enumerate}
}
