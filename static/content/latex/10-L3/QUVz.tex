\chapitre{Groupe fini}
\sousChapitre{Groupe fini}

\contenu{
\texte{
Soit $V$ un $\R$-espace vectoriel de dimension finie et $GL(V)$ le groupe des automorphismes linéaires. Pour $f\in GL(V)$ et $a\in V$ on considère l'application $A_{f,a}:V\to V,\; A_{f,a}(v)=f(v)+a$.
}
\begin{enumerate}
    \item \question{Montrer que ${\mathcal{A}}:=\{A_{f,a}\mid f\in GL(V),\; a\in V \}$ muni de l'opération de composition est un groupe. Ce groupe s'appelle {\it le groupe des transformations affines}.}
    \item \question{\label{q:2} Trouver deux sous-groupes de ${\mathcal{A}}$ isomorphes à $(GL(V),\cdot)$ et à $(V,+)$, respectivement.}
    \item \question{Montrer que ${\mathcal{A}}$ est produit semi-direct des deux sous-groupes obtenus dans \ref{q:2}.}
\end{enumerate}
}
