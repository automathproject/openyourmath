\chapitre{Forme bilinéaire}
\sousChapitre{Forme bilinéaire}

\contenu{
\texte{
Soit $H$ un espace hermitien et $a$ un endomorphisme de $H$.
}
\begin{enumerate}
    \item \question{Montrer que $b=a a ^*$ est un endomorphisme auto-adjoint
positif.}
    \item \question{Montrer en utilisant une diagonalisation qu'il existe un unique
 endomorphisme auto-adjoint positif $c$ de $H$ tel que $c^2=b$.}
    \item \question{On suppose désormais que $a$ est inversible. Montrer que $a$
 s'écrit de fa{\c c}on unique sous la forme $a=hu$ o{ù} $h$ est un
 endomorphisme auto-adjoint positif et $u$ un endomorphisme unitaire.}
    \item \question{Déterminer $h$ et $u$ pour l'endomorphisme $a$ dont la matrice
 dans la base canonique de $\Cc^2$ est $$\left[
  \begin{array}{cc}
1&i\\ i&1
  \end{array}
\right]$$}
\end{enumerate}
}
