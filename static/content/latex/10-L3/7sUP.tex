}
\begin{enumerate}
    \item \question{Soit $A$ un anneau commutatif et $I\subset A$ un idéal.
Montrer que $I[X] $ est un idéal de $A[X]$. Montrer que $A[X]/I[X]\simeq A/I[X]$. Montrer que si $I$ est premier, $I[X]$ est premier.}
    \item \question{Soit $p$ un nombre premier. Montrer que $\Z[X]/(p)\simeq \mathbb{F}_p[X]$.}
    \item \question{Montrer que les idéaux premiers non-nuls de $\Z[X]$ sont :
  \begin{enumerate}}
    \item \question{$(p)$, où $p\in \N$ est un nombre premier ;}
    \item \question{$(R(X))$, où $R\in \Z[X]$ est un polyn\^ome irréductible ;}
    \item \question{$(p,R(X))$, où $p\in \N$ est un nombre premier et $R\in \Z[X]$ est un polyn\^ome irréductible dont la réduction modulo $p$, $\overline{R}(X)\in \mathbb{F}_p[X]$, est un polyn\^ome irréductible.}
\end{enumerate}
