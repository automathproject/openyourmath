\begin{enumerate}
    \item \question{Pour les anneaux de polyn\^omes sur un corps $F$ montrer que $F[x]/(x^2-1)\simeq F[x]/(x^2-4),\; \; F[x]/(x^2+1)\simeq F[x]/(x^2+2x+2)\, ,$ $F[x,y]/(x+y)\simeq F[x],\; F[x]/(x+1)\simeq F$.}
    \item \question{Montrer que $\Q[x]/(x^2-2)\simeq  \Q[\sqrt{2}],\; \Z[x]/(x^2+1)\simeq  \Z[i].$ Déduire du dernier isomorphisme que $(x^2+1)$ est un idéal premier non-maximal.}
\end{enumerate}
