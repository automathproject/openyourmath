\contenu{
\texte{
Soit $I$ un id\'eal d'un anneau $A$. On note par $(a)=a\cdot A$ l'id\'eal principal engendr\'e par $a$. Montrer que :
}
\begin{enumerate}
    \item \question{$I=A$ si et seulement si $I$ contient une unit\'e;}
    \item \question{$(a)=A$ ssi $a$ est inversible;}
    \item \question{Un anneau $A$ est  un corps ssi  $(0)$ est le seul id\'eal propre de $A$.}
\end{enumerate}
}
