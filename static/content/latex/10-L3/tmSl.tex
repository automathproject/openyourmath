\chapitre{Extension de corps}
\sousChapitre{Extension de corps}

\contenu{
\texte{
Soit $E$ une extension du corps $k$. On dit que $E$ est une {\it extension quadratique} de $k$ si $[E:k]=2$.
}
\begin{enumerate}
    \item \question{On suppose $k$ de caractéristique $\neq 2$. Montrer que les extensions quadratiques de $k$ (s'il en existe) sont des corps de rupture des polyn\^omes irréductibles de la forme $X^2-a,\; a\in k$. En déduire toutes les extensions quadratiques de $\Q$ à isomorphisme près.}
    \item \question{On suppose $k$ de caractéristique $2$. Montrer que les extensions quadratiques de $k$ (s'il en existe) sont des corps de rupture des polyn\^omes irréductibles de l'une des deux formes : $X^2-a$ ou $X^2-X-a,\; a\in k$. Une extension quadratique du premier type et une extension quadratique du deuxième type peuvent-elles être isomorphes au dessus de $k$ ?}
\end{enumerate}
}
