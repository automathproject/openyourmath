\texte{
Pour quelle valeur de $\lambda$ les polyn\^omes suivants ont-ils un zéro en commun ?
}
\begin{enumerate}
    \item \question{$X^3-\lambda X+2$ et $X^2+\lambda X+2$ ;}
    \item \question{$X^3+\lambda X^2-9$ et $X^3+\lambda X-3$.}
\end{enumerate}
}
