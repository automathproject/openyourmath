\chapitre{Sous-groupe, morphisme}
\sousChapitre{Sous-groupe, morphisme}

\contenu{
\texte{
Soit $V_4:=\{ e,\; (12)(34),\; (13)(24),\; (23)(14) \}\subset S_4$.
}
\begin{enumerate}
    \item \question{Montrer que c'est un sous-groupe distingué de $S_4$.}
    \item \question{Montrer que $V_4\simeq \Z/2 \Z \times \Z/2 \Z$.}
    \item \question{Soit $H=\{ e,\; (12)(34) \}$. Montrer que $H \vartriangleleft V_4$ mais que $H$ n'est pas un sous-groupe distingué de $S_4$.}
    \item \question{Montrer que $S_4/V_4$ est isomorphe à $S_3$.}
\end{enumerate}
}
