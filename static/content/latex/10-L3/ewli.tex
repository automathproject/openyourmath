\contenu{
\texte{
Soit $A$ un anneau commutatif et soit 
$$
Nil\; (A):=\{a\in A \mid \exists n\in N,\; a^n=0 \}\; .
$$
}
\begin{enumerate}
    \item \question{Montrer que $Nil\; (A)$ est un idéal de $A$.}
    \item \question{Montrer que si $P$ est un idéal premier de $A$, alors $Nil\; (A)\subset P$.}
    \item \question{Montrer que pour tout $x\not\in Nil\; (A)$ il existe un idéal premier $P$ 
tel que $x\not\in P$ ({\it Indication}: Utiliser le théorème de Zorn).}
    \item \question{En déduire que 
$$
Nil\; (A)=\bigcap_{P\mbox{ premier }}P\; .
$$}
\end{enumerate}
}
