\begin{enumerate}
    \item \question{Montrer que si $A[X]$ est principal, alors $A$ est principal.}
    \item \question{Montrer que si $A[X]$ est principal est si $A$ n'est pas un corps, alors il existe un corps $K$ tel que $K[X]$ soit un corps.}
    \item \question{En déduire que $A[X]$ est principal si et seulement si $A$ est un corps.
{\it Remarque}: Ceci implique que si $A[X]$ est principal alors $A[X]$ est euclidien.}
\end{enumerate}
