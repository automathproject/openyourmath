\chapitre{Anneau, corps}

\contenu{
\texte{
Soit $A$ un anneau.  D\'emontrer que :
}
\begin{enumerate}
    \item \question{$\forall a\in A\ \ 0_A
\cdot a=0_A$;}
\reponse{Ecrire $(0+a)a=a.a$ d'une part ($0$ est neutre pour $+$) et
    $(0+a).a=0.a+a.a$ (distributivité).}
    \item \question{$(-1_A)\cdot a=-a$;}
\reponse{$(-1).a+a=(-1+1).a=0.a=0$ (distributivité, puis question précédente)}
    \item \question{$|A|\geq 2\ \Longleftrightarrow\ 1_A\neq 0_A$ dans $A$.}
\reponse{Si $|A|=1$, $1=0$. Si $1=0$, $\forall a\in A, a=1.a=0.a=0$, donc
    $A=\{0\}$.}
\end{enumerate}
}
