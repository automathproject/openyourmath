\contenu{
\texte{
Soit $G$ un sous-groupe du groupe symétrique $S_n,\; n\geq 2$, contenant une permutation impaire. Montrer que $GA_n=S_n$ et en déduire que $G$ contient au moins un sous-groupe distingué d'indice 2.
}
}
