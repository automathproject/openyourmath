\texte{
Soient $n \in \Nn^*$ et $x_1, \ldots, x_n \in ]0,  + \infty[$.
}
\begin{enumerate}
    \item \question{En utilisant la concavit\'e du $\log$, montrer que
$ (x_1 \ldots x_n)^{\frac 1n} \leq \frac{x_1 + \ldots + x_n}n$.}
    \item \question{Montrer que $ (x_1 \ldots x_n)^{\frac 1n} \geq \frac 1{\frac 1{x_1} + \ldots +
\frac 1{x_n}}$.}
    \item \question{En d\'eduire que $ n ! \leq (\frac{n + 1}2)^n$.}
\end{enumerate}
}
