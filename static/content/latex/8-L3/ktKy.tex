\chapitre{Théorème de Sylow}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Soit $G$ un $p$-groupe agissant sur un ensemble fini $E$.
 Montrer que le cardinal de l'ensemble des points fixes de l'action
 est congru, modulo $p$, au cardinal de $E$.}
    \item \question{En considérant une action de $G$ sur lui-même, montrer que le théorème de Burnside : 
 le centre d'un $p$-groupe non réduit à l'élément neutre
 n'est pas réduit à l'élément neutre.}
\end{enumerate}
