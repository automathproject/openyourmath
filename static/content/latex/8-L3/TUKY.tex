\contenu{
\texte{
Soit $f:G  \rightarrow H$  un morphisme de groupes finis. Soit $G^\prime$ un
sous-groupe de $G$ d'ordre premier \`a l'ordre de $H$. Montrer que $G^\prime\subset
\hbox{\rm ker}(f)
$.
}
\indication{R\'esulte de l'exercice \ref{ex:le12}.}
}
