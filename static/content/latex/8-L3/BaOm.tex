\chapitre{Théorème de Sylow}

\contenu{
\texte{
Soit $p$ un nombre premier et $m$ un entier non multiple de $p$.
Soit $G$ un groupe de cardinal $|G|=p^dm$.
}
\begin{enumerate}
    \item \question{Montrer que le nombre de $p$-Sylow de $G$ divise $m$.}
    \item \question{Montrer que pour tout $0\leq i\leq d$, $G$ possède un sous groupe d'ordre $p^i$.}
\end{enumerate}
}
