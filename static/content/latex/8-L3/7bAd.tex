\contenu{
\texte{
Soit $G$ un groupe d'ordre $6$.
}
\begin{enumerate}
    \item \question{Montrer que $G$ admet un élément $\tau$ d'ordre $2$ et un élément $\sigma$ d'ordre $3$.}
    \item \question{Quelles sont les valeurs possibles de $\tau\sigma\tau$ ?}
    \item \question{Déterminer, dans chacun des cas précédents, la structure de $G$ à isomorphisme près.}
\end{enumerate}
}
