\chapitre{Sous-groupe distingué}
\sousChapitre{Sous-groupe distingué}

\contenu{
\texte{
Soient $T=\left\{\left(\begin{smallmatrix}
a&b\\0&c\end{smallmatrix}\right) :
a,c\in\Rr\setminus\left\{0\right\},b\in
\Rr\right\}$ et $U=\left\{\left(\begin{smallmatrix} 1&b\\0&1\end{smallmatrix}\right) : b\in
\Rr\right\}.$\\
}
\begin{enumerate}
    \item \question{Montrer que $T$ est un sous-groupe de $\text{GL}_2(\Rr).$}
    \item \question{Montrer que $U$ est un sous-groupe distingu\'e de $T.$}
\end{enumerate}
}
