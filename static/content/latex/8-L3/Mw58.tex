\texte{
On travaille dans $GL(2,\mathbb{F}_3)$.
}
\begin{enumerate}
    \item \question{Soit $$t=\begin{pmatrix}1&1\cr 0&1\end{pmatrix} \quad \text{ et } \quad s=\begin{pmatrix}1&0\cr 0&-1\end{pmatrix}.$$
Calculer $tst^{-1}s^{-1}$.}
    \item \question{Déterminer $D(GL(2,k))$.
Noter que ce calcul ne suffit pas pour déterminer $D(SL(2,k))$.}
\end{enumerate}
