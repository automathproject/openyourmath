\texte{

\begin{enumerate}
    \item \question{Montrer que le groupe des matrices triangulaires supérieures de diagonale identité est résoluble.}
    \item \question{Montrer que si $H$ est un sous-groupe distingué d'un groupe $G$, alors $G$ est résoluble si et seulement si $H$ et $G/H$ le sont.
(On pourra commencer par le cas où $G/H$ est abélien).}
    \item \question{Montrer qu'un $p$-groupe est résoluble.}
\end{enumerate}
