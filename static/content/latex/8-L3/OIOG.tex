\contenu{
\texte{
On appelle groupe dihédral $D_{2n}$ le groupe des isométries d'un polygone régulier à $n$ côtés.
}
\begin{enumerate}
    \item \question{Déterminer (par exemple à l'aide d'une action de groupe) le cardinal puis la liste des éléments de $D_{2n}$.}
    \item \question{On suppose $n$ impair. Déterminer les $2$-Sylow de $D_n$ et vérifier qu'ils sont conjugués.}
    \item \question{On suppose $n=6$. Déterminer un $2$-Sylow de $D_6$.
Déterminer deux sous-groupes d'ordre $2$ de $D_6$ non conjugués dans $D_6$.}
\end{enumerate}
}
