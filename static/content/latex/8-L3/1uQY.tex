\texte{

}
\begin{enumerate}
    \item \question{Montrer que, après avoir fixé un générateur de $(\Z/11\Z)^\star$,
 la donnée d'un morphisme de groupes de $\Z/5\Z$ dans $Aut(\Z/11\Z)$
 revient à la donnée d'un morphisme de groupes de $\Z/5\Z$ dans $\Z/10\Z$.}
    \item \question{En déduire une structure de produit semi-direct sur $\Z/11\Z\rtimes\Z/5\Z$.}
    \item \question{Montrer que toutes les structures de produit semi-direct donnent des groupes isomorphes.
 \textit{On pourra montrer que si $\phi, \psi\in Hom(\Z/5\Z,Aut(\Z/11\Z)$ 
 alors il existe $\gamma\in Aut(\Z/11\Z)$ tel que $\psi (h) = \gamma\circ\phi (h)\circ \gamma^{-1}$.}}
    \item \question{Montrer que tous les morphismes de $\Z/p\Z\to Aut (\Z/q\Z)$ sont de la forme 
 $t\mapsto \{x\mapsto k^t x\}$ où $k$ est un élément de $(\Z/q\Z)^\star$ d'ordre $p$.}
\end{enumerate}
