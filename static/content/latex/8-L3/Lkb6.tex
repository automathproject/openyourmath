\contenu{
\texte{
Soit $G$ un groupe d'ordre $10$.
}
\begin{enumerate}
    \item \question{Montrer que $G$ est un produit semi-direct.}
    \item \question{Déterminer les automorphismes d'ordre $2$ de $\Z/5\Z$.}
    \item \question{En déduire les deux possibilités pour les classes d'isomorphismes de $G$.}
\end{enumerate}
}
