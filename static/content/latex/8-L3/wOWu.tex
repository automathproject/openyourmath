\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Montrer que les automorphismes du groupe $(\Z/n\Z,+)$ sont obtenus par multiplication par un inversible de 
 $(\Z/n\Z,\times)$.}
    \item \question{Décrire un isomorphisme de $\Z/10\Z$ sur $(\Z/11\Z)^\star$.}
    \item \question{Montrer que si $G$ et $H$ sont deux groupes d'ordre premiers entre eux, alors
 $$Aut(G\times H)=Aut(G)\times Aut (H).$$}
    \item \question{En déduire le groupe des automorphismes de $\Z/133\Z$.}
    \item \question{Soit $p$ un nombre premier et $n$ un entier naturel non nul.
 Montrer que $$Aut((\Z/p\Z)^n)=GL(n,\mathbb{F}_p).$$}
    \item \question{Montrer que $Aut(\Z/2\Z\times\Z/2\Z)\to Bij((1,0),(1,1),(0,1))$
 est un isomorphisme.}
\end{enumerate}
