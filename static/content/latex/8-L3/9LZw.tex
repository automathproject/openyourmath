\texte{
Soit $G$ un groupe tel que l'application $x\rightarrow x^{-1}$ soit un morphisme.
Montrer que $G$ est commutatif.
}
\indication{$(xy)^{-1} = x^{-1} y^{-1} \Rightarrow xy=yx$.}
}
