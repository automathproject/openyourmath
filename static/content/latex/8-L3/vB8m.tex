\chapitre{Sous-groupe distingué}
\sousChapitre{Sous-groupe distingué}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Soit $  G  $ un groupe, $  H  $ un sous-groupe de $  G  .$
Montrer que les propri\'et\'es suivantes sont \'equivalentes~:
\vskip1mm \hskip2mm {\em i)} $  \forall g \in G  :  gHg^{-1}\subset H  .$
\vskip1mm \hskip2mm {\em ii)} $  \forall g \in G  :  gHg^{-1}= H  .$
\vskip1mm \hskip2mm {\em iii)} $  \forall g \in G  :  gH=Hg  .$}
    \item \question{En d\'eduire que tout sous-groupe d'indice $  2  $ est distingu\'e.}
\end{enumerate}
}
