\chapitre{Sous-groupe distingué}
\sousChapitre{Sous-groupe distingué}

\contenu{
\texte{
Montrer que $\textrm{SL}_n(\Rr)$ est un sous-groupe distingu\'e du groupe $\textrm{GL}_n(\Rr)$ et que le groupe quotient est isomorphe \`a $\Rr^\times$.
}
\indication{Le morphisme ``d\'eterminant'' de $\textrm{GL}_n(\Rr)$ dans $\Rr^\times$ est surjectif et de noyau $\textrm{SL}_n(\Rr)$.}
}
