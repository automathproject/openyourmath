\begin{enumerate}
    \item \question{Soit $p$ un nombre premier. Déterminer à isomorphisme près, tous les groupes d'ordre $p$.}
    \item \question{Soit $p$ un nombre premier. Déterminer à isomorphisme près, tous les groupes d'ordre $p^2$.}
    \item \question{Donner des exemples de groupes d'ordre $6$ non abéliens.}
    \item \question{Déterminer l'ordre des groupes diédraux $D_n$.}
    \item \question{Déterminer l'ordre des groupes alternés $\mathcal{A}_n$.}
\end{enumerate}
