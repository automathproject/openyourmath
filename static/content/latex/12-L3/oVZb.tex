\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Montrez que $d(x,y)=|x-y|$ est bien une distance sur
l'ensemble des r\'eels.}
    \item \question{Pour tout couple d'\'el\'ements
$X=(x_1,...,x_n)$ et $Y=(y_1,...,y_n)$ de $\mathbb{R}^n$, on
d\'efinit $d(X,Y)=\sup_{i=1..n}|x_i-y_i|$. Montrez que $d$ est
bien une distance sur $\mathbb{R}^n$.}
    \item \question{Faire de m\^eme avec
$d(X,Y)=|x_1-y_1|+|x_2-y_2|+...+|x_n-y_n|$.}
\end{enumerate}
}
