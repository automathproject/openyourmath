\contenu{
\texte{
Déterminer les extremums de la fonction $f(x,y,z) = 2x+3y+2z$ sur
l'intersection du plan d'équation $x+z=1$ avec le cylindre
${\cal Z}= \{x^2+y^2 = 2\}\subset \Rr^3$.
}
}
