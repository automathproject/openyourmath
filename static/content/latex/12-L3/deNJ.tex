\contenu{
\texte{
Soit $g(x,y,z)=xyz -32$, ${\cal S} =\{(x,y,z)\in \Rr^3 \; ;\;
g(x,y,z) =0 \}$ et soit $f: \Rr^3 \to \Rr$ l'application
$f(x,y,z) = xy+2yz + 2xz$. Déterminer $\min \{f(x,y,z) \; ;
(x,y,z) \in S\}$.
}
}
