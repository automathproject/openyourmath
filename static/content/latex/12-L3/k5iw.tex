\contenu{
\texte{
Trouvez les points critiques des fonctions suivantes et
d\'eterminez si ce sont des minima locaux, des maxima locaux ou
des points selle.
}
\begin{enumerate}
    \item \question{$f(x,y)=x^3+6x^2+3y^2-12xy+9x$ ;}
    \item \question{$f(x,y)=\sin x+y^2-2y+1$ ;}
    \item \question{$f(x,y,z)=\cos 2x\cdot \sin y+z^2$ ;}
    \item \question{$f(x,y,z)=(x+y+z)^2$.}
\end{enumerate}
}
