\chapitre{Extremum, extremum lié}

\contenu{
\texte{
Déterminer les extremums (locaux et/ou globaux) de:
}
\begin{enumerate}
    \item \question{$f(x,y) = x^2 -y^2$, $(x,y) \in \Rr^2$.}
    \item \question{$f(x,y) = x^3 -y^3$, $(x,y) \in \Rr^2$.}
    \item \question{$f(x,y) = x^3 +y^3-3xy$, $(x,y) \in \Rr^2$.}
    \item \question{$f(x,y) = x^2 +y^2-2xy+1$, $(x,y) \in \Rr^2$.}
\end{enumerate}
}
