\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Déterminer le plan tangent en $A(1,-1,-1)$
à la surface paramétrée par $(u,v)\mapsto (u+v^2,u^2-v^2,v)$.}
    \item \question{Déterminer le plan tangent en $A(1,1,-1)$
à la surface d'équation $x^5+y^5+z^5=1$.}
    \item \question{Soit $U$ un ouvert de $\Rr^2$ et $f~:U\to \Rr$ une application de classe $\mathcal{C}^\infty$.
Déterminer le plan tangent au graphe de $f$ en chacun de ses points.}
\end{enumerate}
