\contenu{
\texte{
Pour quelle(s) valeur(s) de $k$ l'origine est-elle un
puits pour l'équation $x' = Ax$~?
}
\begin{enumerate}
    \item \question{$\begin{pmatrix} a & -k \cr k &2 \cr \end{pmatrix}$,}
    \item \question{$\begin{pmatrix} 3 & 0 \cr k &-4 \cr \end{pmatrix}$,}
    \item \question{$\begin{pmatrix} k^2 & 1 \cr 0 &k \cr \end{pmatrix}$,}
    \item \question{$\begin{pmatrix} 0 & -1 &0 \cr 1&0&0 \cr -1 &0 &k
\cr \end{pmatrix}$.}
\end{enumerate}
}
