\contenu{
\texte{
On considère l'équation différentielle 
$$(E)\qquad y''+y'+y=0$$
}
\begin{enumerate}
    \item \question{Résoudre l'équation différentielle $(E)$ et étudier le comportement des
solu\-tions en
$+\infty$ et $-\infty$.}
    \item \question{Soit $f$ une fonction de classe ${\cal C}^2$ sur
${\Rr}$ telle que
$f''+f'+f$ soit
$T$-périodique. Montrer que $f(x+T)-f(x)$ est solution de $(E)$.
En déduire que si $f$ est bornée sur ${\Rr}$, $f$ est elle-même
$T$-périodique.}
\end{enumerate}
}
