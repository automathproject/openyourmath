\contenu{
\texte{
Soit $f : {[a,b]} \to \R$ de classe $\mathcal{C}^2$. On note $M = \sup|f''|$ et on suppose $M>0$.
}
\begin{enumerate}
    \item \question{Montrer que : $\forall\ x \in {]a,b[}$, on a $\left|f'(x) - \frac {f(b)-f(a)}{b-a}\right|
    < M\frac {b-a}2$.}
\reponse{Formule de Taylor pour calculer $f(a)$ et $f(b)$ à partir de $f(x)$.}
    \item \question{Si $\left|f'(a) - \frac {f(b)-f(a)}{b-a}\right| = M\frac {b-a}2$,
    montrer que $f$
    est polynomiale de degré inférieur ou égal à 2.}
\reponse{\'Etudier $f(x) - f(a) - (x-a)f'(a) \pm M(x-a)^2/2$.}
\end{enumerate}
}
