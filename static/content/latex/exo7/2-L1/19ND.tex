\chapitre{Suite}
\sousChapitre{Convergence}

\contenu{
\texte{
Soit $(u_n)$ une suite r{\'e}elle convergente, et
$S_n = \frac 1{2^n} \sum_{p=0}^n\,C_n^pu_p$.
{\'E}tudier la suite $(S_n)$.
}
\reponse{
$\lim S_n = \lim u_n$.\par
Si $u_n = (3i/2)^n$, alors $(u_n)$ diverge, mais $S_n \xrightarrow[n\to\infty]{} 0$.
}
}
