\chapitre{Continuité, limite et étude de fonctions réelles}

\contenu{
\texte{
Soit $f:[0,1]\rightarrow {\Rr}$ continue, \'{e}valuer :
$$\lim\limits_{n\rightarrow \infty }\sum\limits_{k=1}^{n}(-1)^{k}f\left(\frac{k}{n}\right). $$
}
