\texte{
\'Etudier la convergence et calculer la limite \'{e}ventuelle de la suite
$(u_{n})_{n\in \Nn^{*}}$ d\'{e}finie par :
$$\forall n\in \Nn^{*},u_{n}=\prod\limits_{k=1}^{n}(1+\frac{k}{n^{2}}).$$
}
}
