\begin{enumerate}
    \item \question{$\frac 1{\sin^2x} - \frac 1{\sh^2x}                      \to \frac 23$ lorsque  $x\to 0$.}
    \item \question{$\frac {\sin x\sh x - \tan x\tanh x}{\sh^4x - \tanh^4x}       \to -> -\frac1{12}$ lorsque  $x\to 0$.}
    \item \question{$(\ch x)^\alpha - (\sh x)^\alpha                           \to

    \item \question{$\frac{\cos\pi x}{4x^2-9}                                 \to  -> \frac\pi{12}$ lorsque  $x\to 3/2$.}
    \item \question{$\frac{\sin 3x}{1-2\cos x}                                \to-\sqrt3$ lorsque  $x\to \pi/3$.}
    \item \question{$\frac{e^{\sin x}-e^x}{\sin x-x}                          \to 1$ lorsque  $x\to 0$.}
