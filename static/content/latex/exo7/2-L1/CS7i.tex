\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Calculs}

\contenu{
\texte{
Calculer les dérivées des fonctions~:
}
\begin{enumerate}
    \item \question{$x \mapsto \sqrt{1+x^2 \sin^2 x},\qquad  x \mapsto \frac{\exp(1/x)+1} {\exp(1/x)-1}.$}
    \item \question{$x \mapsto \log (\frac{1+\sin(x)} {1-\sin(x)}),\qquad  x \mapsto (x(x-2))^{1/3}.$}
\end{enumerate}
}
