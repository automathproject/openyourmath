\chapitre{Suite}
\sousChapitre{Convergence}

\contenu{
\texte{
On donne la suite $(u_n)$ définie par~:
$$ u_1= \sqrt2 \quad\mbox{et}\quad u_n = \sqrt{2-u_{n-1}}.$$
En étudiant les suites $(u_{2n})$ et $(u_{2n+1})$,
montrer que la suite $(u_n)$ est convergente.
}
}
