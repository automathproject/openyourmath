\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soit $f$ une fonction dérivable sur $[0,1]$, vérifiant~:
}
\begin{enumerate}
    \item \question{[i)]  $0\leq f'\leq2$~;}
    \item \question{[ii)]  $f'$ est décroissante~;}
    \item \question{[iii)]  $f(0)=0$ et $f(1)=1$.}
\end{enumerate}
}
