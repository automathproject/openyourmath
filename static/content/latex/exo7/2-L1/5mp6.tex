\chapitre{Calcul d'intégrales}

\contenu{
\texte{
Calculer les primitives suivantes :
$$
\int e^x\cos x dx \quad ; \quad
\int \frac{\ln x}{x^n}dx \ n\in\Nn \quad ; \quad
\int x \text{Arctan}\,x dx \quad ; \quad
\int (x^2+x+1)e^x dx.
$$
}
