\chapitre{Calcul d'intégrales}
\sousChapitre{Théorie}

\contenu{
\texte{
Soient $f \in C^{1} ([a, b], \Rr)$ et $I_{n} = \int_{a}^{b}{f (t)\sin (nt)dt}$.
}
\begin{enumerate}
    \item \question{A l'aide d'une int\'egration par parties, montrer que $I_{n} \rightarrow 0$.}
    \item \question{Montrer que ceci est encore vrai si $f$ est en escalier.}
    \item \question{En d\'eduire que le r\'esultat subsiste pour $f$ continue par morceaux.}
\end{enumerate}
}
