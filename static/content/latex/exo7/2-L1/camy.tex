\contenu{
\texte{
Soit $(u_n)$ définie par : $u_1 \in \R$, $u_{n+1} = \frac1{ne^{u_n}}$.
Quelle est la nature de la série $\sum u_n$ ?
}
\reponse{
Pour $n > 2$, $u_{n+1} < \frac1n$
     donc $u_{n+2} > \frac1{(n+1)e^{1/n}} \sim \frac1n$ donc la série
     diverge.
}
}
