\texte{
D\'eveloppements limit\'es ~en $0$ de :
}
\begin{enumerate}
    \item \question{$\cos x. \ln (1+x)$ \`a l'ordre $4$.}
    \item \question{$\displaystyle{\frac{1}{\cos x}}$ \`a l'ordre $4$.}
    \item \question{$\arcsin \left ( \ln(1+x^2) \right )$ \`a l'ordre $6$.}
    \item \question{$\displaystyle{\frac{\sinh x-x}{x^3}}$ ~\`a l'ordre $4$.}
    \item \question{$(1+x)^{\frac{1}{1+x}}$ \`a l'ordre $3$.}
\end{enumerate}
}
