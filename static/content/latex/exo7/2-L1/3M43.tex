\texte{
On pose $u_n = \sum_{k=1}^{2n} (-1)^k\sqrt k$. Donner un équivalent de $u_n$
quand $n\to\infty$.
(Regrouper les termes deux par deux puis comparer à une intégrale)
}
\reponse{
$\sqrt{\frac n2}$.
}
}
