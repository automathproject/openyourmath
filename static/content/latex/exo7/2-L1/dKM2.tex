\contenu{
\texte{
Soit $f:\R \to \R$ trois fois dérivable en $a$.
\'Etudier $\lim_{h\to0}\frac1{h^4}
\begin{vmatrix} 1  &f(a)     &f(a+h)  \cr
          1  &f(a+h)   &f(a+2h) \cr
          1  &f(a+2h)  &f(a+3h) \cr \end{vmatrix}$.
}
\reponse{
$f'(a)f'''(a) - f''(a)^2$.
}
}
