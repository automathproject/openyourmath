\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Fonctions convexes}

\contenu{
\texte{
Soit $f : {[a,b]} \to {\R}$ continue telle que :
$\forall\ x,y \in {[a,b]},\ f\left(\frac {x+y}2\right) \le \frac {f(x) + f(y)}2$.

Montrer que $f$ est convexe.
}
}
