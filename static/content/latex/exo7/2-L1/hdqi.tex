\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $(a_k)$ une suite de réels telle que $\sum_{k=0}^n a_k = 0$.
\'Etudier la convergence de
$ \int_{t=0}^{+\infty} \sum_{k=0}^n a_k\cos(a_kt)\frac{d t}t$.
}
}
