\chapitre{Continuité, limite et étude de fonctions réelles}
\sousChapitre{Continuité : théorie}

\contenu{
\texte{
Soit $f : {[0,+\infty[} \to \R$ une fonction continue ayant une limite
finie en $+\infty$.
}
\begin{enumerate}
    \item \question{Montrer que $f$ est bornée.}
    \item \question{Montrer que $f$ admet un maximum ou un minimum absolu, mais pas nécéssairement les deux.}
    \item \question{Montrer que $f$ est uniformément continue.}
\end{enumerate}
}
