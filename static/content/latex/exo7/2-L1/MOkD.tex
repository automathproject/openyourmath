\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Autre}

\contenu{
\texte{
Soit $f : {[a,b]} \to \R$ $n$ fois dérivable.
}
\begin{enumerate}
    \item \question{Si $f$ s'annule en $n+1$ points distincts dans $[a,b]$,
      montrer qu'il existe $c \in {]a,b[}$ tel que $f^{(n)}(c) = 0$.}
    \item \question{Si $f(a) = f'(a) = \dots = f^{(n-1)}(a) = f(b) = 0$,
      montrer qu'il existe $c \in {]a,b[}$ tel que $f^{(n)}(c) = 0$.}
\end{enumerate}
}
