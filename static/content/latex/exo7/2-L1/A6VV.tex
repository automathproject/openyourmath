\chapitre{Développement limité}
\sousChapitre{Autre}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Montrer que $\frac1x\ln(\ch x)$ admet en $+\infty$ un développement limité
    généralisé à tout ordre.}
    \item \question{En déduire le développement limité de $(\ch x)^{1/x}$ en $+\infty$ à
    un ordre $n$ quelconque.}
\reponse{
$e\left(1 - \frac{\ln2}x + \frac{\ln^22}{2!\,x^2} - \dots
                       + (-1)^n\frac{\ln^n2}{n!\,x^n}\right)
                       + o (x^{-n})$.
}
\end{enumerate}
