\chapitre{Suite}
\sousChapitre{Convergence}

\contenu{
\texte{
Soit $(u_n)$ une suite r{\'e}elle. On appelle {\it valeur d'adh{\'e}rence\/}
toute limite d'une sous-suite convergente extraite de~$(u_n)$.
}
\begin{enumerate}
    \item \question{Quelles sont les valeurs d'adh{\'e}rence d'une suite convergente ?}
    \item \question{Quelles sont les valeurs d'adh{\'e}rence de la suite $(\cos(n\pi/3))$ ?}
    \item \question{Montrer que si la suite $(u_n)$ est born{\'e}e et diverge, elle a au moins deux
     valeurs d'adh{\'e}rence.}
\end{enumerate}
}
