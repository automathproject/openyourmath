\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $n \ge 0$ un entier. Montrer que l'int\'egrale
$$I_n = \int_0^\infty t^n \exp(-t^2) dt$$
est convergente. La calculer en fonction de $n$, sachant que $I_0
= \sqrt{\pi}/2$.
}
}
