\sousChapitre{Formule de Taylor}

\contenu{
\texte{
Soit $f:\R \to \R$ une fonction de classe $\mathcal{C}^n$ telle
que $f$ et $f^{(n)}$ sont bornées sur $\R$.
On veut montrer que les dérivées intermédiaires sont aussi bornées sur $\R$.
}
\begin{enumerate}
    \item \question{Cas $n = 2$ : Utiliser la formule de Taylor-Lagrange à l'ordre 2.}
    \item \question{Cas général : Utiliser l'exercice \ref{diffinie}.}
\end{enumerate}
}
