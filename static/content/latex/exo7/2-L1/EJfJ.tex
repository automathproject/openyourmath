\chapitre{Dérivabilité des fonctions réelles}
\sousChapitre{Fonctions convexes}

\contenu{
\texte{
Soit $f : \R \to \R$ convexe et $x < y < z$.
Montrer que $\begin{vmatrix} 1 &x &f(x) \cr 1 &y &f(y) \cr 1 &z &f(z) \cr\end{vmatrix} > 0$.
}
}
