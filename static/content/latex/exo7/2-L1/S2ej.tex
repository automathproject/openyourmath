\chapitre{Calcul d'intégrales}
\sousChapitre{Intégrale impropre}

\contenu{
\texte{
Soit $f : {\R^+} \to {\R^+}$ de classe $\mathcal{C}^1$, intégrable.
}
\begin{enumerate}
    \item \question{On suppose $f' \le 1$. Montrer que $f(x)\to0$ lorsque $x\to+\infty$.}
    \item \question{Est-ce encore vrai si on suppose seulement $f'\le 1+g$ avec $g$ intégrable~?}
\end{enumerate}
}
