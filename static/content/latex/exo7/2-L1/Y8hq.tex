\texte{
On consid{\`e}re l'{\'e}quation : $x^n + x^{n-1} + \dots + x - 1 = 0$.
}
\begin{enumerate}
    \item \question{Prouver qu'il existe une unique racine positive, $a_n$.}
    \item \question{Montrer que la suite $(a_n)$ est d{\'e}croissante.}
    \item \question{Montrer que $a_n \xrightarrow[n\to\infty ]{} \frac12$ (calculer $a_n^{n+1} - 1$).}
\end{enumerate}
}
