\chapitre{Propriétés de R}
\sousChapitre{Maximum, minimum, borne supérieure}

\contenu{
\texte{
Soit $A$ une partie non vide et bornée de $\Rr$. Montrer que $\mbox{sup}\{|x-y|,\;(x,y)\in A^2\}=\mbox{sup }A-\mbox{inf }A$.
}
\reponse{
$\forall(x,y)\in A^2,\;|y-x|\leq\mbox{sup }A-\mbox{inf }A$ et
$\forall\varepsilon>0,\;\exists(x,y)\in A^2/\;|y-x|>\mbox{sup }A-\mbox{inf }A-\varepsilon$.
}
}
