\chapitre{Théorème de Sylow}
\sousChapitre{Théorème de Sylow}

\contenu{
\texte{
Soit $\mathcal{P}_n$ un polygone régulier à $n$ côtés dans le plan euclidien orienté.
On appelle groupe diédral $D_{n}$ le groupe des isométries de $\mathcal{P}_n$.
}
\begin{enumerate}
    \item \question{Rappeler sans démonstration la liste complète des éléments de $D_{n}$.}
    \item \question{Décrire un sous-groupe cyclique d'ordre $n$ dans $D_{n}$.}
    \item \question{On écrit $n=2^kn'$ où $n'$ est impair. En considérant un polygone régulier inscrit, décrire une application injective de $D_{2^{k}}$ dans $D_{n}$. Faire une figure dans le cas $n=12$.}
    \item \question{Décrire pour chaque diviseur premier $p$ de $2n$, un $p$-Sylow de $D_{n}$.}
\end{enumerate}
}
