\chapitre{Sous-groupe distingué}
\sousChapitre{Sous-groupe distingué}

\contenu{
\texte{
Soit $G$ un groupe.
}
\begin{enumerate}
    \item \question{Un sous-groupe $H$ de $G$ est distingué si: $\forall x\in G, xH=Hx$, ce qui est équivalent à dire que $H$ est le noyau d'un morphisme de $G$ dans un groupe.
Rappeler la démonstration de cette équivalence.}
    \item \question{Si $H$ est un sous-groupe d'indice 2 de $G$, montrer que $H$ est distingué.}
    \item \question{Si $G$ est abélien, montrer que tout sous-groupe de $G$  est distingué.}
    \item \question{Le centre de $G$ est l'ensemble $Z(G)=\{z\in G: \forall x\in G, xz=zx\}$. Montrer que $Z(G)$ est un sous-groupe distingué.}
\end{enumerate}
}
