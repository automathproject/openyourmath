\chapitre{Action de groupe}

\contenu{
\texte{
Soit $G$ un groupe, $H$ un sous-groupe d'indice $n$ dans $G$.
}
\begin{enumerate}
    \item \question{A l'aide des classes à gauche modulo $H$ dans $G$, construire un
homomorphisme $\phi: G \mapsto{\mathcal S}_n$.}
    \item \question{Montrer que si $N \subset H$ et $N$ est normal dans $G$, on a
 $N < {\rm Ker}\ \phi < H$.}
    \item \question{En déduire que tout groupe fini $G$ est isomorphe à un
sous-groupe de ${\mathcal S}_n$.}
\end{enumerate}
}
