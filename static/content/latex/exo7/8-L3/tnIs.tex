\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Décrire un $2$-Sylow de $GL_3(\mathbb{F}_2)$.}
    \item \question{Soit $A=\begin{pmatrix}1&1&0\\1&0&1\\0&1&0\end{pmatrix}$.
Montrer que le polynôme minimal de $A$ est irréductible de degré 3.
En déduire que $GL_3(\mathbb{F}_2)\cap\mathbb{F}_2[A]$ est un $7$-Sylow de $GL_3(\mathbb{F}_2)$.}
    \item \question{Déterminer un $3$-Sylow de $GL_3(\mathbb{F}_2)$ à l'aide de la matrice 
$B=\begin{pmatrix}1&0&0\\0&1&1\\0&1&0\end{pmatrix}$.}
\end{enumerate}
