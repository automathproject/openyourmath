\texte{

}
\begin{enumerate}
    \item \question{Montrer que $\mathcal{S}_3$ est résoluble.}
    \item \question{Montrer que le groupe $D_4$ des permutations de profil $(.,.)(.,.)$ est un groupe abélien d'ordre $4$ distingué dans $A_4$.
En déduire que $\mathcal{A}_4$ et donc $\mathcal{S}_4$ sont résolubles.}
    \item \question{On suppose désormais $n\geq 3$. Soit $c$ un $3$ cycle. En considérant $c^2$, montrer que $c$ est un commutateur dans $\mathcal{S}_n$.
En déduire le sous groupe dérivé $D(\mathcal{S}_n)$.}
    \item \question{Montrer que pour $n\geq 5$, $\mathcal{A}_n$ et $\mathcal{S}_n$ ne sont pas résolubles.}
\end{enumerate}
