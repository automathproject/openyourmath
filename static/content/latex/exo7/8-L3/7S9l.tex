\contenu{
\texte{
On considère l'action d'un groupe $G$ sur un ensemble $E$.
}
\begin{enumerate}
    \item \question{Montrer qu'un sous-ensemble de $E$ est globalement stable par $G$ si
et seulement s'il est réunion d'orbites.}
    \item \question{Montrer que deux éléments dans la même orbite ont des stabilisateurs
conjugués.}
    \item \question{Montrer que deux éléments conjugués dans le groupe $G$ fixent le même nombre d'éléments.}
\end{enumerate}
}
