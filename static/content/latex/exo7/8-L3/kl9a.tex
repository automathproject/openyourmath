\contenu{
\texte{
Soient $G$ un groupe fini et $m$ un entier premier \`a l'ordre de $G$. 
Montrer que pour tout $a\in G$ l'\'equation $x^m=a$ admet une unique solution.
}
\indication{$x^m=a \Leftrightarrow x=a^u$ o\`u $um+v|G|=1$.}
}
