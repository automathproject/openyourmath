\chapitre{Espace métrique complet, espace de Banach}
\sousChapitre{Espace métrique complet, espace de Banach}

\contenu{
\texte{
Soit $E$ un espace de Banach et $F$ un sous-espace fermé de $E$.
}
\begin{enumerate}
    \item \question{Montrer que $N(\bar x)=\inf_{y\in F} \Vert x+y\Vert=d(x,F)$ définit une norme
sur l'espace vectoriel quotient $E/F$.}
    \item \question{Montrer à l'aide du critère sur les séries que $E/F$ muni de $N$ est un
espace de Banach.}
\end{enumerate}
}
