\chapitre{Espace topologique, espace métrique}
\sousChapitre{Espace topologique, espace métrique}

\contenu{
\texte{
Soit $M_n({\Rr})$ l'ensemble des matrices carrées d'ordre $n$, muni de la
distance $d(A,B)=\max_{i,j}\vert a_{i,j}-b_{i,j}\vert$ où $A=(a_{i,j})$ et
$B=(b_{i,j})$.
}
\begin{enumerate}
    \item \question{Montrer que l'ensemble des matrices inversibles est un ouvert dense de
$M_n({\Rr})$.}
    \item \question{Dans le cas $n=2$, décider si les ensembles suivants sont ouverts,
fermés, ni ouverts ni fermés :
 
 ${\cal A}$ = matrices ayant deux valeurs propres distinctes et $>0$.

 ${\cal B}$ = matrices ayant deux valeurs propres $>0$.}
\end{enumerate}
}
