\chapitre{Connexité}
\sousChapitre{Connexité}

\contenu{
\texte{
D\'eterminer les parties connexes de
$$\{(x,y)\in \Rr^2 \; ; x\neq y\} \quad  \text{et  de} \quad
\{(z,w) \in \Cc^2 \; ; z\neq w\}\; .$$
}
\reponse{
Dans $\Rr^2$ il y a deux composantes connexes : $\{(x,y)\in \Rr^2 \; ; x > y\}$ et $\{(x,y)\in \Rr^2 \; ; x < y\}$.
Dans $\Cc^2$ il n'y en a qu'une seule : $\{(z,w) \in \Cc^2 \; ; z\neq w\}
$
}
}
