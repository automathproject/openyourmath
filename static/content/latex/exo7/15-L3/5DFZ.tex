\chapitre{Espace topologique, espace métrique}
\sousChapitre{Espace topologique, espace métrique}

\contenu{
\texte{
Soit $X=\{ a,b,c,d \}$. Lesquelles parmi les collections de
sous-ensembles suivants d\'eterminent une topologie sur $X$ ?
Justifier.
}
\begin{enumerate}
    \item \question{$\emptyset$, $X$, $\{a\}$, $\{ b\}$, $\{a,c\}$, $\{a,b,c\}$,
  $\{a,b\}$;}
\reponse{d\'efinit une topologie.}
    \item \question{$\emptyset$, $X$, $\{a\}$, $\{ b\}$, $\{a,b\}$, $\{
  b,d\}$;}
\reponse{ne d\'efinit pas une topologie,
car $\{a\}\cup \{ b,d\}=\{a,b,d\}$ n'est pas dans la collection.}
    \item \question{$\emptyset$, $X$, $\{a,c,d\}$, $\{b,c,d\}$.}
\reponse{ne d\'efinit pas une topologie, car  $\{a,c,d\}\cap
\{b,c,d\}=\{c,d\}$ n'est pas dans la collection.}
\end{enumerate}
}
