\contenu{
\texte{
On dit qu'un espace topologique possède la propriété du point fixe si
toute fonction continue de $X$ dans $X$ admet un point fixe.
}
\begin{enumerate}
    \item \question{Montrer qu'un espace topologique possédant cette propriété est
nécessai\-rement connexe.}
    \item \question{Montrer que si
$X$ a cette propriété, tout
$Y$ homéomorphe à
$X$ la possède aussi.}
    \item \question{Montrer ainsi que ${\mathbb{S}^1}$ n'est pas homéomorphe à un segment.}
\end{enumerate}
}
