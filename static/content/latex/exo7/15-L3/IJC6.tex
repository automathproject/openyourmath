\contenu{
\texte{
Soit $E=C([0,1])$, $\mu(x)=\int_0^1 x(t)\ dt$, $\mu_n(x)={1\over n}\sum_{k=1}^n
x({k\over n})$.
}
\begin{enumerate}
    \item \question{Calculer $\Vert\mu\Vert$ et $\Vert\mu_n\Vert$.}
    \item \question{Montrer que $\mu_n(x)$ converge vers $\mu(x)$ pour tout $x$ dans $E$, mais
que $\Vert\mu-\mu_n\Vert=2$.}
\end{enumerate}
}
