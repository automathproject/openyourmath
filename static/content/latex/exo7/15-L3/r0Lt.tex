\chapitre{Espace métrique complet, espace de Banach}
\sousChapitre{Espace métrique complet, espace de Banach}

\contenu{
\texte{
On considère $E$ l'espace des séries convergentes, et on pose
$$\Vert\xi\Vert=\sup_n\vert\sum_{k=1}^n\xi_k\vert$$
}
\begin{enumerate}
    \item \question{Vérifier que ceci définit une norme sur $E$ pour laquelle il est complet.}
    \item \question{L'espace $l^1$ des séries absolument convergentes est un sous-espace de
$E$; montrer que les normes $\Vert.\Vert$ et $\Vert.\Vert_1$
 ne sont pas équivalentes sur $l^1$ (en considérant une série de terme général
$\xi_k={{(-1)^k}\over {k^\alpha}}$.)}
    \item \question{Montrer que $l^1$ est dense dans $(E,\Vert.\Vert)$.}
\end{enumerate}
}
