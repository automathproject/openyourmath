\chapitre{Espace vectoriel normé}
\sousChapitre{Espace vectoriel normé}

\contenu{
\texte{
Soit $(E,N)$ un espace vectoriel normé. Montrer :
$$
\forall
(x,y)\in(E\setminus\{0\})^2,\;N(x-y)\geq\frac{1}{2}\sup(N(x),N(y))\cdot
N\left(\frac{x}{N(x)}-\frac{y}{N(y)}\right).
$$
}
}
