\begin{enumerate}
    \item \question{Soit $X$ un espace $T_2$ et $K$ un sous ensemble quasi
compact. Montrer que $K$ est fermé.}
    \item \question{Donner la définition d'un espace $T_4$.
Montrer qu'un espace $X$ est $T_4$ si et seulement si~:
pour tout fermé $A$ contenu dans un ouvert $U$ il
existe un ouvert $V$ tel que $A \subset V \subset
\overline{V} \subset U$.

espace Hausdorff.}
    \item \question{Montrer que $\{\, x\in X \mid f(x) = g(x) \,\}$
est fermé.}
    \item \question{Montrer que si $f$ et $g$ co\"\i ncident
sur un ensemble dense dans $X$, alors $f=g$.}
