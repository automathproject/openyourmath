\contenu{
\texte{
Dans $\Rr^2$ on considère l'ensemble $A$ des points dont une coordonnée au
moins est irrationnelle.
}
\begin{enumerate}
    \item \question{Soit  $\alpha\in {\Rr\backslash \Qq}$; décrire l'ensemble $A\cap\{(x,y)\in
{\Rr^2}, x=\alpha\}.$}
    \item \question{Montrer que $A$ est connexe par arcs (plus précisément deux points de $A$
peuvent être reliés par une ligne polygonale).}
\end{enumerate}
}
