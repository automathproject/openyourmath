\contenu{
\texte{
Soit $f : \Cc\to\Cc$, $z=x+iy\mapsto x^3y^2+ix^2y^3$ et $g=f_{\Rr} : \Rr^2\to\Rr^2$ l'application réelle associée.
}
\begin{enumerate}
    \item \question{En quels points de $\Rr^2$, $g$ est-elle différentiable ?}
    \item \question{En quels points de $\Rr^2$, $g$ vérifie-t-elle les équations de Cauchy-Riemann ?}
    \item \question{En quels points de $\Cc$, $f$ est-elle $\Cc$-dérivable ?}
    \item \question{En quels points de $\Cc$, $f$ est-elle holomorphe ?}
\end{enumerate}
}
