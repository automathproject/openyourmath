\contenu{
\texte{
Trouver des domaines de définition $\Omega $ (le plus grand
possible) et des fonctions holomorphes $f=u+iv$ sur $\Omega $  étant donné
la partie réelle $u$ ou la partie imaginaire $v$.
}
\begin{enumerate}
    \item \question{$u=x^2-y^2+5x+y-{y\over x^2+y^2}$}
    \item \question{$u=e^x(x\cos y-y\sin y)+2\sin x\sh y+x^3-3xy^2+y$}
    \item \question{$v=\ln{(x^2+y^2)}+x-2y$}
\end{enumerate}
}
