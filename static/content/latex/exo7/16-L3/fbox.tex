\chapitre{Formule de Cauchy}
\sousChapitre{Formule de Cauchy}

\contenu{
\texte{
Soit \(U\subset \C\) un ouvert. Soit \(L\subset \C\) une droite. Soit \(f:U\to \C\) une fonction continue telle que \(f|_{U\setminus L}\) est holomorphe. Montrer que \(f\) est holomorphe.
}
\indication{Utiliser le théorème de Morera et faire une disjonction de cas selon la position du triangle par rapport à \(L\).}
}
