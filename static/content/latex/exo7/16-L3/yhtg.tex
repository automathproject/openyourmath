\contenu{
\texte{
En évaluant $\int_Ce^z\ dz$ sur le cercle unité, montrer que
$$\int_0^{2\pi}e^{\cos\theta }\cos{(\theta +\sin\theta )}\ d\theta =
\int_0^{2\pi}e^{\cos\theta }\sin{(\theta +\sin\theta )}\ d\theta =
0$$
}
}
