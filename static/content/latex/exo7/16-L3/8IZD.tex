\texte{
Soit $\mathbb{H}ox{Log} z$ la détermination principale du logarithme dans 
$\Cc^-$.
}
\begin{enumerate}
    \item \question{On considère $z=e^{2i\pi\over3}$; les nombres complexes $\mathbb{H}ox{Log} (z^2)$ et $2\mathbb{H}ox{Log}
z$ sont-ils égaux ?}
    \item \question{On considère $z=e^{3i\pi\over4}$; les nombres complexes $z^{2i}$, $(z^2)^i$ et $(z^i)^2$ sont-ils égaux ?}
\end{enumerate}
}
