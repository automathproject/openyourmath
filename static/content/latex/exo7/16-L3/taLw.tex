\chapitre{Formule de Cauchy}
\sousChapitre{Formule de Cauchy}

\contenu{
\texte{
On considère la série entière $f(z)=\sum_{n\ge 1} {z^n\over n}$.
}
\begin{enumerate}
    \item \question{Montrer que $f$ admet un prolongement analytique sur un ouvert que
l'on déterminera.}
    \item \question{Déterminer le développement en série entière de ce prolongement en
$i$. Quel est son rayon de convergence ?}
\end{enumerate}
}
