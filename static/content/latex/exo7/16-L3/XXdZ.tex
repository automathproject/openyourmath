\contenu{
\texte{
Si $f$ est une fonction méromorphe sur un ouvert $D$ de $\Cc$, on notera $P(f)$ le lieu de ses pôles.
}
\begin{enumerate}
    \item \question{Soit $D$ un ouvert de $\Cc$ et $f,g$ deux applications méromorphes sur $D$.
Montrer qu'on a équivalence entre
\begin{enumerate}}
    \item \question{$f=g$}
    \item \question{$\{z\in D-P(f)-P(g), f(z)=g(z)\}$ a un point d'accumulation dans $D$.}
\end{enumerate}
}
