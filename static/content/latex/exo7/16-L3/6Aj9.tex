\chapitre{Fonction logarithme et fonction puissance}
\sousChapitre{Fonction logarithme et fonction puissance}

\contenu{
\texte{
On considère $f(z)=\sqrt{z^2}=(z^2)^{1/2}$ définie à l'aide de la
détermination principale du logarithme.
}
\begin{enumerate}
    \item \question{Trouver le domaine de définition et donner une expression
explicite de $f$.}
    \item \question{Peut-on prolonger $f$ par continuité sur un ouvert plus grand ?}
    \item \question{Mêmes questions en prenant la détermination ${\rm log}z=\mathrm{Log}
z+2i\pi$.}
\end{enumerate}
}
