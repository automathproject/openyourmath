\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Les formes bilinéaires symétriques données par les formes quadratiques suivantes sur $\mathbb{F}_7^3$ sont-elles équivalentes ?
$$q(x,y,z)=x^2+6y^2+2z^2$$
et $$Q(x,y,z)=xy+4z^2.$$}
    \item \question{Montrer que deux formes quadratiques équivalentes sur un espace $E$ prennent les mêmes valeurs dans $k$.}
    \item \question{La forme $q$ prend-elle toutes les valeurs de $\mathbb{F}_7$ ? Vérifier qu'elle prend les valeurs $3$ et $5$.}
\end{enumerate}
