\contenu{
\texte{
Soit $K$ un corps et $\overline{K}$ une cl\^oture algébrique de $K$. Soient $a,b\in \overline{K}\setminus K$. Etablir l'équivalence des conditions :
}
\begin{enumerate}
    \item \question{il existe un automorphisme $\phi $ de $\overline{K}$ au dessus de $K$ tel que $\phi (a)=b$ ;}
    \item \question{$a$ et $b$ ont le même polyn\^ome minimal $f(X)\in K[X]$.}
\end{enumerate}
}
