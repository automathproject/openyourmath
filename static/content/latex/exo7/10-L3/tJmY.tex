\contenu{
\texte{
On dit que $\quad a \mod n$ est inversible si 
il existe $b \mod n$ tel que $ab \equiv 1 \mod n$.
}
\begin{enumerate}
    \item \question{Trouver  tous les \'el\'ements inversibles modulo $5$, $6$, $9$, $11$.}
    \item \question{Trouver $\hbox{pgcd} (107, 281)$  et sa representation lin\'eaire
en utilisant {\it l'algorithme d'Euclide}.}
    \item \question{Trouver l'inverse de $107\mod 281$ et l'inverse de $281 \mod 107$.}
    \item \question{Montrer que $\quad a \mod n$ est inversible ssi $a$ et $n$
sont premiers entre eux.}
\end{enumerate}
}
