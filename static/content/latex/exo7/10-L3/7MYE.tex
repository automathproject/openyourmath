\contenu{
\texte{
Soit $K\subset E$ une extension finie.
}
\begin{enumerate}
    \item \question{Montrer que la trace $Tr^E_K$ est une forme linéaire sur $E$ en tant que $K$-espace vectoriel.}
    \item \question{Calculer la norme $N^E_K(\alpha \beta )$ en fonction de $N^E_K(\alpha )$ et de  $N^E_K(\beta )$. Calculer  $N^E_K(a)$, où $a\in K$.}
\end{enumerate}
}
