\contenu{
\texte{
Les \'el\'ements inversibles d'un anneau $A$
forment le groupe multiplicatif $(A^{\times}, \cdot)$.
}
\begin{enumerate}
    \item \question{Trouver $A^{\times}$ pour les anneaux 1. et 2. 
de l'exercice \ref{exodessus}.}
    \item \question{Trouver le groupe $\Zz[\sqrt{-1}]^{\times}$ en utilisant 
la norme complexe.}
    \item \question{Montrer que le groupe $\Zz[\sqrt{2}]^{\times}$ est infini.}
\end{enumerate}
}
