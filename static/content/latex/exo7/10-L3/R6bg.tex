\texte{
Dans le groupe symétrique $S_4$ trouver les sous-groupes suivants
}
\begin{enumerate}
    \item \question{le sous-groupe des éléments $\sigma$ tels que l'image par $\sigma$ de l'ensemble $\{ 1,2 \}$ est $\{ 1,2 \}$.}
    \item \question{le sous-groupe des éléments $\tau $ tels que si $a\equiv b$ mod $2$ alors $\tau(a)\equiv \tau(b)$ mod $2$. ({\it{Indication}} : $(13)(24)$ fait partie de ce sous-groupe).}
\end{enumerate}
}
