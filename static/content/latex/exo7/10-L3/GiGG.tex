\contenu{
\texte{
Soit $f(x)\in A[x]$ un polyn\^ome
sur un anneau $A$. Supposons que $(x-1)\,|\,f(x^n)$. Montrer que
$(x^n-1)\,|\,f(x^n)$.
}
\reponse{
Notons $f(x^{n})=P(x-1)$. Alors $f(1)=0\cdot P(1)=0$ et donc $(x-1)|f$.
  Notons $f=Q(x-1)$. On a alors $f(x^{n})=Q(x^{n})(x^{n}-1)$. $(x^{n}-1)$
  divise bien $f$.
}
}
