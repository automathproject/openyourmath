\contenu{
\texte{
Trouver  toutes les solutions des \'equations :
}
\begin{enumerate}
    \item \question{$ax+b=c\ $ ($a,b,c\in K$, $K$ est un corps);}
\reponse{une seule solution $x=a^{-1}(c-b)$}
    \item \question{$2x\equiv 3 \mod 10\ \ $  et  $\ 2x\equiv 6 \mod 10$ dans l'anneau 
$\Zz_{10}=\Zz/10\Zz$.}
\reponse{pas de solution, et deux solutions. Attention, dans $\Zz/10\Zz$, on ne
    peut pas inverser $2$. Ecrire $2x=3+10k$ pour obtenir que $2|3$, et
    $2x=6+10k$ pour simplifier par $2$... dans $\Rr$.}
\end{enumerate}
}
