\texte{
\label{ptfermat}
Soit $p$ un nombre premier et $a$ un nombre premier \`a $p$.
Montrer que :
}
\begin{enumerate}
    \item \question{$am\equiv an \mod p$ ssi  $m\equiv n \mod p$;}
    \item \question{La suite $a,\ 2a, \ 3a,\dots , (p-1)a \mod p$
est une permutation de la suite $1,\ 2, \ 3,\dots , (p-1) \mod p$;}
    \item \question{$a^{p-1}\equiv 1 \mod p$.}
\end{enumerate}
}
