\texte{

}
\begin{enumerate}
    \item \question{Soit $G$ le groupe engendré par deux éléments $a,b$, défini par les relations $a^m=b^2=e,\; (ab)^2=e$, où $m\in \N^*\cup \{\infty  \}$. Montrer que $G\simeq D_m$, où $D_m$ est le groupe diédral de degré $m$.}
    \item \question{Montrer que pour tout groupe $H$ engendré par deux éléments $\alpha,\; \beta$, qui vérifient les relations $\alpha^m=\beta^2=e,\; (\alpha\beta)^2=e$ il existe un epimorphisme de $D_m$ dans $H$.}
\end{enumerate}
