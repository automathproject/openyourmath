\contenu{
\texte{
Développer $(a+b+c+d)^2$ et $(a+b+c)^3$.
}
\reponse{
$$(a+b+c+d)^2=a^2+b^2+c^2+d^2+2(ab+ac+ad+bc+bd+cd)$$ et

$$(a+b+c)^3=a^3+b^3+c^3+3(a^2b+ab^2+a^2c+ac^2+b^2c+bc^2)+6abc.$$
}
}
