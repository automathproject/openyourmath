\texte{
Soit $f : {[0,1]} \to {[0,1]}$ croissante.
On note $A = \{ x \in {[0,1]}$ tq $f(x) \le x \}$.
}
\begin{enumerate}
    \item \question{D{\'e}montrer que $A$ n'est pas vide.}
    \item \question{D{\'e}montrer que $f(A) \subset A$.}
    \item \question{Soit $a = \inf(A)$. Montrer que $f(a)$ minore $A$.}
    \item \question{En d{\'e}duire que $f(a) = a$.}
\end{enumerate}
}
