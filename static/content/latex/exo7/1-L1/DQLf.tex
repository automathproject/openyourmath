\texte{
Soit $f \in \mathcal{L}(E)$ tel que $f^3 = \mathrm{id}_E$.
}
\begin{enumerate}
    \item \question{Montrer que $\mathrm{Ker}(f-\mathrm{id}) \oplus \Im(f-\mathrm{id}) = E$.}
    \item \question{Montrer que $\mathrm{Ker}(f-\mathrm{id}) = \Im(f^2 + f + \mathrm{id})$
 et $\mathrm{Im}(f-\mathrm{id}) = \mathrm{Ker}(f^2 + f + \mathrm{id})$.}
\end{enumerate}
}
