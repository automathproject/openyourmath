\chapitre{Polynôme, fraction rationnelle}
\sousChapitre{Pgcd}

\contenu{
\texte{
Trouver tous les polynômes $U$ et $V$ de ${\bf R}[X]$ tels que $AU+BV$ soit un
pgcd de~$A$ et~$B$ avec
$A=X^4-2X^3-2X^2+10X-7$ et $B=X^4-2X^3-3X^2+13X-10$.
}
}
