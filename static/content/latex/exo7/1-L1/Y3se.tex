\contenu{
\texte{
Soit $f : E \to F$ une application linéaire et $E_1$, $E_2$ deux
sous-espaces vectoriels de $E$, $F_1$, $F_2$ deux sous-espaces vectoriels
de $F$.
Que pouvez-vous-dire de $f(E_1+E_2)$, $f(E_1\cap E_2)$, $f^{-1}(F_1+F_2)$, $f^{-1}(F_1\cap F_2)$ ?
}
}
