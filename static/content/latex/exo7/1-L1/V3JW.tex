\contenu{
\texte{
Soit $M \in \mathcal{M}_n(K)$ {\it non scalaire\/} telle que $\mathrm{tr} M = 0$.
}
\begin{enumerate}
    \item \question{Montrer qu'il existe une matrice colonne $X_1$ telle que $MX_1$ ne soit
    pas colinéaire à $X_1$.}
    \item \question{En déduire que $M$ est semblable à une matrice
    $N = \begin{pmatrix} 0      &\dots  \cr
                   \vdots &M_1    \cr
    \end{pmatrix}$
    où $M_1 \in \mathcal{M}_{n-1}(K)$ et $\mathrm{tr} M_1 = 0$.}
    \item \question{Montrer que $M$ est semblable à une matrice à diagonale nulle.}
    \item \question{Montrer qu'il existe
    $A,B \in \mathcal{M}_n(K)$ telles que $M = AB-BA$.}
\end{enumerate}
}
