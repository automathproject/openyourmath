\contenu{
\texte{
On considère la suite réelle $(u_n)_{n\in\N}$ définie par $u_0=2$ et pour tout $n\in\N,\: u_{n+1}=\sqrt{u_n}$.
}
\begin{enumerate}
    \item \question{Montrer que pour tout $n\in\N$, $u_n>1$.}
    \item \question{Montrer que pour tout réel $a\in]1;+\infty[$, on a $\frac{1}{\sqrt a+1} \leq \frac12$.}
    \item \question{En déduire que pour tout $n\in\N$, on a $u_{n+1}-1\leq \frac12\left(u_n-1\right)$.}
    \item \question{Montrer que pour tout $n\in\N$, $u_n-1 \leq \left(\frac12\right)^n$. En déduire la limite de la suite $(u_n)$.}
\end{enumerate}
}
