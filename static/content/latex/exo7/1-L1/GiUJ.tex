\contenu{
\texte{
Soit $X$ un ensemble. Si $A \subset X$ on note $\chi _A$ la fonction caract\'eristique
associ\'ee. Montrer que ${\Phi} : {\mathcal{P} (X)} \to {\mathcal{F} (X, \left\{ 0, 1 \right\})}$, ${A} \mapsto {\chi _A}$ est bijective.
}
}
