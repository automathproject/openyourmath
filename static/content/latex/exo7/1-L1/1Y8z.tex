\contenu{
\texte{
Comment faut-il choisir $m \in \C$ pour que l'{\'e}quation :
$z^2 - (2+im)z - (1+im) = 0$ admette deux racines imaginaires conjugu{\'e}es ?
}
\reponse{
$m=2i$.
}
