\contenu{
\texte{
Soient $P,Q \in { K[X]}$.
}
\begin{enumerate}
    \item \question{Montrer que si $P(X^n)$ est divisible par $X - 1$, alors $P$ est divisible par $X-1$\quad $(n\in\N)$.}
    \item \question{Montrer que si $P(X^3) + XQ(X^3)$ est divisible par $X^2+X+1$, alors $P$ et $Q$
    sont divisibles par $X-1$.}
\end{enumerate}
}
