\texte{
Soient $E,F$ deux $ K$-ev et $f \in \mathcal{L}(E,F)$.
}
\begin{enumerate}
    \item \question{Montrer que si $H$ est un sev de $E$, alors \\
    $\dim f(H) = \dim H - \dim (H\cap\mathrm{Ker} f)$.}
    \item \question{Montrer que si $K$ est un sev de $F$, alors\\
    $\dim f^{-1}(K) = \dim (K\cap\Im f) + \dim(\mathrm{Ker} f)$.}
\end{enumerate}
