\texte{
Soit $ A\in M_2(\R ) .$ On nomme commutant de $
A $ et on note
 $ C(A) $ l'ensemble des $ B\in M_2(\R ) $ telles que $ AB=BA .$
}
\begin{enumerate}
    \item \question{Montrer que  $ C(A) $ et un sous espace vectoriel de $ M_2(\R ) .$}
    \item \question{Montrer que pour tout $ k \in \N  , A^k\in C(A) .$}
\end{enumerate}
}
