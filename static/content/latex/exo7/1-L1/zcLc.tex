\sousChapitre{Binôme de Newton et combinaison}

\contenu{
\texte{
En utilisant la formule du binôme de Newton, montrer que
$\sum\limits_{k = 0}^n (-1)^k C_n^k = 0$. En d\'eduire la valeur de
$\sum\limits_{0 \leq 2k \leq n} C_n^{2k}$.
}
}
