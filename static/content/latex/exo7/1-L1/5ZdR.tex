\chapitre{Application linéaire}
\sousChapitre{Définition}

\contenu{
\texte{
Soit l'application $f~:\mathbb{R}^3\rightarrow \mathbb{R}^3$ donn\'ee par~:
$$
f(x, y, z) = (x + 2y + z, 2x  + y + 3z, -x -y -z).
$$
}
\begin{enumerate}
    \item \question{Justifier que $f$ est lin\'eaire.}
    \item \question{Donner la matrice de $A$ dans la base canonique de $\mathbb{R}^3$.}
    \item \question{\begin{enumerate}}
    \item \question{[(a)] D\'eterminer une base et la dimension du noyau de $f$, not\'e $\ker f$.}
    \item \question{[(b)] L'application $f$ est-elle injective~?}
\end{enumerate}
}
