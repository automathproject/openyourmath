\contenu{
\texte{
D\'emontrer que $C_n^kC_{n-k}^{p-k}=C_p^kC_n^p$ (pour $0\le k \le p \le n$).
En d\'eduire que
$$ \sum_{k=0}^{n}C_n^kC_{n-k}^{p-k}=2^pC_n^p.$$
}
}
