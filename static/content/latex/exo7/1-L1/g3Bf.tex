\chapitre{Nombres complexes}
\sousChapitre{Autre}

\contenu{
\texte{
On d\'efinit une fonction $f$ de $\mathbb C-\{i\}$ dans $\mathbb C-\{1\}$
en posant $$f(z)={z+i \over z-i}.$$
}
\begin{enumerate}
    \item \question{On suppose $z$ r\'eel. Quel est le module de $f(z)$~?}
    \item \question{Trouver les nombres complexes $z$ tels que $f(z)=z$.}
\end{enumerate}
}
