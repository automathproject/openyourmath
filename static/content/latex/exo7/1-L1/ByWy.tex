\contenu{
\texte{
Soit $f \in \mathcal{L}(E)$ tel que $f^3 = 0$.
}
\begin{enumerate}
    \item \question{Montrer que $\mathrm{rg} f + \mathrm{rg} f^2 \le \dim E$.}
    \item \question{Montrer que $2\mathrm{rg} f^2 \le \mathrm{rg} f$
     (appliquer le théorème du rang à $f_{|\Im f}$).}
\end{enumerate}
}
