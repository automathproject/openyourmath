\texte{
Soient $a$ et $b$ deux entiers tels que $a\ge b\ge 1$ et $\mathrm{pgcd}(a,b)=1$.
}
\begin{enumerate}
    \item \question{Montrer que $\text{ pgcd}(a+b,a-b)=1$ ou $2$,}
    \item \question{Si pgcd$(a,b)=1 $, montrer que pgcd$(a+b,ab)=1$,}
    \item \question{Si pgcd$(a,b)=1 $, montrer que pgcd$(a+b,a^2+b^2)=1$ ou $2$.}
\end{enumerate}
