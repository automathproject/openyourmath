\chapitre{Espace vectoriel}
\sousChapitre{Définition, sous-espace}

\contenu{
\texte{
Soient $F,G,H$ trois sous-espaces d'un espace vectoriel $E$.
Comparer $F \cap (G + (F \cap H))$ et $(F \cap G) + (F \cap H)$.
}
\reponse{
Il y a égalité.
}
}
