\texte{
\'Etudier la liberté des familles suivantes :
}
\begin{enumerate}
    \item \question{$E=\{  {\text{fcts}} : \R    \to    \R \}$, ${\cal F}=(\sin, \cos)$.}
    \item \question{$E=\{ {\text{fcts}} : {\R^{+*}} \to \R \}$, ${\cal F}=(f_a : x \longmapsto x^a),\ a\in\R$.}
    \item \question{$E=\{ {\text{fcts}} : \R      \to  \R \}$, ${\cal F}=(f_a : x \longmapsto |x-a|),\ a\in\R$.}
\end{enumerate}
}
