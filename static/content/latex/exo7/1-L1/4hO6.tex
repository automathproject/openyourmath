\texte{
Soit $P_n$ une assertion dépendant de $n \in \N$ telle que:
}
\begin{enumerate}
    \item \question{$P_0$ est vraie;}
    \item \question{$\forall n \in \N, P_n \Rightarrow 
(P_{n+3} \text{ et } P_{n+4})$.}
\indication{Considérer le sous-ensemble (sous-monoïde) de $\N$ suivant :
\[ \{3k+4l\mid k,l \in \N\}.\]
Pour le cas général, considérer le pgcd de $a$ et $b$.}
\end{enumerate}
}
