\chapitre{Matrice}
\sousChapitre{Propriétés élémentaires, généralités}

\contenu{
\texte{
On rappelle qu'une matrice carr\'ee $A$ d'ordre $n$ est dite
{\em sym\'etrique} si $a_{i,j}=a_{j,i}, \forall\, i,j$, et {\em
antisym\'etrique} si $a_{i,j}=-a_{j,i}$.
}
\begin{enumerate}
    \item \question{Combien y a-t-il de matrices antisym\'etriques diagonales\,?}
    \item \question{Montrer que $A {\,}^t A$ est sym\'etrique pour toute matrice
carr\'ee $A$.}
    \item \question{Montrer que si $A, B$ sont sym\'etriques, leur produit
$C = AB$ est sym\'etrique si et seulement si $AB = BA$. Que dire
si elles sont antisym\'etriques\,? Si l'une est sym\'etrique et
l'autre antisym\'etrique\,?}
    \item \question{Soit $P$ un polyn\^ome. Montrer que si $A$ est
sym\'etrique, $P(A)$ l'est aussi. Que dire si $A$ est
antisym\'etrique\,?}
\end{enumerate}
}
