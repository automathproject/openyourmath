\contenu{
\texte{
Soient $U, V, W$ des s.e.v. d'un e.v. $E$, v\'erifiant $(I) :   U\cap V=\{0\}=(U+V)\cap W$.
}
\begin{enumerate}
    \item \question{D\'emontrer que $V\cap W=\{0\}=U\cap(V+W)$.\\}
    \item \question{Montrer que $(I)$ \'equivaut \`a $$(II) :  (\forall x \in U+V+W)    (\exists ! (u,v,w) \in U\times V \times W)   (x=u+v+w).$$}
\end{enumerate}
}
