\contenu{
\texte{
$A$ et $B$ étant des parties d'un ensemble~$E$,
démontrer les lois de Morgan~:
$$\complement A \cup\complement B = \complement(A \cap B) \quad\mbox{et}
\quad\complement A \cap\complement B = \complement(A \cup B).$$
}
}
