\texte{
Soit $E =  K_3[X]$,
$F = \{ P \in E$ tq $P(0) = P(1) = P(2) = 0 \}$,
$G = \{ P \in E$ tq $P(1) = P(2) = P(3) = 0 \}$,
et $H = \{ P \in E$ tq $P(X) = P(-X) \}$.
}
\begin{enumerate}
    \item \question{Montrer que $F \oplus G = \{ P \in E$ tq $P(1) = P(2) = 0 \}$.}
    \item \question{Montrer que $F \oplus G \oplus H = E$.}
\end{enumerate}
}
