\texte{
Soit $A \in \mathcal{M}{n+1}(\Q)$ telle que $a_{ij} = C_{j-1}^{i-1}$.
Interpréter $A$ comme la matrice d'un endomorphisme simple de $\Q_n[X]$.
En déduire la matrice $A^{-1}$.
}
}
