}
\begin{enumerate}
    \item \question{Soit $n$ un entier naturel dont le reste de la division euclidienne par 5 vaut
$2$ ou $3$, montrer que $n^2 +1$ est divisible par $5$.}
    \item \question{Montrer que pour tout entier naturel $n$, l'entier $n^5-n$ est
divisible par $5$.}
\end{enumerate}
