\contenu{
\texte{
D\'eterminer les entiers $n \in \Nn$ tels que :
}
\begin{enumerate}
    \item \question{$n|n + 8$.}
    \item \question{$n-1|n + 11$.}
    \item \question{$n-3|n^3-3$.}
\end{enumerate}
}
