\texte{
Comparer les diff\'erentes phrases (sont-elles \'equivalentes, contraires, quelles sont
celles qui impliquent les autres...)
}
\begin{enumerate}
    \item \question{$ (\forall x) (\exists y)/ (x \leq y)$.}
    \item \question{$ (\forall x) (\forall y) (x \leq y)$.}
    \item \question{$ (\exists x) (\exists y)/(x \leq y)$.}
    \item \question{$ (\exists x)/(\forall y) (x \leq y)$.}
    \item \question{$ (\exists x)/(\forall y) (y<x)$.}
    \item \question{$ (\exists x) (\exists y)/ (y<x)$.}
    \item \question{$ (\forall x) (\exists y)/(x  =  y)$.}
\end{enumerate}
}
