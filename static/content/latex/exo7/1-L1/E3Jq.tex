\contenu{
\texte{
On définit une suite $(C_n)_{n\in\N}$ par $C_0=1$ et pour tout naturel $n$,  
$C_{n+1} = \sum_{k=0}^{n}C_kC_{n-k}$.
}
\begin{enumerate}
    \item \question{Calculer les cinq premiers termes de la suite;}
    \item \question{Montrer par récurrence que pour tout $n\geq 0$,  $C_n \geq 2^{n-1}$;}
    \item \question{Montrer par récurrence forte que pour tout $n\geq 0$,  $C_n \geq 3^{n-2}$;}
    \item \question{Tenter de montrer par une récurrence similaire à la précédente que pour tout $n\geq 0$,  $C_n \geq 4^{n-2}$. À quel endroit ceci échoue-t-il ? Pourquoi est-il heureux que cela échoue ?}
\end{enumerate}
}
