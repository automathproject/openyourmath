\contenu{
\texte{
Soit $E$ un $ K$-ev de dimension finie, $F_1$, $F_2$, \dots, $F_n$
des sev de $E$ tels que $F_1 + \dots + F_n = E$.
Montrer qu'il existe des sev $G_1 \subset F_1$, \dots, $G_n \subset F_n$
tels que $G_1 \oplus G_2 \oplus \dots \oplus G_n = E$.
}
}
