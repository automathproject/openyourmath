\texte{
Soit $A = \{ m/n \in \Q \text{ tel que } n \text{ est impair} \}$.
}
\begin{enumerate}
    \item \question{Montrer que $A$ est un sous-anneau de $\Q$.}
    \item \question{Chercher les {\'e}l{\'e}ments inversibles dans $A$.}
    \item \question{Montrer que les id{\'e}aux de $A$ sont tous principaux engendr{\'e}s
    par les nom\-bres de la forme $2^k$, $k\in \N$.}
\end{enumerate}
}
