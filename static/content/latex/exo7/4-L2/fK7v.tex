\chapitre{Groupe, anneau, corps}
\sousChapitre{Anneau}

\contenu{
\texte{
Soient les ensembles
$${L=\left\{
\left (
\begin{array}{ll}
x&0\\
0&0
\end{array}
\right )\in \mathcal{M}_2(\Rr): x\in \Rr
\right\}\text{ et }
M=\left\{
\left (
\begin{array}{ll}
x&x\\
-x&-x
\end{array}
\right )\in \mathcal{M}_2(\Rr): x\in \Rr
\right\}}$$

\'Etudier si, munis des lois usuelles, $L$ et  $M$ sont des anneaux, des corps.
}
}
