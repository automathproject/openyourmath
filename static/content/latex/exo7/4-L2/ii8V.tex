\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Sous-espace stable}

\contenu{
\texte{
Soit $E$ un $\C$-ev de dimension finie et $u \in \mathcal{L}(E)$.
}
\begin{enumerate}
    \item \question{Montrer qu'il existe une droite vectorielle stable par $u$.}
    \item \question{Montrer qu'il existe un hyperplan stable par $u$
    (considérer $\Im(u-\lambda\mathrm{id})$ où $\lambda$ est une valeur propre de $u$).}
    \item \question{Donner un exemple où ces propriétés sont en défaut pour un $\R$-ev.}
\end{enumerate}
}
