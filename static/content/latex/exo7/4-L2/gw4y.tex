\sousChapitre{Espace vectoriel euclidien de dimension 3}

\contenu{
\texte{
Soit $E$ un espace vectoriel euclidien orienté de dimension 3.
Trouver les applications $f \in \mathcal{L}(E)$ vérifiant :
}
\begin{enumerate}
    \item \question{$f(\vec u\wedge\vec v) = f(\vec u)\wedge f(\vec v)$.}
    \item \question{$f(\vec u\wedge\vec v) =-f(\vec u)\wedge f(\vec v)$.}
\end{enumerate}
}
