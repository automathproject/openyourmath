\chapitre{Groupe, anneau, corps}
\sousChapitre{Anneau}

\contenu{
\texte{
Un id{\'e}al $I$ d'un anneau $A$ est dit premier si~: $\forall\ x,y\in A,\ xy\in I \Rightarrow  x\in I$ ou $y\in I$.
}
\begin{enumerate}
    \item \question{Quels sont les id{\'e}aux premiers de~$\Z$~?}
    \item \question{Montrer que si $A$ est commutatif non nul et si tous les id{\'e}aux de~$A$ sont premiers alors $A$ est un corps.}
\reponse{
$A$ est int{\`e}gre car $\{0\}$ est premier et si $a\in A\setminus\{0\}$
alors $a\times a\in(a^2)$ qui est premier donc $a^2$ divise $a$ d'o{\`u} $a$ est inversible.
}
\end{enumerate}
}
