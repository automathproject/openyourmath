\contenu{
\texte{
Soit $A=(a_{ij})_{\substack{1\leq i\leq n\\1\leq j\leq n}}$ une matrice symétrique réelle.
Montrer que ses valeurs propres $\lambda_{1},\ldots,\lambda_{n}$ vérifient
$$
  \sum_{i=1}^{n}{\lambda_{i}}=\sum_{i=1}^{n}{a_{i,j}^{2}}
$$
}
}
