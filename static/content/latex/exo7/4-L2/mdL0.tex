\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Polynôme caractéristique, théorème de Cayley-Hamilton}

\contenu{
\texte{
Soit $C = \begin{pmatrix} a_1\cr \vdots\cr a_n \end{pmatrix} \in \mathcal{M}_{n,1}(\R)$ et $M = C{}^tC$.
}
\begin{enumerate}
    \item \question{Chercher le rang de $M$.}
\reponse{1 si $C \ne 0$, 0 si $C = 0$.}
    \item \question{En déduire le polynôme caractéristique de $M$.}
\reponse{$\dim(E_0) \ge n-1  \Rightarrow  X^{n-1}$ divise $\chi_M  \Rightarrow 
             \chi_M = (-1)^n(X^n - (a_1^2+\dots+a_n^2)X^{n-1})$.}
    \item \question{$M$ est-elle diagonalisable ?}
\reponse{Oui.}
\end{enumerate}
}
