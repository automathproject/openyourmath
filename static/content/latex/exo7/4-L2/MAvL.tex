\begin{enumerate}
    \item \question{Soient $A,B \in \mathcal{M}_n(\R)$ telles que $AB = BA$.
    Montrer que $\det(A^2 + B^2) \ge 0$.}
    \item \question{Chercher $A,B$ ne commutant pas telles que $\det(A^2 + B^2) < 0$.}
\reponse{
$A = \begin{pmatrix} 1        &\lambda \cr
                            \lambda  &1       \cr \end{pmatrix}$,\quad
             $B = \begin{pmatrix} 0        &\lambda \cr
                            -\lambda &0       \cr \end{pmatrix}$,\quad
             $|\lambda| > \frac 12$.
}
\end{enumerate}
