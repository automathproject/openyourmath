\contenu{
\texte{
Soit $f:\Rr^3\times\Rr^3\to \Rr$ d\'efinie de la
mani\`ere suivante : si $u=(x,y,z)$ et
$u'=(x',y',z')$ alors
$${f(u,u')=2xx'+yy'+2zz'+xy'+yx'+xz'+zx'+yz'+zy'.}$$
}
\begin{enumerate}
    \item \question{Montrer que $f$ est un produit scalaire sur l'espace vectoriel canonique $\Rr^3$.}
    \item \question{Soit $P$ le sous-espace vectoriel de $\Rr^3$ d'\'equation cart\'esienne $2x-y+z=0$.
    \begin{enumerate}}
    \item \question{D\'eterminer l'orthogonal du sous-espace vectoriel $P$.}
    \item \question{D\'eterminer un sous-espace vectoriel de $\Rr^3$ dont l'orthogonal est $P$.}
\end{enumerate}
}
