\texte{
$A$ est nomm\'e id\'eal de $A$
lorsque pour tout $x\in J$ et tout $a\in A$ le produit $ax$ appartient \`a $J$.
}
\begin{enumerate}
    \item \question{Trouver tous les id\'eaux d'un corps $\mathbb{K}$.}
    \item \question{Montrer que tout id\'eal de $\Zz$ est de la forme $a\Zz ,$ 
o\`u $a\in \Zz .$}
    \item \question{On note $D $ l'ensemble des rationnels $x$ tels que il 
existe $k\in \Nn $ tel que $x10^{k}\in \Zz .$
Montrer que tout id\'eal de $D $ est de la forme
$aD$ o\`u $a\in D.$}
\end{enumerate}
