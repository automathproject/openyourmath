\contenu{
\texte{
Montrer que si $ n\in \Nn^{*}, A\in M_{n}({\Rr})$ :
$$\mathrm{rg}(A)=n\Rightarrow \mathrm{rg}(\mathrm{Com}(A))=n ; $$
$$\mathrm{rg}(A)=n-1\Rightarrow \mathrm{rg}(\mathrm{Com}(A))=1 ; $$
$$\mathrm{rg}(A)\leq n-2\Rightarrow \mathrm{rg}(\mathrm{Com}(A))=0. $$
}
}
