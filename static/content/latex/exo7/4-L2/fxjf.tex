\texte{
Soient $  A  $ et $  B  $ deux matrices sym\'etriques positives.
Soit $  k \in \N ^* .$
}
\begin{enumerate}
    \item \question{Montrer que tout vecteur propre de $  A^k  $ est vecteur propre de $  A  .$}
    \item \question{Si $  A^k = B^k   $ alors $  A = B  .$}
    \item \question{Que se passe-t-il sans l'hypoth\`ese $  A  $ et $  B  $ sym\'etriques positives ?}
\end{enumerate}
}
