\sousChapitre{Produit scalaire, norme}

\contenu{
\texte{
Soit $E=\mathcal{C}^1([0,1],\R)$ et $\varphi (f,g)= \int_{[0,1]} fg+f'g'$.
}
\begin{enumerate}
    \item \question{Montrer que $\varphi$ est un produit scalaire.}
    \item \question{Soit $V=\{f\in E\, |\, f(0)=f(1)=0\}$ et $W=\{f\in E \, |\, f''=f\}$. Montrer que $V$ et $W$
    sont supplémentaires orthogonaux et exprimer la projection orthogonale sur $W$.}
    \item \question{Soit $E_{\alpha \beta}=\{f\in E\, |\, f(0)=\alpha \text{ et } f(1)=\beta\}$. Déterminer
    $\inf\limits_{f\in E_{\alpha \beta}}  \int_{[0,1]} f^2+f'^2$.}
\reponse{
$\pi(f)(t) = f(0)\frac{\sh(1-t)}{\sh(1)} + f(1)\frac{\sh(t)}{\sh(1)}.$
L'inf est atteint pour la fonction~$f\in W$ telle
    que $f(0) = \alpha$ et $f(1) = \beta$, soit
    $f(t) = \alpha\frac{\sh(1-t)}{\sh(1)} + \beta\frac{\sh(t)}{\sh(1)}$
    et $inf=\frac{(\alpha^2+\beta^2)\ch(1) - 2\alpha\beta}{\sh(1)}$.
}
\end{enumerate}
}
