\chapitre{Groupe, anneau, corps}
\sousChapitre{Anneau}

\contenu{
\texte{
Soit $A$ un anneau. On appelle {\it caract{\'e}ristique de\/} $A$ l'ordre de
$1$ dans le groupe additif $(A,+)$.
On suppose $A$ de caract{\'e}ristique finie, $n$.
}
\begin{enumerate}
    \item \question{Montrer que : $\forall\ x \in A,\ nx = 0$.}
    \item \question{Si $A$ est int{\`e}gre, montrer que $n$ est un nombre premier.}
    \item \question{Si $A$ est int{\`e}gre et commutatif, montrer que $x  \mapsto x^n$ est un morphisme
    d'anneau.}
\end{enumerate}
}
