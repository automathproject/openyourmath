\chapitre{Groupe, anneau, corps}
\sousChapitre{Groupe de permutation}

\contenu{
\texte{
Soit $G$ un sous-groupe de $\mathcal{S}_n.$
}
\begin{enumerate}
    \item \question{Montrer que si $G$ est d'ordre impair alors $G$ ne contient aucune permutation impaire.}
    \item \question{Montrer que si $G$ contient au moins une permutation impaire,
alors
$G$ contient autant de permutations paires que de
permutations impaires.}
\end{enumerate}
}
