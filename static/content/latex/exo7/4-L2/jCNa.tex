\contenu{
\texte{
Soient $x_0$,$x_1$,...,$x_n$, $n+1$ réels distincts, et
  $y_0$,$y_1$,...,$y_n$, $n+1$ réels (distincts ou non).
  
  Montrer qu'il existe un unique polynôme $P$ tel que :
  $$
  \forall i\in\{0,...,n\}\quad P(x_i)=y_i
  $$
}
}
