\texte{
Soit $E$ un $ K$-ev de dimension finie et $F,G$ deux sev de $E$ tels que
$F\oplus G = E$.
}
\begin{enumerate}
    \item \question{Montrer que $F^\circ \oplus G^\circ = E^*$.}
    \item \question{Montrer que $F^\circ$ est naturellement isomorphe à $G^*$ et
 $G^\circ$ à $F^*$.}
\end{enumerate}
}
