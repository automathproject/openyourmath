\chapitre{Réduction d'endomorphisme, polynôme annulateur}
\sousChapitre{Sous-espace stable}

\contenu{
\texte{
Soient $E$ un $K$-espace vectoriel, $f$ et $g$ des
endomorphismes de $E$ tels que $f\circ g=g\circ f$
et $P$ un polyn\^ome de $K[X].$
}
\begin{enumerate}
    \item \question{Montrer que $P(g)$ et $f$ commutent.}
    \item \question{Montrer que le noyau et l'image de l'endomorphisme $P(g)$ sont stables par $f$. Donner des cas particuliers de cette situation.}
\end{enumerate}
}
