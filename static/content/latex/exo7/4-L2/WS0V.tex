\chapitre{Groupe, anneau, corps}
\sousChapitre{Groupe, sous-groupe}

\contenu{
\texte{
Si $G$ est un groupe, on appelle centre de $G$ et on note $Z (G)$ l'ensemble
$\left\{ x \in G / \forall y \in G, xy = yx\right\}$.
}
\begin{enumerate}
    \item \question{Montrer que $Z (G)$ est un sous-groupe de $G$.}
    \item \question{Montrer que $G$ est commutatif ssi $Z (G) = G$.}
    \item \question{Calculer $Z (\sigma_3)$.}
\end{enumerate}
}
