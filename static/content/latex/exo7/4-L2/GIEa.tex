\contenu{
\texte{
Soient $p,q$ deux projecteurs orthogonaux.
}
\begin{enumerate}
    \item \question{Montrer que $p\circ q \circ p$ est auto-adjoint.}
    \item \question{Montrer que
    $(\Im p + \mathrm{Ker} q) \mathop{\oplus}\limits^\perp (\mathrm{Ker} p \cap \Im q)  = E$.}
    \item \question{En déduire que $p\circ q$ est diagonalisable.}
\reponse{
$(p\circ q)_{|\Im p} = (p\circ q \circ p)_{|\Im p}$ est
             diagonalisable et $(p\circ q)_{|\mathrm{Ker} q + (\mathrm{Ker} p\cap \Im q)} = 0$
             donc tout vecteur de $E$ est somme de vecteurs propres pour
             $p\circ q$.
}
\end{enumerate}
}
