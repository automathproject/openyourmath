\texte{
Soit $G$ un groupe fini tel que : $\forall\ x \in G,\ x^2 = e$.
}
\begin{enumerate}
    \item \question{Montrer que $G$ est commutatif (consid{\'e}rer $(xy)(xy)$).}
    \item \question{Soit $H$ un sous-groupe de $G$ et $x \in G \setminus H$.
    On note $K$ le sous groupe engendr{\'e} par $H \cup \{x\}$.
    \par
    Montrer que $\mathrm{Card}\, K = 2\mathrm{Card}\, H$.}
    \item \question{En d{\'e}duire que $\mathrm{Card}\, G$ est une puissance de 2.}
\end{enumerate}
}
