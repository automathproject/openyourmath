\texte{
Soient $u,v$ deux endomorphisme d'un espace vectoriel $E$ non nul tels que
$u\circ v - v\circ u = \mathrm{id}_E$.
}
\begin{enumerate}
    \item \question{Simplifier $u^k\circ v - v\circ u^k$ pour $k\in\N$ puis
    $P(u)\circ v - v\circ P(u)$ pour $P\in K[X]$.}
    \item \question{Montrer que $u$ et $v$ n'ont pas de polyn√¥mes minimaux.}
\reponse{
Pour $P\in K[X]$ on a $P(u)\circ v - v\circ P(u) = P'(u)$.
}
\end{enumerate}
