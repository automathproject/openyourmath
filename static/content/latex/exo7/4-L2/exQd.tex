\texte{
Soit $E =  K_n[X]$ et $ u : E \to  E, P  \mapsto {X^nP(1/X).}$
}
\begin{enumerate}
    \item \question{Déterminer $u\circ u$. En déduire que $u$ est diagonalisable.}
    \item \question{Donner une base de vecteurs propres de $u$.}
\end{enumerate}
