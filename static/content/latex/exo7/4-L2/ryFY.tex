\texte{
Le but de cet exercice est de déterminer tous les morphismes d'anneaux de $\Z/10\Z$ dans un $\Z/n\Z$.
}
\begin{enumerate}
    \item \question{Montrer que tout morphisme de groupes de $\Z/10\Z$ dans un $\Z/n\Z$ est déterminé par l'image de $[1]_{10}$.}
    \item \question{Quelles sont les valeurs possibles pour l'image de $[1]_{10}$ par un morphisme d'anneaux de $\Z/10\Z$ dans un $\Z/n\Z$ ?}
    \item \question{Déterminer tous les morphismes d'anneaux de $\Z/10\Z$ dans un $\Z/n\Z$.}
\end{enumerate}
}
