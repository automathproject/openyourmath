\chapitre{Déterminant, système linéaire}
\sousChapitre{Système linéaire, rang}

\contenu{
\texte{
\'Etudier l'existence de solutions du système :
$$\left\{\begin{array}{ccccccccc} 4x &+&3y &+&2z &+& t &=& a \cr
             x &+&4y &+&3z &+&2t &=& b \cr
            2x &+& y &+&4z &+&3t &=& c \cr
            3x &+&2y &+& z &+&4t &=& d.\cr\end{array}\right.$$
}
\reponse{
Système de Cramer.
}
}
