\contenu{
\texte{
Soit $f\in\mathrm{End}(\R^{3})$ tel que $f^{2}=0$. Montrer qu'il existe $\alpha\in{(\R^{3})}^{*}$
et $v\in\R^{3}$ tels que
 $$
 \forall x\in \R^{3} \;\; f(x)=\alpha(x)\,v.
 $$
\textit{\small(Indication : commencer par montrer que $\mathrm{rg}(f)\leq 1$)}
}
}
