\chapitre{Groupe, anneau, corps}
\sousChapitre{Groupe, sous-groupe}

\contenu{
\texte{
Soient $ (G, \star)$ et $ (H, \triangle)$ deux groupes. On d\'efinit sur $G \times H$
la loi $\heartsuit$ par $ (x, y) \heartsuit (x', y') = (x \star x', y \triangle y')$.
}
\begin{enumerate}
    \item \question{Montrer que $ (G \times H, \heartsuit)$ est un groupe.}
    \item \question{Si $G$ est de cardinal 2, dresser la table de $G \times G$ et la reconna\^{\i}tre
parmi les exemples des exercices pr\'ec\'edents.}
\end{enumerate}
}
