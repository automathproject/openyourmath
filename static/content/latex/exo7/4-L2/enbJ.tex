\chapitre{Déterminant, système linéaire}
\sousChapitre{Autre}

\contenu{
\texte{
Soient $u,n \in \mathcal{L}(E)$ deux endomorphismes d'un $\C$-ev de dimension finie,
$u$ inversible, $n$ nilpotent, avec $u\circ n = n \circ u$.
}
\begin{enumerate}
    \item \question{Démontrer que $\det n = 0$.}
    \item \question{Chercher le polynôme caractéristique de $n$.
    En déduire que $\det(\text{id}_E + n ) = 1$.}
    \item \question{Démontrer que $\det( u + n ) = \det u$.}
\end{enumerate}
}
