\contenu{
\texte{
Soit $J = \begin{pmatrix}I_n&0\cr0&-I_p\cr\end{pmatrix}$, et $M = \begin{pmatrix}A&B\cr C&D\end{pmatrix}$
telle que ${}^tMJM = J$.
Montrer que $A$ et $D$ sont inversibles.
}
\reponse{
On a ${}^tAA - {}^tCC = I_n$.

Soit $X$ tel que $AX = 0$. Donc ${}^tXX = -{}^t(CX)(CX)$, donc $X = 0$.
}
}
