\texte{
Soit $G$ un groupe additif et $f : G \to {G'}$ un morphisme de groupes.
}
\begin{enumerate}
    \item \question{Montrer que pour tout sous-groupe $H$ de $G$ on a : $f^{-1}(f(H)) = H + \mathrm{Ker}\,f$.}
    \item \question{Montrer que pour tout sous-groupe $H'$ de $G'$ on a : $f(f^{-1}(H')) = H' \cap \Im\,f$.}
\end{enumerate}
}
