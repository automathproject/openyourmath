\chapitre{Espace euclidien, espace normé}
\sousChapitre{Projection, symétrie}

\contenu{
\texte{
Soit $E$ un espace euclidien de dimension $n$.
Soit $f \in {\cal O}(E)$, et $s$ une réflexion par rapport à un hyperplan $H$.
Soit $\vec u \in H^\perp, \vec u \ne \vec0$.
}
\begin{enumerate}
    \item \question{Montrer que $f\circ s \circ f^{-1}$ est aussi une symétrie et en donner la
    base.}
    \item \question{En déduire que $f$ et $s$ commutent si et seulement si $\vec u$ est vecteur
    propre de $f$.}
    \item \question{Quel est le centre de $O(E)$ ?}
\end{enumerate}
}
