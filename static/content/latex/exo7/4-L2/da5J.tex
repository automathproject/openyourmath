\contenu{
\texte{
Soit $E$ un $ K$-ev de dimension $n$ et $F$ un sev de $E^*$ dimension $p$.

On note $F^\bot = \{ \vec x \in E \text{ tq }
                     \forall\ f \in F \text{ on a } f(\vec x) = 0 \}$.
Chercher $\dim F^\bot$.
}
}
