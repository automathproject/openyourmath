}
\begin{enumerate}
    \item \question{$$\left\{ \begin{aligned}\dot3x + \dot7y &= \dot3\cr \dot6x - \dot7y &= \dot0.\end{aligned}\right.$$}
\reponse{$x=\dot{\overline{25}}$, $y=\dot{\overline{32}}$.}
    \item \question{$x^2 - \dot{\overline{31}}x + \dot{\overline{18}} = \dot0$.}
\reponse{$x=\dot{\overline{15}}$ ou $\dot{\overline{16}}$.}
\indication{\begin{enumerate}
\item
\item $\dot6^2 = -\dot1$.
\end{enumerate}}
\end{enumerate}
