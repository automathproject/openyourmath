\chapitre{Probabilité discrète}
\sousChapitre{Probabilité et dénombrement}

\contenu{
\texte{
Les trois mousquetaires (donc quatre personnes) ont mélangé leurs bottes dans 
le couloir de l'auberge. D'Artagnan se lève le premier et prend deux bottes 
au hasard. Calculer la probabilité pour que :
}
\begin{enumerate}
    \item \question{Les deux bottes soient les siennes.}
    \item \question{Les deux bottes forment une paire (une paire est la réunion d'un pied droit et d'un pied gauche).}
    \item \question{Les deux bottes soient deux pieds droits.}
    \item \question{Les deux bottes appartiennent à deux personnes différentes.}
\end{enumerate}
}
