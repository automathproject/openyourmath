\chapitre{Probabilité discrète}
\sousChapitre{Probabilité et dénombrement}

\contenu{
\texte{
On lance deux dés à 6 faces. Décrire l'ensemble $\Omega$ des résultats
possibles et la probabilité $P$ associée à cette expérience. Donner la
probabilité d'obtenir:
}
\begin{enumerate}
    \item \question{un double,}
    \item \question{au plus un nombre pair,}
    \item \question{exactement un nombre pair,}
    \item \question{deux nombres qui se suivent.}
\end{enumerate}
}
