\chapitre{Fonction convexe}

\contenu{
\texte{
Soit $f:{\Rr}\rightarrow {\Rr }$ continue telle que :
$$\forall (x,y)\in {\Rr}^{2}, f\left( \frac{x+y}{2}\right) \leq \frac{%
f(x)+f(y)}{2}. $$
Montrer que $f$ est convexe.
}
}
