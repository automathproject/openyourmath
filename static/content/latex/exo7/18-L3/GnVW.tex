\texte{
Montrer que :
$$\forall n\in \Nn^{*},\forall (x_{1},...,x_{n})\in \left( {\Rr}^{+*}\right)^{n},
1+\left( \prod\limits_{k=1}^{n}x_{k}\right) ^{\frac{1}{n}}\leq
\left( 1+\prod\limits_{k=1}^{n}x_{k}\right) ^{\frac{1}{n}}. $$
}
