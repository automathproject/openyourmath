\contenu{
\texte{
Soit $f:I\rightarrow {\Rr}$ convexe ou $I$ est un intervalle ouvert de ${\Rr}$,
d\'{e}rivable en $x_{0}\in I$ et telle que $f^{\prime}(x_{0})=0.$
Montrer que $x_{0}$ minimise $f$ sur $I.$
}
}
