\chapitre{Fonction convexe}
\sousChapitre{Fonction convexe}

\contenu{
\texte{
Soit $g\in C({\Rr},{\Rr})$, montrer que $g$ est convexe si et seulement
si :
$$\forall h\in CM([0,1],{\Rr}), g\left( \int_{0}^{1}h\right) \leq
\int_{0}^{1}g(h). $$
}
}
