\chapitre{Différentiabilité, calcul de différentielles}
\sousChapitre{Différentiabilité, calcul de différentielles}

\contenu{
\texte{

}
\begin{enumerate}
    \item \question{Que peut-on dire de la diff\'erentiabilit\'e de
l'application $f:\Rr^2 \to \Rr$ d\'efinie par $f(x_1,x_2) = \|x\|
_\infty = \max{(|x_1|, |x_2|)}$?}
    \item \question{G\'en\'eraliser ceci \`a
$f:{\cal F} \to \Rr$, $f(x) = \|x\| _\infty$, avec ${\cal F} =
\Rr^n$ ou ${\cal F}$ l'ensemble des suites convergentes vers zero.}
\end{enumerate}
}
