\chapitre{Equation différentielle}
\sousChapitre{Résolution d'équation différentielle du premier ordre}

\contenu{
\texte{
Pour chacune des \'equations diff\'erentielles qui suit : \'ecrire la solution
passant par le point M(.,.) et tracer sommairement le graphe de la solution.
}
\begin{enumerate}
    \item \question{$y'+2xy=0,\ \ \ M=(0,1) $,}
    \item \question{$y'+y \tan x = \sin x \cos x \ \ \ M=(\frac{\pi}{4},0)$,}
    \item \question{$x(x^2-1)y'+2y=x^2$, On d\'eterminera $a,b,c \in \Rr$ tels que
$\frac{1}{x(x^2-1)}=\frac{a}{x}+\frac{b}{x-1}+\frac{c}{x+1}$.}
\end{enumerate}
}
