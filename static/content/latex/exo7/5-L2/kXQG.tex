\texte{
Soient $E = \mathcal{C}([a,b]\to\R)$, $(f_n)$ une suite de fonctions de $E$
et $f \in E$.
Comparer les {\'e}nonc{\'e}s :
$$1 : \|f_n-f\|_1 \xrightarrow[n\to\infty]{} 0
  \qquad
  2 : \|f_n-f\|_2 \xrightarrow[n\to\infty]{}0  \qquad
  3 : \|f_n-f\|_\infty \xrightarrow[n\to\infty]{} 0.$$
}
}
