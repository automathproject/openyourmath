\contenu{
\texte{
Soit $A$ une partie d'un espace vectoriel norm\'e $E$. On
rappelle que la fronti\`ere de $A$ est l'ensemble $\mathrm{Fr}(A)=\bar A-
\stackrel{\circ}{A}$. Montrer que~:
}
\begin{enumerate}
    \item \question{$ \mathrm{Fr}(A)=\{x\in E \mid \forall \epsilon>0, B(x,\epsilon)\cap A
\neq\emptyset \text{\ et\ } B(x,\epsilon)\cap C_A\neq\emptyset\}$}
    \item \question{$\mathrm{Fr}(A)=\mathrm{Fr}(C_A)$}
    \item \question{$A$ est ferm\'e si et seulement si $\mathrm{Fr}(A)$ est inclus dans $A$.}
    \item \question{$A$ est ouvert si et seulement si $\mathrm{Fr}(A)\cap A=\emptyset$.}
\end{enumerate}
}
