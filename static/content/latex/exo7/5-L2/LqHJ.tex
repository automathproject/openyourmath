\contenu{
\texte{
En posant $t=\arctan x,$ r\'{e}soudre :
$$y''(x)+\frac{2x}{1+x^{2}}y^{\prime }(x)+\frac{y(x)}{(1+x^{2})^{2}}=0. $$
}
\reponse{
R\'{e}ponse : $x\rightarrow \frac{\lambda x+\mu
}{\sqrt{1+x^{2}}},\left( \lambda ,\mu \right) \in {\Rr}^{2}.$
}
}
