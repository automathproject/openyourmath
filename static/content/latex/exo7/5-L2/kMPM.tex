\texte{
Chercher $\lim_{x\to0}  \int_{t=x}^{2x} \frac{\cos t\ln(1+t^2)}{\sin^2t\sh t}\,d t$.
}
\reponse{
$= \int_{t=x}^{2x} \left({\frac1t - \frac{5t}6 +  o(t)}\right)d t \to \ln 2$.
}
}
