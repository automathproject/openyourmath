\texte{

}
\begin{enumerate}
    \item \question{Soit $I(x) =  \int_{t=0}^{+\infty} e^{-t^2}\cos(2xt)\,d t$.
    Prouver que $I$ est de classe $\mathcal{C}^1$ sur $\R$.}
    \item \question{Chercher une relation simple entre $I$ et $I'$.}
    \item \question{En déduire la valeur de $I(x)$
    $\Bigl($on admet que $I(0) = \frac{\sqrt\pi}2\Bigr)$.}
\reponse{
$I'(x) = -2xI(x)$.
$I(x) = \frac{\sqrt\pi}2 e^{-x^2}$.
}
\end{enumerate}
