\texte{
Soit $E$ un evn de dimension infinie.
}
\begin{enumerate}
    \item \question{Soit $F$ un sev de dimension finie et $a\in E\setminus F$.
  \begin{enumerate}}
    \item \question{Montrer qu'il existe $b\in F$ tel que $\|a-b\| = d(a,F)$.}
    \item \question{En d{\'e}duire qu'il existe $c\in E$ tel que $\|c\| = 1 = d(c,F)$.}
\end{enumerate}
