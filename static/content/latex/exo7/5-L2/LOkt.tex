\chapitre{Suite et série de fonctions}
\sousChapitre{Continuité, dérivabilité}

\contenu{
\texte{
Soient ${f_n} : D \to \R$ des fonctions continues convergeant vers une
fonction continue $f$ et $(x_n)$ une suite d'éléments de $D$ convergeant vers
$x \in D$.
}
\begin{enumerate}
    \item \question{Si les fonctions $f_n$ convergent uniformément, montrer que
    $f_n(x_n) \to f(x)$ lorsque $n\to\infty$.}
    \item \question{Donner un contre-exemple lorsqu'il y a seulement convergence simple.}
\end{enumerate}
}
