\contenu{
\texte{
On admet que $\sum_{n=1}^\infty \frac 1{n^2} = \frac{\pi^2}6$.
Calculer $ \int_{t=0}^1 \ln(t)\ln(1-t)\,d t$.
}
\reponse{
$=\sum_{n=1}^\infty  \int_{t=0}^1 -\frac{t^n\ln t}n\,d t
          = \sum_{n=1}^\infty \frac1{n(n+1)^2} = 2-\frac{\pi^2}6$.
}
}
