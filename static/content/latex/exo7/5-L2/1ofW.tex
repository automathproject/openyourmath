\texte{
Trouver les solutions r\'eelles des \'equations diff\'erentielles
suivantes :
}
\begin{enumerate}
    \item \question{$y'(t)+2y(t) = 0$ ;}
    \item \question{$\displaystyle{\frac{dx}{dt}-x=0}$ ;}
    \item \question{$y'(x)+2y(x) = 0 \text{ avec } (y-y')(0)=0$.}
\end{enumerate}
}
