\begin{enumerate}
    \item \question{$y' = y(1+y)$.}
\reponse{$y = -1 + \frac1{1-\lambda e^x}$ ou $y = -1$.}
    \item \question{$y' = \sin x\cos y$.}
\reponse{$y \equiv 2\arctan(\lambda e^{-\cos x}) - \frac\pi2 (\mathrm{mod}\,\pi)$.}
    \item \question{$2yy'\sqrt x = \sqrt{y^2-1}$.}
\reponse{$y = \pm\sqrt{1+(\sqrt x+\lambda)^2}$ ou $y = \pm1$.}
    \item \question{$1+xy' = e^y$, condition initiale : $y(1)=1$.}
\reponse{$y = -\ln\bigl(1-x(1-1/e)\bigr)$.}
    \item \question{$y' = \sqrt{|y|}$ : étudier les problèmes de raccordements.}
\reponse{$y=\left(\lambda+\frac x2\right)\left|\lambda+\frac x2\right|$ ou $y = 0$.}
\end{enumerate}
