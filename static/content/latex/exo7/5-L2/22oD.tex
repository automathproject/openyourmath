\begin{enumerate}
    \item \question{$y'' = \sin y$, $y(0) = \frac\pi2$, $y'(0) = \sqrt 2$.}
\reponse{$y=4\arctan((\sqrt2-1) e^x)$.}
    \item \question{$2(2a-y)y'' = y'^2$.}
\reponse{$y=2a-(\lambda x+\mu)^{2/3}$.}
    \item \question{$yy''=y'^2-y^2$ : poser $z=y'/y$.}
\reponse{$y=\exp\left(-\frac {x^2}2 + \lambda x + \mu\right)$.}
\end{enumerate}
