\sousChapitre{Intégrale de Riemann dépendant d'un paramètre}

\contenu{
\texte{
On pose $\varphi(x) =  \int_{t=0}^1 e^{-x/t}\,d t$.
}
\begin{enumerate}
    \item \question{Montrer que $\varphi$ est de classe $\mathcal{C}^\infty$ sur $\R^{+*}$.}
    \item \question{Vérifier que $\varphi''(x) = \frac{e^{-x}}x$.}
\end{enumerate}
