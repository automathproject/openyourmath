\texte{
R\'{e}soudre sur ${\Rr }$ :
\begin{eqnarray*}
x\frac{dy(x)}{dx}-2y(x) &=&x^{4}, \\
y"(x)-4y(x) &=&4e^{-2x}, \\
y"(x)-2y^{\prime }(x)+y(x) &=&e^{x}\sin x.
\end{eqnarray*}
}
