\chapitre{Fonction de plusieurs variables}
\sousChapitre{Différentiabilité}

\contenu{
\texte{
Soit $f:\rbrack 0,1\lbrack \times \rbrack 0,1\lbrack \to \R 
$, 
$$
f(x,y)=\left\{ 
        \begin{array}{cc}
        x(1-y) & \mbox{ si }x\leq y \\
        y(1-x) & \mbox{ si }x > y
        \end{array}
        \right . 
$$
Etudier la continuit\'e et la diff\'erentiabilit\'e de $f$.
}
}
