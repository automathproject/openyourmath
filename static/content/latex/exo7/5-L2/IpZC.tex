\contenu{
\texte{
Développer $f(x) = \frac x{1-x-x^2}$ en série entière en utilisant
la relation : $(1-x-x^2)f(x) = x$.
}
\reponse{
$a_n = a_{n-1} + a_{n-2}  \Rightarrow 
          a_n = \frac1{\sqrt5}\left(\left(\frac{1+\sqrt5}2\right)^n
              - \left(\frac{1-\sqrt5}2\right)^n\right)$.
}
}
