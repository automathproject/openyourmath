\texte{
Soit $A$ une partie compacte d'un evn $E$ et $f : A \to A$ telle que :
$\forall\ x,y \in A,\ d(f(x),f(y)) \ge d(x,y)$.
}
\begin{enumerate}
    \item \question{Soit $a \in A$ et $(a_n)$ la suite d{\'e}finie par :
    $a_0 = a$, $a_{n+1} = f(a_n)$. Montrer que $a$ est valeur d'adh{\'e}rence de la
    suite $(a_n)$.}
    \item \question{Soient $a,b \in A$. Montrer que $d(f(a),f(b)) = d(a,b)$.}
    \item \question{Montrer que $f(A) = A$.}
\end{enumerate}
}
