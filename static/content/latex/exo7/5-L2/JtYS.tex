\chapitre{Equation différentielle}
\sousChapitre{Résolution d'équation différentielle du deuxième ordre}

\contenu{
\texte{
R\'esoudre en posant $z (t) = y (e^t)$ ou $y (-e^t)$ suivant le signe de $x$, les
\'equations diff\'erentielles (d'Euler) suivantes :
}
\begin{enumerate}
    \item \question{$x^2y''-2y = x$.}
    \item \question{$x^2y'' + xy' + y = x \ln \left|x\right|$.}
\end{enumerate}
}
