\chapitre{Fonction de plusieurs variables}

\contenu{
\texte{
On consid\`ere une suite $(u_n)_n$, de terme g\'en\'eral $u_n\in\R^2$.
}
\begin{enumerate}
    \item \question{Donner la d\'efinition de convergence pour une telle suite. (Ceci est une question de cours!)}
    \item \question{Soit la suite de terme g\'en\'eral $u_n=(\tanh(n), \cos(n) \exp(-n^2))$. \'Etudier sa convergence.}
\end{enumerate}
}
