\texte{
Soit $E$ un evn et $F$ un sev de $E$.
}
\begin{enumerate}
    \item \question{Montrer que $\overline{F}$ est un sev de $E$.}
    \item \question{Si $E$ est de dimension finie, montrer que $F = \overline F$.}
    \item \question{Dans le cas g{\'e}n{\'e}ral, montrer que $\mathring F = \varnothing$ ou $F = E$.}
\end{enumerate}
