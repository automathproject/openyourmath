\texte{
R\'esoudre et raccorder \'eventuellement :
}
\begin{enumerate}
    \item \question{$xy'-2y = x^4$.}
    \item \question{$x (1 + x^2)y' = y$.}
    \item \question{$ (x^2 + 1)y' + (x-1)^2y = x^3-x^2 + x + 1$.}
    \item \question{$ (e^x-1)y' + (e^x + 1)y = 3 + 2e^x$.}
\end{enumerate}
}
