\texte{
Calculer les limites : $\lim_{x\to0} \int_x^{3x} \frac t{\tan^2t}\,dt$
    et $\lim_{x\to0} \frac1{x^3} \int_0^x \frac {t^2}{t+e^{3t}}\,dt$.
}
