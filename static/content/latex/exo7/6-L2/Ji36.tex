\chapitre{Courbes planes}
\sousChapitre{Courbes paramétrées}

\contenu{
\texte{
Pour chacune des courbes suivantes, d\'eterminer la
tangente en tout point, les points d'inflexion et de
rebroussement, les branches infinies, les points doubles\,;
construire et tracer la courbe.
}
\begin{enumerate}
    \item \question{$\displaystyle{ x = \sin 4t, \quad y = \cos 3t}$\,;}
    \item \question{$\displaystyle{ x = t - \sin t, \quad y = 1-\cos t}$\,;}
    \item \question{$\displaystyle{ x = \cos^3 t, \quad y = \sin ^3 t}$\,;}
    \item \question{$\displaystyle{ x = \cos t, \quad y = {\sin^2 t \over 2+\sin t}}$\,;}
    \item \question{$\displaystyle{ x = {t^3 \over 1-t^2}, \quad y = {1+t\over
(1-t)^2}}$\,;}
    \item \question{$\displaystyle{ x = \cos^3 t+\sin t, \quad y = \sin^3t+\cos t}$\,;}
    \item \question{$\displaystyle{ x = 3 \cos t -2\sin^3 t, \quad y = \cos4t}$.}
\end{enumerate}
}
