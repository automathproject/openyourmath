\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Soit $E$ un espace affine et $F$ et $G$ deux sous espaces affines de direction supplémentaire dans $\vec{E}$.
}
\begin{enumerate}
    \item \question{Décrire la symétrie $s$ par rapport à $F$ parallèlement à $G$. Est-elle bijective ?}
    \item \question{Soit $M$ un point de $E$. Quel est le milieu du segment $[M,s(M)] $?}
    \item \question{Déterminer $ker(\vec{s}-Id_{\vec{E}})$ et $ker(\vec{s}+Id_{\vec{E}})$.}
    \item \question{Soit $\vec{u}$ un vecteur de $G$. Déterminer l'application $s\circ t_{\vec{u}}\circ s^{-1}$.}
\end{enumerate}
}
