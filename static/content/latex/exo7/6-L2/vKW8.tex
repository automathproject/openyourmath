\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Soit $E$ un espace affine et $f$ une application affine de $E$ dans $E$.
}
\begin{enumerate}
    \item \question{Montrer que si $f\circ f = f$ alors $f$ est une projection affine.}
    \item \question{Montrer que si $f\circ f = id$ alors $f$ est une sym\'etrie affine.}
\end{enumerate}
}
