\texte{
Soient $M$, $M'$ deux points d'une parabole $\cal P$ tels que
$(MM')$ passe par le foyer $F$. Quels sont :
}
\begin{enumerate}
    \item \question{L'enveloppe des droites $(MM')$ ?}
\reponse{$F$.}
    \item \question{Le lieu des milieux des segments $[M,M']$ ?}
\reponse{$M = (t^2/2p,t)$, $M' = (t'^2/2p,t')  \Rightarrow  tt' = -p^2  \Rightarrow 
    \begin{cases} x = \frac p4\left(u^2+\frac1{u^2}\right)\cr
           y = \frac p2\left(u-\frac1u\right)\end{cases}$
    avec $u = \frac tp$. (Parabole passant par $F$)}
    \item \question{L'enveloppe des m√©diatrices de $[M,M']$ ?}
\reponse{$\begin{cases} x = \frac{3p}4\left(u^2+\frac1{u^2}\right)\cr
                     y = -\frac p4\left(u-\frac1u\right)^3.\cr\end{cases}$}
\end{enumerate}
}
