\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Soient $A, B, C$ trois points non align\'es d'un plan affine.
}
\begin{enumerate}
    \item \question{Soit $f$ une application
affine telle que $f (A) = A$, $f (B) = B$ et $f (C) = C$. Montrer que $f = id$.}
    \item \question{Soient $f$ et $g$ affines telles que $f (A) = g (A)$, $f (B) = g (B)$ et $f (C) = g (C)$.
Que peut-on dire ?}
    \item \question{Soit $f$ affine telle que $f (A) = B$, $f (B) = C$ et $f (C) = A$. Que peut-on dire ?}
\end{enumerate}
}
