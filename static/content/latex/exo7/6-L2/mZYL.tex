\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Propriétés des triangles}

\contenu{
\texte{
Soit $ABC$ un triangle, et $M \in (ABC)$. On note $\alpha,\beta,\gamma$ les
aires des triangles $MBC$, $MCA$, $MAB$.
}
\begin{enumerate}
    \item \question{On suppose que $M$ est dans l'enveloppe convexe de $\{A,B,C\}$.\par
    Montrer que : $(\alpha = \beta = \gamma) \iff
                   M = \text{Bar}(A:1,\ B:1,\ C:1)$.}
    \item \question{Quels sont tous les points du plan $(ABC)$ tels que
    $\alpha = \beta = \gamma$ ?}
\reponse{
$G$ et les symétriques de $A,B,C$ par rapport aux milieux des côtés opposés.
}
\end{enumerate}
}
