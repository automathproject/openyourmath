\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
Soit $A$ et $B$ deux points d'un plan affine euclidien.
Déterminer suivant la valeur de la constante~$k$,
}
\begin{enumerate}
    \item \question{l'ensemble des points $M$ du plan tels que $MA^2+MB^2=k$}
    \item \question{l'ensemble des points $M$ du plan tels que $MA^2-MB^2=k$}
    \item \question{et l'ensemble des points $M$ du plan tels que $MA/MB=k$}
    \item \question{l'ensemble des points $M$ du plan tels que $MA/MB<k$}
\end{enumerate}
}
