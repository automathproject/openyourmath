\chapitre{Géométrie affine dans le plan et dans l'espace}
\sousChapitre{Géométrie affine dans le plan et dans l'espace}

\contenu{
\texte{
On se donne $2$ droites $D_1$ et $D_2$ ayant comme vecteurs directeurs respectifs
 $ \vec{u_1}$ et $\vec{u_2}$.
}
\begin{enumerate}
    \item \question{Perpendiculaire commune \`a ces deux droites.

\begin{enumerate}}
    \item \question{On suppose que $ \vec{u_1}$ et $\vec{u_2}$ ne sont pas colin\'eaires et 
  on note $  \vec{n}:=\vec{u_1} \wedge \vec{u_2} $.



\begin{enumerate}}
    \item \question{Montrer que le plan $P_1$ contenant $D_1$ et admettant  $\vec{n}$ comme vecteur directeur et
le plan $P_2$ contenant $D_2$ et admettant  $\vec{n}$ comme vecteur directeur se coupent en une droite
$\Delta$.}
    \item \question{Montrer que $\Delta$ est une perpendiculaire commune \`a $D_1$ et $D_2$
 (c'est \`a dire $\Delta$ coupe  $D_1$  et $D_2$, et 
 est  orthogonale \`a $D_1$ et \`a  $D_2$).}
    \item \question{Montrer que $\Delta$ est la seule  perpendiculaire commune \`a $D_1$ et $D_2$.}
\end{enumerate}
}
